198  double
199  value(const std::map<field, double> &fields_value) override
200  {
201  const double &phase_order_cahn_hilliard =
202  fields_value.at(field::phase_order_cahn_hilliard);
203  return mobility_cahn_hilliard_constant *
204  (1 - phase_order_cahn_hilliard * phase_order_cahn_hilliard) *
205  (1 - phase_order_cahn_hilliard * phase_order_cahn_hilliard);
206  }

213  void
214  vector_value(const std::map<field, std::vector<double>> &field_vectors,
215  std::vector<double> &property_vector) override
216  {
217  const std::vector<double> &phase_order_cahn_hilliard =
218  field_vectors.at(field::phase_order_cahn_hilliard);
219  for (unsigned int i = 0; i < property_vector.size(); ++i)
220  property_vector[i] =
221  mobility_cahn_hilliard_constant *
222  (1 - phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]) *
223  (1 - phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]);
224  }

234  double

236  {

261  }
