272 ParticlePointLineForce<dim>::find_projection_point(const Point<3> &point_p,
273  const Point<3> &point_a,
274  const Point<3> &point_b)
275 {
276  Tensor<1, 3> vector_ab = point_b - point_a;
277  Tensor<1, 3> vector_ap = point_p - point_a;
278 
279  Point<3> projection =
280  point_a + ((vector_ap * vector_ab) / (vector_ab * vector_ab)) * vector_ab;
281 
282  return projection;
