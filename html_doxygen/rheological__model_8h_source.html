613 template <int dim>
614 inline double
615 calculate_shear_rate_magnitude(const Tensor<2, dim> shear_rate)
616 {
617  double shear_rate_magnitude = 0;
618  for (unsigned int i = 0; i < dim; ++i)
619  {
620  for (unsigned int j = 0; j < dim; ++j)
621  {
622  shear_rate_magnitude += (shear_rate[i][j] * shear_rate[j][i]);
623  }
624  }
625  shear_rate_magnitude = sqrt(0.5 * shear_rate_magnitude);
626  return shear_rate_magnitude;
627 }
