120 /**
121  * @brief Extrapolates vector of solution to time_vector[0] using previous solution times and previous solution
122  *
123  * @tparam The type of the variable being extrapolated (e.g. double, Tensor<1,dim>)
124  * @param time_vector The vector of times. The solution will be extrapolated to time 0. It should be of size number_of_previous_solutions+1
125  * @param solution_vector The vector of solutions. The solution will be extrapolated to index 0. It should be at least of size number_of_previous_solutions
126  * @param number_of_previous_solutions The number of previous solutions to use to extrapolate
127  * @param extrapolated_solution The vector of extrapolated solutions
128  */
129 
130 template <typename DataType>
131 inline void
132 bdf_extrapolate(const std::vector<double> &time_vector,
133  const std::vector<std::vector<DataType>> &solution_vector,
134  const unsigned int number_of_previous_solutions,
135  std::vector<DataType> &extrapolated_solution)
136 {
137  // If only one previous solution is used, we don't make a Lagrange
138  // polynomial but use the previous value
139  if (number_of_previous_solutions == 1)
140  {
141  for (unsigned int q = 0; q < extrapolated_solution.size(); ++q)
142  extrapolated_solution[q] = solution_vector[0][q];
143  return;
144  }
145 
146  // Otherwise we extrapolate with a Lagrange polynomial
147  for (unsigned int q = 0; q < extrapolated_solution.size(); ++q)
148  {
149  // Set extrapolated solution to zero
150  extrapolated_solution[q] = 0;
151 
152  // For all the previous solutions which we will use to extrapolate
153  for (unsigned int p = 0; p < number_of_previous_solutions; ++p)
154  {
155  // Factor is the weight of the previous solution fixed by the Lagrange
156  // Polynomial
157  double factor = 1;
158  for (unsigned int k = 0; k < number_of_previous_solutions; ++k)
159  {
160  if (p != k)
161  {
162  // The time vector also contains the time of the solution to
163  // be extrapoled to, hence the +1
164  factor *= (time_vector[0] - time_vector[k + 1]) /
165  (time_vector[p + 1] - time_vector[k + 1]);
166  }
167  }
168  // Calculate the contribution of this time step to the extrapolated
169  // solution
170  extrapolated_solution[q] += solution_vector[p][q] * factor;
171  }
172  }
173 }
