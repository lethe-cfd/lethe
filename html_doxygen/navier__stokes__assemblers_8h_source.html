231  inline Tensor<1, dim>
232  get_kinematic_viscosity_gradient(
233  const Tensor<2, dim> &velocity_gradient,
234  const Tensor<3, dim> &velocity_hessians,
235  const double shear_rate_magnitude,
236  const double grad_kinematic_viscosity_shear_rate) const
237  {
238  // Calculates an approximation of the shear rate magnitude gradient using
239  // the derived form, since it does not change with rheological models
240  Tensor<1, dim> grad_shear_rate;
241  for (unsigned int d = 0; d < dim; ++d)
242  {
243  if (dim == 2)
244  {
245  for (unsigned int k = 0; k < dim; ++k)
246  {
247  grad_shear_rate[d] +=
248  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
249  shear_rate_magnitude;
250  }
251  grad_shear_rate[d] +=
252  (velocity_gradient[0][1] + velocity_gradient[1][0]) *
253  (velocity_hessians[0][d][1] + velocity_hessians[1][d][0]) /
254  shear_rate_magnitude;
255  }
256  else
257  {
258  for (unsigned int k = 0; k < dim; ++k)
259  {
260  grad_shear_rate[d] +=
261  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
262  shear_rate_magnitude +
263  (velocity_gradient[(k + 1) % dim][(k + 2) % dim] +
264  velocity_gradient[(k + 2) % dim][(k + 1) % dim]) *
265  (velocity_hessians[(k + 1) % dim][d][(k + 2) % dim] +
266  velocity_hessians[(k + 2) % dim][d][(k + 1) % dim]) /
267  shear_rate_magnitude;
268  }
269  }
270  }
271  return grad_shear_rate * grad_kinematic_viscosity_shear_rate;
272  };
