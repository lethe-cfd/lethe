142  if (rpt_parameters.tuning_param.cost_function_type ==
143  Parameters::RPTTuningParameters::CostFunctionType::larachi)
144  {
145  for (unsigned int i = 0; i < measured_counts.size(); i++)
146  {
147  cost_function +=
148  std::pow((calculated_counts[i] - measured_counts[i]) /
149  (calculated_counts[i] + measured_counts[i]),
150  2);
151  }
152  }
153  else if (rpt_parameters.tuning_param.cost_function_type ==
154  Parameters::RPTTuningParameters::CostFunctionType::l1)
155  {
156  for (unsigned int i = 0; i < measured_counts.size(); i++)
157  cost_function += abs(calculated_counts[i] - measured_counts[i]);
158 
159  cost_function /= measured_counts.size();
160  }
161  else if (rpt_parameters.tuning_param.cost_function_type ==
162  Parameters::RPTTuningParameters::CostFunctionType::l2)
163  {
164  for (unsigned int i = 0; i < measured_counts.size(); i++)
165  cost_function +=
166  std::pow((calculated_counts[i] - measured_counts[i]), 2);
167

169  }
