143  template <typename VectorType>
144  void
145  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
146  const VectorType &current_solution,
147  const std::vector<VectorType> &previous_solutions)
148  {
149  fe_values_vof.reinit(cell);
150  this->quadrature_points = fe_values_vof.get_quadrature_points();
151  auto &fe_vof = fe_values_vof.get_fe();
152 
153  if (dim == 2)
154  this->cell_size = std::sqrt(4. * cell->measure() / M_PI) / fe_vof.degree;
155  else if (dim == 3)
156  this->cell_size =
157  pow(6 * cell->measure() / M_PI, 1. / 3.) / fe_vof.degree;
158 
159  fe_values_vof.get_function_values(current_solution,
160  this->present_phase_values);
161  fe_values_vof.get_function_gradients(current_solution,
162  this->phase_gradients);
163  fe_values_vof.get_function_laplacians(current_solution,
164  this->phase_laplacians);
165 
166  fe_values_vof.get_function_gradients(previous_solutions[0],
167  this->previous_phase_gradients);
168 
169 
170  // Gather previous vof values
171  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
172  {
173  fe_values_vof.get_function_values(previous_solutions[p],
174  this->previous_phase_values[p]);
175  }
176 
177  for (unsigned int q = 0; q < this->n_q_points; ++q)
178  {
179  this->JxW[q] = fe_values_vof.JxW(q);
180 
181  for (unsigned int k = 0; k < this->n_dofs; ++k)
182  {
183  // Shape function
184  this->phi[q][k] = fe_values_vof.shape_value(k, q);
185  this->grad_phi[q][k] = fe_values_vof.shape_grad(k, q);
186  this->hess_phi[q][k] = fe_values_vof.shape_hessian(k, q);
187  this->laplacian_phi[q][k] = trace(this->hess_phi[q][k]);
188  }
189  }
190  }

203  template <typename VectorType>
204  void
205  reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
206  const VectorType &current_solution,
207  const std::vector<VectorType> &previous_solutions,
208  const Parameters::ALE<dim> &ale)
209  {
210  fe_values_fd.reinit(cell);
211 
212  fe_values_fd[velocities_fd].get_function_values(current_solution,
213  velocity_values);
214  fe_values_fd[velocities_fd].get_function_gradients(
215  current_solution, velocity_gradient_values);
216 
217  for (unsigned int q = 0; q < this->n_q_points; ++q)
218  {
219  this->velocity_divergences[q] =
220  trace(this->velocity_gradient_values[q]);
221  }
222 
223  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
224  {
225  fe_values_fd[velocities_fd].get_function_values(
226  previous_solutions[p], this->previous_velocity_values[p]);
227  }
228 
229  if (!ale.enabled())
230  return;
231 
232  // ALE enabled, so extract the ALE velocity and subtract it from the
233  // velocity obtained from the fluid dynamics
234  Tensor<1, dim> velocity_ale;
235  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
236  ale.velocity;
237  Vector<double> velocity_ale_vector(dim);
238 
239  for (unsigned int q = 0; q < n_q_points; ++q)
240  {
241  velocity_ale_function->vector_value(quadrature_points[q],
242  velocity_ale_vector);
243  for (unsigned int d = 0; d < dim; ++d)
244  velocity_ale[d] = velocity_ale_vector[d];
245 
246  velocity_values[q] -= velocity_ale;
247 
248  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
249  {
250  this->previous_velocity_values[p][q] -= velocity_ale;
251  }
252  }
253  }
