166  inline Tensor<1, 3>
167  find_projection(const Tensor<1, 3> &vector_a, const Tensor<1, 3> &vector_b)
168  {
169  Tensor<1, 3> vector_c;
170  vector_c = ((vector_a * vector_b) / (vector_b.norm_square())) * vector_b;
171 
172  return vector_c;
173  }

227  inline void
228  apply_force_and_torque(
229  const std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
230  &forces_and_torques,
231  Tensor<1, 3> &particle_torque,
232  Tensor<1, 3> &particle_force,
233  const Point<3> &point_on_boundary,
234  int boundary_id = 0)
235  {
236  // Getting the values from the forces_and_torques tuple, which are: 1,
237  // normal force, 2, tangential force, 3, tangential torque and 4, rolling
238  // resistance torque
239  Tensor<1, 3> normal_force = std::get<0>(forces_and_torques);
240  Tensor<1, 3> tangential_force = std::get<1>(forces_and_torques);
241  Tensor<1, 3> tangential_torque = std::get<2>(forces_and_torques);
242  Tensor<1, 3> rolling_resistance_torque = std::get<3>(forces_and_torques);
243 
244  // Calculation of total force
245  Tensor<1, 3> total_force = normal_force + tangential_force;
246 
247  calculate_force_and_torque_on_boundary(boundary_id,
248  total_force,
249  point_on_boundary);
250 
251  // Updating the force of particles in the particle handler
252  // Since the forces were calculated on the wall, we use the -= operator
253  particle_force -= total_force;
254 
255  // Updating the torque acting on particles
256  // The torque was direcly calculated on the particle, thus we use the +=
257  // operator
258  particle_torque += tangential_torque + rolling_resistance_torque;
259  }
