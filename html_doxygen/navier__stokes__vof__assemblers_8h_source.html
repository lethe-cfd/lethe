242  inline Tensor<1, dim>
243  get_kinematic_viscosity_gradient(
244  const Tensor<2, dim> &velocity_gradient,
245  const Tensor<3, dim> &velocity_hessians,
246  const double shear_rate_magnitude,
247  const double grad_kinematic_viscosity_shear_rate) const
248  {
249  // Calculates an approximation of the shear rate magnitude gradient using
250  // the derived form, since it does not change with rheological models
251  Tensor<1, dim> grad_shear_rate;
252  for (unsigned int d = 0; d < dim; ++d)
253  {
254  if (dim == 2)
255  {
256  for (unsigned int k = 0; k < dim; ++k)
257  {
258  grad_shear_rate[d] +=
259  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
260  shear_rate_magnitude;
261  }
262  grad_shear_rate[d] +=
263  (velocity_gradient[0][1] + velocity_gradient[1][0]) *
264  (velocity_hessians[0][d][1] + velocity_hessians[1][d][0]) /
265  shear_rate_magnitude;
266  }
267  else
268  {
269  for (unsigned int k = 0; k < dim; ++k)
270  {
271  grad_shear_rate[d] +=
272  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
273  shear_rate_magnitude +
274  (velocity_gradient[(k + 1) % dim][(k + 2) % dim] +
275  velocity_gradient[(k + 2) % dim][(k + 1) % dim]) *
276  (velocity_hessians[(k + 1) % dim][d][(k + 2) % dim] +
277  velocity_hessians[(k + 2) % dim][d][(k + 1) % dim]) /
278  shear_rate_magnitude;
279  }
280  }
281  }
282  return grad_shear_rate * grad_kinematic_viscosity_shear_rate;
283  };
