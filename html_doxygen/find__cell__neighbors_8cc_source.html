14 FindCellNeighbors<dim>::find_cell_neighbors(

18  typename DEM::dem_data_structures<dim>::cells_neighbor_list

20 {
21  // The output vectors of the function are cells_local_neighbor_list and
22  // cells_ghost_neighbor_list. The first contains all the local neighbors cells
23  // of all local cells; while the second contains all the ghost cells of all
24  // local cells. They are two vectors with size of the number of active cells.
25  // The first elements of all vectors are the main cells
26  typename DEM::dem_data_structures<dim>::cell_vector local_neighbor_vector;
27  typename DEM::dem_data_structures<dim>::cell_vector ghost_neighbor_vector;
28 
29  // This vector is used to avoid repetition of adjacent cells. For instance if
30  // cell B is recognized as the neighbor of cell A, cell A will not be added to
31  // the neighbor list of cell B again. This is done using the total_cell_list
32  // vector
33  typename DEM::dem_data_structures<dim>::cell_set total_cell_list;
34 
35  // For each cell, the cell vertices are found and used to find adjacent cells.
36  // The reason is to find the cells located on the corners of the main cell.
37  auto v_to_c = GridTools::vertex_to_cell_map(triangulation);
38 
39  // Looping over cells
40  for (const auto &cell : triangulation.active_cell_iterators())
41  {
42  // If the cell is owned by the processor
43  if (cell->is_locally_owned())
44  {
45  // The first element of each vector is the cell itself.
46  local_neighbor_vector.push_back(cell);
47 
48  total_cell_list.insert(cell);
49 
50  for (unsigned int vertex = 0; vertex < cell->n_vertices(); ++vertex)
51  {
52  for (const auto &neighbor : v_to_c[cell->vertex_index(vertex)])
53  {
54  if (neighbor->is_locally_owned())
55  {
56  auto search_iterator = total_cell_list.find(neighbor);
57  auto local_search_iterator =
58  std::find(local_neighbor_vector.begin(),
59  local_neighbor_vector.end(),
60  neighbor);
61 
62  // If the cell (neighbor) is a local cell and not present
63  // in the total_cell_list vector, it will be added as the
64  // neighbor of the main cell
65  // ("cell") and also to the total_cell_list to avoid
66  // repetition for next cells.
67  if (search_iterator == total_cell_list.end() &&
68  local_search_iterator == local_neighbor_vector.end())
69  {
70  local_neighbor_vector.push_back(neighbor);
71  }
72 
73  // If the neighbor cell is a ghost, it should be added in
74  // the ghost_neighbor_vector container
75  }
76  else if (neighbor->is_ghost())
77  {
78  auto ghost_search_iterator =
79  std::find(ghost_neighbor_vector.begin(),
80  ghost_neighbor_vector.end(),
81  neighbor);
82  if (ghost_search_iterator == ghost_neighbor_vector.end())
83  {
84  if (ghost_neighbor_vector.empty())
85  {
86  ghost_neighbor_vector.push_back(cell);
87  }
88 
89  ghost_neighbor_vector.push_back(neighbor);
90  }
91  }
92  }
93  }
94  }
95  if (!local_neighbor_vector.empty())
96  cells_local_neighbor_list.push_back(local_neighbor_vector);
97  if (!ghost_neighbor_vector.empty())
98  cells_ghost_neighbor_list.push_back(ghost_neighbor_vector);
99 
100  local_neighbor_vector.clear();
101  ghost_neighbor_vector.clear();
102  }
103 }

321 void
322 FindCellNeighbors<dim>::get_periodic_neighbor_list(
323  const typename Triangulation<dim>::active_cell_iterator &cell,
324  const std::map<unsigned int, std::vector<unsigned int>>
325  &coinciding_vertex_groups,
326  const std::map<unsigned int, unsigned int> &vertex_to_coinciding_vertex_group,
327  const std::vector<std::set<typename Triangulation<dim>::active_cell_iterator>>
328  &v_to_c,
329  typename DEM::dem_data_structures<dim>::cell_vector &periodic_neighbor_list)
330 {
331  // Loop over vertices of the cell
332  for (unsigned int vertex = 0; vertex < cell->n_vertices(); ++vertex)
333  {
334  // Get global id of vertex
335  unsigned int vertex_id = cell->vertex_index(vertex);
336 
337  // Check if vertex is at periodic boundary, should be a key if so
338  if (vertex_to_coinciding_vertex_group.find(vertex_id) !=
339  vertex_to_coinciding_vertex_group.end())
340  {
341  // Get the coinciding vertex key to the current vertex
342  unsigned int coinciding_vertex_key =
343  vertex_to_coinciding_vertex_group.at(vertex_id);
344 
345  // Store the neighbor cells in list
346  for (auto coinciding_vertex :
347  coinciding_vertex_groups.at(coinciding_vertex_key))
348  {
349  // Skip the current vertex since we want only cells linked
350  // to the periodic vertices
351  if (coinciding_vertex != vertex_id)
352  {
353  // Loop over all periodic neighbor
354  for (const auto &neighbor_id : v_to_c[coinciding_vertex])
355  {
356  periodic_neighbor_list.push_back(neighbor_id);
357  }
358  }
359  }
360  }
361  }
