249  template <int dim>
250  void
251  Nitsche<dim>::parse_parameters(ParameterHandler &prm)
252  {
253  prm.enter_subsection("nitsche");
254  {
255  const std::string op = prm.get("verbosity");
256  if (op == "verbose")
257  verbosity = Verbosity::verbose;
258  if (op == "quiet")
259  verbosity = Verbosity::quiet;
260 
261  number_solids = prm.get_integer("number of solids");
262  nitsche_solids.resize(number_solids);
263 
264  for (unsigned int i_solid = 0; i_solid < number_solids; ++i_solid)
265  nitsche_solids[i_solid]->parse_parameters(prm, i_solid);
266  }
267  prm.leave_subsection();
268  }
