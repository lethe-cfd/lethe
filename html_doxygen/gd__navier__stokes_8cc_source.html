61 template <int dim>
62 void
63 GDNavierStokesSolver<dim>::setup_assemblers()
64 {
65  this->assemblers.clear();
66 
67  // Buoyant force
68  if (this->simulation_parameters.multiphysics.buoyancy_force)
69  {
70  this->assemblers.push_back(
71  std::make_shared<BuoyancyAssembly<dim>>(this->simulation_control));
72  }
73 
74  // ALE
75  if (this->simulation_parameters.ale.enabled())
76  {
77  this->assemblers.push_back(
78  std::make_shared<NavierStokesAssemblerALE<dim>>(
79  this->simulation_control, this->simulation_parameters.ale));
80  }
81 
82  if (this->simulation_parameters.multiphysics.VOF)
83  {
84  // Time-stepping schemes
85  if (is_bdf(this->simulation_control->get_assembly_method()) &&
86  this->simulation_parameters.physical_properties_manager
87  .density_is_constant())
88  {
89  this->assemblers.push_back(
90  std::make_shared<GLSNavierStokesVOFAssemblerBDF<dim>>(
91  this->simulation_control));
92  }
93  else if (is_bdf(this->simulation_control->get_assembly_method()))
94  {
95  this->assemblers.push_back(
96  std::make_shared<
97  GLSIsothermalCompressibleNavierStokesVOFAssemblerBDF<dim>>(
98  this->simulation_control));
99  }
100  if (!this->simulation_parameters.physical_properties_manager
101  .density_is_constant())
102  {
103  this->assemblers.push_back(
104  std::make_shared<
105  GLSIsothermalCompressibleNavierStokesVOFAssemblerCore<dim>>(
106  this->simulation_control, this->simulation_parameters));
107  }
108  else
109  {
110  // Core assembler
111  this->assemblers.push_back(
112  std::make_shared<GLSNavierStokesVOFAssemblerCore<dim>>(
113  this->simulation_control, this->simulation_parameters));
114  }
115  }
116  else
117  {
118  // Time-stepping schemes
119  if (is_bdf(this->simulation_control->get_assembly_method()))
120  {
121  this->assemblers.push_back(
122  std::make_shared<GLSNavierStokesAssemblerBDF<dim>>(
123  this->simulation_control));
124  }
125 
126  // Velocity sources term
127  if (this->simulation_parameters.velocity_sources.type ==
128  Parameters::VelocitySource::VelocitySourceType::srf)
129  {
130  this->assemblers.push_back(
131  std::make_shared<GLSNavierStokesAssemblerSRF<dim>>(
132  this->simulation_parameters.velocity_sources));
133  }
134 
135  if (this->simulation_parameters.physical_properties_manager
136  .is_non_newtonian())
137  {
138  // Core assembler with Non newtonian viscosity
139  this->assemblers.push_back(
140  std::make_shared<GDNavierStokesAssemblerNonNewtonianCore<dim>>(
141  this->simulation_control, gamma));
142  }
143  else
144  {
145  // Core default assembler
146  if ((this->simulation_parameters.stabilization
147  .use_default_stabilization == true) ||
148  this->simulation_parameters.stabilization.stabilization ==
149  Parameters::Stabilization::NavierStokesStabilization::grad_div)
150  this->assemblers.push_back(
151  std::make_shared<GDNavierStokesAssemblerCore<dim>>(
152  this->simulation_control, gamma));
153 
154  else
155  throw std::runtime_error(
156  "Using the GD solver with a stabilization other than the grad_div "
157  "stabilization will lead to an unstable block solver that is unable to converge");
158  }
159  }
160 }

174 template <int dim>
175 void
176 GDNavierStokesSolver<dim>::assemble_system_matrix()
177 {
178  TimerOutput::Scope t(this->computing_timer, "Assemble matrix");
179  // this->simulation_control->set_assembly_method(this->time_stepping_method);
180 
181  this->system_matrix = 0;
182  setup_assemblers();
183 
184  auto scratch_data = NavierStokesScratchData<dim>(
185  this->simulation_parameters.physical_properties_manager,
186  *this->fe,
187  *this->cell_quadrature,
188  *this->mapping,
189  *this->face_quadrature);
190 
191  if (this->simulation_parameters.multiphysics.VOF)
192  {
193  const DoFHandler<dim> *dof_handler_vof =
194  this->multiphysics->get_dof_handler(PhysicsID::VOF);
195  scratch_data.enable_vof(
196  dof_handler_vof->get_fe(),
197  *this->cell_quadrature,
198  *this->mapping,
199  this->simulation_parameters.multiphysics.vof_parameters.phase_filter);
200  }
201 
202 
203  WorkStream::run(
204  this->dof_handler.begin_active(),
205  this->dof_handler.end(),
206  *this,
207  &GDNavierStokesSolver::assemble_local_system_matrix,
208  &GDNavierStokesSolver::copy_local_matrix_to_global_matrix,
209  scratch_data,
210  StabilizedMethodsTensorCopyData<dim>(this->fe->n_dofs_per_cell(),
211  this->cell_quadrature->size()));
212 
213 
214  system_matrix.compress(VectorOperation::add);
215 
216  // Finally we move pressure mass matrix into a separate matrix:
217  pressure_mass_matrix.reinit(sparsity_pattern.block(1, 1));
218  pressure_mass_matrix.copy_from(system_matrix.block(1, 1));
219 
220  // Note that settings this pressure block to zero is not identical to
221  // not assembling anything in this block, because this operation here
222  // will (incorrectly) delete diagonal entries that come in from
223  // hanging node constraints for pressure DoFs. This means that our
224  // whole system matrix will have rows that are completely
225  // zero. Luckily, FGMRES handles these rows without any problem.
226  system_matrix.block(1, 1) = 0;
227 }

231 template <int dim>
232 void
233 GDNavierStokesSolver<dim>::assemble_local_system_matrix(
234  const typename DoFHandler<dim>::active_cell_iterator &cell,
235  NavierStokesScratchData<dim> &scratch_data,
236  StabilizedMethodsTensorCopyData<dim> &copy_data)
237 {
238  copy_data.cell_is_local = cell->is_locally_owned();
239  if (!cell->is_locally_owned())
240  return;
241 
242  scratch_data.reinit(
243  cell,
244  this->evaluation_point,
245  this->previous_solutions,
246  this->forcing_function,
247  this->flow_control.get_beta(),
248  this->simulation_parameters.stabilization.pressure_scaling_factor);
249  if (this->simulation_parameters.multiphysics.VOF)
250  {
251  const DoFHandler<dim> *dof_handler_vof =
252  this->multiphysics->get_dof_handler(PhysicsID::VOF);
253  typename DoFHandler<dim>::active_cell_iterator phase_cell(
254  &(*(this->triangulation)),
255  cell->level(),
256  cell->index(),
257  dof_handler_vof);
258 
259  scratch_data.reinit_vof(
260  phase_cell,
261  *this->multiphysics->get_solution(PhysicsID::VOF),
262  *this->multiphysics->get_filtered_solution(PhysicsID::VOF),
263  *this->multiphysics->get_previous_solutions(PhysicsID::VOF));
264  }
265 
266  if (this->simulation_parameters.multiphysics.heat_transfer)
267  {
268  const DoFHandler<dim> *dof_handler_ht =
269  this->multiphysics->get_dof_handler(PhysicsID::heat_transfer);
270  scratch_data.enable_heat_transfer(dof_handler_ht->get_fe(),
271  *this->cell_quadrature,
272  *this->mapping);
273  }
274 
275  scratch_data.calculate_physical_properties();
276  copy_data.reset();
277 
278 
279  for (auto &assembler : this->assemblers)
280  {
281  assembler->assemble_matrix(scratch_data, copy_data);
282  }
283 
284 
285  cell->get_dof_indices(copy_data.local_dof_indices);
286 }

439 template <int dim>
440 void
441 GDNavierStokesSolver<dim>::assemble_L2_projection()
442 {
443  system_matrix = 0;
444  auto &system_rhs = this->system_rhs;
445  system_rhs = 0;
446  FEValues<dim> fe_values(*this->mapping,
447  *this->fe,
448  *this->cell_quadrature,
449  update_values | update_quadrature_points |
450  update_JxW_values);
451  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
452  const unsigned int n_q_points = this->cell_quadrature->size();
453  FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
454  Vector<double> local_rhs(dofs_per_cell);
455  std::vector<Vector<double>> initial_velocity(n_q_points,
456  Vector<double>(dim + 1));
457  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
458  const FEValuesExtractors::Vector velocities(0);
459  const FEValuesExtractors::Scalar pressure(dim);
460 
461  Tensor<1, dim> rhs_initial_velocity_pressure;
462  double rhs_initial_pressure;
463 
464  std::vector<Tensor<1, dim>> phi_u(dofs_per_cell);
465  std::vector<double> phi_p(dofs_per_cell);
466 
467  for (const auto &cell : this->dof_handler.active_cell_iterators())
468  {
469  if (cell->is_locally_owned())
470  {
471  fe_values.reinit(cell);
472  local_matrix = 0;
473  local_rhs = 0;
474  this->simulation_parameters.initial_condition->uvwp.vector_value_list(
475  fe_values.get_quadrature_points(), initial_velocity);
476  for (unsigned int q = 0; q < n_q_points; ++q)
477  {
478  for (unsigned int k = 0; k < dofs_per_cell; ++k)
479  {
480  phi_p[k] = fe_values[pressure].value(k, q);
481  phi_u[k] = fe_values[velocities].value(k, q);
482  }
483 
484  // Establish the rhs tensor operator
485  for (int i = 0; i < dim; ++i)
486  {
487  const unsigned int component_i =
488  this->fe->system_to_component_index(i).first;
489  rhs_initial_velocity_pressure[i] =
490  initial_velocity[q](component_i);
491  }
492  rhs_initial_pressure = initial_velocity[q](dim);
493 
494  for (unsigned int i = 0; i < dofs_per_cell; ++i)
495  {
496  // Matrix assembly
497  for (unsigned int j = 0; j < dofs_per_cell; ++j)
498  {
499  local_matrix(i, j) +=
500  (phi_u[j] * phi_u[i]) * fe_values.JxW(q);
501  local_matrix(i, j) +=
502  (phi_p[j] * phi_p[i]) * fe_values.JxW(q);
503  }
504  local_rhs(i) += (phi_u[i] * rhs_initial_velocity_pressure +
505  phi_p[i] * rhs_initial_pressure) *
506  fe_values.JxW(q);
507  }
508  }
509 
510  cell->get_dof_indices(local_dof_indices);
511  this->nonzero_constraints.distribute_local_to_global(
512  local_matrix,
513  local_rhs,
514  local_dof_indices,
515  system_matrix,
516  system_rhs);
517  }
518  }
519  system_matrix.compress(VectorOperation::add);
520  system_rhs.compress(VectorOperation::add);
521 }

524 template <int dim>
525 void
526 GDNavierStokesSolver<dim>::setup_dofs_fd()
527 {
528  TimerOutput::Scope t(this->computing_timer, "Setup DOFs");
529 
530  system_matrix.clear();
531 
532  this->dof_handler.distribute_dofs(*this->fe);
533  // DoFRenumbering::Cuthill_McKee(this->dof_handler);
534 
535 
536  std::vector<unsigned int> block_component(dim + 1, 0);
537  block_component[dim] = 1;
538  DoFRenumbering::component_wise(this->dof_handler, block_component);
539 
540 
541  // To be used to replace the above part once 9.2 release is out and TRAVIS-CI
542  // version is updated
543  dofs_per_block =
544  DoFTools::count_dofs_per_fe_block(this->dof_handler, block_component);
545 
546  unsigned int dof_u = dofs_per_block[0];
547  unsigned int dof_p = dofs_per_block[1];
548 
549  this->locally_owned_dofs.resize(2);
550  this->locally_owned_dofs[0] =
551  this->dof_handler.locally_owned_dofs().get_view(0, dof_u);
552  this->locally_owned_dofs[1] =
553  this->dof_handler.locally_owned_dofs().get_view(dof_u, dof_u + dof_p);
554 
555  IndexSet locally_relevant_dofs_acquisition;
556  DoFTools::extract_locally_relevant_dofs(this->dof_handler,
557  locally_relevant_dofs_acquisition);
558  this->locally_relevant_dofs.resize(2);
559  this->locally_relevant_dofs[0] =
560  locally_relevant_dofs_acquisition.get_view(0, dof_u);
561  this->locally_relevant_dofs[1] =
562  locally_relevant_dofs_acquisition.get_view(dof_u, dof_u + dof_p);
563 
564  FEValuesExtractors::Vector velocities(0);
565 
566  // Non-zero constraints
567  auto &nonzero_constraints = this->nonzero_constraints;
568  {
569  nonzero_constraints.clear();
570  nonzero_constraints.reinit(locally_relevant_dofs_acquisition);
571 
572 
573  DoFTools::make_hanging_node_constraints(this->dof_handler,
574  nonzero_constraints);
575  for (unsigned int i_bc = 0;
576  i_bc < this->simulation_parameters.boundary_conditions.size;
577  ++i_bc)
578  {
579  if (this->simulation_parameters.boundary_conditions.type[i_bc] ==
580  BoundaryConditions::BoundaryType::noslip)
581  {
582  VectorTools::interpolate_boundary_values(
583  *this->mapping,
584  this->dof_handler,
585  this->simulation_parameters.boundary_conditions.id[i_bc],
586  dealii::Functions::ZeroFunction<dim>(dim + 1),
587  nonzero_constraints,
588  this->fe->component_mask(velocities));
589  }
590  else if (this->simulation_parameters.boundary_conditions.type[i_bc] ==
591  BoundaryConditions::BoundaryType::slip)
592  {
593  std::set<types::boundary_id> no_normal_flux_boundaries;
594  no_normal_flux_boundaries.insert(
595  this->simulation_parameters.boundary_conditions.id[i_bc]);
596  VectorTools::compute_no_normal_flux_constraints(
597  this->dof_handler,
598  0,
599  no_normal_flux_boundaries,
600  nonzero_constraints);
601  }
602  else if (this->simulation_parameters.boundary_conditions.type[i_bc] ==
603  BoundaryConditions::BoundaryType::function)
604  {
605  VectorTools::interpolate_boundary_values(
606  *this->mapping,
607  this->dof_handler,
608  this->simulation_parameters.boundary_conditions.id[i_bc],
609  NavierStokesFunctionDefined<dim>(
610  &this->simulation_parameters.boundary_conditions
611  .bcFunctions[i_bc]
612  .u,
613  &this->simulation_parameters.boundary_conditions
614  .bcFunctions[i_bc]
615  .v,
616  &this->simulation_parameters.boundary_conditions
617  .bcFunctions[i_bc]
618  .w),
619  nonzero_constraints,
620  this->fe->component_mask(velocities));
621  }
622 
623  else if (this->simulation_parameters.boundary_conditions.type[i_bc] ==
624  BoundaryConditions::BoundaryType::periodic)
625  {
626  DoFTools::make_periodicity_constraints(
627  this->dof_handler,
628  this->simulation_parameters.boundary_conditions.id[i_bc],
629  this->simulation_parameters.boundary_conditions.periodic_id[i_bc],
630  this->simulation_parameters.boundary_conditions
631  .periodic_direction[i_bc],
632  nonzero_constraints);
633  }
634  }
635  }
636  nonzero_constraints.close();
637 
638  {
639  this->zero_constraints.clear();
640  this->zero_constraints.reinit(locally_relevant_dofs_acquisition);
641 
642  DoFTools::make_hanging_node_constraints(this->dof_handler,
643  this->zero_constraints);
644 
645  for (unsigned int i_bc = 0;
646  i_bc < this->simulation_parameters.boundary_conditions.size;
647  ++i_bc)
648  {
649  if (this->simulation_parameters.boundary_conditions.type[i_bc] ==
650  BoundaryConditions::BoundaryType::slip)
651  {
652  std::set<types::boundary_id> no_normal_flux_boundaries;
653  no_normal_flux_boundaries.insert(
654  this->simulation_parameters.boundary_conditions.id[i_bc]);
655  VectorTools::compute_no_normal_flux_constraints(
656  this->dof_handler,
657  0,
658  no_normal_flux_boundaries,
659  this->zero_constraints);
660  }
661  else if (this->simulation_parameters.boundary_conditions.type[i_bc] ==
662  BoundaryConditions::BoundaryType::periodic)
663  {
664  DoFTools::make_periodicity_constraints(
665  this->dof_handler,
666  this->simulation_parameters.boundary_conditions.id[i_bc],
667  this->simulation_parameters.boundary_conditions.periodic_id[i_bc],
668  this->simulation_parameters.boundary_conditions
669  .periodic_direction[i_bc],
670  this->zero_constraints);
671  }
672  else // if(nsparam.boundaryConditions.boundaries[i_bc].type==Parameters::noslip
673  // || Parameters::function)
674  {
675  VectorTools::interpolate_boundary_values(
676  *this->mapping,
677  this->dof_handler,
678  this->simulation_parameters.boundary_conditions.id[i_bc],
679  dealii::Functions::ZeroFunction<dim>(dim + 1),
680  this->zero_constraints,
681  this->fe->component_mask(velocities));
682  }
683  }
684  }
685  this->zero_constraints.close();
686 
687  this->present_solution.reinit(this->locally_owned_dofs,
688  this->locally_relevant_dofs,
689  this->mpi_communicator);
690 
691  this->evaluation_point.reinit(this->locally_owned_dofs,
692  this->locally_relevant_dofs,
693  this->mpi_communicator);
694 
695  // Initialize vector of previous solutions
696  for (auto &solution : this->previous_solutions)
697  {
698  solution.reinit(this->locally_owned_dofs,
699  this->locally_relevant_dofs,
700  this->mpi_communicator);
701  }
702 
703  this->newton_update.reinit(this->locally_owned_dofs, this->mpi_communicator);
704  this->system_rhs.reinit(this->locally_owned_dofs, this->mpi_communicator);
705  this->local_evaluation_point.reinit(this->locally_owned_dofs,
706  this->mpi_communicator);
707 
708 
709  sparsity_pattern.reinit(this->locally_owned_dofs,
710  this->locally_owned_dofs,
711  this->locally_relevant_dofs,
712  MPI_COMM_WORLD);
713 
714  Table<2, DoFTools::Coupling> coupling(dim + 1, dim + 1);
715  for (unsigned int c = 0; c < dim + 1; ++c)
716  for (unsigned int d = 0; d < dim + 1; ++d)
717  if (!((c == dim) && (d == dim)))
718  coupling[c][d] = DoFTools::always;
719  else
720  coupling[c][d] = DoFTools::always;
721 
722  DoFTools::make_sparsity_pattern(this->dof_handler,
723  coupling,
724  sparsity_pattern,
725  nonzero_constraints,
726  true,
727  Utilities::MPI::this_mpi_process(
728  MPI_COMM_WORLD));
729 
730  sparsity_pattern.compress();
731 
732  system_matrix.reinit(sparsity_pattern);
733  pressure_mass_matrix.reinit(sparsity_pattern.block(1, 1));
734 
735  if (this->simulation_parameters.post_processing.calculate_average_velocities)
736  {
737  AssertThrow(this->simulation_parameters.mesh_adaptation.type ==
738  Parameters::MeshAdaptation::Type::none,
739  ExcMessage(
740  "Time-averaging velocities and calculating reynolds "
741  "stresses are currently unavailable for mesh "
742  "adaptation."));
743 
744  this->average_velocities->initialize_vectors(
745  this->locally_owned_dofs,
746  this->locally_relevant_dofs,
747  this->fe->n_dofs_per_vertex(),
748  this->mpi_communicator);
749 
750  if (this->simulation_parameters.restart_parameters.checkpoint)
751  {
752  this->average_velocities->initialize_checkpoint_vectors(
753  this->locally_owned_dofs,
754  this->locally_relevant_dofs,
755  this->mpi_communicator);
756  }
757  }
758 
759 
760  double global_volume =
761  GridTools::volume(*this->triangulation, *this->mapping);
762 
763  this->pcout << " Number of active cells: "
764  << this->triangulation->n_global_active_cells() << std::endl
765  << " Number of degrees of freedom: "
766  << this->dof_handler.n_dofs() << std::endl;
767  this->pcout << " Volume of triangulation: " << global_volume
768  << std::endl;
769 
770  this->multiphysics->set_dof_handler(PhysicsID::fluid_dynamics,
771  &this->dof_handler);
772  this->multiphysics->set_block_solution(PhysicsID::fluid_dynamics,
773  &this->present_solution);

775  &this->previous_solutions);

1046 template <int dim>
1047 void
1048 GDNavierStokesSolver<dim>::solve_system_GMRES(const bool initial_step,
1049  const double absolute_residual,
1050  const double relative_residual,
1051  const bool renewed_matrix)
1052 {
1053  const AffineConstraints<double> &constraints_used =
1054  initial_step ? this->nonzero_constraints : this->zero_constraints;
1055  const double linear_solver_tolerance =
1056  std::max(relative_residual * this->system_rhs.l2_norm(), absolute_residual);
1057 
1058  if (this->simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
1059  .verbosity != Parameters::Verbosity::quiet)
1060  {
1061  this->pcout << " -Tolerance of iterative solver is : "
1062  << linear_solver_tolerance << std::endl;
1063  }
1064 
1065 
1066  TrilinosWrappers::MPI::BlockVector completely_distributed_solution(
1067  this->locally_owned_dofs, this->mpi_communicator);
1068 
1069  SolverControl solver_control(this->simulation_parameters.linear_solver
1070  .at(PhysicsID::fluid_dynamics)
1071  .max_iterations,
1072  linear_solver_tolerance,
1073  true,
1074  true);
1075 
1076  SolverFGMRES<TrilinosWrappers::MPI::BlockVector> solver(solver_control);
1077 
1078  if (this->simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
1079  .preconditioner == Parameters::LinearSolver::PreconditionerType::ilu)
1080  {
1081  if (renewed_matrix || velocity_ilu_preconditioner == 0 ||
1082  pressure_ilu_preconditioner == 0 || system_ilu_preconditioner == 0)
1083  setup_ILU();
1084  }
1085  else if (this->simulation_parameters.linear_solver
1086  .at(PhysicsID::fluid_dynamics)
1087  .preconditioner ==
1088  Parameters::LinearSolver::PreconditionerType::amg)
1089  {
1090  if (renewed_matrix || velocity_amg_preconditioner == 0 ||
1091  pressure_amg_preconditioner == 0 || system_amg_preconditioner == 0)
1092  setup_AMG();
1093  }
1094 
1095  {
1096  TimerOutput::Scope t(this->computing_timer, "Solve linear system");
1097  if (this->simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
1098  .preconditioner == Parameters::LinearSolver::PreconditionerType::ilu)
1099  solver.solve(this->system_matrix,
1100  this->newton_update,
1101  this->system_rhs,
1102  *system_ilu_preconditioner);
1103  else if (this->simulation_parameters.linear_solver
1104  .at(PhysicsID::fluid_dynamics)
1105  .preconditioner ==
1106  Parameters::LinearSolver::PreconditionerType::amg)
1107  solver.solve(this->system_matrix,
1108  this->newton_update,
1109  this->system_rhs,
1110  *system_amg_preconditioner);
1111  else
1112  AssertThrow(
1113  this->simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
1114  .preconditioner ==
1115  Parameters::LinearSolver::PreconditionerType::ilu ||
1116  this->simulation_parameters.linear_solver
1117  .at(PhysicsID::fluid_dynamics)
1118  .preconditioner ==
1119  Parameters::LinearSolver::PreconditionerType::amg,
1120  ExcMessage(
1121  "This linear solver does not support this preconditioner. Only <ilu> and <amg> preconditioners are supported."));
1122 
1123  if (this->simulation_parameters.linear_solver.at(PhysicsID::fluid_dynamics)
1124  .verbosity != Parameters::Verbosity::quiet)
1125  {
1126  this->pcout << " -Iterative solver took : "
1127  << solver_control.last_step() << " steps " << std::endl;
1128  }
1129 
1130  constraints_used.distribute(this->newton_update);
1131  }
