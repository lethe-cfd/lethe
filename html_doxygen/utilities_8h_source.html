173 inline double
174 calculate_point_property(const double phase,
175  const double property0,
176  const double property1)
177 {
178  double property_eq = phase * property1 + (1 - phase) * property0;
179 
180  // Limit parameters value (patch)
181  // TODO see if necessary after compression term is added in the
182  // VOF advection equation
183  const double property_min = std::min(property0, property1);
184  const double property_max = std::max(property0, property1);
185  if (property_eq < property_min)
186  {
187  property_eq = property_min;
188  }
189  if (property_eq > property_max)
190  {
191  property_eq = property_max;
192  }
193 
194  return property_eq;
195 }

219 inline double
220 calculate_point_property_cahn_hilliard(const double phase_cahn_hilliard,
221  const double property0,
222  const double property1)
223 {
224  double phase = phase_cahn_hilliard;
225 
226  if (std::abs(phase_cahn_hilliard) > 1)
227  {
228  phase = sgn(phase_cahn_hilliard);
229  }
230 
231  double property_avg = (property0 + property1) * 0.5;
232  double property_diff = (property0 - property1) * 0.5;
233 
234  double property_eq = phase * property_diff + property_avg;
235 
236  return property_eq;
237 }
