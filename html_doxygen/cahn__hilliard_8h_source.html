16  *
17  * Implementation of the Cahn-Hilliard equations as an auxiliary physics.
18  * Equations solved:
19  * dPhi/dt + u * gradPhi = div(M(Phi)*grad eta)
20  * eta - f(Phi) + epsilon^2 * div(grad Phi) = 0
21  * with Phi the phase field parameter (or phase order), eta the chemical
22  * potential, and M the mobility function and epsilon the interface thickness.
23  * The phase field parameter must not be confused with the order (respectively
24  * phase_cahn_hilliard_order and potential_cahn_hilliard_order) of the finite
25  elements related to the
26  * phase field parameter and the chemical potential.
27  */

60 public:
61  CahnHilliard(MultiphysicsInterface<dim> *multiphysics_interface,
62  const SimulationParameters<dim> &p_simulation_parameters,
63  std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
64  p_triangulation,
65  std::shared_ptr<SimulationControl> p_simulation_control)
66  : AuxiliaryPhysics<dim, TrilinosWrappers::MPI::Vector>(
67  p_simulation_parameters.non_linear_solver.at(PhysicsID::cahn_hilliard))
68  , multiphysics(multiphysics_interface)
69  , computing_timer(p_triangulation->get_communicator(),
70  this->pcout,
71  TimerOutput::summary,
72  TimerOutput::wall_times)
73  , simulation_parameters(p_simulation_parameters)
74  , triangulation(p_triangulation)
75  , simulation_control(p_simulation_control)
76  , dof_handler(*triangulation)
77  {
78  if (simulation_parameters.mesh.simplex)
79  {
80  // for simplex meshes
81  fe = std::make_shared<FESystem<dim>>(
82  FE_SimplexP<dim>(
83  simulation_parameters.fem_parameters.phase_cahn_hilliard_order),
84  1,
85  FE_SimplexP<dim>(
86  simulation_parameters.fem_parameters.potential_cahn_hilliard_order),
87  1);
88  mapping = std::make_shared<MappingFE<dim>>(*fe);
89  cell_quadrature = std::make_shared<QGaussSimplex<dim>>(
90  std::max(
91  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
92  simulation_parameters.fem_parameters
93  .potential_cahn_hilliard_order) +
94  1);
95  face_quadrature = std::make_shared<QGaussSimplex<dim - 1>>(
96  std::max(
97  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
98  simulation_parameters.fem_parameters
99  .potential_cahn_hilliard_order) +
100  1);
101  ;
102  }
103  else
104  {
105  // Usual case, for quad/hex meshes
106  fe = std::make_shared<FESystem<dim>>(
107  FE_Q<dim>(
108  simulation_parameters.fem_parameters.phase_cahn_hilliard_order),
109  1,
110  FE_Q<dim>(
111  simulation_parameters.fem_parameters.potential_cahn_hilliard_order),
112  1);
113  mapping = std::make_shared<MappingQ<dim>>(
114  std::max(
115  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
116  simulation_parameters.fem_parameters.potential_cahn_hilliard_order),
117  simulation_parameters.fem_parameters.qmapping_all);
118  cell_quadrature = std::make_shared<QGauss<dim>>(
119  std::max(
120  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
121  simulation_parameters.fem_parameters
122  .potential_cahn_hilliard_order) +
123  1);
124  face_quadrature = std::make_shared<QGauss<dim - 1>>(
125  std::max(
126  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
127  simulation_parameters.fem_parameters
128  .potential_cahn_hilliard_order) +
129  1);
130  }
131 
132  // Allocate solution transfer
133  solution_transfer =
134  std::make_shared<parallel::distributed::
135  SolutionTransfer<dim, TrilinosWrappers::MPI::Vector>>(
136  dof_handler);
137 
138  // Set size of previous solutions using BDF schemes information
139  previous_solutions.resize(maximum_number_of_previous_solutions());
140 
141  // Prepare previous solutions transfer
142  previous_solutions_transfer.reserve(previous_solutions.size());
143  for (unsigned int i = 0; i < previous_solutions.size(); ++i)
144  {
145  previous_solutions_transfer.emplace_back(
146  parallel::distributed::
147  SolutionTransfer<dim, TrilinosWrappers::MPI::Vector>(
148  this->dof_handler));
149  }
150 
151  // Change the behavior of the timer for situations when you don't want
152  // outputs
153  if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
154  this->computing_timer.disable_output();
155  }
