29 template <int dim>
30 void
31 DisableContacts<dim>::update_local_and_ghost_cell_set(
32  const DoFHandler<dim> &background_dh)
33 {
34  local_and_ghost_cells.clear();
35  for (const auto &cell : background_dh.active_cell_iterators())
36  {
37  if (cell->is_locally_owned() || cell->is_ghost())
38  {
39  local_and_ghost_cells.insert(cell);
40  }
41  }
42 }

44 template <int dim>
45 void
46 DisableContacts<dim>::calculate_granular_temperature_and_solid_fraction(
47  const Particles::ParticleHandler<dim> &particle_handler,
48  const std::set<typename DoFHandler<dim>::active_cell_iterator>
49  &local_and_ghost_cells_with_particles,
50  Vector<double> &granular_temperature_average,
51  Vector<double> &solid_fractions)
52 {
53  // Iterating through the active cells in cell set with particles
54  for (const auto &cell : local_and_ghost_cells_with_particles)
55  {
56  // Particles in the cell
57  auto particles_in_cell = particle_handler.particles_in_cell(cell);
58  const unsigned int n_particles_in_cell =
59  particle_handler.n_particles_in_cell(cell);
60 
61  // Initialize variables for solid fraction computation
62  double solid_fraction = 0.0;
63  double solid_volume = 0.0;
64  const double cell_volume = cell->measure();
65 
66  // Initialize variables for granular temperature computation
67  double granular_temperature_cell = 0.0;
68  Tensor<1, dim> velocity_cell_average;
69  Tensor<1, dim> cell_velocity_fluctuation_squared_average;
70 
71  // First loop over particles in cell to compute the sum of
72  // particle velocity and the solid volume of the current cell
73  for (auto particles_in_cell_iterator = particles_in_cell.begin();
74  particles_in_cell_iterator != particles_in_cell.end();
75  ++particles_in_cell_iterator)
76  {
77  // Get particle properties
78  auto particle_properties =
79  particles_in_cell_iterator->get_properties();
80  const double dp = particle_properties[DEM::PropertiesIndex::dp];
81 
82  for (int d = 0; d < dim; ++d)
83  {
84  // Get the particle velocity component (v_x, v_y & v_z if dim = 3)
85  int v_axis = DEM::PropertiesIndex::v_x + d;
86 
87  // Add the velocity component value
88  velocity_cell_average[d] += particle_properties[v_axis];
89  }
90 
91  solid_volume += M_PI * pow(dp, dim) / (2.0 * dim);
92  }
93 
94  // Calculate average velocity in the cell (sum/n_particles)
95  velocity_cell_average /= n_particles_in_cell;
96 
97  // Calculate solid fraction of cell
98  solid_fraction = solid_volume / cell_volume;
99 
100  // Second loop over particle to compute the sum of the cell velocity
101  // fluctuations
102  for (auto particles_in_cell_iterator = particles_in_cell.begin();
103  particles_in_cell_iterator != particles_in_cell.end();
104  ++particles_in_cell_iterator)
105  {
106  auto particle_properties =
107  particles_in_cell_iterator->get_properties();
108 
109  for (int d = 0; d < dim; ++d)
110  {
111  // Get the particle velocity component (v_x, v_y & v_z if dim = 3)
112  int v_axis = DEM::PropertiesIndex::v_x + d;
113 
114  cell_velocity_fluctuation_squared_average[d] +=
115  Utilities::fixed_power<2>(particle_properties[v_axis] -
116  velocity_cell_average[d]);
117  }
118  }
119 
120  // Calculate average granular temperature in the cell
121  for (int d = 0; d < dim; ++d)
122  {
123  cell_velocity_fluctuation_squared_average[d] /= n_particles_in_cell;
124  granular_temperature_cell +=
125  cell_velocity_fluctuation_squared_average[d] / dim;
126  }
127 
128  // Store the average granular temperature and solid fraction with
129  // active cell index
130  granular_temperature_average[cell->active_cell_index()] =
131  granular_temperature_cell;
132  solid_fractions[cell->active_cell_index()] = solid_fraction;
133  }
134 }

136 template <int dim>
137 void
138 DisableContacts<dim>::identify_mobility_status(
139  const DoFHandler<dim> &background_dh,
140  const Particles::ParticleHandler<dim> &particle_handler,
141  const unsigned int n_active_cells,
142  MPI_Comm mpi_communicator)
143 {
144  // Reset cell status containers
145  cell_mobility_status.clear();
146 
147  // Get a copy of the active & ghost cells set to iterate over and remove cell
148  // of the set when the mobility status is known to avoid unnecessary
149  // iterations for next loops. We don't want to modify the original set since
150  // it is only updated when there's load balancing or reading of checkpoints.
151  auto local_and_ghost_cells_copy = local_and_ghost_cells;
152 
153  // Create dummy dofs for background dof handler for the mobility_at_nodes
154  const FE_Q<dim> fe(1);
155  const unsigned int dofs_per_cell = fe.dofs_per_cell;
156 
157  // Get locally owned and relevant dofs
158  const IndexSet locally_owned_dofs = background_dh.locally_owned_dofs();
159  IndexSet locally_relevant_dofs =
160  DoFTools::extract_locally_relevant_dofs(background_dh);
161 
162  // Reinit all value of mobility at nodes as inactive (0)
163  mobility_at_nodes.reinit(locally_owned_dofs,
164  locally_relevant_dofs,
165  mpi_communicator);
166  mobility_at_nodes = 0;
167 
168  // Check if the cell is empty (n_particle = 0), if so, nodes and cells are
169  // flagged as empty mobility status (3)
170  for (auto cell = local_and_ghost_cells_copy.begin();
171  cell != local_and_ghost_cells_copy.end();)
172  {
173  // Check if the cell has any particles
174  if (particle_handler.n_particles_in_cell(*cell) == 0)
175  {
176  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
177  (*cell)->get_dof_indices(local_dof_indices);
178 
179  // Assign mobile status to cell in map
180  const unsigned int cell_id = (*cell)->active_cell_index();
181  cell_mobility_status.insert({cell_id, mobility_status::inactive});
182 
183  // Remove cell from cell set and iterate to the following cell, the
184  // erase function returns the next iterator.
185  cell = local_and_ghost_cells_copy.erase(cell);
186 
187  // Assign empty status to all nodes
188  for (auto node_id : local_dof_indices)
189  {
190  mobility_at_nodes(node_id) = mobility_status::empty;
191  }
192  }
193  else
194  {
195  // Since erase() is not called, we need to increment the iterator
196  ++cell;
197  }
198  }
199 
200  // Update ghost values of mobility_at_nodes
201  mobility_at_nodes.update_ghost_values();
202 
203  // Calculate the average granular temperature and solid fraction for each
204  // cells currently in the local_and_ghost_cells_copy set (no empty cells)
205  Vector<double> granular_temperature_average(n_active_cells);
206  Vector<double> solid_fractions(n_active_cells);
207  calculate_granular_temperature_and_solid_fraction(
208  particle_handler,
209  local_and_ghost_cells_copy,
210  granular_temperature_average,
211  solid_fractions);
212 
213  // Check if the cell is mobile by criteria:
214  // * granular temperature > threshold or
215  // * solid fraction of cell < threshold or
216  // * is next to an empty cell
217  // If so, nodes are flagged and cells are stored with mobile status (2)
218  for (auto cell = local_and_ghost_cells_copy.begin();
219  cell != local_and_ghost_cells_copy.end();)
220  {
221  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
222  (*cell)->get_dof_indices(local_dof_indices);
223 
224  // Check if the cell has any empty neighbor cell by the value at nodes
225  bool has_empty_neighbor = false;
226  for (auto node_id : local_dof_indices)
227  {
228  if (mobility_at_nodes[node_id] == mobility_status::empty)
229  {
230  has_empty_neighbor = true;
231  break; // No need to check the other nodes
232  }
233  }
234 
235  // Check if the cell is mobile by criteria
236  // (granular temperature or solid fraction or empty neighbor)
237  const unsigned int cell_id = (*cell)->active_cell_index();
238  if (granular_temperature_average[cell_id] >
239  granular_temperature_threshold ||
240  solid_fractions[cell_id] < solid_fraction_threshold ||
241  has_empty_neighbor)
242  {
243  // Assign mobile status to cell in map
244  cell_mobility_status.insert({cell_id, mobility_status::mobile});
245 
246  // Remove cell from cell set and iterate to the following cell
247  cell = local_and_ghost_cells_copy.erase(cell);
248 
249  // Assign mobile status to nodes but don't overwrite empty nodes.
250  // This prevents assigning mobile status to the empty cells in the
251  // next check (additional mobile layer) since we don't verify again if
252  // cell is empty or not.
253  for (auto node_id : local_dof_indices)
254  {
255  // Possible cases: (nodes are initialized to inactive (0))
256  // node = max(mobile (2), inactive (0)) = mobile (2)
257  // node = max(mobile (2), empty (3)) = empty (3)
258  mobility_at_nodes(node_id) =
259  std::max((int)mobility_status::mobile,
260  mobility_at_nodes[node_id]);
261  }
262  }
263  else
264  {
265  // Since erase() is not called, we need to increment the iterator
266  ++cell;
267  }
268  }
269 
270  mobility_at_nodes.update_ghost_values();
271 
272  // Check if the cell is mobile by neighbor (at least a node is flagged as
273  // mobile from previous check), this is the additional mobile layer.
274  // If so, cells are stored in map as mobile status (2) and nodes that are not
275  // mobile are flagged as active (1)
276  for (auto cell = local_and_ghost_cells_copy.begin();
277  cell != local_and_ghost_cells_copy.end();)
278  {
279  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
280  (*cell)->get_dof_indices(local_dof_indices);
281 
282  // Check if the cell has any mobile nodes to know if we need to iterate
283  // over the next cell (no call of erase() function)
284  bool has_mobile_node = false;
285 
286  for (auto node_id : local_dof_indices)
287  {
288  // Check if node is mobile and assign mobile status to the cell
289  if (mobility_at_nodes[node_id] == mobility_status::mobile)
290  {
291  // Assign mobile status to cell in map
292  const unsigned int cell_id = (*cell)->active_cell_index();
293  cell_mobility_status.insert({cell_id, mobility_status::mobile});
294 
295  // Remove cell from cell set and iterate to the following cell,
296  // also label it as mobile to avoid double iteration
297  cell = local_and_ghost_cells_copy.erase(cell);
298  has_mobile_node = true;
299 
300  // Assign active status to nodes except mobile because
301  // this will cause to propagate the mobile status to the
302  // neighbors in this loop since the mobility check at node
303  // is executed in the same container that we are assigning new
304  // mobility status
305  for (auto node_id : local_dof_indices)
306  {
307  // Possible cases: (nodes are initialized to inactive (0))
308  // node = max(active (1), inactive (0)) = active (1)
309  // node = max(active (1), empty (3)) = empty (3)
310  // node = max(active (1), mobile (2)) = mobile (2)
311  mobility_at_nodes[node_id] =
312  std::max((int)mobility_status::active,
313  mobility_at_nodes[node_id]);
314  }
315  break; // No need to check the other nodes
316  }
317  }
318 
319  // Since erase() is not called (no mobile node), we need to increment the
320  // iterator
321  if (!has_mobile_node)
322  {
323  ++cell;
324  }
325  }
326 
327  mobility_at_nodes.update_ghost_values();
328 
329  // Check if the cell is active (at least a node is flagged as active from
330  // previous check), this is the layer of active cells
331  // If so, cells are stored with active status in the map (1)
332  for (auto cell = local_and_ghost_cells_copy.begin();
333  cell != local_and_ghost_cells_copy.end();
334  ++cell)
335  {
336  std::vector<types::global_dof_index> local_dofs_indices(dofs_per_cell);
337  (*cell)->get_dof_indices(local_dofs_indices);
338 
339  bool has_active_nodes = false;
340  bool has_mobile_nodes = false;
341 
342  // Check if cell has active and/or mobile nodes
343  for (auto node_id : local_dofs_indices)
344  {
345  has_active_nodes =
346  (mobility_at_nodes[node_id] == mobility_status::active) ||
347  has_active_nodes;
348 
349  has_mobile_nodes =
350  (mobility_at_nodes[node_id] == mobility_status::mobile) ||
351  has_mobile_nodes;
352  }
353 
354  // Active nodes with mobile nodes means that the cell is part of the
355  // additional mobile layer.
356  // Active nodes and no mobile nodes means that the cell is a neighbor
357  // of the additional layer of mobile cells, this is an active cell.
358  if (has_active_nodes && !has_mobile_nodes)
359  {
360  const unsigned int cell_id = (*cell)->active_cell_index();
361  cell_mobility_status.insert({cell_id, mobility_status::active});
362  }
363  }
364 
365  // Store the inactive cells in the map, those are all the remaining cells in
366  // the active & ghost cells set
367  for (auto cell = local_and_ghost_cells_copy.begin();
368  cell != local_and_ghost_cells_copy.end();
369  ++cell)
370  {
371  // Assign mobile status to cell in map
372  const unsigned int cell_id = (*cell)->active_cell_index();
373  cell_mobility_status.insert({cell_id, mobility_status::inactive});
374  }
375 }
