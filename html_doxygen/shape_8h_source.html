555  {
556  Point<dim> gradient{};
557  for (unsigned int d = 0; d < dim; d++)
558  {
559  // For cases where the coordinate is of value 0, we avoid computing the
560  // gradient. That is an issue when the exponent is lower than or equal
561  // to 1
562  if (abs(centered_point[d]) > epsilon)
563  gradient[d] = exponents[d] *
564  pow(abs(half_lengths[d]), -exponents[d]) *
565  pow(abs(centered_point[d]), exponents[d]) /
566  (centered_point[d] + DBL_MIN);
567  else
568  gradient[d] = exponents[d] *
569  pow(abs(half_lengths[d]), -exponents[d]) *
570  pow(abs(epsilon), exponents[d]) / (epsilon + DBL_MIN);
571  }
572  return gradient;
573  }
