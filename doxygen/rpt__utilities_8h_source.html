36 /**
37  * @brief Create grid of the reactor vessel (cylinder).
38  */
39 template <int dim>
40 void
41 attach_grid_to_triangulation(Triangulation<dim> &triangulation,
42  const Parameters::RPTParameters &parameters,
43  const unsigned int n_refinement)
44 {
45  // Generate cylinder (needs rotation and shift to get origin at the bottom
46  // with z towards top)
47  GridGenerator::subdivided_cylinder(triangulation,
48  5,
49  parameters.reactor_radius,
50  parameters.reactor_height / 2.);
51  Tensor<1, dim> axis({0, 1, 0});
52  GridTools::rotate(axis, M_PI_2, triangulation);
53  const Tensor<1, dim> shift_vector({0, 0, parameters.reactor_height / 2.});
54  GridTools::shift(shift_vector, triangulation);
55 
56  // Add cylindrical manifold
57  const Tensor<1, dim> direction({0, 0, 1});
58  const CylindricalManifold<dim, dim> manifold(direction, {0, 0, 1});
59  triangulation.set_manifold(0, manifold);
60 
61  // Refine all cells
62  triangulation.prepare_coarsening_and_refinement();
63  triangulation.refine_global(n_refinement);
64 }

72 template <int dim>
73 std::vector<Point<dim>>
74 read_positions(std::string &positions_filename)
75 {
76  // Read text file with positions and store it in vector
77  std::ifstream particle_file(positions_filename);
78 
79  std::string skip;
80  std::getline(particle_file, skip); // Skip header line
81  std::vector<double> values;
82  std::copy(std::istream_iterator<double>(particle_file),
83  std::istream_iterator<double>(),
84  std::back_inserter(values));
85 
86  unsigned int number_of_positions = values.size() / dim;
87 
88  // Extract positions
89  std::vector<Point<dim>> positions;
90  for (unsigned int i = 0; i < number_of_positions; i++)
91  {
92  Point<dim> point(values[dim * i],
93  values[dim * i + 1],
94  values[dim * i + 2]);
95 
96  positions.push_back(point);
97  }
98 
99  return positions;
100 }

134 template <int dim>
135 std::vector<Detector<dim>>
136 assign_detector_positions(Parameters::DetectorParameters &detector_parameters)
137 {
138  // Read text file with detector positions and store it in vector
139  std::ifstream detector_file(detector_parameters.detector_positions_file);
140 
141  std::string skip;
142  std::getline(detector_file, skip);
143  // Skip header line if the header is present
144  if (!isalpha(skip[0]))
145  {
146  detector_file.seekg(0, std::ios::beg);
147  }
148  std::vector<double> values;
149  std::copy(std::istream_iterator<double>(detector_file),
150  std::istream_iterator<double>(),
151  std::back_inserter(values));
152 
153  // Get the number of detector (2 positions for 1 detector, face and middle)
154  int number_of_detector = values.size() / (2 * dim);
155 
156  // Extract positions, create point objects and detectors
157  std::vector<Detector<dim>> detectors;
158  for (int i = 0; i < number_of_detector; i++)
159  {
160  Point<dim> face_point(values[2 * dim * i],
161  values[2 * dim * i + 1],
162  values[2 * dim * i + 2]);
163  Point<dim> middle_point(values[2 * dim * i + dim],
164  values[2 * dim * i + dim + 1],
165  values[2 * dim * i + dim + 2]);
166 
167  Detector<dim> detector(detector_parameters, i, face_point, middle_point);
168 
169  detectors.push_back(detector);
170  }
171  return detectors;
172 }
