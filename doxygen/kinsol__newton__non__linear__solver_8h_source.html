67 template <typename VectorType>
68 void
69 KinsolNewtonNonLinearSolver<VectorType>::solve(const bool is_initial_step)
70 {
71 #ifdef DEAL_II_WITH_SUNDIALS
72  bool first_step = is_initial_step;
73 
74  PhysicsSolver<VectorType> *solver = this->physics_solver;
75 
76  auto &local_evaluation_point = solver->get_local_evaluation_point();
77 
78  VectorType &evaluation_point = solver->get_evaluation_point();
79  VectorType &present_solution = solver->get_present_solution();
80 
81  typename SUNDIALS::KINSOL<VectorType>::AdditionalData additional_data;
82  additional_data.function_tolerance = this->params.tolerance;
83  additional_data.maximum_non_linear_iterations = this->params.max_iterations;
84  additional_data.step_tolerance = this->params.tolerance;
85 
86  // Kinsol solution strategy, the default one is line search
87  if (this->params.kinsol_strategy ==
88  Parameters::NonLinearSolver::KinsolStrategy::normal_newton)
89  {
90  additional_data.strategy =
91  SUNDIALS::KINSOL<VectorType>::AdditionalData::SolutionStrategy::newton;
92  }
93  else if (this->params.kinsol_strategy ==
94  Parameters::NonLinearSolver::KinsolStrategy::line_search)
95  {
96  additional_data.strategy = SUNDIALS::KINSOL<
97  VectorType>::AdditionalData::SolutionStrategy::linesearch;
98  }
99  else if (this->params.kinsol_strategy ==
100  Parameters::NonLinearSolver::KinsolStrategy::picard)
101  {
102  additional_data.strategy =
103  SUNDIALS::KINSOL<VectorType>::AdditionalData::SolutionStrategy::picard;
104  }
105 
106  SUNDIALS::KINSOL<VectorType> nonlinear_solver(additional_data);
107 
108  nonlinear_solver.reinit_vector = [&](VectorType &x) {
109  x.reinit(local_evaluation_point);
110  };
111 
112  nonlinear_solver.residual = [&](const VectorType &evaluation_point_for_kinsol,
113  VectorType &residual) {
114  solver->pcout << "Computing residual vector..." << std::endl;
115  evaluation_point = evaluation_point_for_kinsol;
116  solver->apply_constraints();
117  solver->assemble_system_rhs();
118  auto &current_residual = solver->get_system_rhs();
119  residual = current_residual;
120  solver->pcout << " -Residual: " << residual.l2_norm() << std::endl;
121  return 0;
122  };
123 
124  nonlinear_solver.setup_jacobian =
125  [&](const VectorType &present_solution_for_kinsol,
126  const VectorType & /*current_f*/) {
127  solver->pcout << "Computing jacobian matrix..." << std::endl;
128  evaluation_point = present_solution_for_kinsol;
129  solver->assemble_system_matrix();
130  solver->setup_preconditioner();
131  return 0;
132  };
133 
134  nonlinear_solver.solve_with_jacobian = [&](const VectorType & /* residual */,
135  VectorType &dst,
136  const double /* tolerance */) {
137  solver->pcout << "Solving linear system..." << std::endl;
138  solver->solve_linear_system(first_step);
139  dst = solver->get_newton_update();
140  return 0;
141  };
142 
143  local_evaluation_point = present_solution;
144 
145  nonlinear_solver.solve(local_evaluation_point);
146  present_solution = local_evaluation_point;
147 #else
148  (void)is_initial_step;
149  throw std::runtime_error(
150  "Kinsol newton nonlinear solver requires DEAL_II to be compiled with SUNDIALS");
151 #endif // DEAL_II_WITH_SUNDIALS
152 }
