113  /**
114  * @brief Carries out calculation of the rolling resistance torque using the
115  * constant model
116  *
117  * @param particle_properties Particle properties
118  * @param effective_rolling_friction_coefficient Effective rolling friction
119  * coefficient
120  * @param normal_force_norm Normal force norm
121  * @return rolling resistance torque
122  */
123  inline Tensor<1, 3>
124  constant_resistance(const ArrayView<const double> &particle_properties,
125  const double effective_rolling_friction_coefficient,
126  const double normal_force_norm,
127  const Tensor<1, 3> & /*normal_contact_vector*/)
128  {
129  // Getting the angular velocity of particle in the vector format
130  Tensor<1, 3> angular_velocity;
131  for (int d = 0; d < 3; ++d)
132  {
133  angular_velocity[d] =
134  particle_properties[DEM::PropertiesIndex::omega_x + d];
135  }
136 
137  // Calculation of particle-wall angular velocity (norm of the
138  // particle angular velocity)
139  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
140 
141  double omega_value = angular_velocity.norm();
142  if (omega_value != 0)
143  {
144  particle_wall_angular_velocity = angular_velocity / omega_value;
145  }
146 
147  // Calculation of rolling resistance torque
148  Tensor<1, 3> rolling_resistance_torque =
149  -effective_rolling_friction_coefficient *
150  (particle_properties[DEM::PropertiesIndex::dp] * 0.5) *
151  normal_force_norm * particle_wall_angular_velocity;
152 
153  return rolling_resistance_torque;
154  }

156  /**
157  * @brief Carries out calculation of the rolling resistance torque using the
158  * viscous model
159  *
160  * @param particle_properties Particle properties
161  * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
162  * @param normal_force_norm Normal force norm
163  * @return rolling resistance torque
164  */
165  inline Tensor<1, 3>
166  viscous_resistance(const ArrayView<const double> &particle_properties,
167  const double effective_rolling_friction_coefficient,
168  const double normal_force_norm,
169  const Tensor<1, 3> &normal_contact_vector)
170  {
171  // Getting the angular velocity of particle in the vector format
172  Tensor<1, 3> angular_velocity;
173  for (int d = 0; d < 3; ++d)
174  {
175  angular_velocity[d] =
176  particle_properties[DEM::PropertiesIndex::omega_x + d];
177  }
178 
179  // Calculation of particle-wall angular velocity (norm of the
180  // particle angular velocity)
181  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
182 
183  double omega_value = angular_velocity.norm();
184  if (omega_value != 0)
185  {
186  particle_wall_angular_velocity = angular_velocity / omega_value;
187  }
188 
189  Tensor<1, 3> v_omega =
190  cross_product_3d(angular_velocity,
191  particle_properties[DEM::PropertiesIndex::dp] * 0.5 *
192  normal_contact_vector);
193 
194  // Calculation of rolling resistance torque
195  Tensor<1, 3> rolling_resistance_torque =
196  -effective_rolling_friction_coefficient *
197  particle_properties[DEM::PropertiesIndex::dp] * 0.5 * normal_force_norm *
198  v_omega.norm() * particle_wall_angular_velocity;
199 
200  return rolling_resistance_torque;
201  }
