35 {
36 public:

40  {}

42  /**
43  * @brief assemble_matrix Interface for the call to matrix assembly
44  * @param scratch_data Scratch data containing the information for Cahn-Hilliard
45  * equations.
46  * It is important to note that the scratch data has to have been re-inited
47  * before calling for matrix assembly.
48  * @param copy_data Destination where the local_rhs and local_matrix are copied to.
49  */
50 
51  virtual void

63  */
64 
65  virtual void
66  assemble_rhs(CahnHilliardScratchData<dim> &scratch_data,

74 /**
75  * @brief Class that assembles the core of the Cahn-Hilliard equation :
76  * \f$ \frac{d \phi}{dt} + (u \cdot \nabla) \phi - \nabla \cdot (M(\phi)\nabla
77  * \eta) = 0 \\ \eta - \frac{\lambda}{\epsilon^2}(\phi^3 - \phi) + \lambda
78  * \nabla^2 \phi + \xi \nabla^2 \eta = 0 \f$
79  *
80  * @tparam dim An integer that denotes the number of spatial dimensions
81  *
82  * @ingroup assemblers
83  */
84 
85

87 class CahnHilliardAssemblerCore : public CahnHilliardAssemblerBase<dim>
