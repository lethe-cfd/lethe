150  void
151  parse(ParameterHandler &prm)
152  {
153  dimensionality.parse_parameters(prm);
154  test.parse_parameters(prm);
155 
156  for (auto physics_name : physics_names)
157  {
158  PhysicsID physics_id = get_physics_id(physics_name);
159  linear_solver[physics_id].parse_parameters(prm, physics_name);
160  non_linear_solver[physics_id].parse_parameters(prm, physics_name);
161  }
162 
163  mesh_adaptation.parse_parameters(prm);
164  mesh.parse_parameters(prm);
165  mesh_box_refinement->parse_parameters(prm);
166  nitsche->parse_parameters(prm);
167  physical_properties.parse_parameters(prm, dimensionality);
168  multiphysics.parse_parameters(prm);
169  timer.parse_parameters(prm);
170  fem_parameters.parse_parameters(prm);
171  laser_parameters->parse_parameters(prm);
172  forces_parameters.parse_parameters(prm);
173  post_processing.parse_parameters(prm);
174  flow_control.parse_parameters(prm);
175  restart_parameters.parse_parameters(prm);
176  boundary_conditions.parse_parameters(prm);
177  boundary_conditions_ht.parse_parameters(prm);
178  boundary_conditions_tracer.parse_parameters(prm);
179  boundary_conditions_vof.parse_parameters(prm);
180  boundary_conditions_cahn_hilliard.parse_parameters(prm);
181  manifolds_parameters.parse_parameters(prm);
182  initial_condition->parse_parameters(prm);
183  analytical_solution->parse_parameters(prm);
184  source_term.parse_parameters(prm);
185  simulation_control.parse_parameters(prm);
186  velocity_sources.parse_parameters(prm);
187  particlesParameters->parse_parameters(prm);
188  multiphysics.parse_parameters(prm);
189  constrain_solid_domain.parse_parameters(prm);
190  stabilization.parse_parameters(prm);
191  ale.parse_parameters(prm);
192  evaporation.parse_parameters(prm);
193  tracer_drift_velocity.parse_parameters(prm);
194 
195  physical_properties_manager.initialize(physical_properties);
196 
197 
198  // Check consistency of parameters parsed in different subsections
199  if (multiphysics.VOF && physical_properties.number_of_fluids != 2)
200  {
201  throw std::logic_error(
202  "Inconsistency in .prm!\n with VOF = true\n use: number of fluids = 2");
203  }
204 
205  if (not(multiphysics.VOF) && post_processing.postprocessed_fluid ==
206  Parameters::FluidIndicator::fluid1)
207  {
208  throw std::logic_error(
209  "Inconsistency in .prm!\n when VOF = false"
210  "\n use (default value): set postprocessed fluid = both"
211  "\n or: set postprocessed fluid = fluid 0");
212  }
213 
214  if (physical_properties.number_of_fluids == 2 &&
215  (!multiphysics.VOF && !multiphysics.cahn_hilliard))
216  {
217  throw std::logic_error(
218  "Inconsistency in .prm!\n "
219  "Number of fluids in 'physical properties' was set to 2,\n "
220  "but neither VOF or cahn hilliard is enabled in the 'multiphysics'.\n ");
221  }
222 
223  // Interface physical property models consistency check
224  if (multiphysics.vof_parameters.surface_tension_force.enable)
225  {
226  std::string constant_surface_tension_model(
227  " subsection fluid-fluid interaction\n"
228  " set first fluid id = 0\n"
229  " set second fluid id = 1\n"
230  " set surface tension model = constant\n"
231  " set surface tension coefficient = $value_of_coefficient\n"
232  " end\n");
233 
234  std::string linear_surface_tension_model(
235  " subsection fluid-fluid interaction\n"
236  " set first fluid id = 0\n"
237  " set second fluid id = 1\n"
238  " set surface tension model = linear\n"
239  " set surface tension coefficient = $value_of_coefficient\n"
240  " set temperature-driven surface tension gradient = $value_of_gradient\n"
241  " end\n");
242 
243  std::string phase_change_surface_tension_model(
244  " subsection fluid-fluid interaction\n"
245  " set first fluid id = 0\n"
246  " set second fluid id = 1\n"
247  " set surface tension model = phase change\n"
248  " set surface tension coefficient = $value_of_coefficient\n"
249  " set temperature-driven surface tension gradient = $value_of_gradient\n"
250  " set solidus temperature = $value_of_solidus_temperature\n"
251  " set liquidus temperature = $value_of_liquidus_temperature\n"
252  " end\n");
253  if (!multiphysics.vof_parameters.surface_tension_force
254  .enable_marangoni_effect) // constant surface tension model
255  {
256  if (physical_properties.number_of_material_interactions == 0)
257  {
258  throw std::logic_error(
259  "Inconsistency in .prm!\n "
260  "In subsection VOF, with surface tension force enabled,\n "
261  "but no material interactions specified in\n "
262  "subsection physical properties.\n "
263  "Use:\n\n"
264  " set number of material interactions = 1\n"
265  " subsection material interaction 0\n"
266  " set type = fluid-fluid\n" +
267  constant_surface_tension_model + " end\n");
268  }
269  else if (multiphysics.VOF &&
270  multiphysics
271  .heat_transfer) // disabled Marangoni effect error
272  {
273  if (!is_constant_surface_tension_model(
274  physical_properties.material_interactions))
275  {
276  throw std::logic_error(
277  "Inconsistency in .prm!\n "
278  "In subsection multiphysics, VOF and heat transfer enabled,\n "
279  "and in subsection physical properties, a non-constant surface\n "
280  "tension model, but Marangoni effect disabled in subsection\n "
281  "surface tension force of subsection VOF. This is necessary to account\n "
282  "for Marangoni effect. In subsection VOF, use:\n\n "
283  " subsection surface tension force\n"
284  " set enable = true\n"
285  " set enable marangoni effect = true\n"
286  " end\n");
287  }
288  }
289  else
290  {
291  if (no_fluid_fluid_interaction_error(
292  physical_properties.material_interactions))
293  {
294  throw std::logic_error(
295  "Inconsistency in .prm!\n "
296  "in subsection VOF, surface tension force enabled,\n "
297  "but no fluid-fluid material interactions specified in \n "
298  "subsection physical properties\n "
299  "Use:\n\n"
300  " subsection material interaction $material_interaction_id\n"
301  " set type = fluid-fluid\n" +
302  constant_surface_tension_model + " end\n");
303  }
304  }
305  }
306  else // non-constant surface tension model
307  {
308  if (physical_properties.number_of_material_interactions == 0)
309  {
310  throw std::logic_error(
311  "Inconsistency in .prm!\n "
312  "In subsection VOF, marangoni effect enabled,\n "
313  "but no material interactions specified in subsection physical\n "
314  "properties. This is necessary to account for Marangoni \n "
315  "effect. In subsection physical properties, use:\n\n"
316  " set number of material interactions = 1\n"
317  " subsection material interaction 0\n"
318  " set type = fluid-fluid\n" +
319  linear_surface_tension_model +
320  "\n"
321  "or: \n\n"
322  " set number of material interactions = 1\n"
323  " subsection material interaction 0\n"
324  " set type = fluid-fluid\n" +
325  phase_change_surface_tension_model + " end\n");
326  }
327  else
328  {
329  if (no_fluid_fluid_interaction_error(
330  physical_properties.material_interactions))
331  {
332  throw std::logic_error(
333  "Inconsistency in .prm!\n "
334  "In subsection VOF, Marangoni effect enabled,\n "
335  "but no fluid-fluid material interactions specified in subsection\n "
336  "physical properties. This is necessary to account for Marangoni\n "
337  "effect. In subsection physical properties, use:\n\n"
338  " subsection material interaction $material_interaction_id\n"
339  " set type = fluid-fluid\n" +
340  linear_surface_tension_model +
341  "\n"
342  "or:\n\n"
343  " set number of material interactions = 1\n"
344  " subsection material interaction 0\n"
345  " set type = fluid-fluid\n" +
346  phase_change_surface_tension_model + " end\n");
347  }
348  if (is_constant_surface_tension_model(
349  physical_properties.material_interactions))
350  {
351  throw std::logic_error(
352  "Inconsistency in .prm!\n "
353  "In subsection VOF, Marangoni effect enabled,\n "
354  "but a constant surface tension model is specified in subsection\n "
355  "physical properties. This is necessary to account for Marangoni \n "
356  "effect. In subsection physical properties, use:\n\n"
357  " subsection material interaction $material_interaction_id\n"
358  " set type = fluid-fluid\n" +
359  linear_surface_tension_model +
360  "\n"
361  "or:\n\n"
362  " set number of material interactions = 1\n"
363  " subsection material interaction 0\n"
364  " set type = fluid-fluid\n" +
365  phase_change_surface_tension_model + " end\n");
366  }
367  }
368  }
369  }
370 
371  if (multiphysics.cahn_hilliard && multiphysics.VOF)
372  {
373  throw std::runtime_error(
374  "Cannot solve a multiphase problem using VOF and Cahn-Hilliard at the same time");
375  }
376 
377  if (multiphysics.cahn_hilliard &&
378  physical_properties.number_of_material_interactions == 0)
379  {
380  throw std::logic_error(
381  "Inconsistency in .prm!\n "
382  "cahn hilliard = true, \n "
383  "but no fluid-fluid material interactions specified in\n "
384  "subsection physical properties\n "
385  "Use:\n\n"
386  " subsection material interaction $material_interaction_id\n"
387  " set type = fluid-fluid\n"
388  " subsection fluid-fluid interaction\n"
389  " set first fluid id = 0\n"
390  " set second fluid id = 1\n"
391  " set cahn hilliard mobility model = constant\n"
392  " set cahn hilliard mobility coefficient = $value_of_coefficient\n"
393  " end\n"
394  " end\n");
395  }
396  if (laser_parameters->activate_laser &&
397  (laser_parameters->laser_type ==
398  Parameters::Laser<
399  dim>::LaserType::gaussian_heat_flux_vof_interface ||
400  laser_parameters->laser_type ==
401  Parameters::Laser<
402  dim>::LaserType::uniform_heat_flux_vof_interface) &&
403  !multiphysics.VOF)
404  {
405  throw std::logic_error(
406  "At the moment, the laser surface heat flux is not implemented for 1 fluid simulations."
407  "Please enable the VOF auxiliary physic in the 'multiphysics' subsection, \n"
408  "specify a 2nd fluid in the 'physical properties' subsection,\n"
409  "and define appropriate initial conditions in the 'initial conditions' subsection.");
410  }
411 
412  if (!multiphysics.heat_transfer && constrain_solid_domain.enable)
413  {
414  throw std::logic_error(
415  "Inconsistency in .prm!\n "
416  "The apply constraints on a solid domain feature is enabled, however\n "
417  "'heat transfer' was not set to 'true' in the 'multiphysics' subsection.\n ");
418  }
419 
420  if (constrain_solid_domain.enable &&
421  physical_properties_manager.get_number_of_fluids() <
422  constrain_solid_domain.number_of_constraints)
423  {
424  std::string n_constraints =
425  Utilities::to_string(constrain_solid_domain.number_of_constraints);
426  std::string n_fluids = Utilities::to_string(
427  physical_properties_manager.get_number_of_fluids());
428  throw std::logic_error(
429  "Inconsistency in .prm!\n "
430  "The number of constraints (" +
431  n_constraints + ") is greater than the number of fluids (" +
432  n_fluids +
433  ").\n "
434  "Only 1 constraint per fluid can be declared.\n ");
435  }
436 
437  if (constrain_solid_domain.enable && multiphysics.cahn_hilliard)
438  {
439  throw std::logic_error(
440  "Inconsistency in .prm!\n "
441  "The current implementation for constraining solid domains with\n "
442  "temperature is not implemented for Cahn Hilliard simulations.\n ");
443  }

446  inline bool
447  no_fluid_fluid_interaction_error(
448  std::vector<Parameters::MaterialInteractions> &material_interactions)
449  {
450  for (const Parameters::MaterialInteractions &material_interaction :
451  material_interactions)
452  {
453  if (material_interaction.material_interaction_type ==
454  Parameters::MaterialInteractions::MaterialInteractionsType::
455  fluid_fluid)
456  {
457  return false;
458  }
459  }
460  return true;
461  }
462 
463  inline bool
464  is_constant_surface_tension_model(
465  std::vector<Parameters::MaterialInteractions> &material_interactions)
466  {
467  for (const Parameters::MaterialInteractions &material_interaction :
468  material_interactions)
469  {
470  if (material_interaction.surface_tension_model !=
471  Parameters::MaterialInteractions::SurfaceTensionModel::constant)
472  {
473  return false;
474  }
475  }
476  return true;
477  }
