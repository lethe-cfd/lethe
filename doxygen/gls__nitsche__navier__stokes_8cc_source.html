64 template <int dim, int spacedim>
65 template <bool assemble_matrix>
66 void
67 GLSNitscheNavierStokesSolver<dim, spacedim>::assemble_nitsche_restriction()
68 {
69  TimerOutput::Scope t(this->computing_timer, "assemble Nitsche restriction");
70 
71  Assert(
72  !this->simulation_parameters.physical_properties_manager.is_non_newtonian(),
73  RequiresConstantViscosity("assemble_nitsche_restriction"));
74 
75  // Viscosity for stabilization constant
76  const double kinematic_viscosity =
77  this->simulation_parameters.physical_properties_manager
78  .get_kinematic_viscosity_scale();
79 
80  // Time steps and inverse time steps which is used for stabilization constant
81  std::vector<double> time_steps_vector =
82  this->simulation_control->get_time_steps_vector();
83  const double dt = time_steps_vector[0];
84  const double sdt = 1. / dt;
85 
86  for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
87  {
88  std::shared_ptr<Particles::ParticleHandler<spacedim>> &solid_ph =
89  solids[i_solid]->get_solid_particle_handler();
90 
91 
92  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
93 
94  std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
95 
96  FullMatrix<double> local_matrix(dofs_per_cell, dofs_per_cell);
97  dealii::Vector<double> local_rhs(dofs_per_cell);
98 
99  Tensor<1, spacedim> velocity;
100  Function<spacedim> *solid_velocity =
101  solids[i_solid]->get_solid_velocity();
102 
103  solid_velocity->set_time(this->simulation_control->get_current_time());
104 
105  // Penalization terms
106  const double beta =
107  this->simulation_parameters.nitsche->nitsche_solids[i_solid]->beta;
108 
109  // Loop over all local particles
110  auto particle = solid_ph->begin();
111  while (particle != solid_ph->end())
112  {
113  local_matrix = 0;
114  local_rhs = 0;
115 
116 
117 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 4)
118  const auto &cell =
119  particle->get_surrounding_cell(*this->triangulation);
120 #else
121  const auto &cell = particle->get_surrounding_cell();
122 #endif
123  double h_cell = 0;
124  if (dim == 2)
125  h_cell = std::sqrt(4. * cell->measure() / M_PI) /
126  this->velocity_fem_degree;
127  else if (dim == 3)
128  h_cell = pow(6 * cell->measure() / M_PI, 1. / 3.) /
129  this->velocity_fem_degree;
130  const double penalty_parameter =
131  1. / std::pow(h_cell * h_cell, double(dim) / double(spacedim));
132  const auto &dh_cell =
133  typename DoFHandler<spacedim>::cell_iterator(*cell,
134  &this->dof_handler);
135  dh_cell->get_dof_indices(fluid_dof_indices);
136 
137  const auto pic = solid_ph->particles_in_cell(cell);
138  Assert(pic.begin() == particle, ExcInternalError());
139  for (const auto &p : pic)
140  {
141  const ArrayView<const double> properties = p.get_properties();
142 
143  velocity = 0;
144  const auto &ref_q = p.get_reference_location();
145  const auto &real_q = p.get_location();
146  const auto &JxW = properties[0];
147 
148  for (unsigned int k = 0; k < dofs_per_cell; ++k)
149  {
150  const auto comp_k =
151  this->fe->system_to_component_index(k).first;
152  if (comp_k < spacedim)
153  {
154  // Get the velocity at non-quadrature point (particle in
155  // fluid)
156  auto &evaluation_point = this->evaluation_point;
157  velocity[comp_k] +=
158  evaluation_point[fluid_dof_indices[k]] *
159  this->fe->shape_value(k, ref_q);
160  }
161  }
162 
163  const double u_mag = velocity.norm();
164 
165 
166  const double tau =
167  this->simulation_control->get_assembly_method() ==
168  Parameters::SimulationControl::TimeSteppingMethod::steady ?
169  1. / std::sqrt(std::pow(2. * u_mag / h_cell, 2) +
170  9 * std::pow(4 * kinematic_viscosity /
171  (h_cell * h_cell),
172  2)) :
173  1. /
174  std::sqrt(
175  std::pow(sdt, 2) + std::pow(2. * u_mag / h_cell, 2) +
176  9 * std::pow(4 * kinematic_viscosity / (h_cell * h_cell),
177  2));
178 
179  for (unsigned int i = 0; i < dofs_per_cell; ++i)
180  {
181  const auto comp_i =
182  this->fe->system_to_component_index(i).first;
183  if (comp_i < spacedim)
184  {
185  if (assemble_matrix)
186  {
187  for (unsigned int j = 0; j < dofs_per_cell; ++j)
188  {
189  const auto comp_j =
190  this->fe->system_to_component_index(j).first;
191  if (comp_i == comp_j)
192  {
193  local_matrix(i, j) +=
194  penalty_parameter * beta *
195  this->fe->shape_value(i, ref_q) *
196  this->fe->shape_value(j, ref_q) * JxW;
197 
198  // Partial jacobian of the strong residual
199  // term so that Nitsche's IB is taken into
200  // account in the stabilization
201  local_matrix(i, j) +=
202  penalty_parameter * beta *
203  this->fe->shape_value(j, ref_q) *
204  (this->fe->shape_grad(i, ref_q) *
205  velocity) *
206  tau * JxW;
207  }
208  }
209  }
210  // Regular residual
211  local_rhs(i) += -penalty_parameter * beta *
212  velocity[comp_i] *
213  this->fe->shape_value(i, ref_q) * JxW +
214  penalty_parameter * beta *
215  solid_velocity->value(real_q, comp_i) *
216  this->fe->shape_value(i, ref_q) * JxW;
217 
218  // Residual due taking into account Nitsche IB in the
219  // strong residual used for GLS stabilization
220  const double strong_residual =
221  -penalty_parameter * beta *
222  (solid_velocity->value(real_q, comp_i) -
223  velocity[comp_i]);
224 
225  const double test_function =
226  this->fe->shape_grad(i, ref_q) * velocity;
227 
228  local_rhs(i) +=
229  -tau * (strong_residual)*test_function * JxW;
230  }
231  }
232  }
233  const AffineConstraints<double> &constraints_used =
234  this->zero_constraints;
235  auto &system_rhs = this->system_rhs;
236  constraints_used.distribute_local_to_global(local_matrix,
237  local_rhs,
238  fluid_dof_indices,
239  this->system_matrix,
240  system_rhs);
241  particle = pic.end();
242  }
243  this->system_matrix.compress(VectorOperation::add);
244  this->system_rhs.compress(VectorOperation::add);
245  }
246 }

248 template <>
249 Tensor<1, 3>
250 GLSNitscheNavierStokesSolver<2, 3>::calculate_forces_on_solid(
251  const unsigned int i_solid)
252 {
253  std::shared_ptr<Particles::ParticleHandler<3>> &solid_ph =
254  solids[i_solid]->get_solid_particle_handler();
255 
256  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;
257 
258  std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
259 
260  Tensor<2, 3> velocity_gradient;
261  double pressure;
262  Tensor<1, 3> normal_vector;
263  Tensor<2, 3> fluid_stress;
264  Tensor<2, 3> fluid_pressure;
265  Tensor<1, 3> force; // to be changed for a vector of tensors when
266  // allowing multiple solids
267 
268 
269  Assert(
270  !this->simulation_parameters.physical_properties_manager.is_non_newtonian(),
271  RequiresConstantViscosity("assemble_nitsche_restriction"));
272 
273  const double kinematic_viscosity =
274  this->simulation_parameters.physical_properties_manager
275  .get_kinematic_viscosity_scale();
276 
277  // Loop over all local particles
278  auto particle = solid_ph->begin();
279  while (particle != solid_ph->end())
280  {
281 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 4)
282  const auto &cell = particle->get_surrounding_cell(*this->triangulation);
283 #else
284  const auto &cell = particle->get_surrounding_cell();
285 #endif
286  const auto &dh_cell =
287  typename DoFHandler<3>::cell_iterator(*cell, &this->dof_handler);
288  dh_cell->get_dof_indices(fluid_dof_indices);
289 
290  const auto pic = solid_ph->particles_in_cell(cell);
291  Assert(pic.begin() == particle, ExcInternalError());
292 
293  // Generate FEField function to evaluate values and gradients
294  // at the particle location
295  auto &evaluation_point = this->evaluation_point;
296 
297 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 4)
298  Functions::
299  FEFieldFunction<3, DoFHandler<3, 3>, TrilinosWrappers::MPI::Vector>
300  fe_field(this->dof_handler, evaluation_point);
301 #else
302  Functions::FEFieldFunction<3, TrilinosWrappers::MPI::Vector> fe_field(
303  this->dof_handler, evaluation_point);
304 #endif
305 
306 
307  fe_field.set_active_cell(dh_cell);
308 
309  for (const auto &p : pic)
310  {
311  const ArrayView<const double> properties = p.get_properties();
312 
313  velocity_gradient = 0;
314  pressure = 0;
315  const auto &q = p.get_location();
316  const auto &JxW = properties[0];
317  normal_vector[0] = -properties[1];
318  normal_vector[1] = -properties[2];
319  normal_vector[2] = -properties[3];
320 
321  for (int k = 0; k < 3; ++k)
322  {
323  velocity_gradient[k] = fe_field.gradient(q, k);
324  }
325 
326  pressure = fe_field.value(q, 3);
327 
328  for (int d = 0; d < 2; ++d)
329  {
330  fluid_pressure[d][d] = pressure;
331  }
332 
333  fluid_stress = kinematic_viscosity *
334  (velocity_gradient + transpose(velocity_gradient)) -
335  fluid_pressure;
336  force += fluid_stress * normal_vector * JxW;
337  }
338 
339  particle = pic.end();
340  }
341  force = Utilities::MPI::sum(force, this->mpi_communicator);
342  return force;
343 }

433 {

435  solids[i_solid]->get_solid_particle_handler();
436 
437  const unsigned int dofs_per_cell = this->fe->dofs_per_cell;

439  std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
440 
441  // Penalization terms
442  const double beta =
443  this->simulation_parameters.nitsche->nitsche_solids[i_solid]->beta;
444  Tensor<1, spacedim> velocity;
445  Function<spacedim> *solid_velocity = solids[i_solid]->get_solid_velocity();
446 
447 
448  Tensor<1, 3> torque;
449  torque = 0;
450 
451  // Todo center of rotation should be parameter passed.
452  Point<spacedim> center_of_rotation =
453  this->simulation_parameters.nitsche->nitsche_solids[i_solid]->cor;
454 
455  // Loop over all local particles
456  auto particle = solid_ph->begin();
457  while (particle != solid_ph->end())
458  {
459 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 4)
460  const auto &cell = particle->get_surrounding_cell(*this->triangulation);
461 #else
462  const auto &cell = particle->get_surrounding_cell();
463 #endif
464  double h_cell = 0;
465  if (dim == 2)
466  h_cell =
467  std::sqrt(4. * cell->measure() / M_PI) / this->velocity_fem_degree;
468  else if (dim == 3)
469  h_cell =
470  pow(6 * cell->measure() / M_PI, 1. / 3.) / this->velocity_fem_degree;
471  const double penalty_parameter =
472  1. / std::pow(h_cell * h_cell, double(dim) / double(spacedim));
473  const auto &dh_cell =
474  typename DoFHandler<spacedim>::cell_iterator(*cell, &this->dof_handler);
475  dh_cell->get_dof_indices(fluid_dof_indices);
476 
477  const auto pic = solid_ph->particles_in_cell(cell);
478  Assert(pic.begin() == particle, ExcInternalError());
479  for (const auto &p : pic)
480  {
481  const ArrayView<const double> properties = p.get_properties();
482 
483  Tensor<1, spacedim> force;
484  force = 0;
485  velocity = 0;
486  const auto &ref_q = p.get_reference_location();
487  const auto &real_q = p.get_location();
488  const auto &JxW = properties[0];
489 
490  for (unsigned int k = 0; k < dofs_per_cell; ++k)
491  {
492  const auto comp_k = this->fe->system_to_component_index(k).first;
493  if (comp_k < spacedim)
494  {
495  // Get the velocity at non-quadrature point (particle in
496  // fluid)
497  auto &evaluation_point = this->evaluation_point;
498  velocity[comp_k] += evaluation_point[fluid_dof_indices[k]] *
499  this->fe->shape_value(k, ref_q);
500  }
501  }
502  for (unsigned int i = 0; i < dofs_per_cell; ++i)
503  {
504  const auto comp_i = this->fe->system_to_component_index(i).first;
505  if (comp_i < spacedim)
506  {
507  force[comp_i] -=
508  penalty_parameter * beta * this->fe->shape_value(i, ref_q) *
509  JxW *
510  (solid_velocity->value(real_q, comp_i) - velocity[comp_i]);
511  }
512  }
513  // Calculate torque on particle location
514  auto distance = real_q - center_of_rotation;
515 
516  if (dim == 2)
517  {
518  torque[0] = 0.;
519  torque[1] = 0.;
520  torque[2] += distance[0] * force[1] - distance[1] * force[0];
521  }
522  else if (dim == 3)
523  {
524  torque[0] += distance[1] * force[2] - distance[2] * force[1];
525  torque[1] += distance[2] * force[0] - distance[0] * force[2];
526  torque[2] += distance[0] * force[1] - distance[1] * force[0];
527  }
528  }
529  particle = pic.end();
530  }
531  torque = Utilities::MPI::sum(torque, this->mpi_communicator);
532  return torque;
533 }

538  const unsigned int i_solid,
539  bool first_solid_forces)

542

561  solid_indices,
562  independent_column_names,
563  force,
564  dependent_column_names,
565  this->simulation_parameters.simulation_control.log_precision);
566

579  table.write_text(std::cout);
580  }
581

599  solid_forces_table[i_solid].add_value("f_z", 0.);
600 
601  // Precision
602  solid_forces_table[i_solid].set_precision(
603  "f_x", this->simulation_parameters.forces_parameters.output_precision);
604  solid_forces_table[i_solid].set_precision(
605  "f_y", this->simulation_parameters.forces_parameters.output_precision);
606  solid_forces_table[i_solid].set_precision(
607  "f_z", this->simulation_parameters.forces_parameters.output_precision);
608 
609  std::string filename_force =
610  this->simulation_parameters.simulation_control.output_folder +
611  this->simulation_parameters.nitsche->nitsche_solids[i_solid]
612  ->force_output_name +
613  "_" + Utilities::int_to_string(i_solid, 2) + ".dat";
614  std::ofstream output_force(filename_force.c_str());
615 
616  solid_forces_table[i_solid].write_text(output_force);
617 }

702 template <int dim, int spacedim>
703 void
704 GLSNitscheNavierStokesSolver<dim, spacedim>::solve()
705 {
706  MultithreadInfo::set_thread_limit(1);
707 
708  // Fluid setup
709  read_mesh_and_manifolds(
710  *this->triangulation,
711  this->simulation_parameters.mesh,
712  this->simulation_parameters.manifolds_parameters,
713  this->simulation_parameters.restart_parameters.restart,
714  this->simulation_parameters.boundary_conditions);
715 
716  this->setup_dofs();
717 
718  // Solid setup
719  if (!this->simulation_parameters.restart_parameters.restart)
720  {
721  TimerOutput::Scope t(this->computing_timer,
722  "Nitsche solid mesh and particles");
723  for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
724  {
725  solids[i_solid]->initial_setup();
726 
727  // Output initial configuration, if output_frequency!=0
728  if (this->simulation_control->is_output_iteration())
729  {
730  output_solid_particles(i_solid);
731  output_solid_triangulation(i_solid);
732  }
733  }
734  if constexpr (dim == spacedim)
735  {
736  // Parse the nitsche solids to the multiphysics interface
737  this->multiphysics->set_solid(&solids);
738  }
739  }
740 
741  this->set_initial_condition(
742  this->simulation_parameters.initial_condition->type,
743  this->simulation_parameters.restart_parameters.restart);
744 
745 
746 
747  // Time integration
748  while (this->simulation_control->integrate())
749  {
750  this->simulation_control->print_progression(this->pcout);
751 
752  {
753  TimerOutput::Scope t(this->computing_timer, "Nitsche particles motion");
754  for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
755  {
756  if (this->simulation_parameters.nitsche->nitsche_solids[i_solid]
757  ->enable_particles_motion)
758  {
759  // Particle and solid displacement is explicit, thus it must go
760  // from t to t+dt
761  const double time_step =
762  this->simulation_control->get_time_step();
763  const double initial_time =
764  this->simulation_control->get_current_time() - time_step;
765 
766  solids[i_solid]->update_temperature_time(time_step);
767  solids[i_solid]->integrate_velocity(time_step, initial_time);
768  solids[i_solid]->move_solid_triangulation(time_step,
769  initial_time);
770  }
771  }
772  if constexpr (dim == spacedim)
773  {
774  // Parse the nitsche solids to the multiphysics interface
775  this->multiphysics->set_solid(&solids);
776  }
777  }
778  if (this->simulation_control->is_at_start())
779  this->iterate();
780  else
781  {
782  this->refine_mesh();
783  this->iterate();
784  }
785 
786  this->postprocess(false);
787  bool first_solid_forces(true);
788  bool first_solid_torques(true);
789 
790  for (unsigned int i_solid = 0;
791  i_solid < this->simulation_parameters.nitsche->number_solids;
792  ++i_solid)
793  {
794  if (this->simulation_parameters.nitsche->nitsche_solids[i_solid]
795  ->calculate_force_on_solid)
796  {
797  postprocess_solid_forces(i_solid, first_solid_forces);
798  first_solid_forces = false;
799  }
800  if (this->simulation_parameters.nitsche->nitsche_solids[i_solid]
801  ->calculate_torque_on_solid)
802  {
803  // for 22 and 33 simulations
804  //(to be implemented for 23 simulations)
805  if (dim == spacedim)
806  {
807  postprocess_solid_torques(i_solid, first_solid_torques);
808  first_solid_torques = false;
809  }
810  }
811 
812  if (this->simulation_control->is_output_iteration())
813  {
814  output_solid_particles(i_solid);
815  output_solid_triangulation(i_solid);
816  }
817  }
818  this->finish_time_step();
819  }
820  if (this->simulation_parameters.test.enabled)
821  {
822  for (unsigned int i_solid = 0; i_solid < solids.size(); ++i_solid)
823  {
824  solids[i_solid]->print_particle_positions();
825  }
826  }
827  this->finish_simulation();
828 }
