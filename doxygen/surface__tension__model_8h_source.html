140 /**
141  * @brief Linear surface tension. The surface tension is given by:
142  * sigma_0 + dsimga/dT * (T-T_0).
143  */
144 class SurfaceTensionLinear : public SurfaceTensionModel
145 {
146 public:
147  /**
148  * @brief Default constructor
149  * @param p_surface_tension_parameters The parameters for the surface tension
150  * model, including the surface tension coefficient sigma_0 and surface
151  * tension gradient dsimga/dT at the reference temperature T_0
152  */
153  SurfaceTensionLinear(
154  const Parameters::SurfaceTensionParameters &p_surface_tension_parameters)
155  : surface_tension_coefficient(
156  p_surface_tension_parameters.surface_tension_coefficient)
157  , T_0(p_surface_tension_parameters.T_0)
158  , surface_tension_gradient(
159  p_surface_tension_parameters.surface_tension_gradient)
160  {
161  this->model_depends_on[field::temperature] = true;
162  }
163 
164  /**
165  * @brief value Calculates the surface tension coefficient
166  * @param field_vectors Vectors of the fields on which the surface tension
167  * depends. In this case, it depends on the temperature.
168  * @return value of the physical property calculated with the fields_value
169  */
170  double
171  value(const std::map<field, double> &fields_value) override
172  {
173  AssertThrow(fields_value.find(field::temperature) != fields_value.end(),
174  PhysicialPropertyModelFieldUndefined("SurfaceTensionLinear",
175  "temperature"));
176  const double temperature = fields_value.at(field::temperature);
177  return surface_tension_coefficient +
178  surface_tension_gradient * (temperature - T_0);
179  }
180 
181  /**
182  * @brief vector_value Calculates the vector of surface tension coefficient.
183  * @param field_vectors Vectors of the fields on which the surface tension
184  * depends. In this case, it depends on the temperature.
185  * @param property_vector Vectors of the surface tension coefficient values
186  */
187  void
188  vector_value(const std::map<field, std::vector<double>> &field_vectors,
189  std::vector<double> &property_vector) override
190  {
191  AssertThrow(field_vectors.find(field::temperature) != field_vectors.end(),
192  PhysicialPropertyModelFieldUndefined("SurfaceTensionLinear",
193  "temperature"));
194  const std::vector<double> &temperature =
195  field_vectors.at(field::temperature);
196  for (unsigned int i = 0; i < property_vector.size(); ++i)
197  property_vector[i] = surface_tension_coefficient +
198  surface_tension_gradient * (temperature[i] - T_0);
199  }
200 
201  /**
202  * @brief jacobian Calculates the jacobian (the partial derivative) of the
203  * surface tension coefficient with respect to a field
204  * @param field_values Value of the various fields on which the property may
205  * depend.
206  * @param id Indicator of the field with respect to which the jacobian
207  * should be calculated.
208  * @return value of the partial derivative of the surface tension coefficient
209  * with respect to the field.
210  */
211  double
212  jacobian(const std::map<field, double> & /*field_values*/, field id) override
213  {
214  if (id == field::temperature)
215  return surface_tension_gradient;
216  else
217  return 0;
218  }
219 
220  /**
221  * @brief vector_jacobian Calculates the derivative of the surface tension
222  * coefficient with respect to a field.
223  * @param field_vectors Vector for the values of the fields used to evaluate
224  * the property.
225  * @param id Identifier of the field with respect to which a derivative should
226  * be calculated.
227  * @param jacobian vector of the value of the derivative of the surface
228  * tension coefficient with respect to the field id.
229  */
230  void
231  vector_jacobian(
232  const std::map<field, std::vector<double>> & /*field_vectors*/,
233  const field id,
234  std::vector<double> &jacobian_vector) override
235  {
236  if (id == field::temperature)
237  std::fill(jacobian_vector.begin(),
238  jacobian_vector.end(),
239  surface_tension_gradient);
240  else
241  std::fill(jacobian_vector.begin(), jacobian_vector.end(), 0);
242  }
243 
244 protected:
245  const double surface_tension_coefficient;
246  const double T_0;
247  const double surface_tension_gradient;
248 };

273  /**
274  * @brief value Calculates the surface tension coefficient.
275  * @param field_vectors Vectors of the fields on which the surface tension
276  * depends. In this case, it depends on the temperature.
277  * @return value of the physical property calculated with the fields_value.
278  */
279  double
280  value(const std::map<field, double> &fields_value) override
281  {
282  AssertThrow(fields_value.find(field::temperature) != fields_value.end(),
283  PhysicialPropertyModelFieldUndefined(
284  "SurfaceTensionPhaseChange", "temperature"));
285  const double temperature = fields_value.at(field::temperature);
286 
287  double surface_tension;
288 
289  if (temperature < T_solidus)
290  surface_tension = 0.0;
291  else if (temperature > T_liquidus)
292  surface_tension = surface_tension_coefficient +
293  surface_tension_gradient * (temperature - T_0);
294  else
295  {
296  const double l_frac =
297  calculate_liquid_fraction(temperature, T_solidus, T_liquidus);
298 
299  surface_tension = (surface_tension_coefficient +
300  surface_tension_gradient * (temperature - T_0)) *
301  l_frac;
302  }
303  return surface_tension;
304  }
