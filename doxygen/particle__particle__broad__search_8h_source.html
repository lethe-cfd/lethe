149  &particles_to_evaluate,
150  std::vector<types::particle_index> &contact_pair_candidates_container)
151  {
152  // Create a arbitrary temporary empty container
153  if (contact_pair_candidates_container.empty())
154  {
155  contact_pair_candidates_container.reserve(40);
156  }
157 
158  // Store particle ids from the selected particle iterator
159  for (auto particle_iterator = particle_begin;
160  particle_iterator != particles_to_evaluate.end();
161  ++particle_iterator)
162  {
163  contact_pair_candidates_container.emplace_back(
164  particle_iterator->get_id());
