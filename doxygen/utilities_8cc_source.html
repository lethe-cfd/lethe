440 std::string
441 get_last_value_of_parameter(const std::string &file_name,
442  const std::string &parameter_name)
443 {
444  std::string return_value;
445 
446  std::ifstream x_file(file_name);
447  AssertThrow(x_file.fail() == false, ExcIO());
448 
449  while (x_file)
450  {
451  // Get one line and then match a regex to it that matches the parameter
452  // we are looking for. Before we do that, strip spaces from the front
453  // and back of the line:
454  std::string line;
455  std::getline(x_file, line);
456 
457  while ((line.size() > 0) && (line[0] == ' ' || line[0] == '\t'))
458  line.erase(0, 1);
459  while ((line.size() > 0) &&
460  (line[line.size() - 1] == ' ' || line[line.size() - 1] == '\t'))
461  line.erase(line.size() - 1, std::string::npos);
462 
463  std::match_results<std::string::const_iterator> matches;
464  const std::string regex =
465  "set[ \t]+" + parameter_name + "[ \t]*=[ \t]*(.*)";
466  if (std::regex_match(line, matches, std::regex(regex)))
467  {
468  // Since the line as a whole matched, the 'matches' variable needs to
469  // contain two entries: [0] denotes the whole string, and [1] the
470  // one that was matched by the '(.*)' expression.
471  Assert(matches.size() == 2, dealii::ExcInternalError());
472  return_value = std::string(matches[1].first, matches[1].second);
473  }
474  }
475 
476  return return_value;
477 }

524 unsigned int
525 get_dimension(const std::string &file_name)
526 {
527  const std::string dimension =
528  get_last_value_of_parameter(file_name, "dimension");
529 
530  if (dimension.size() > 0)
531  {
532  // Extracted from ASPECT
533  // A common problem is that people have .prm files that were generated
534  // on Windows, but then run this on Linux where the line endings are
535  // different. This is pernicious because it means that the conversion
536  // of a string such as "2\r" to an integer fails, but if we print
537  // this string, it comes out completely garbled because it contains
538  // a carriage-return without a newline -- so the error message looks
539  // like this:
540  //
541  // >. While reading the dimension from the input file, ASPECT found a
542  // string that can not be converted to an integer: <2
543  //
544  // Note how the end of the error message overwrites the beginning
545  // of the line.
546  //
547  // To avoid this kind of error, specifically test up front that the
548  // text in question does not contain '\r' characters. If we are on
549  // linux, then this kind of character would means that the line endings
550  // are wrong. On the other hand, if we are on windows, then the
551  // getline command we have used in finding 'dimension' would have
552  // filtered it out. So its presence points to a problem.
553 
554  AssertThrow(
555  dimension.find('\r') == std::string::npos,
556  dealii::ExcMessage(
557  "It appears that your input file uses Windows-style "
558  "line endings ('\\r\\n') but you are running on a system where "
559  "the C++ run time environment expects input files to have "
560  "Unix-style line endings ('\\n'). You need to convert your "
561  "input file to use the correct line endings before running "
562  "ASPECT with it."));
563  try
564  {
565  return dealii::Utilities::string_to_int(dimension);
566  }
567  catch (...)
568  {
569  AssertThrow(false,
570  dealii::ExcMessage(
571  "While reading the dimension from the input file, "
572  "Lethe found a string that can not be converted to "
573  "an integer: <" +
574  dimension + ">."));
575  return 0; // we should never get here.
576  }
577  }
578  else
579  {
580  AssertThrow(
581  false,
582  dealii::ExcMessage(
583  "While reading the dimension from the input file, "
584  "Lethe found a value that is neither 2 or 3. Since August 2023, "
585  "Lethe requires that the user explicitly specify the dimension of the problem within the parameter file. This can be achieved by adding set dimension = 2 or set dimension = 3 within the parameter file"));
586  }
587 }
