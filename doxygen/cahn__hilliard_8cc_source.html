291 template <int dim>
292 std::pair<double, double>
293 CahnHilliard<dim>::calculate_L2_error()
294 {
295  auto mpi_communicator = triangulation->get_communicator();
296 
297  FEValues<dim> fe_values(*this->mapping,
298  *fe,
299  *this->cell_quadrature,
300  update_values | update_gradients |
301  update_quadrature_points | update_JxW_values);
302 
303  const FEValuesExtractors::Scalar phase_order(0);
304  const FEValuesExtractors::Scalar chemical_potential(1);
305  const unsigned int n_q_points = this->cell_quadrature->size();
306 
307  std::vector<Vector<double>> q_exactSol(n_q_points, Vector<double>(2));
308 
309  std::vector<double> local_phase_order_values(n_q_points);
310  std::vector<double> local_chemical_potential_values(n_q_points);
311 
312  auto &exact_solution =
313  simulation_parameters.analytical_solution->cahn_hilliard;
314  exact_solution.set_time(simulation_control->get_current_time());
315 
316  double l2_error_phase_order = 0.;
317  double l2_error_chemical_potential = 0.;
318 
319  for (const auto &cell : dof_handler.active_cell_iterators())
320  {
321  if (cell->is_locally_owned())
322  {
323  fe_values.reinit(cell);
324  fe_values[phase_order].get_function_values(evaluation_point,
325  local_phase_order_values);
326  fe_values[chemical_potential].get_function_values(
327  evaluation_point, local_chemical_potential_values);
328 
329  // Get the exact solution at all Gauss points
330  exact_solution.vector_value_list(fe_values.get_quadrature_points(),
331  q_exactSol);
332 
333  for (unsigned int q = 0; q < n_q_points; q++)
334  {
335  // Find the values of x and u_h (the finite element solution) at
336  // the quadrature points
337  double phase_order_sim = local_phase_order_values[q];
338  double phase_order_exact = q_exactSol[q][0];
339 
340  l2_error_phase_order += (phase_order_sim - phase_order_exact) *
341  (phase_order_sim - phase_order_exact) *
342  fe_values.JxW(q);
343 
344  double chemical_potential_sim =
345  local_chemical_potential_values[q];
346  double chemical_potential_exact = q_exactSol[q][1];
347 
348  l2_error_chemical_potential +=
349  (chemical_potential_sim - chemical_potential_exact) *
350  (chemical_potential_sim - chemical_potential_exact) *
351  fe_values.JxW(q);
352  }
353  }
354  }
355  l2_error_phase_order =
356  Utilities::MPI::sum(l2_error_phase_order, mpi_communicator);
357  l2_error_chemical_potential =
358  Utilities::MPI::sum(l2_error_chemical_potential, mpi_communicator);
359 
360  return std::make_pair(std::sqrt(l2_error_phase_order),
361  std::sqrt(l2_error_chemical_potential));
362 }

365 template <int dim>
366 void
367 CahnHilliard<dim>::calculate_phase_statistics()
368 {
369  auto mpi_communicator = triangulation->get_communicator();
370 
371  FEValues<dim> fe_values(*mapping,
372  *fe,
373  *cell_quadrature,
374  update_values | update_gradients |
375  update_quadrature_points | update_JxW_values);
376 
377  const FEValuesExtractors::Scalar phase_order(0);
378 
379  const unsigned int n_q_points = cell_quadrature->size();
380  std::vector<double> local_phase_order_values(n_q_points);
381 
382  double integral = 0;
383  double max_phase_value = std::numeric_limits<double>::min();
384  double min_phase_value = std::numeric_limits<double>::max();
385 
386  for (const auto &cell : dof_handler.active_cell_iterators())
387  {
388  if (cell->is_locally_owned())
389  {
390  fe_values.reinit(cell);
391  fe_values[phase_order].get_function_values(present_solution,
392  local_phase_order_values);
393 
394  for (unsigned int q = 0; q < n_q_points; q++)
395  {
396  integral += local_phase_order_values[q] * fe_values.JxW(q);
397  max_phase_value =
398  std::max(local_phase_order_values[q], max_phase_value);
399  min_phase_value =
400  std::min(local_phase_order_values[q], min_phase_value);
401  }
402  }
403  }
404 
405  integral = Utilities::MPI::sum(integral, mpi_communicator);
406  double global_volume = GridTools::volume(*triangulation, *mapping);
407  double phase_average = integral / global_volume;
408 
409 
410  // Console output
411  if (simulation_parameters.post_processing.verbosity ==
412  Parameters::Verbosity::verbose)
413  {
414  announce_string(this->pcout, "Phase statistics");
415  this->pcout << "Min: " << min_phase_value << std::endl;
416  this->pcout << "Max: " << max_phase_value << std::endl;
417  this->pcout << "Average: " << phase_average << std::endl;
418  this->pcout << "Integral: " << integral << std::endl;
419  }
420 
421  statistics_table.add_value("time", simulation_control->get_current_time());
422  statistics_table.add_value("min", min_phase_value);
423  statistics_table.add_value("max", max_phase_value);
424  statistics_table.add_value("average", phase_average);
425  statistics_table.add_value("integral", integral);
426 }

862

864  i_bc <
865  this->simulation_parameters.boundary_conditions_cahn_hilliard.size;
866  ++i_bc)

868  ComponentMask mask(2, true);

885  .phi),
886  nonzero_constraints,
887  mask);
888  }
889  }
890  }

895  zero_constraints.clear();

902  ++i_bc)
903  {
904  if (this->simulation_parameters.boundary_conditions_cahn_hilliard

908  VectorTools::interpolate_boundary_values(
909  this->dof_handler,
910  this->simulation_parameters.boundary_conditions_cahn_hilliard

915  }
916  }

1026 void
1027 CahnHilliard<dim>::solve_linear_system(const bool initial_step,
1028  const bool /*renewed_matrix*/)
1029 {
1030  TimerOutput::Scope t(this->computing_timer, "Solve linear system");
1031 
1032  auto mpi_communicator = triangulation->get_communicator();
1033 
1034  const AffineConstraints<double> &constraints_used =
1035  initial_step ? nonzero_constraints : this->zero_constraints;
1036 
1037  const double absolute_residual =
1038  simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1039  .minimum_residual;
1040  const double relative_residual =
1041  simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1042  .relative_residual;
1043 
1044  const double linear_solver_tolerance =
1045  std::max(relative_residual * system_rhs.l2_norm(), absolute_residual);
1046 
1047  if (this->simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1048  .verbosity != Parameters::Verbosity::quiet)
1049  {

1051  << linear_solver_tolerance << std::endl;
1052  }
1053 
1054  const double ilu_fill =
1055  simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1056  .ilu_precond_fill;
1057  const double ilu_atol =
1058  simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1059  .ilu_precond_atol;
1060  const double ilu_rtol =
1061  simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1062  .ilu_precond_rtol;
1063  TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
1064  ilu_fill, ilu_atol, ilu_rtol, 0);
1065 
1066  TrilinosWrappers::PreconditionILU ilu_preconditioner;
1067 
1068  ilu_preconditioner.initialize(system_matrix, preconditionerOptions);
1069 
1070  GlobalVectorType completely_distributed_solution(locally_owned_dofs,
1071  mpi_communicator);
1072 
1073  SolverControl solver_control(simulation_parameters.linear_solver
1074  .at(PhysicsID::cahn_hilliard)
1075  .max_iterations,
1076  linear_solver_tolerance,
1077  true,
1078  true);
1079 
1080  TrilinosWrappers::SolverGMRES::AdditionalData solver_parameters(
1081  false,
1082  simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1083  .max_krylov_vectors);
1084 
1085 
1086  TrilinosWrappers::SolverGMRES solver(solver_control, solver_parameters);
1087 
1088 
1089  solver.solve(system_matrix,
1090  completely_distributed_solution,
1091  system_rhs,
1092  ilu_preconditioner);
1093 
1094  if (simulation_parameters.linear_solver.at(PhysicsID::cahn_hilliard)
1095  .verbosity != Parameters::Verbosity::quiet)
1096  {
1097  this->pcout << " -Iterative solver took : " << solver_control.last_step()
1098  << " steps " << std::endl;
1099  }
1100 
1101  constraints_used.distribute(completely_distributed_solution);
1102  newton_update = completely_distributed_solution;
1103 }

1106 template <typename VectorType>
1107 std::pair<Tensor<1, dim>, Tensor<1, dim>>
1108 CahnHilliard<dim>::calculate_barycenter(const GlobalVectorType &solution,
1109  const VectorType &solution_fd)
1110 {
1111  const MPI_Comm mpi_communicator = this->triangulation->get_communicator();
1112 
1113  FEValues<dim> fe_values_cahn_hilliard(*this->mapping,
1114  *this->fe,
1115  *this->cell_quadrature,
1116  update_values |
1117  update_quadrature_points |
1118  update_JxW_values);
1119 
1120  const DoFHandler<dim> *dof_handler_fd =
1121  multiphysics->get_dof_handler(PhysicsID::fluid_dynamics);
1122 
1123  FEValues<dim> fe_values_fd(*this->mapping,
1124  dof_handler_fd->get_fe(),
1125  *this->cell_quadrature,
1126  update_values);
1127 
1128  const unsigned int n_q_points = this->cell_quadrature->size();
1129  std::vector<double> phase_cahn_hilliard_values(n_q_points);
1130  std::vector<Tensor<1, dim>> velocity_values(n_q_points);
1131  std::vector<Point<dim>> quadrature_locations(n_q_points);
1132 
1133  const FEValuesExtractors::Vector velocity(0);
1134  const FEValuesExtractors::Scalar phase_order(0);
1135 
1136  Tensor<1, dim> barycenter_location;
1137  Tensor<1, dim> barycenter_velocity;
1138  double volume = 0;
1139 
1140 
1141  std::map<field, std::vector<double>> fields;
1142 
1143  for (const auto &cell : this->dof_handler.active_cell_iterators())
1144  {
1145  if (cell->is_locally_owned())
1146  {
1147  fe_values_cahn_hilliard.reinit(cell);
1148  quadrature_locations =
1149  fe_values_cahn_hilliard.get_quadrature_points();
1150  fe_values_cahn_hilliard[phase_order].get_function_values(
1151  solution, phase_cahn_hilliard_values);
1152 
1153  // Get fluid dynamics active cell iterator
1154  typename DoFHandler<dim>::active_cell_iterator cell_fd(
1155  &(*(this->triangulation)),
1156  cell->level(),
1157  cell->index(),
1158  dof_handler_fd);
1159 
1160  fe_values_fd.reinit(cell_fd);
1161  fe_values_fd[velocity].get_function_values(solution_fd,
1162  velocity_values);
1163 
1164  for (unsigned int q = 0; q < n_q_points; q++)
1165  {
1166  const double JxW = fe_values_cahn_hilliard.JxW(q);
1167 
1168 
1169  volume += (1 - phase_cahn_hilliard_values[q]) * 0.5 * JxW;
1170  barycenter_location += (1 - phase_cahn_hilliard_values[q]) * 0.5 *
1171  quadrature_locations[q] * JxW;
1172  barycenter_velocity += (1 - phase_cahn_hilliard_values[q]) * 0.5 *
1173  velocity_values[q] * JxW;
1174  }
1175  }
1176  }
1177 
1178  volume = Utilities::MPI::sum(volume, mpi_communicator);
1179  barycenter_location =
1180  Utilities::MPI::sum(barycenter_location, mpi_communicator) / volume;
1181  barycenter_velocity =
1182  Utilities::MPI::sum(barycenter_velocity, mpi_communicator) / volume;
1183 
1184  return std::pair<Tensor<1, dim>, Tensor<1, dim>>(barycenter_location,
1185  barycenter_velocity);
1186 }
