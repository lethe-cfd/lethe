31  FEFaceValues<dim> fe_face_values(fe,
32  face_quadrature_formula,
33  update_values | update_quadrature_points |
34  update_normal_vectors);
35  Tensor<1, dim> normal_vector;
36  Point<dim> quad_point;
37 
38  // Store information of the cell
39  boundaries_information.cell = cell;
40  boundaries_information.boundary_id = cell->face(face_id)->boundary_id();
41 
42  // Store information of the periodic cell
43  boundaries_information.periodic_cell = cell->periodic_neighbor(face_id);
44

46  fe_face_values.reinit(cell, face_id);
47  boundaries_information.normal_vector = fe_face_values.normal_vector(0);
48  boundaries_information.point_on_face = fe_face_values.quadrature_point(0);
49 
50  // Find the normal vector of the periodic boundary face and point with the
51  // reinit FE face values

53  cell->periodic_neighbor_face_no(face_id));
54  normal_vector = fe_face_values.normal_vector(0);
55  quad_point = fe_face_values.quadrature_point(0);
56  boundaries_information.periodic_normal_vector = normal_vector;
57  boundaries_information.point_on_periodic_face = quad_point;

121 template <int dim>
122 void
123 PeriodicBoundariesManipulator<dim>::check_and_move_particles(
124  const periodic_boundaries_cells_info_struct<dim> &boundaries_cells_content,
125  typename Particles::ParticleHandler<dim>::particle_iterator_range
126  &particles_in_cell,
127  bool &particles_in_pb0_cell,
128  bool &particle_has_been_moved)
129 {
130  for (auto particle = particles_in_cell.begin();
131  particle != particles_in_cell.end();
132  ++particle)
133  {
134  // Get the current particle location
135  Point<dim> particle_position = particle->get_location();
136 
137  // Initialize points and normal vector related of the cell that contains
138  // the particle
139  Point<dim> point_on_face, point_on_periodic_face;
140  Tensor<1, dim> normal_vector, distance_between_faces;
141  if (particles_in_pb0_cell)
142  {
143  point_on_face = boundaries_cells_content.point_on_face;
144  normal_vector = boundaries_cells_content.normal_vector;
145  distance_between_faces = constant_periodic_offset;
146  }
147  else
148  {
149  point_on_face = boundaries_cells_content.point_on_periodic_face;
150  normal_vector = boundaries_cells_content.periodic_normal_vector;
151  distance_between_faces = -constant_periodic_offset;
152  }
153 
154  // Calculate distance between particle position and the face of
155  // boundary cell d = n•(pt_particle - pt_face)
156  double distance_with_face =
157  scalar_product(particle_position - point_on_face, normal_vector);
158 
159  // If distance >= 0, particle is outside of cell (or on face).
160  // If so, particle location is modified to get moved into the periodic
161  // cell.
162  if (distance_with_face >= 0.0)
163  {
164  // Move particle outside the current cell to the periodic cell.
165  particle_position += distance_between_faces;
166  particle->set_location(particle_position);
167 
168  // Update flag to indicate that particle has been moved
169  particle_has_been_moved = true;
170  }
171  }
172 }
