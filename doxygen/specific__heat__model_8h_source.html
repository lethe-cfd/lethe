119 /**
120  * @brief This model takes into account the phase change of a material
121  * by considering the latent heat into the specific heat over a phase change
122  * interval determined by [T_solidus,T_liquidus].
123  */
124 class PhaseChangeSpecificHeat : public SpecificHeatModel
125 {
126 public:
127  /**
128  * @brief Default constructor
129  */
130  PhaseChangeSpecificHeat(
131  const Parameters::PhaseChange p_phase_change_parameters)
132  : param(p_phase_change_parameters)
133  {
134  this->model_depends_on[field::temperature] = true;
135  this->model_depends_on[field::temperature_p1] = true;
136  this->model_depends_on[field::temperature_p2] = true;
137  }
138 
139  /**
140  * @brief value Calculates the value of the phase change specific heat.
141  * @param fields_value Value of the various field on which the specific heat depends.
142  * @return value of the specific heat. */
143  double
144  value(const std::map<field, double> &fields_value) override
145  {
146  const double temperature = fields_value.at(field::temperature);
147  const double temperature_p1 = fields_value.at(field::temperature_p1);
148 
149  const double temperature_p2 = fields_value.at(field::temperature_p2);
150 
151 
152  // Gather information required from the simulation control to have the time
153  // history
154  std::shared_ptr<SimulationControl> simulation_control =
155  get_simulation_control();
156 
157  auto method = simulation_control->get_assembly_method();
158 
159  std::vector<double> time_steps_vector =
160  simulation_control->get_time_steps_vector();
161 
162  Vector<double> bdf_coefs = bdf_coefficients(method, time_steps_vector);
163 
164 
165  // If change between the temperature is insufficient, backtrack to the first
166  // order implementation
167  if (method != Parameters::SimulationControl::TimeSteppingMethod::bdf1 &&
168  std::abs(temperature - temperature_p2) < 1e-6)
169  method = Parameters::SimulationControl::TimeSteppingMethod::bdf1;
170 
171  switch (method)
172  {
173  case Parameters::SimulationControl::TimeSteppingMethod::bdf1:
174  if (temperature > temperature_p1)
175  {
176  const double dT = std::max(temperature - temperature_p1, 1e-6);
177  return (enthalpy(temperature) - enthalpy(temperature - dT)) / dT;
178  }
179  else
180  {
181  const double dT = std::max(temperature_p1 - temperature, 1e-6);
182  return (enthalpy(temperature + dT) - enthalpy(temperature)) / dT;
183  }
184  break;
185 
186  case Parameters::SimulationControl::TimeSteppingMethod::bdf2:
187  {
188  const double enthalpy_current = enthalpy(temperature);
189  const double enthalpy_p1 = enthalpy(temperature_p1);
190  const double enthalpy_p2 = enthalpy(temperature_p2);
191  const double dH = bdf_coefs[0] * enthalpy_current +
192  bdf_coefs[1] * enthalpy_p1 +
193  bdf_coefs[2] * enthalpy_p2;
194  const double dT = bdf_coefs[0] * temperature +
195  bdf_coefs[1] * temperature_p1 +
196  bdf_coefs[2] * temperature_p2;
197  return dH / dT;
198  break;
199  }
200 
201  default:
202  throw(std::runtime_error(
203  "Other time integration schemes are not supported by the phase change model"));
204  }
205  }

208  /**
209  * @brief vector_value Calculates the vector of specific heat.
210  * @param field_vectors Vector of fields on which the specific heat may depend.
211  * @param property_vector Vector of specific_heat values.
212  */
213  void
214  vector_value(const std::map<field, std::vector<double>> &field_vectors,
215  std::vector<double> &property_vector) override
216  {
217  const std::vector<double> &temperature_vec =
218  field_vectors.at(field::temperature);
219  const std::vector<double> &p1_temperature_vec =
220  field_vectors.at(field::temperature_p1);
221  const std::vector<double> &p2_temperature_vec =
222  field_vectors.at(field::temperature_p2);
223 
224  const unsigned int n_values = temperature_vec.size();
225 
226  Assert(n_values == p2_temperature_vec.size(),
227  SizeOfFields(n_values, p2_temperature_vec.size()));
228  Assert(n_values == p1_temperature_vec.size(),
229  SizeOfFields(n_values, p1_temperature_vec.size()));
230  Assert(n_values == property_vector.size(),
231  SizeOfFields(n_values, property_vector.size()));
232 
233  // Gather information required from the simulation control to have the time
234  // history
235  std::shared_ptr<SimulationControl> simulation_control =
236  get_simulation_control();
237  std::vector<double> time_steps_vector =
238  simulation_control->get_time_steps_vector();
239 
240  for (unsigned int i = 0; i < n_values; ++i)
241  {
242  const double temperature = temperature_vec[i];
243  const double temperature_p1 = p1_temperature_vec[i];
244  const double temperature_p2 = p2_temperature_vec[i];
245 
246  // If change between the temperature is insufficient, backtrack to the
247  // first order implementation
248  auto method = simulation_control->get_assembly_method();
249 
250  if (method != Parameters::SimulationControl::TimeSteppingMethod::bdf1 &&
251  std::abs(temperature - temperature_p2) < 1e-6)
252  method = Parameters::SimulationControl::TimeSteppingMethod::bdf1;
253 
254  Vector<double> bdf_coefs = bdf_coefficients(method, time_steps_vector);
255 
256  switch (method)
257  {
258  case Parameters::SimulationControl::TimeSteppingMethod::bdf1:
259  if (temperature > temperature_p1)
260  {
261  const double dT =
262  std::max(temperature - temperature_p1, 1e-6);
263  property_vector[i] =
264  (enthalpy(temperature) - enthalpy(temperature - dT)) / dT;
265  }
266  else
267  {
268  const double dT =
269  std::max(temperature_p1 - temperature, 1e-6);
270  property_vector[i] =
271  (enthalpy(temperature + dT) - enthalpy(temperature)) / dT;
272  }
273  break;
274 
275  case Parameters::SimulationControl::TimeSteppingMethod::bdf2:
276  {
277  const double enthalpy_current = enthalpy(temperature);
278  const double enthalpy_p1 = enthalpy(temperature_p1);
279  const double enthalpy_p2 = enthalpy(temperature_p2);
280  const double dH = bdf_coefs[0] * enthalpy_current +
281  bdf_coefs[1] * enthalpy_p1 +
282  bdf_coefs[2] * enthalpy_p2;
283  const double dT = bdf_coefs[0] * temperature +
284  bdf_coefs[1] * temperature_p1 +
285  bdf_coefs[2] * temperature_p2;
286  property_vector[i] = dH / dT;
287  break;
288  }
289 
290  default:
291  throw(std::runtime_error(
292  "Other time integration schemes are not supported by the phase change model"));
293  }
294  }
295  }

327  /**
328  * @brief enthalpy Calculates the enthalpy of a phase change material for a temperature T
329  * The enthalpy is defined as :
330  *
331  * Pure liquid
332  * ------------
333  * if (T>T_liquidus) : H = cp_solid * T_solidus + 0.5*(cp_solid+cp_liquid) *
334  * (T_liquidus-T_solidus) + latent_enthalpy + cp_liquid * (T-T_liquidus)
335  *
336  * Liquid-solid mixture
337  * -----------------
338  * else if (T>T_solidus) : cp_solid * T_solidus + 0.5*(cp_solid+cp_liquid) *
339  * (T-T_solidus) + liquid_fraction * latent_enthalpy
340  *
341  * Pure solid
342  * ------------
343  * else : cp_solid * T
344  *
345  * @param T temperature at which to calculate the enthalpy
346  * @return Value of the enthalpy
347  */
348 
349  inline double
350  enthalpy(const double T)
351  {
352  if (T > param.T_liquidus)
353  return (param.cp_s * param.T_solidus +
354  0.5 * (param.cp_l + param.cp_s) *
355  (param.T_liquidus - param.T_solidus) +
356  param.latent_enthalpy + (T - param.T_liquidus) * param.cp_l);
357  else if (T > param.T_solidus)
358  {
359  liquid_fraction = calculate_liquid_fraction(T, param);
360  return (param.cp_s * param.T_solidus +
361  0.5 * (param.cp_l + param.cp_s) * (T - param.T_solidus) +
362  param.latent_enthalpy * liquid_fraction);
363  }
364  else
365  return param.cp_s * T;
366  }
367 
368 private:
369  const Parameters::PhaseChange param;
370  double liquid_fraction;
371 };
