87 /**
88  * @brief Filter phase fraction with a hyperbolic tangent function
89  * for VOF simulations.
90  *
91  * The filtered phase is defined as:
92  *
93  * \f$\phi' = 0.5 \tanh(\beta(\phi-0.5)) + 0.5\f$
94  *
95  * where \f$\phi'\f$ is the filtered phase fraction value and \f$\beta\f$
96  * is a model parameter that enables sharper definition when increased.
97  *
98  * The filter allows a sharper definition of the interface and clamps the value
99  * of the phase fraction between 0 and 1.
100  */
101 class VolumeOfFluidTanhFilter : public VolumeOfFluidFilterBase
102 {
103 public:
104  /**
105  * @brief Constructor of the hyperbolic tangent phase fraction filter.
106  *
107  * @param[in] beta Value of the \f$\beta\f$ parameter modulating the
108  * interface sharpness.
109  */
110  VolumeOfFluidTanhFilter(const double beta)
111  : beta(beta)
112  {}
113 
114  /**
115  * @brief Computes the value of the filtered phase fraction.
116  *
117  * @param[in] unfiltered_phase Value of the phase fraction before applying the
118  * filter.
119  *
120  * @return Value of the computed phase fraction after applying the filter.
121  */
122  virtual double
123  filter_phase(const double &unfiltered_phase) override
124  {
125  return 0.5 * tanh(beta * (unfiltered_phase - 0.5)) + 0.5;
126  }
127 
128 private:
129  /**
130  * User-defined parameter that influences the sharpness of the interface
131  * between fluids. A greater value leads to a sharper interface. The
132  * recommended value is \f$\beta=20\f$.
133  */
134  const double beta;
135 };
