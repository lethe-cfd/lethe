382 template <int dim>
383 std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
384 ParticleWallNonLinearForce<dim>::calculate_nonlinear_contact_force_and_torque(
385  particle_wall_contact_info<dim> &contact_info,
386  const ArrayView<const double> &particle_properties)
387 {
388  // i is the particle, j is the wall.
389  // we need to put a minus sign infront of the normal_vector to respect the
390  // convention (i -> j)
391  Tensor<1, 3> normal_vector = -contact_info.normal_vector;
392  const unsigned int particle_type =
393  particle_properties[DEM::PropertiesIndex::type];
394 
395  // Calculation of model parameters (beta, sn and st). These values
396  // are used to consider non-linear relation of the contact force to
397  // the normal overlap
398  double radius_times_overlap_sqrt =
399  sqrt(particle_properties[DEM::PropertiesIndex::dp] * 0.5 *
400  contact_info.normal_overlap);
401  double model_parameter_sn = 2 *
402  this->effective_youngs_modulus[particle_type] *
403  radius_times_overlap_sqrt;
404 
405  // Calculation of normal and tangential spring and dashpot constants
406  // using particle and wall properties
407  double normal_spring_constant =
408  1.3333 * this->effective_youngs_modulus[particle_type] *
409  radius_times_overlap_sqrt;
410 
411  // There is no minus sign here since model_parameter_beta is negative or
412  // equal to zero.
413  double normal_damping_constant =
414  1.8257 * this->model_parameter_beta[particle_type] *
415  sqrt(model_parameter_sn * particle_properties[DEM::PropertiesIndex::mass]);
416 
417  // There is a minus sign since the tangential force is applied in the opposite
418  // direction of the tangential_overlap
419  double tangential_spring_constant =
420  -8. * this->effective_shear_modulus[particle_type] *
421  radius_times_overlap_sqrt +
422  DBL_MIN;
423  // TODO add the tangential damping
424 
425  // Calculation of normal force using spring and dashpot normal forces
426  Tensor<1, 3> normal_force =
427  (normal_spring_constant * contact_info.normal_overlap +
428  normal_damping_constant * contact_info.normal_relative_velocity) *
429  normal_vector;
430 
431  // Calculation of tangential force
432  Tensor<1, 3> tangential_force =
433  tangential_spring_constant * contact_info.tangential_overlap;
434 
435  double coulomb_threshold =
436  this->effective_coefficient_of_friction[particle_type] *
437  normal_force.norm();
438 
439  // Check for gross sliding
440  if (tangential_force.norm() > coulomb_threshold)
441  {
442  // Gross sliding occurs and the tangential overlap and tangential
443  // force are limited to Coulomb's criterion
444  tangential_force =
445  coulomb_threshold * (tangential_force / tangential_force.norm());
446 
447  contact_info.tangential_overlap =
448  tangential_force / (tangential_spring_constant + DBL_MIN);
449  }
450 
451  // Calculation torque caused by tangential force
452  // We add the minus sign here since the tangential_force is applied on the
453  // particle is in the opposite direction
454  Tensor<1, 3> tangential_torque =
455  cross_product_3d((0.5 * particle_properties[DEM::PropertiesIndex::dp] *
456  normal_vector),
457  -tangential_force);
458 
459  // Rolling resistance torque
460  Tensor<1, 3> rolling_resistance_torque =
461  (this->*calculate_rolling_resistance_torque)(
462  particle_properties,
463  this->effective_coefficient_of_rolling_friction[particle_type],
464  normal_force.norm(),
465  contact_info.normal_vector);
466 
467  return std::make_tuple(normal_force,
468  tangential_force,
469  tangential_torque,
470  rolling_resistance_torque);
471 }
