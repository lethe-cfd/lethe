46 /**
47  * The general idea behind the algorithm:
48  * It uses the granular temperature to determinate if particles in a cell are
49  * mobile enough that the contact forces are worth computation. A cell having a
50  * granular temperature under a value (default is 1e-4) will have an inactive
51  * status which makes them rejected in the broad search step. This results in no
52  * contact forces or velocity integration in future steps for those particles,
53  * which make less computation cost for the simulation.
54  *
55  * Cells may be flagged as that called mobility status:
56  * - mobile (everything is calculated as if feature is not enabled)
57  * - active (particles with low motion next to mobile particles)
58  * - inactive (particles are not in a neighborâ€™s candidate list of cells around)
59  *
60  * There are some edge cases that need some attention. In these cases, particles
61  * in cells can't be deactivated:
62  *
63  * 1. The solid fraction of the cell is under a value (default = 40%): particles
64  * in the cell may have other forces that can move them and it is supposed that
65  * with this fraction, there is not enough particles around to hold them in
66  * their position.
67  *
68  * 2. Cells having empty cell neighbors: it means that particles are next to a
69  * floating wall/mesh and they may have a change of contact forces from it if
70  * the wall disappears or is moving.
71  *
72  * 3. The cell has cell neighbors which is flagged as mobile from the 3 criteria
73  * mentioned above (granular temperature, solid fractions or floating walls):
74  * this is because motion is badly propagated to particles to the cell around
75  * without this additional "layer" of mobile cells.
76  *
77  * 4. The cell has cell neighbors which is flagged as mobile from the previous
78  * criterion (additional layer of mobile cells): this is again to allow the
79  * propagation of motion, but only the particles in contact with particles
80  * from the mobile cells are considered for the contact force calculation but
81  * their position is not computed at the integration step. Those cells are
82  * flagged as active cells. It works with the assignment and verification of the
83  * mobility status at nodes to check the status of the neighboring cells.
84  *
85  */

92  /**
93  * Mobility status flag used to identify the status at nodes and the status
94  * of the cell:
95  *
96  * inactive (0)
97  * The movement of particles in the cell is considered as negligible,
98  * particles within this cell are not considered in the contact detection
99  * (rejected at the broad search step), so no force calculation or
100  * integration is applied
101  *
102  * active (1)
103  * The movement of particles in the cell is considered as negligible, but
104  * there's at least one neighbor cell that is flagged as mobile, meaning that
105  * particles need to be in contact candidates lists at the broad search step,
106  * particles directly in contact with the mobile cell are also considered in
107  * force calculation, but none of the particles in the cell are integrated.
108  *
109  * mobile (2)
110  * The movement of particles in the cell is significant or there is at least
111  * one neighbor cell that is mobile by criteria (see the
112  * identify_mobility_status() description), particles need to be in contact
113  * candidates lists at the broad search step and particles are treated as
114  * usual (force calculation and integration)
115  *
116  * empty (3)
117  * This status is only used for the node-based mobility status identification,
118  * no cells are flagged as empty, only node can by identify as empty. Without
119  * this identification of the empty cells, we can't identify the cell that
120  * have a empty neighbor cell, which is critical for simulationd using
121  * floating walls or mesh
122  */
123  enum mobility_status : unsigned int
124  {
125  inactive = 0, // used for cells and nodes
126  active = 1, // used for cells and nodes
127  mobile = 2, // used for cells and nodes
128  empty = 3 // used for nodes only
129  };

145  /**
146  * @brief Carries out the identification of the mobility status of each cell
147  * through a node-based identification and check. Only the active and ghost
148  * cells are processed.
149  *
150  * The following 4 checks (search loops) are done:
151  *
152  * 1. Check if the cell is empty (n_particle = 0), if so, nodes and cells are
153  * flagged as empty mobility status (3)
154  *
155  * 2. Check if the cell is mobile by criteria (average granular temperature >
156  * threshold, solid fraction < threshold or has at least one empty node from
157  * previous check), if so, nodes are flagged and cells are stored with mobile
158  * mobility status (2)
159  *
160  * 3. Check if the cell is mobile by neighbor (at least a node is flagged as
161  * mobile from previous check), if so, cells are stored in map as mobile
162  * status (2) and nodes that are not mobile are flagged as active (1)
163  *
164  * 4. Check if the cell is active (at least a node is flagged as active from
165  * previous check), if so, cells are stored with active status in the map (1)
166  *
167  * The remaining cells are inactive (0)
168  *
169  * @param background_dh The dof handler of the background grid
170  *
171  * @param particle_handler The particle handler that contains all the particles
172  *
173  * @param n_active_cells The number of active cells in triangulation
174  *
175  * @param mpi_communicator The MPI communicator
176  */

191  {
192  periodic_node_ids.clear();
193 
194  IndexSet local_lines = constraints.get_local_lines();
195  for (auto i : local_lines)
196  {
197  for (auto j : local_lines)
198  {
199  // Map the values of the periodic nodes (with repetition)
200  if (constraints.are_identity_constrained(i, j))
201  {
202  periodic_node_ids.insert(std::make_pair(i, j));
203  }
204  }

261  /**
262  * @brief Carries out the calculation of the granular temperature and solid
263  * fraction approximation (pcm method) in each active cell. Those values are
264  * criteria for cell mobility
265  *
266  * solid fraction = n particles * particle volume / cell volume
267  *
268  * Granular temperature:
269  * 1. average velocity of cell = sum of particle velocity /
270  * n_particles
271  * 2. average of cell velocity fluctuation squared = sum of particle velocity
272  * fluctuation squared / n_particles
273  * 3. granular temperature = sum of average of cell velocity fluctuation
274  * squared / dim
275  *
276  * @param particle_handler The particle handler that contains all the particles
277  *

279  * cells that empty cells are removed from

294  /**
295  * @brief Assigns the mobility status to node and the periodic coinciding node from the
296  * periodic nodes map. If required, this will prevent overwriting a node
297  * status prevailing over the initial one. For instance, empty status must not
298  * be overwritten by mobile status because all cells with this node will be
299  * part of the additional mobile layer and will propagate wrong mobility
300  * status to the next cells.
301  *
302  * @param node_id The current node index (periodic or not)
303  *
304  * @param mobility_status The mobility status to assign to the periodic node
305  *
306  */
307  inline void
308  assign_node_status(const unsigned int node_id, const int mobility_status)
309  {
310  // Prevailing mobility status of the node and assignation
311  int mobility_status_assigned =
312  std::max(mobility_status, mobility_at_nodes(node_id));
313  mobility_at_nodes(node_id) = mobility_status_assigned;
314 
315  // Check if node has a periodic node and assign the same mobility status if
316  // prevailing on the one on the periodic node
317  auto it = periodic_node_ids.find(node_id);
318  if (it != periodic_node_ids.end())
319  {
320  mobility_at_nodes(it->second) =
321  std::max(mobility_status_assigned, mobility_at_nodes(it->second));
322  }
323  }
324 
325  // Assign active status to nodes except mobile because
326  // this will cause to propagate the mobile status to the
327  // neighbors in this loop since the mobility check at node
328  // is executed in the same container that we are assigning new
329  // mobility status
330  inline void
331  assign_mobility_status(unsigned int cell_id,
332  std::vector<types::global_dof_index> dof_indices,
333  const int cell_status,
334  const int node_status)
335  {
336  cell_mobility_status.insert({cell_id, cell_status});
337 
338  // Assign mobility status to nodes but don't overwrite empty or mobile nodes
339  // in regards of the case.
340  for (auto node_id : dof_indices)
341  {
342  // Prevailing mobility status of the node and assignation
343  int status_assigned = std::max(node_status, mobility_at_nodes(node_id));
344  mobility_at_nodes(node_id) = status_assigned;
345 
346  // Check if node has a periodic node and assign the same mobility status
347  // if prevailing on the one on the periodic node
348  auto it = periodic_node_ids.find(node_id);
349  if (it != periodic_node_ids.end())
350  {
351  mobility_at_nodes(it->second) =
352  std::max(status_assigned, mobility_at_nodes(it->second));
353  }
354  }
355  }
