66 template <typename VectorType>
67 void
68 InexactNewtonNonLinearSolver<VectorType>::solve(const bool is_initial_step)
69 {
70  double global_res;
71  double current_res;
72  double last_res;
73  bool first_step = is_initial_step;
74  unsigned int outer_iteration = 0;
75  last_res = 1e6;
76  current_res = 1e6;
77  global_res = 1e6;
78 
79  // current_res and global_res are different as one is defined based on the l2
80  // norm of the residual vector (current_res) and the other (global_res) is
81  // defined by the physical solver and may differ from the l2_norm of the
82  // residual vector. Only the global_res is compared to the tolerance in order
83  // to evaluate if the nonlinear system is solved. Only current_res is used for
84  // the alpha scheme as this scheme only monitors the convergence of the
85  // non-linear system of equation (the matrix problem).
86 
87  PhysicsSolver<VectorType> *solver = this->physics_solver;
88 
89  auto &evaluation_point = solver->get_evaluation_point();
90  auto &present_solution = solver->get_present_solution();
91 
92  if (!this->params.reuse_matrix)
93  matrix_requires_assembly = true;
94 
95  while ((global_res > this->params.tolerance) &&
96  outer_iteration < this->params.max_iterations)
97  {
98  evaluation_point = present_solution;
99 
100  if (matrix_requires_assembly)
101  solver->assemble_system_matrix();
102  if (this->params.force_rhs_calculation || outer_iteration == 0)
103  solver->assemble_system_rhs();
104 
105  if (outer_iteration == 0)
106  {
107  auto &system_rhs = solver->get_system_rhs();
108  current_res = system_rhs.l2_norm();
109  last_res = current_res;
110  }
111 
112  if (this->params.verbosity != Parameters::Verbosity::quiet)
113  {
114  solver->pcout << "Newton iteration: " << outer_iteration
115  << " - Residual: " << current_res << std::endl;
116  }
117 
118  solver->solve_linear_system(first_step);
119  double last_alpha_res = current_res;
120 
121  unsigned int alpha_iter = 0;
122  for (double alpha = 1.0; alpha > 1e-1; alpha *= 0.5)
123  {
124  auto &local_evaluation_point = solver->get_local_evaluation_point();
125  auto &newton_update = solver->get_newton_update();
126  local_evaluation_point = present_solution;
127  local_evaluation_point.add(alpha, newton_update);
128  solver->apply_constraints();
129  evaluation_point = local_evaluation_point;
130  solver->assemble_system_rhs();
131 
132  auto &system_rhs = solver->get_system_rhs();
133  current_res = system_rhs.l2_norm();
134 
135  if (this->params.verbosity != Parameters::Verbosity::quiet)
136  {
137  solver->pcout << "\talpha = " << std::setw(6) << alpha
138  << std::setw(0) << " res = "
139  << std::setprecision(this->params.display_precision)
140  << std::setw(6) << current_res << std::setw(6)
141  << "\tL^2(dx) = " << std::setw(6)
142  << newton_update.l2_norm() << std::setw(6)
143  << "\tL^infty(dx) = "
144  << std::setprecision(this->params.display_precision)
145  << newton_update.linfty_norm() << std::endl;
146  }
147 
148  // If it's not the first iteration of alpha check if the residual is
149  // smaller then the last alpha iteration. If it's not smaller we fall
150  // back to the last alpha iteration, which is generally the first one.
151  if (alpha_iter != 0 and current_res > last_alpha_res)
152  {
153  alpha = 2 * alpha;
154  local_evaluation_point = present_solution;
155  local_evaluation_point.add(alpha, newton_update);
156  solver->apply_constraints();
157  evaluation_point = local_evaluation_point;
158 
159  if (this->params.verbosity != Parameters::Verbosity::quiet)
160  {
161  solver->pcout
162  << "\t\talpha value was kept at alpha = " << alpha
163  << " since alpha = " << alpha / 2
164  << " increased the residual" << std::endl;
165  }
166  current_res = last_alpha_res;
167  break;
168  }
169  if (current_res < this->params.step_tolerance * last_res ||
170  last_res < this->params.tolerance)
171  {
172  // If the current residual has decreased by less than
173  // the last residual * matrix_tolerance or more than one alpha
174  // iteration was required, the jacobian approximation has become
175  // insufficiently performant and, consequently, it is preferable
176  // to renew the jacobian matrix
177  if (current_res > this->params.matrix_tolerance * last_res ||
178  alpha_iter > 0)
179  matrix_requires_assembly = true;
180  else
181  matrix_requires_assembly = false;
182 
183  break;
184  }
185  last_alpha_res = current_res;
186  alpha_iter++;
187  }
188 
189  global_res = solver->get_current_residual();
190  present_solution = evaluation_point;
191  last_res = current_res;
192  ++outer_iteration;
193  }
194 
195  // If the non-linear solver has not converged abort simulation if
196  // abort_at_convergence_failure=true
197  if ((global_res > this->params.tolerance) &&
198  outer_iteration >= this->params.max_iterations &&
199  this->params.abort_at_convergence_failure)
200  {
201  throw(std::runtime_error(
202  "Stopping simulation because the non-linear solver has failed to converge"));
203  }
204 }
