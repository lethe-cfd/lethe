443 CFDDEMSolver<dim>::cell_weight(
444  const typename parallel::distributed::Triangulation<dim>::cell_iterator &cell,
445  const CellStatus status) const
446 #endif
447 
448 
449 {
450  // Assign no weight to cells we do not own.
451  if (!cell->is_locally_owned())
452  return 0;
453 
454  // This determines how important particle work is compared to cell
455  // work (by default every cell has a weight of 1000).
456  // We set the weight per particle higher to indicate that
457  // the particle load is more important than the fluid load. The optimal
458  // value of this number depends on the application and can range from 0
459  // (cheap particle operations, expensive cell operations) to much larger
460  // than 1000 (expensive particle operations, cheap cell operations, like in
461  // this case). This parameter will need to be tuned for different cases of
462  // CFD-DEM coupling.
463  const unsigned int particle_weight =
464  dem_parameters.model_parameters.load_balance_particle_weight;
465 
466  // This does not use adaptive refinement, therefore every cell
467  // should have the status CELL_PERSIST. However this function can also
468  // be used to distribute load during refinement, therefore we consider
469  // refined or coarsened cells as well.
470 
471 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
472  if (status == parallel::distributed::Triangulation<dim>::CELL_PERSIST ||
473  status == parallel::distributed::Triangulation<dim>::CELL_REFINE)
474 #else
475  if (status == CellStatus::cell_will_persist ||
476  status == CellStatus::cell_will_be_refined)
477 #endif
478 
479  {
480  const unsigned int n_particles_in_cell =
481  this->particle_handler.n_particles_in_cell(cell);
482  return n_particles_in_cell * particle_weight;
483  }
484 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
485  else if (status == parallel::distributed::Triangulation<dim>::CELL_COARSEN)
486 #else
487  else if (status == CellStatus::children_will_be_coarsened)
488 #endif
489  {

883 template <int dim>
884 void
885 CFDDEMSolver<dim>::dem_contact_build(unsigned int counter)
886 {
887  // Check to see if it is contact search step
888  contact_detection_step =
889  check_contact_detection_method(counter,
890  this->cfd_dem_simulation_parameters,
891  displacement,
892  this->particle_handler,
893  this->mpi_communicator,
894  this->simulation_control,
895  contact_detection_step,
896  checkpoint_step,
897  load_balance_step,
898  smallest_contact_search_criterion);
899 
900  // Sort particles in cells
901  // The last statement in the if condition allows the calling of
902  // dem_contact_build for the first DEM iteration in the first CFD time step
903  // directly after reading the dem initial checkpoint files
904 
905  if (contact_detection_step || checkpoint_step || load_balance_step ||
906  (this->simulation_control->is_at_start() && (counter == 0)))
907  {
908  this->pcout << "DEM contact search at dem step " << counter << std::endl;
909  contact_build_number++;
910 
911  // Particles displacement if passing through a periodic boundary
912  if (has_periodic_boundaries)
913  periodic_boundaries_object.execute_particles_displacement(
914  this->particle_handler, periodic_boundaries_cells_information);
915 
916  this->particle_handler.sort_particles_into_subdomains_and_cells();
917 
918  displacement.resize(
919  this->particle_handler.get_max_local_particle_index());
920  force.resize(displacement.size());
921  torque.resize(displacement.size());
922 
923  this->particle_handler.exchange_ghost_particles(true);
924 
925  if (has_disabled_contacts && !this->simulation_control->is_at_start())
926  {
927  // Update the active and ghost cells set (this should be done after a
928  // load balance or a checkpoint, but since the fem-dem code do not
929  // have a specific step for that we do it also when the contact search
930  // is done)
931  disable_contacts_object.update_local_and_ghost_cell_set(
932  this->void_fraction_dof_handler);
933  disable_contacts_object.identify_mobility_status(

935  this->particle_handler,
936  (*this->triangulation).n_active_cells(),
937  this->mpi_communicator);
938  }
939 
940  // Updating moment of inertia container
941  update_moment_of_inertia(this->particle_handler, MOI);
942  }
943  else
944  {
945  this->particle_handler.update_ghost_particles();
946  }
947 
948  // Modify particles contact containers by search sequence
949  if (load_balance_step || checkpoint_step || contact_detection_step ||
950  (this->simulation_control->is_at_start() && (counter == 0)))
951  {
952  // Execute broad search by filling containers of particle-particle
953  // contact pair candidates and containers of particle-wall
954  // contact pair candidates
955  if (!contacts_are_disabled(counter))
956  {
957  contact_manager.execute_particle_particle_broad_search(
958  this->particle_handler, has_periodic_boundaries);
959 
960  contact_manager.execute_particle_wall_broad_search(
961  this->particle_handler,
962  boundary_cell_object,
963  floating_mesh_info,
964  dem_parameters.floating_walls,
965  this->simulation_control->get_current_time());
966  }
967  else // disabling particle contacts are enabled & counter > 1
968  {
969  contact_manager.execute_particle_particle_broad_search(
970  this->particle_handler,
971  disable_contacts_object,
972  has_periodic_boundaries);
973 
974  contact_manager.execute_particle_wall_broad_search(
975  this->particle_handler,
976  boundary_cell_object,
977  floating_mesh_info,
978  dem_parameters.floating_walls,
979  this->simulation_control->get_current_time(),
980  disable_contacts_object);
981  }
982 
983  // Update contacts, remove replicates and add new contact pairs
984  // to the contact containers when particles are exchanged between
985  // processors
986  contact_manager.update_contacts(has_periodic_boundaries);
987 
988  // Updates the iterators to particles in local-local contact
989  // containers
990  contact_manager.update_local_particles_in_cells(this->particle_handler,
991  load_balance_step,
992  has_periodic_boundaries);
993 
994  // Execute fine search by updating particle-particle contact
995  // containers regards the neighborhood threshold
996  contact_manager.execute_particle_particle_fine_search(
997  neighborhood_threshold_squared,
998  has_periodic_boundaries,
999  periodic_offset);
1000 
1001  // Execute fine search by updating particle-wall contact containers
1002  // regards the neighborhood threshold
1003  contact_manager.execute_particle_wall_fine_search(
1004  dem_parameters.floating_walls,
1005  this->simulation_control->get_current_time(),
1006  neighborhood_threshold_squared);
1007 
1008  // Reset different steps. The contact build should be performed everytime
1009  // we restart the simulation or everytime load balancing is performed. At
1010  // the end of the restart step or the load balance step, and after all
1011  // necessary contact build, vector resizing, and solution transfers have
1012  // been performed, this functions are set to false. The checkpoint_step
1013  // remains false for the duration of the simulation while the load
1014  // balancing is reset everytime load balancing is called.
1015  checkpoint_step = false;
1016  load_balance_step = false;
1017  }
