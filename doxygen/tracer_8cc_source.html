310 double
311 Tracer<dim>::calculate_L2_error()
312 {
313  auto mpi_communicator = triangulation->get_communicator();
314 
315  FEValues<dim> fe_values(*mapping,
316  *fe,
317  *cell_quadrature,
318  update_values | update_gradients |
319  update_quadrature_points | update_JxW_values);
320 
321  const unsigned int n_q_points = cell_quadrature->size();
322 
323  std::vector<double> q_exact_solution(n_q_points);
324  std::vector<double> q_scalar_values(n_q_points);
325 
326  auto &exact_solution = simulation_parameters.analytical_solution->tracer;
327  exact_solution.set_time(simulation_control->get_current_time());
328 
329  double l2error = 0.;
330 
331  for (const auto &cell : dof_handler.active_cell_iterators())
332  {
333  if (cell->is_locally_owned())
334  {
335  fe_values.reinit(cell);
336  fe_values.get_function_values(present_solution, q_scalar_values);
337 
338  // Get the exact solution at all gauss points
339  exact_solution.value_list(fe_values.get_quadrature_points(),
340  q_exact_solution);
341 
342  for (unsigned int q = 0; q < n_q_points; q++)
343  {
344  double sim = q_scalar_values[q];
345  double exact = q_exact_solution[q];
346  l2error += (sim - exact) * (sim - exact) * fe_values.JxW(q);
347  }
348  }
349  }
350  l2error = Utilities::MPI::sum(l2error, mpi_communicator);
351  return l2error;

429 template <int dim>
430 void
431 Tracer<dim>::calculate_tracer_statistics()
432 {
433  auto mpi_communicator = triangulation->get_communicator();
434 
435  FEValues<dim> fe_values(*mapping,
436  *fe,
437  *cell_quadrature,
438  update_values | update_gradients |
439  update_quadrature_points | update_JxW_values);
440 
441  const unsigned int n_q_points = cell_quadrature->size();
442  std::vector<double> q_tracer_values(n_q_points);
443 
444  double volume_integral = 0;
445  double max_tracer_value = DBL_MIN;
446  double min_tracer_value = DBL_MAX;
447 
448  for (const auto &cell : dof_handler.active_cell_iterators())
449  {
450  if (cell->is_locally_owned())
451  {
452  fe_values.reinit(cell);
453  fe_values.get_function_values(present_solution, q_tracer_values);
454 
455  for (unsigned int q = 0; q < n_q_points; q++)
456  {
457  volume_integral += q_tracer_values[q] * fe_values.JxW(q);
458  max_tracer_value = std::max(q_tracer_values[q], max_tracer_value);
459  min_tracer_value = std::min(q_tracer_values[q], min_tracer_value);
460  }
461  }
462  }
463  volume_integral = Utilities::MPI::sum(volume_integral, mpi_communicator);
464  double global_volume = GridTools::volume(*triangulation, *mapping);
465  double tracer_average = volume_integral / global_volume;
466 
467  double variance_integral = 0;
468 
469  for (const auto &cell : dof_handler.active_cell_iterators())
470  {
471  if (cell->is_locally_owned())
472  {
473  fe_values.reinit(cell);
474  fe_values.get_function_values(present_solution, q_tracer_values);
475 
476  for (unsigned int q = 0; q < n_q_points; q++)
477  {
478  variance_integral += (q_tracer_values[q] - tracer_average) *
479  (q_tracer_values[q] - tracer_average) *
480  fe_values.JxW(q);
481  }
482  }
483  }
484 
485  variance_integral = Utilities::MPI::sum(variance_integral, mpi_communicator);
486  double tracer_variance = variance_integral / global_volume;
487  double tracer_std_deviation = std::sqrt(tracer_variance);
488 
489  // Console output
490  if (simulation_parameters.post_processing.verbosity ==
491  Parameters::Verbosity::verbose)
492  {
493  this->pcout << "Tracer statistics : " << std::endl;
494  this->pcout << "\t Min : " << min_tracer_value << std::endl;
495  this->pcout << "\t Max : " << max_tracer_value << std::endl;
496  this->pcout << "\t Average : " << tracer_average << std::endl;
497  this->pcout << "\t Std-Dev : " << tracer_std_deviation << std::endl;
498  }
499 
500  statistics_table.add_value("time", simulation_control->get_current_time());
501  statistics_table.add_value("min", min_tracer_value);
502  statistics_table.add_value("max", max_tracer_value);
503  statistics_table.add_value("average", tracer_average);
504  statistics_table.add_value("std-dev", tracer_std_deviation);
505 }

805

808  const AffineConstraints<double> &constraints_used =

813  const double relative_residual =
814  simulation_parameters.linear_solver.at(PhysicsID::tracer).relative_residual;

817  std::max(relative_residual * system_rhs.l2_norm(), absolute_residual);
818

824  }
825

827  simulation_parameters.linear_solver.at(PhysicsID::tracer).ilu_precond_fill;
828  const double ilu_atol =
829  simulation_parameters.linear_solver.at(PhysicsID::tracer).ilu_precond_atol;
830  const double ilu_rtol =
831  simulation_parameters.linear_solver.at(PhysicsID::tracer).ilu_precond_rtol;
832  TrilinosWrappers::PreconditionILU::AdditionalData preconditionerOptions(
833  ilu_fill, ilu_atol, ilu_rtol, 0);
834 
835  TrilinosWrappers::PreconditionILU ilu_preconditioner;
836 
837  ilu_preconditioner.initialize(system_matrix, preconditionerOptions);
838 
839  GlobalVectorType completely_distributed_solution(locally_owned_dofs,
840  mpi_communicator);
841 
842  SolverControl solver_control(
843  simulation_parameters.linear_solver.at(PhysicsID::tracer).max_iterations,
844  linear_solver_tolerance,
845  true,
846  true);
847 
848  TrilinosWrappers::SolverGMRES::AdditionalData solver_parameters(
849  false,
850  simulation_parameters.linear_solver.at(PhysicsID::tracer)
851  .max_krylov_vectors);
852 
853

855

861

863  Parameters::Verbosity::quiet)

866  << " steps " << std::endl;
867  }
868

870  newton_update = completely_distributed_solution;
