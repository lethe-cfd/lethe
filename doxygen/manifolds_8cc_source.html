237 attach_manifolds_to_triangulation(
238  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
239  Parameters::Manifolds manifolds)
240 {
241  for (unsigned int i = 0; i < manifolds.types.size(); ++i)
242  {
243  if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::spherical)
244  {
245  // Create a point using the parameter file input
246  Point<spacedim> circle_center(
247  entry_string_to_tensor<spacedim>(manifolds.manifold_point[i]));
248 
249  static const SphericalManifold<dim, spacedim> manifold_description(
250  circle_center);
251  triangulation.set_manifold(manifolds.id[i], manifold_description);
252  }
253  else if (manifolds.types[i] ==
254  Parameters::Manifolds::ManifoldType::cylindrical)
255  {
256  if constexpr (spacedim == 3)
257  {
258  // Create a point using the parameter file input
259  Point<spacedim> point_on_axis(
260  entry_string_to_tensor<spacedim>(manifolds.manifold_point[i]));
261 
262  // Create a tensor representing the direction of the length of the
263  // cylinder
264  Tensor<1, spacedim> cylinder_axis(
265  entry_string_to_tensor<spacedim>(
266  manifolds.manifold_direction[i]));
267 
268  static const CylindricalManifold<dim, spacedim>
269  manifold_description(cylinder_axis, point_on_axis);
270  triangulation.set_manifold(manifolds.id[i], manifold_description);
271  }
272  else
273  throw std::runtime_error(
274  "Cylindrical manifolds are not supported in 2D");
275  }
276  else if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::iges)
277  {
278  attach_cad_to_manifold(triangulation,
279  manifolds.cad_files[i],
280  manifolds.id[i]);
281  }
282  else if (manifolds.types[i] == Parameters::Manifolds::ManifoldType::none)
283  {
284  }
285  else
286  throw std::runtime_error("Unsupported manifolds type");
287  }
