16  *
17  * Implementation of the Cahn-Hilliard equations as an auxiliary physics.
18  * Equations solved:
19  * dPhi/dt + u * gradPhi = div(M(Phi)*grad eta)
20  * eta - f(Phi) + epsilon^2 * div(grad Phi) = 0
21  * with Phi the phase field parameter (or phase order), eta the chemical
22  * potential, and M the mobility function and epsilon the interface thickness.
23  * The phase field parameter must not be confused with the order (respectively
24  * phase_cahn_hilliard_order and potential_cahn_hilliard_order) of the finite
25  elements related to the
26  * phase field parameter and the chemical potential.
27  */

63  CahnHilliard(MultiphysicsInterface<dim> *multiphysics_interface,
64  const SimulationParameters<dim> &p_simulation_parameters,
65  std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
66  p_triangulation,
67  std::shared_ptr<SimulationControl> p_simulation_control)
68  : AuxiliaryPhysics<dim, GlobalVectorType>(
69  p_simulation_parameters.non_linear_solver.at(PhysicsID::cahn_hilliard))
70  , multiphysics(multiphysics_interface)
71  , computing_timer(p_triangulation->get_communicator(),
72  this->pcout,
73  TimerOutput::summary,
74  TimerOutput::wall_times)
75  , simulation_parameters(p_simulation_parameters)
76  , triangulation(p_triangulation)
77  , simulation_control(p_simulation_control)
78  , dof_handler(*triangulation)
79  {
80  if (simulation_parameters.mesh.simplex)
81  {
82  // for simplex meshes
83  fe = std::make_shared<FESystem<dim>>(
84  FE_SimplexP<dim>(
85  simulation_parameters.fem_parameters.phase_cahn_hilliard_order),
86  1,
87  FE_SimplexP<dim>(
88  simulation_parameters.fem_parameters.potential_cahn_hilliard_order),
89  1);
90  mapping = std::make_shared<MappingFE<dim>>(*fe);
91  cell_quadrature = std::make_shared<QGaussSimplex<dim>>(
92  std::max(
93  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
94  simulation_parameters.fem_parameters
95  .potential_cahn_hilliard_order) +
96  1);
97  face_quadrature = std::make_shared<QGaussSimplex<dim - 1>>(
98  std::max(
99  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
100  simulation_parameters.fem_parameters
101  .potential_cahn_hilliard_order) +
102  1);
103  ;
104  }
105  else
106  {
107  // Usual case, for quad/hex meshes
108  fe = std::make_shared<FESystem<dim>>(
109  FE_Q<dim>(
110  simulation_parameters.fem_parameters.phase_cahn_hilliard_order),
111  1,
112  FE_Q<dim>(
113  simulation_parameters.fem_parameters.potential_cahn_hilliard_order),
114  1);
115  mapping = std::make_shared<MappingQ<dim>>(std::max(
116  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
117  simulation_parameters.fem_parameters.potential_cahn_hilliard_order));
118  cell_quadrature = std::make_shared<QGauss<dim>>(
119  std::max(
120  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
121  simulation_parameters.fem_parameters
122  .potential_cahn_hilliard_order) +
123  1);
124  face_quadrature = std::make_shared<QGauss<dim - 1>>(
125  std::max(
126  simulation_parameters.fem_parameters.phase_cahn_hilliard_order,
127  simulation_parameters.fem_parameters
128  .potential_cahn_hilliard_order) +
129  1);
130  }
131 
132  // Allocate solution transfer
133  solution_transfer = std::make_shared<
134  parallel::distributed::SolutionTransfer<dim, GlobalVectorType>>(
135  dof_handler);
136 
137  // Set size of previous solutions using BDF schemes information
138  previous_solutions.resize(maximum_number_of_previous_solutions());
139 
140  // Prepare previous solutions transfer
141  previous_solutions_transfer.reserve(previous_solutions.size());
142  for (unsigned int i = 0; i < previous_solutions.size(); ++i)
143  {
144  previous_solutions_transfer.emplace_back(
145  parallel::distributed::SolutionTransfer<dim, GlobalVectorType>(
146  this->dof_handler));
147  }
148

153  }
