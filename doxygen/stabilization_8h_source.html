24 /**
25  * @brief Calculate the stabilization parameter for the Navier-Stokes equations
26  * in steady-state
27  * @return Value of the stabilization parameter - tau
28  *
29  * @param u_mag Magnitude of the velocity
30  *
31  * @param kinematic_viscosity Kinematic viscosity
32  *
33  * @param h Cell size; it should be calculated using the diameter of a sphere of
34  * equal volume to that of the cell.
35  */
36 inline double
37 calculate_navier_stokes_gls_tau_steady(const double u_mag,
38  const double kinematic_viscosity,
39  const double h)
40 {
41  return 1. / std::sqrt(Utilities::fixed_power<2>(2. * u_mag / h) +
42  9 * Utilities::fixed_power<2>(4 * kinematic_viscosity /
43  (h * h)));
44 }
