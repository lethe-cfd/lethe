43 template <int dim>
44 void
45 PlaneInsertion<dim>::find_inplane_cells(
46  const parallel::distributed::Triangulation<dim> &triangulation,
47  Point<3> plane_point,
48  Tensor<1, 3> plane_normal_vector)
49 {
50  plane_cells_for_insertion.clear();
51 
52  // Looping through cells
53  for (const auto &cell : triangulation.active_cell_iterators())
54  {
55  // If the cell is owned by the processor
56  if (cell->is_locally_owned())
57  {
58  // Vector that goes from the plane reference point to the vertex 0.
59  Tensor<1, 3> connecting_vector_ref =
60  point_nd_to_3d(cell->vertex(0)) - plane_point;
61 
62  // Normal distance between vertex 0 and the plane.
63  double vertex_wall_distance_ref =
64  connecting_vector_ref * plane_normal_vector;
65 
66  // Loop over all the vertices of the cell
67  for (unsigned int vertex_id = 1; vertex_id < cell->n_vertices();
68  ++vertex_id)
69  {
70  // Vector that goes from the plane reference point to the vertex
71  // n-th
72  Tensor<1, 3> connecting_vector =
73  point_nd_to_3d(cell->vertex(vertex_id)) - plane_point;
74 
75  // Normal distance between vertex n-th and the plane
76  double vertex_wall_distance =
77  connecting_vector * plane_normal_vector;
78 
79  // If the multiplication of the two distances is negative, then at
80  // least one of vertex of the cell is on both sides of the plane.
81  // If the multiplication is equal to zero, then one vertex is on
82  // the plane.
83  if (vertex_wall_distance_ref * vertex_wall_distance <= 0)
84  {
85  plane_cells_for_insertion.insert(cell);
86  break; // If so, we break the loop, and go to the next cell
87  }
88  }
89  }
90  }
91 }

107 void
108 PlaneInsertion<dim>::insert(
109  Particles::ParticleHandler<dim> &particle_handler,
110  const parallel::distributed::Triangulation<dim> &triangulation,
111  const DEMSolverParameters<dim> &dem_parameters)
112 {
113  if (particles_of_each_type_remaining == 0 &&
114  this->current_inserting_particle_type !=
115  dem_parameters.lagrangian_physical_properties.particle_type_number - 1)
116  {
117  particles_of_each_type_remaining =
118  dem_parameters.lagrangian_physical_properties.number.at(
119  ++current_inserting_particle_type);
120  }
121  if (particles_of_each_type_remaining > 0)
122  {
123  if (mark_for_update)
124  {
125  find_inplane_cells(
126  triangulation,
127  dem_parameters.insertion_info.insertion_plane_point,
128  dem_parameters.insertion_info.insertion_plane_normal_vector);
129  find_centers_of_inplane_cells();
130  mark_for_update = false;
131  }
132 
133  MPI_Comm communicator = triangulation.get_communicator();
134  ConditionalOStream pcout(
135  std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0);
136 
137  unsigned int this_mpi_process =
138  Utilities::MPI::this_mpi_process(communicator);
139  unsigned int total_mpi_process =
140  Utilities::MPI::n_mpi_processes(communicator);
141  // List of the empty cells at this time step
142  std::set<typename Triangulation<dim>::active_cell_iterator>
143  empty_cells_on_proc;
144 
145  // Loop over the cell inplane
146  for (const auto &cell : plane_cells_for_insertion)
147  {
148  // if the cell is empty ...
149  if (particle_handler.n_particles_in_cell(cell) == 0)
150  {
151  // ... we add the cell in std::set
152  empty_cells_on_proc.insert(cell);
153  }
154  }
155 
156  // Processor zero knows how many particles every processor wants to insert
157  auto number_of_particles_to_insert_per_core =
158  Utilities::MPI::gather(communicator, empty_cells_on_proc.size(), 0);
159 
160  std::vector<int> starting_IDs_on_every_proc(total_mpi_process);
161  std::vector<int> remained_particles_for_every_proc(total_mpi_process);
162  // Only processor zero does this
163  if (this_mpi_process == 0)
164  {
165  int starting_id_on_proc =
166  particle_handler.get_next_free_particle_index();
167 
168  // we loop over the vector that we have created using the gather.
169  // Every integer in that vector gives the number of empty_cell of a
170  // certain processor.
171  for (unsigned int i = 0; i < total_mpi_process; ++i)
172  {
173  // We subtract those integer from the following variable
174  particles_of_each_type_remaining -=
175  number_of_particles_to_insert_per_core[i];
176 
177  // If the particles_of_each_type_remaining variable become
178  // negative, this means that we are trying to insert too many
179  // particle in the simulation. Thus, we need to decrease the
180  // number of empty_cell we want to insert in at this time step.
181  if (particles_of_each_type_remaining < 0)
182  {
183  // At this point, we know the current processor have to many
184  // available cell to insert in (or just the right amount). The
185  // excess number of available cell on this processor is equal
186  // to the positive value of the
187  // particles_of_each_type_remaining variable since that
188  // variable became negative on this processor.
189 
190  // Thus, we decrease the number of particle to insert on the
191  // current processor. To do so, we add the negative variable
192  // to the number of available cells which guarantee that the
193  // number of inserted particles in the simulation will reach
194  // its maximum on this processor.
195 
196  number_of_particles_to_insert_per_core[i] +=
197  particles_of_each_type_remaining;
198 
199  // We put the particles_of_each_type_remaining variable back
200  // to zero since we have just lower the number of cell
201  // available on this processor by the exceeding amount.
202  particles_of_each_type_remaining = 0;
203 
204  starting_IDs_on_every_proc[i] = starting_id_on_proc;
205  starting_id_on_proc +=
206  number_of_particles_to_insert_per_core[i];
207 
208  // Loop over the remaining processor and put their number of
209  // available cell to zero.
210  for (unsigned int j = i + 1; j < (total_mpi_process); ++j)
211  {
212  number_of_particles_to_insert_per_core[j] = 0;
213  starting_IDs_on_every_proc[j] = 0;
214  }
215  break;
216  }
217  starting_IDs_on_every_proc[i] = starting_id_on_proc;
218  starting_id_on_proc += number_of_particles_to_insert_per_core[i];
219  }
220  std::fill(remained_particles_for_every_proc.begin(),
221  remained_particles_for_every_proc.end(),
222  particles_of_each_type_remaining);
223  }
224 
225  // Now, processor zero knows how many particles will be inserted by every
226  // processor without exceeding the maximum number of particles.
227 
228  // We now need to scatter 3 information per processor, first how
229  // many particles it needs to insert. Second what is the ID of the
230  // first particle it will insert at this insertion step. The starting ID
231  // is mandatory, since we don't want two particles to have the same ID,
232  // and we don't want to use the global_insertion method because it uses an
233  // all_gather, which has a high computational cost. Third, the
234  // particles_of_each_type_remaining variable needs to be updated on
235  // processors other than processor 0 otherwise the insert function will
236  // become incoherent.
237 
238  // This scatters the number of particles to insert
239  unsigned int number_of_particles_to_insert_on_this_core =
240  Utilities::MPI::scatter(communicator,
241  number_of_particles_to_insert_per_core,
242  0);
243 
244  // This scatters the starting ID on every proc for the insertion
245  unsigned int starting_ID_on_proc =
246  Utilities::MPI::scatter(communicator, starting_IDs_on_every_proc, 0);
247 
248  // This scatters particles_of_each_type_remaining on every proc
249  particles_of_each_type_remaining =
250  Utilities::MPI::scatter(communicator,
251  remained_particles_for_every_proc,
252  0);
253 
254  // We yet didn't choose which empty cells won't be used anymore.
255 
256  while (empty_cells_on_proc.size() >
257  number_of_particles_to_insert_on_this_core)
258  {
259  auto it = empty_cells_on_proc
260  .begin(); // Get the iterator of the first element
261  empty_cells_on_proc.erase(it); // Erase the first element
262  }
263 
264  // A vector of vectors, which contains all the properties of all inserted
265  // particles at each insertion step
266  std::vector<std::vector<double>> particle_properties;
267 
268  this->assign_particle_properties(
269  dem_parameters,
270  number_of_particles_to_insert_on_this_core,
271  current_inserting_particle_type,
272  particle_properties);
273 
274  // This is to iterate over the particle_properties vector
275  unsigned int i = 0;
276 
277  // Loop over the empty cells we have kept.
278  for (const auto &cell : empty_cells_on_proc)
279  {
280  // Position of the center of the cell
281  Point<dim> insertion_location =
282  cells_centers.at(cell->active_cell_index());
283 
284  // Generate the random point of insertion
285  insertion_location(0) +=
286  static_cast<double>(rand()) * maximum_range_for_randomness;
287 
288  insertion_location(1) +=
289  static_cast<double>(rand()) * maximum_range_for_randomness;
290 
291  if constexpr (dim == 3)
292  {
293  insertion_location(2) +=
294  static_cast<double>(rand()) * maximum_range_for_randomness;
295  }
296 
297  // Insertion
298  Point<dim> ref_point;
299  particle_handler.insert_particle(insertion_location,
300  ref_point,
301  starting_ID_on_proc++,
302  cell,
303  particle_properties.at(i++));
304  }
305  }
306 }
