121  /** This function is used to find the projection of vector_a on

175  /**
176  * Carries out applying the calculated force and torque on the particle in
177  * contact with the given wall, for both non-linear and linear contact force
178  * calculations
179  *
180  * @param forces_and_torques A tuple which contains: 1, normal force, 2,
181  * tangential force, 3, tangential torque and 4, rolling resistance torque of
182  * a contact pair
183  * @param particle_torque Torque acting on particle
184  * @param particle_force Force acting on particle
185  * @param point_on_boundary Contact point on the wall
186  * @param boundary_id ID of the boundary
187  */
188  inline void
189  apply_force_and_torque(
190  const std::tuple<Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>, Tensor<1, 3>>
191  &forces_and_torques,
192  Tensor<1, 3> &particle_torque,
193  Tensor<1, 3> &particle_force,
194  const Point<3> &point_on_boundary,
195  int boundary_id = 0)
196  {
197  // Getting the values from the forces_and_torques tuple, which are: 1,
198  // normal force, 2, tangential force, 3, tangential torque and 4, rolling
199  // resistance torque
200  Tensor<1, 3> normal_force = std::get<0>(forces_and_torques);
201  Tensor<1, 3> tangential_force = std::get<1>(forces_and_torques);
202  Tensor<1, 3> tangential_torque = std::get<2>(forces_and_torques);
203  Tensor<1, 3> rolling_resistance_torque = std::get<3>(forces_and_torques);
204 
205  // Calculation of total force
206  Tensor<1, 3> total_force = normal_force + tangential_force;
207 
208  calculate_force_and_torque_on_boundary(boundary_id,
209  total_force,
210  point_on_boundary);
211 
212  // Updating the force of particles in the particle handler
213  // Since the forces were calculated on the wall, we use the -= operator
214  particle_force -= total_force;
215 
216  // Updating the torque acting on particles
217  // The torque was direcly calculated on the particle, thus we use the +=
218  // operator
219  particle_torque += tangential_torque + rolling_resistance_torque;
220  }
