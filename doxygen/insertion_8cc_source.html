127 void
128 Insertion<dim>::calculate_insertion_domain_maximum_particle_number(
129  const DEMSolverParameters<dim> &dem_parameters,
130  const ConditionalOStream &pcout)
131 {
132  // Getting properties as local parameters
133  const auto insertion_information = dem_parameters.insertion_info;
134 
135  // Checking if the insertion directions are valid (no repetition)
136  int axis_sum = 0;
137  if constexpr (dim == 2)
138  {
139  axis_sum = insertion_information.direction_sequence[0] +
140  insertion_information.direction_sequence[1];
141 
142  AssertThrow(
143  axis_sum == 1,
144  ExcMessage("Invalid insertion directions: 2 directions are the same "));
145  }
146  if constexpr (dim == 3)
147  {
148  axis_sum = insertion_information.direction_sequence[0] +
149  insertion_information.direction_sequence[1] +
150  insertion_information.direction_sequence[2];
151 
152  AssertThrow(
153  axis_sum == 3,
154  ExcMessage(
155  "Invalid insertion directions: at least 2 directions are the same "));
156  }
157 
158 
159  // This variable is used for calculation of the maximum number of particles
160  // that can fit in the chosen insertion box
161  int maximum_particle_number = 1;
162 
163  number_of_particles_directions.resize(dim);
164  axis_min.resize(dim);
165  axis_max.resize(dim);
166 
167  std::vector<unsigned int> axis_list = {
168  insertion_information.direction_sequence[0],
169  insertion_information.direction_sequence[1]};
170 
171  if constexpr (dim == 3)
172  {
173  axis_list.push_back(insertion_information.direction_sequence[2]);
174  }
175 
176  // Assigning the minimum and maximum positions of the insertion box in respect
177  // to the axis order
178  for (unsigned int axis : axis_list)
179  {
180  switch (axis)
181  {
182  case 0:
183  axis_min[0] = insertion_information.insertion_box_point_1(0);
184  axis_max[0] = insertion_information.insertion_box_point_2(0);
185  break;
186  case 1:
187  axis_min[1] = insertion_information.insertion_box_point_1(1);
188  axis_max[1] = insertion_information.insertion_box_point_2(1);
189  break;
190  case 2:
191  axis_min[2] = insertion_information.insertion_box_point_1(2);
192  axis_max[2] = insertion_information.insertion_box_point_2(2);
193  break;
194  default:
195  AssertThrow(false,
196  ExcMessage("Insertion direction must be 0, 1 or 2"));
197  }
198 
199  // Assign max number of particles according to the direction and calculate
200  // the total max number (maximum_particle_number = max_x * max_y * max_z)
201  int number_of_particles = static_cast<int>(
202  (axis_max[axis] - axis_min[axis]) /
203  (insertion_information.distance_threshold * this->maximum_diameter));
204  number_of_particles_directions[axis] = number_of_particles;
205  maximum_particle_number *= number_of_particles;
206  }
207 
208  // If the inserted number of particles at this step exceeds the maximum
209  // number, a warning is printed
210  if (insertion_information.inserted_this_step > maximum_particle_number)
211  {
212  pcout << "Warning: the requested number of particles for insertion ("
213  << insertion_information.inserted_this_step
214  << ") is higher than maximum expected number of particles ("
215  << maximum_particle_number << ")" << std::endl;
216 
217  // Updating the number of inserted particles at each step
218  inserted_this_step = maximum_particle_number;
219  }
220  else
221  {
222  inserted_this_step = insertion_information.inserted_this_step;
223  }
224 }

228 void
229 Insertion<dim>::find_cells_in_removing_box(
230  const parallel::distributed::Triangulation<dim> &triangulation)
231 {
232  // Clearing the containers
233  in_removal_box.clear();
234  edge_of_removal_box.clear();
235  bool partially_inside, completely_inside;
236  // Looping through cells
237  for (const auto &cell : triangulation.active_cell_iterators())
238  {
239  if (cell->is_locally_owned())
240  {
241  partially_inside = false;
242  completely_inside = true;
243  // Loop through its vertices
244  for (unsigned int vertex_id = 0; vertex_id < cell->n_vertices();
245  ++vertex_id)
246  {
247  Point<3> vertex = point_nd_to_3d(cell->vertex(vertex_id));
248 
249  // Check if the n_th vertex is in the box
250  if (p_min[0] <= vertex[0] && vertex[0] <= p_max[0] &&
251  p_min[1] <= vertex[1] && vertex[1] <= p_max[1] &&
252  p_min[2] <= vertex[2] && vertex[2] <= p_max[2])
253  {
254  partially_inside = true;
255  }
256  else
257  {
258  completely_inside = false;
259  }
260  }
261  // If the cell is completely inside the clearing box, both bool will
262  // be true. We need to add the cell iterator to the
263  // in_removal_box container.
264 
265  // If the cell is partially inside, at least one of the "if" will have
266  // failed, thus the second bool will be false but the first one true.
267  // We need to add the cell iterator to the edge_of_removal_box
268  // container.
269 
270  // If the cell is outside, both bool will be false, and we do nothing.
271  if (completely_inside)
272  in_removal_box.insert(cell);
273  else if (!completely_inside && partially_inside)
274  edge_of_removal_box.insert(cell);
275  }
276  }

283 {

289  to_remove_iterators.reserve(particle_handler.n_locally_owned_particles());
290 
291  // Loop over the first container
292  for (const auto &cell_in_box : in_removal_box)
293  {
294  // Check if this cell has particles
295  auto particles_in_cell = particle_handler.particles_in_cell(cell_in_box);
296 
297  if (!particles_in_cell.empty())
298  {
299  // Loop over the particle in the cell
300  for (auto particle_in_cell = particles_in_cell.begin();
301  particle_in_cell != particles_in_cell.end();
302  ++particle_in_cell)
303  {
304  // Since we know the cell is fully inside the box, we can
305  // remove every particles in it.
306 
307  to_remove_iterators.push_back(particle_in_cell);
308  }
309  }
310  }
311 
312  // Loop over the second container
313  for (auto cell_edge_of_removal_box = edge_of_removal_box.begin();
314  cell_edge_of_removal_box != edge_of_removal_box.end();
315  ++cell_edge_of_removal_box)
316  {
317  // Check if this cell has particle
318  auto particles_in_cell =
319  particle_handler.particles_in_cell(*cell_edge_of_removal_box);
320  const bool particles_exist_in_cell = !particles_in_cell.empty();
321 
322  if (particles_exist_in_cell)
323  {
324  // Loop over the particles in the cell
325  for (auto particle_in_cell = particles_in_cell.begin();
326  particle_in_cell != particles_in_cell.end();
327  ++particle_in_cell)
328  {
329  // We need to check if the particle is inside since the cell
330  // is at the edge.
331  Point<3> particle_position =
332  point_nd_to_3d(particle_in_cell->get_location());
333  if (p_min[0] <= particle_position[0] &&
334  p_max[0] >= particle_position[0] &&
335  p_min[1] <= particle_position[1] &&
336  p_max[1] >= particle_position[1] &&
337  p_min[2] <= particle_position[2] &&
338  p_max[2] >= particle_position[2])
339  {
340  to_remove_iterators.push_back(particle_in_cell);
341  }
342  }
343  }
344  }
345  particle_handler.remove_particles(to_remove_iterators);
346 }
