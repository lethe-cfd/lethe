49 // Carries out assigning the properties of inserted particles. The output vector
50 // is used in insert_global_particles as input argument
51 template <int dim>
52 void
53 Insertion<dim>::assign_particle_properties(
54  const DEMSolverParameters<dim> &dem_parameters,
55  const unsigned int &inserted_this_step_this_proc,
56  const unsigned int &current_inserting_particle_type,
57  std::vector<std::vector<double>> &particle_properties)
58 {
59  // Clearing and resizing particle_properties
60  particle_properties.reserve(inserted_this_step_this_proc);
61 
62  // Getting properties as local parameters
63  // TODO: MAYBE CHANGE THE INPUT TO PHYSICAL PROPERTIES DIRECTLY
64  auto physical_properties = dem_parameters.lagrangian_physical_properties;
65 
66  distributions_objects[current_inserting_particle_type]
67  ->particle_size_sampling(inserted_this_step_this_proc);
68 
69  // A loop is defined over the number of particles which are going to be
70  // inserted at this step
71  for (unsigned int particle_counter = 0;
72  particle_counter < inserted_this_step_this_proc;
73  ++particle_counter)
74  {
75  double type = current_inserting_particle_type;
76  // We make sure that the diameter is positive
77  double diameter =
78  std::abs(distributions_objects[current_inserting_particle_type]
79  ->particle_sizes[particle_counter]);
80  double density =
81  physical_properties.density_particle[current_inserting_particle_type];
82  double vel_x = dem_parameters.insertion_info.vel_x;
83  double vel_y = dem_parameters.insertion_info.vel_y;
84  double vel_z = dem_parameters.insertion_info.vel_z;
85  double omega_x = dem_parameters.insertion_info.omega_x;
86  double omega_y = dem_parameters.insertion_info.omega_y;
87  double omega_z = dem_parameters.insertion_info.omega_z;
88  double fem_force_x = 0.;
89  double fem_force_y = 0.;
90  double fem_force_z = 0.;
91  double fem_torque_x = 0.;
92  double fem_torque_y = 0.;
93  double fem_torque_z = 0.;
94  double mass = density * 4. / 3. * M_PI *
95  Utilities::fixed_power<3, double>(diameter * 0.5);
96  double volumetric_contribution = 0.;
97 
98  std::vector<double> properties_of_one_particle{type,
99  diameter,
100  vel_x,
101  vel_y,
102  vel_z,
103  omega_x,
104  omega_y,
105  omega_z,
106  fem_force_x,
107  fem_force_y,
108  fem_force_z,
109  fem_torque_x,
110  fem_torque_y,
111  fem_torque_z,
112  mass,
113  volumetric_contribution};
114 
115  particle_properties.push_back(properties_of_one_particle);
116  properties_of_one_particle.clear();
117  }
118 }

120 template <int dim>
121 void
122 Insertion<dim>::calculate_insertion_domain_maximum_particle_number(
123  const DEMSolverParameters<dim> &dem_parameters,
124  const ConditionalOStream &pcout)
125 {
126  // Getting properties as local parameters
127  const auto insertion_information = dem_parameters.insertion_info;
128 
129  // Checking if the insertion directions are valid (no repetition)
130  int axis_sum = 0;
131  if constexpr (dim == 2)
132  {
133  axis_sum = insertion_information.axis_0 + insertion_information.axis_1;
134 
135  AssertThrow(
136  axis_sum == 1,
137  ExcMessage("Invalid insertion directions: 2 directions are the same "));
138  }
139  if constexpr (dim == 3)
140  {
141  axis_sum = insertion_information.axis_0 + insertion_information.axis_1 +
142  insertion_information.axis_2;
143 
144  AssertThrow(
145  axis_sum == 3,
146  ExcMessage(
147  "Invalid insertion directions: at least 2 directions are the same "));
148  }
149 
150 
151  // This variable is used for calculation of the maximum number of particles
152  // that can fit in the chosen insertion box
153  int maximum_particle_number = 1;
154 
155  number_of_particles_directions.resize(dim);
156  axis_min.resize(dim);
157  axis_max.resize(dim);
158 
159  std::vector<unsigned int> axis_list = {insertion_information.axis_0,
160  insertion_information.axis_1};
161 
162  if constexpr (dim == 3)
163  {
164  axis_list.push_back(insertion_information.axis_2);
165  }
166 
167  // Assigning the minimum and maximum positions of the insertion box in respect
168  // to the axis order
169  for (unsigned int axis : axis_list)
170  {
171  switch (axis)
172  {
173  case 0:
174  axis_min[0] = insertion_information.x_min;
175  axis_max[0] = insertion_information.x_max;
176  break;
177  case 1:
178  axis_min[1] = insertion_information.y_min;
179  axis_max[1] = insertion_information.y_max;
180  break;
181  case 2:
182  axis_min[2] = insertion_information.z_min;
183  axis_max[2] = insertion_information.z_max;
184  break;
185  default:
186  AssertThrow(false,
187  ExcMessage("Insertion direction must be 0, 1 or 2"));
188  }
189 
190  // Assign max number of particles according to the direction and calculate
191  // the total max number (maximum_particle_number = max_x * max_y * max_z)
192  int number_of_particles = static_cast<int>(
193  (axis_max[axis] - axis_min[axis]) /
194  (insertion_information.distance_threshold * this->maximum_diameter));
195  number_of_particles_directions[axis] = number_of_particles;
196  maximum_particle_number *= number_of_particles;
197  }
198 
199  // If the inserted number of particles at this step exceeds the maximum
200  // number, a warning is printed
201  if (insertion_information.inserted_this_step > maximum_particle_number)
202  {
203  pcout << "Warning: the requested number of particles for insertion ("
204  << insertion_information.inserted_this_step
205  << ") is higher than maximum expected number of particles ("
206  << maximum_particle_number << ")" << std::endl;
207 
208  // Updating the number of inserted particles at each step
209  inserted_this_step = maximum_particle_number;
210  }
211  else
212  {
213  inserted_this_step = insertion_information.inserted_this_step;
214  }
215 }
