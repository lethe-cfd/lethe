164 /**
165  * @brief Used in multiphasic simulations to calculates the equivalent properties for a given phase. Method called in quadrature points loop
166  *
167  * @param phase Phase value for the given quadrature point
168  *
169  * @param property0 Property value for the fluid with index 0 (fluid for phase = 0)
170  *
171  * @param property1 Property value for the fluid with index 1 (fluid for phase = 1)
172  */
173 inline double
174 calculate_point_property(const double phase,
175  const double property0,
176  const double property1)
177 {
178  double property_eq = phase * property1 + (1 - phase) * property0;
179 
180  // Limit parameters value (patch)
181  // TODO see if necessary after compression term is added in the
182  // VOF advection equation
183  const double property_min = std::min(property0, property1);
184  const double property_max = std::max(property0, property1);
185  if (property_eq < property_min)
186  {
187  property_eq = property_min;
188  }
189  if (property_eq > property_max)
190  {
191  property_eq = property_max;
192  }
193 
194  return property_eq;
195 }

197 /**
198  * @brief Used in the calculate_properties_ch to retrieve the sign of the phase parameter
199  * @tparam T val argument's type
200  * @param val value of the variable for which we want to evaluate the sign
201  * @return an integer -1 or 1 depending of the sign of the phase parameter
202  */
203 template <typename T>
204 int
205 sgn(T val)
206 {
207  return (static_cast<T>(0) < val) - (val < static_cast<T>(0));
208 }

211 /**
212  * @brief Clip a properties between a lower and a higher value
213  * @tparam T argument's type
214  * @param[in,out] lower Lower admissible value
215  * @param[in,out] upper Upper admissible value
216  * @return Clipped variable that is not below the lower limit and not above the upper limit
217  *
218  */
219 template <typename T>
220 T
221 clip(const T &n, const T &lower, const T &upper)
222 {
223  return std::max(lower, std::min(n, upper));
224 }

226 /**
227  * @brief Used in multiphasic simulations to calculates the equivalent properties for a given phase given by the Cahn-Hilliard equations. Method called in quadrature points loop
228  *
229  * @param phase Phase value for the given quadrature point
230  *
231  * @param property0 Property value for the fluid with index 0 (fluid for phase = -1)
232  *
233  * @param property1 Property value for the fluid with index 1 (fluid for phase = 1)
234  */
235 inline double
236 calculate_point_property_cahn_hilliard(const double phase_cahn_hilliard,
237  const double property0,
238  const double property1)
239 {
240  double phase = phase_cahn_hilliard;
241 
242  if (std::abs(phase_cahn_hilliard) > 1)
243  {
244  phase = sgn(phase_cahn_hilliard);
245  }
246 
247  double property_avg = (property0 + property1) * 0.5;
248  double property_diff = (property0 - property1) * 0.5;
249 
250  double property_eq = phase * property_diff + property_avg;
251 
252  return property_eq;
253 }
