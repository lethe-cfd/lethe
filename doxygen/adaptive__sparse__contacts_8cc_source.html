44 template <int dim>
45 void
46 AdaptiveSparseContacts<dim>::calculate_granular_temperature_and_solid_fraction(
47  const Particles::ParticleHandler<dim> &particle_handler,
48  const std::set<typename DoFHandler<dim>::active_cell_iterator>
49  &local_and_ghost_cells_with_particles,
50  Vector<double> &granular_temperature_average,
51  Vector<double> &solid_fractions)
52 {
53  // Iterating through the active cells in cell set with particles
54  for (const auto &cell : local_and_ghost_cells_with_particles)
55  {
56  // Particles in the cell
57  auto particles_in_cell = particle_handler.particles_in_cell(cell);
58  const unsigned int n_particles_in_cell =
59  particle_handler.n_particles_in_cell(cell);
60 
61  // Initialize variables for solid fraction computation
62  double solid_fraction = 0.0;
63  double solid_volume = 0.0;
64  const double cell_volume = cell->measure();
65 
66  // Initialize variables for granular temperature computation
67  double granular_temperature_cell = 0.0;
68  Tensor<1, dim> velocity_cell_average;
69  Tensor<1, dim> cell_velocity_fluctuation_squared_average;
70 
71  // First loop over particles in cell to compute the sum of
72  // particle velocity and the solid volume of the current cell
73  for (auto particles_in_cell_iterator = particles_in_cell.begin();
74  particles_in_cell_iterator != particles_in_cell.end();
75  ++particles_in_cell_iterator)
76  {
77  // Get particle properties
78  auto particle_properties =
79  particles_in_cell_iterator->get_properties();
80  const double dp = particle_properties[DEM::PropertiesIndex::dp];
81 
82  for (int d = 0; d < dim; ++d)
83  {
84  // Get the particle velocity component (v_x, v_y & v_z if dim = 3)
85  int v_axis = DEM::PropertiesIndex::v_x + d;
86 
87  // Add the velocity component value
88  velocity_cell_average[d] += particle_properties[v_axis];
89  }
90 
91  solid_volume += M_PI * Utilities::fixed_power<dim>(dp) / (2.0 * dim);
92  }
93 
94  // Calculate average velocity in the cell (sum/n_particles)
95  velocity_cell_average /= n_particles_in_cell;
96 
97  // Calculate solid fraction of cell
98  solid_fraction = solid_volume / cell_volume;
99 
100  // Second loop over particle to compute the sum of the cell velocity
101  // fluctuations
102  for (auto particles_in_cell_iterator = particles_in_cell.begin();
103  particles_in_cell_iterator != particles_in_cell.end();
104  ++particles_in_cell_iterator)
105  {
106  auto particle_properties =
107  particles_in_cell_iterator->get_properties();
108 
109  for (int d = 0; d < dim; ++d)
110  {
111  // Get the particle velocity component (v_x, v_y & v_z if dim = 3)
112  int v_axis = DEM::PropertiesIndex::v_x + d;
113 
114  cell_velocity_fluctuation_squared_average[d] +=
115  Utilities::fixed_power<2>(particle_properties[v_axis] -
116  velocity_cell_average[d]);
117  }
118  }
119 
120  // Calculate average granular temperature in the cell
121  for (int d = 0; d < dim; ++d)
122  {
123  cell_velocity_fluctuation_squared_average[d] /= n_particles_in_cell;
124  granular_temperature_cell +=
125  cell_velocity_fluctuation_squared_average[d] / dim;
126  }
127 
128  // Store the average granular temperature and solid fraction with
129  // active cell index
130  granular_temperature_average[cell->active_cell_index()] =
131  granular_temperature_cell;
132  solid_fractions[cell->active_cell_index()] = solid_fraction;
133  }
134 }

136 template <int dim>
137 void
138 AdaptiveSparseContacts<dim>::identify_mobility_status(
139  const DoFHandler<dim> &background_dh,
140  const Particles::ParticleHandler<dim> &particle_handler,
141  const unsigned int n_active_cells,
142  MPI_Comm mpi_communicator)
143 {
144  // Reset cell status containers
145  cell_mobility_status.clear();
146 
147  // Get a copy of the active & ghost cells set to iterate over and remove cell
148  // of the set when the mobility status is known. It avoids unnecessary
149  // iteration for next loops. We don't want to modify the original set since
150  // it is only updated when there's load balancing or reading of checkpoints.
151  auto local_and_ghost_cells_copy = local_and_ghost_cells;
152 
153  // Create dummy dofs for background dof handler for the mobility_at_nodes
154  const FE_Q<dim> fe(1);
155  const unsigned int dofs_per_cell = fe.dofs_per_cell;
156 
157  // Get locally owned and relevant dofs
158  const IndexSet locally_owned_dofs = background_dh.locally_owned_dofs();
159  IndexSet locally_relevant_dofs =
160  DoFTools::extract_locally_relevant_dofs(background_dh);
161 
162  // Reinit all value of mobility at nodes as inactive
163  mobility_at_nodes.reinit(locally_owned_dofs,
164  locally_relevant_dofs,
165  mpi_communicator);
166  mobility_at_nodes = 0;
167 
168  // If the advection of particles setting is enabled (useful for CFD-DEM),
169  // mobility status are different: inactive and static_active status are
170  // advected and advected_active since they are not handled in the same way as
171  // in DEM only. They have a special handling in the velocity integration step
172  // because cell averaged velocity and acceleration are applied to particles,
173  // so they need this different status. The criteria for those status are the
174  // same as for DEM.
175  mobility_status inactive_status = (!advect_particles_enabled) ?
176  mobility_status::inactive :
177  mobility_status::advected;
178  mobility_status active_status = (!advect_particles_enabled) ?
179  mobility_status::static_active :
180  mobility_status::advected_active;
181 
182  // Check if the cell is empty (n_particle = 0), if so, nodes and cells are
183  // flagged as empty mobility status
184  for (auto cell = local_and_ghost_cells_copy.begin();
185  cell != local_and_ghost_cells_copy.end();)
186  {
187  // Check if the cell has any particles
188  if (particle_handler.n_particles_in_cell(*cell) == 0)
189  {
190  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
191  (*cell)->get_dof_indices(local_dof_indices);
192 
193  // Assign inactive status to cell and empty status to nodes
194  assign_mobility_status((*cell)->active_cell_index(),
195  local_dof_indices,
196  mobility_status::inactive,
197  mobility_status::empty);
198 
199  // Remove cell from cell set and iterate to the following cell,
200  // erase() function returns the next iterator
201  cell = local_and_ghost_cells_copy.erase(cell);
202  }
203  else
204  {
205  // Since erase() is not called, iterator is incremented
206  ++cell;
207  }
208  }
209 
210  // Update ghost values of mobility_at_nodes
211  mobility_at_nodes.update_ghost_values();
212 
213  // Calculate the average granular temperature and solid fraction for each
214  // cell currently in the local_and_ghost_cells_copy set (empty cells are
215  // already removed)
216  Vector<double> granular_temperature_average(n_active_cells);
217  Vector<double> solid_fractions(n_active_cells);
218  calculate_granular_temperature_and_solid_fraction(
219  particle_handler,
220  local_and_ghost_cells_copy,
221  granular_temperature_average,
222  solid_fractions);
223 
224  // Check if the cell is mobile by criteria:
225  // * granular temperature > threshold or
226  // * solid fraction of cell < threshold or
227  // * is next to an empty cell
228  // If so, nodes are flagged and cells are stored with mobile status
229  for (auto cell = local_and_ghost_cells_copy.begin();
230  cell != local_and_ghost_cells_copy.end();)
231  {
232  const unsigned int cell_id = (*cell)->active_cell_index();
233  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
234  (*cell)->get_dof_indices(local_dof_indices);
235 
236  // Check if the cell has any empty neighbor cell by the value at nodes
237  bool has_empty_node = false;
238  for (auto node_id : local_dof_indices)
239  {
240  if (mobility_at_nodes(node_id) == mobility_status::empty)
241  {
242  has_empty_node = true;
243  break; // No need to check other nodes
244  }
245  }
246 
247  // Check if the cell is mobile by criteria (granular temperature, solid
248  // fraction, next to empty cell)
249  if (granular_temperature_average[cell_id] >
250  granular_temperature_threshold ||
251  solid_fractions[cell_id] < solid_fraction_threshold || has_empty_node)
252  {
253  // Assign mobile status to cell and nodes
254  assign_mobility_status(cell_id,
255  local_dof_indices,
256  mobility_status::mobile,
257  mobility_status::mobile);
258 
259  // Remove cell from cell set
260  cell = local_and_ghost_cells_copy.erase(cell);
261  }
262  else
263  {
264  ++cell;
265  }
266  }
267 
268  mobility_at_nodes.update_ghost_values();
269 
270  // Check if the cell is mobile by neighbor (at least a node is flagged as
271  // mobile from previous check), this is the additional mobile layer.
272  // If so, cells are stored in map as mobile status and nodes that are not
273  // mobile are flagged as active
274  for (auto cell = local_and_ghost_cells_copy.begin();
275  cell != local_and_ghost_cells_copy.end();)
276  {
277  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
278  (*cell)->get_dof_indices(local_dof_indices);
279 
280  // Check if the cell has any mobile nodes by the value at nodes
281  bool has_mobile_node = false;
282  for (auto node_id : local_dof_indices)
283  {
284  if (mobility_at_nodes(node_id) == mobility_status::mobile)
285  {
286  has_mobile_node = true;
287  break;
288  }
289  }
290 
291  if (has_mobile_node)
292  {
293  // Assign mobile status to cell and active to nodes
294  assign_mobility_status((*cell)->active_cell_index(),
295  local_dof_indices,
296  mobility_status::mobile,
297  active_status);
298 
299  // Remove cell from cell set and iterate to the following cell
300  cell = local_and_ghost_cells_copy.erase(cell);
301  }
302  else
303  {
304  ++cell;
305  }
306  }
307 
308  mobility_at_nodes.update_ghost_values();
309 
310  // Check if the cell is active (at least a node is flagged as active from
311  // previous check), this is the layer of active cells
312  // If so, cells are stored with active status in the map
313  for (auto cell = local_and_ghost_cells_copy.begin();
314  cell != local_and_ghost_cells_copy.end();)
315  {
316  std::vector<types::global_dof_index> local_dofs_indices(dofs_per_cell);
317  (*cell)->get_dof_indices(local_dofs_indices);
318 
319  // Check if cell has active nodes
320  bool has_active_nodes = false;
321  for (auto node_id : local_dofs_indices)
322  {
323  if (mobility_at_nodes(node_id) == (int)active_status)
324  {
325  has_active_nodes = true;
326  break;
327  }
328  }
329 
330  if (has_active_nodes)
331  {
332  // Assign active status to cell (no need to assign active to nodes)
333  assign_mobility_status((*cell)->active_cell_index(), active_status);
334  cell = local_and_ghost_cells_copy.erase(cell);
335  }
336  else
337  {
338  ++cell;
339  }
340  }
341 
342  // Store the inactive cells in the map, those are all the remaining cells
343  // in the local & ghost cells set
344  for (auto cell = local_and_ghost_cells_copy.begin();
345  cell != local_and_ghost_cells_copy.end();
346  ++cell)
347  {
348  // Assign inactive status to cell in map
349  assign_mobility_status((*cell)->active_cell_index(), inactive_status);
350  }
351 }

353 template <int dim>
354 void
355 AdaptiveSparseContacts<dim>::update_average_velocities_acceleration(
356  Particles::ParticleHandler<dim> &particle_handler,
357  const Tensor<1, 3> &g,
358  const std::vector<Tensor<1, 3>> &force,
359  const double dt)
360 {
361  cell_velocities_accelerations.clear();
362 
363  // Tensors for velocity and acceleration * dt computation
364  Tensor<1, 3> velocity_cell_average;
365  Tensor<1, 3> acc_dt_cell_average;
366 
367  const Tensor<1, 3> dt_g = g * dt;
368 
369  // Loop over all the cells even if they are not mobile to reset the force
370  // and torque values
371  for (auto cell : local_and_ghost_cells)
372  {
373  if (cell->is_locally_owned())
374  {
375  const unsigned int n_particles_in_cell =
376  particle_handler.n_particles_in_cell(cell);
377 
378  velocity_cell_average.clear();
379  acc_dt_cell_average.clear();
380 
381  if (n_particles_in_cell > 0)
382  {
383  auto particles_in_cell = particle_handler.particles_in_cell(cell);
384  for (auto &particle : particles_in_cell)
385  {
386  // Get particle properties
387  auto particle_properties = particle.get_properties();
388  types::particle_index particle_id =
389  particle.get_local_index();
390 
391  double dt_mass_inverse =
392  dt / particle_properties[DEM::PropertiesIndex::mass];
393 
394  // Calculate the acceleration of the particle times the time
395  // step
396  Tensor<1, 3> acc_dt_particle =
397  dt_g + force[particle_id] * dt_mass_inverse;
398 
399  // Add up the acc. * dt for the average
400  acc_dt_cell_average += acc_dt_particle;
401 
402  for (int d = 0; d < dim; ++d)
403  {
404  // Add up the current velocity for the average velocity
405  velocity_cell_average[d] +=
406  particle_properties[DEM::PropertiesIndex::v_x + d] +
407  acc_dt_particle[d];
408  }
409  }
410 
411  // Compute the average velocity and acceleration * dt
412  velocity_cell_average /= n_particles_in_cell;
413  acc_dt_cell_average /= n_particles_in_cell;
414  }
415 
416  // Insert the average pair in map
417  cell_velocities_accelerations.insert(std::make_pair(
418  cell, std::make_pair(velocity_cell_average, acc_dt_cell_average)));
419  }
420  }
421 }
