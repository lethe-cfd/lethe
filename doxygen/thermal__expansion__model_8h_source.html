114 /**
115  * @brief Implements a phase-dependant thermal expansion coefficient
116  */
117 class ThermalExpansionPhaseChange : public ThermalExpansionModel
118 {
119 public:
120  /**
121  * @brief Default constructor
122  */
123  ThermalExpansionPhaseChange(const Parameters::PhaseChange phase_change_params)
124  : p_phase_change_params(phase_change_params)
125  {
126  this->model_depends_on[field::temperature] = true;
127  }
128 
129  /**
130  * @brief value Calculates the value of the thermal expansion coefficient
131  * @param fields_value Value of the various field on which the thermal expansion coefficient depends.
132  * @return value of the thermal expansion coefficient calculated with the fields_value.
133  */
134  double
135  value(const std::map<field, double> &fields_value) override
136  {
137  if (fields_value.at(field::temperature) > p_phase_change_params.T_liquidus)
138  return p_phase_change_params.thermal_expansion_l;
139  else
140  return p_phase_change_params.thermal_expansion_s;
141  };
142 
143  /**
144  * @brief vector_value Calculates the vector value of thermal expansion coefficients
145  * @param field_vectors Vector of properties on which the thermal expansion coefficients depend
146  * @param property_vector Values of the thermal expansion coefficients
147  */
148  void
149  vector_value(const std::map<field, std::vector<double>> &field_vectors,
150  std::vector<double> &property_vector) override
151  {
152  const std::vector<double> &T = field_vectors.at(field::temperature);
153  for (unsigned int i = 0; i < property_vector.size(); ++i)
154  {
155  // Thermal expansion of solid phase
156  if (T[i] > p_phase_change_params.T_liquidus)
157  property_vector[i] = p_phase_change_params.thermal_expansion_l;
158  else
159  property_vector[i] = p_phase_change_params.thermal_expansion_s;
160  // Thermal expansion of liquid phase
161  }
162  };
163 
164  /**
165  * @brief jacobian Calculates the jacobian (the partial derivative) of the thermal expansion with respect to a field
166  * @param field_values Value of the various fields on which the property may depend.
167  * @param id Indicator of the field with respect to which the jacobian
168  * should be calculated
169  * @return value of the partial derivative of the thermal expansion with respect to the field.
170  */
171 
172  double
173  jacobian(const std::map<field, double> &field_values, field id) override
174  {
175  if (id == field::temperature)
176  return this->numerical_jacobian(field_values, field::temperature);
177  else
178  return 0;
179  };
180 
181  /**
182  * @brief vector_jacobian Calculate the derivative of the thermal expansion with respect to a field
183  * @param field_vectors Vector for the values of the fields used to evaluate the property
184  * @param id Identifier of the field with respect to which a derivative should be calculated
185  * @param jacobian Vector of the value of the derivative of the thermal expansion with respect to the field id
186  */
187 
188  void
189  vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
190  const field id,
191  std::vector<double> &jacobian_vector) override
192  {
193  vector_numerical_jacobian(field_vectors, id, jacobian_vector);
194  };
195 
196 private:
197  Parameters::PhaseChange p_phase_change_params;
198 };
