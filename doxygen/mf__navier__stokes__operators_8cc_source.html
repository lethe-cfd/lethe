19 /**
20  * @brief Creates and fills a table that works as bool dof mask object
21  * needed for the sparsity pattern and computation of the system matrix
22  * of the coarse level in case FE_Q_iso_Q1 elements are used.
23  *
24  * @tparam dim An integer that denotes the number of spatial dimensions.
25  * @tparam spacedim Space dimension.
26  * @param[in] fe Type of finite element used.
27  * @param[in] quadrature Quadrature being used.
28  * @return Table<2, bool> Boolean mask with pairs of dof indices.
29  */
30 template <int dim, int spacedim>
31 const Table<2, bool>
32 create_bool_dof_mask(const FiniteElement<dim, spacedim> &fe,
33  const Quadrature<dim> &quadrature)
34 {
35  const auto compute_scalar_bool_dof_mask = [&quadrature](const auto &fe) {
36  Table<2, bool> bool_dof_mask(fe.dofs_per_cell, fe.dofs_per_cell);
37  MappingQ1<dim, spacedim> mapping;
38  FEValues<dim> fe_values(mapping, fe, quadrature, update_values);
39 
40  Triangulation<dim, spacedim> tria;
41  GridGenerator::hyper_cube(tria);
42 
43  fe_values.reinit(tria.begin());
44  for (unsigned int i = 0; i < fe.dofs_per_cell; ++i)
45  for (unsigned int j = 0; j < fe.dofs_per_cell; ++j)
46  {
47  double sum = 0;
48  for (unsigned int q = 0; q < quadrature.size(); ++q)
49  sum += fe_values.shape_value(i, q) * fe_values.shape_value(j, q);
50  if (sum != 0)
51  bool_dof_mask(i, j) = true;
52  }
53 
54  return bool_dof_mask;
55  };
56 
57  Table<2, bool> bool_dof_mask(fe.dofs_per_cell, fe.dofs_per_cell);
58 
59  if (fe.n_components() == 1)
60  {
61  bool_dof_mask = compute_scalar_bool_dof_mask(fe);
62  }
63  else
64  {
65  const auto scalar_bool_dof_mask =
66  compute_scalar_bool_dof_mask(fe.base_element(0));
67 
68  for (unsigned int i = 0; i < fe.n_dofs_per_cell(); ++i)
69  for (unsigned int j = 0; j < fe.n_dofs_per_cell(); ++j)
70  if (scalar_bool_dof_mask[fe.system_to_component_index(i).second]
71  [fe.system_to_component_index(j).second])
72  bool_dof_mask[i][j] = true;
73  }
74 
75 
76  return bool_dof_mask;
77 }

79 /**
80  * @brief Matrix free helper function
81  *
82  * @tparam dim An integer that denotes the number of spatial dimensions.
83  * @tparam Number Abstract type for number across the class (i.e., double).
84  * @param function Function to evaluate.
85  * @param p_vectorized Batch of points to evaluate function at.
86  * @return VectorizedArray<Number> Batch of evaluated values.

88 template <int dim, typename Number>
89 VectorizedArray<Number>
90 evaluate_function(const Function<dim> &function,
91  const Point<dim, VectorizedArray<Number>> &p_vectorized)
92 {
93  VectorizedArray<Number> result;
94  for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)
95  {
96  Point<dim> p;
97  for (unsigned int d = 0; d < dim; ++d)
98  p[d] = p_vectorized[d][v];
99  result[v] = function.value(p);
100  }
101  return result;
102 }

105  * @brief Matrix free helper function
106  *
107  * @tparam dim An integer that denotes the number of spatial dimensions.
108  * @tparam Number Abstract type for number across the class (i.e., double).
109  * @tparam components Number of solution components.

111  * @param p_vectorized Batch of points to evaluate function at.
112  * @return Tensor<1, components, VectorizedArray<Number>> Batch of evaluated values.

114 template <int dim, typename Number, int components>
115 Tensor<1, components, VectorizedArray<Number>>
116 evaluate_function(const Function<dim> &function,
117  const Point<dim, VectorizedArray<Number>> &p_vectorized)
118 {
119  Tensor<1, components, VectorizedArray<Number>> result;
120  for (unsigned int v = 0; v < VectorizedArray<Number>::size(); ++v)

122  Point<dim> p;
123  for (unsigned int d = 0; d < dim; ++d)
124  p[d] = p_vectorized[d][v];

126  result[d][v] = function.value(p, d);
127  }
128  return result;

163 void
164 NavierStokesOperatorBase<dim, number>::reinit(
165  const Mapping<dim> &mapping,
166  const DoFHandler<dim> &dof_handler,
167  const AffineConstraints<number> &constraints,
168  const Quadrature<dim> &quadrature,
169  const Function<dim> *forcing_function,
170  const double kinematic_viscosity,
171  const StabilizationType stabilization,
172  const unsigned int mg_level,
173  std::shared_ptr<SimulationControl> simulation_control)
174 {
175  this->system_matrix.clear();
176  this->constraints.copy_from(constraints);
177 
178  typename MatrixFree<dim, number>::AdditionalData additional_data;
179  additional_data.mapping_update_flags =
180  (update_values | update_gradients | update_JxW_values |
181  update_quadrature_points | update_hessians);
182  additional_data.mg_level = mg_level;
183 
184  matrix_free.reinit(
185  mapping, dof_handler, this->constraints, quadrature, additional_data);
186 
187  this->fe_degree = dof_handler.get_fe().degree;
188 
189  this->forcing_function = forcing_function;
190 
191  this->kinematic_viscosity = kinematic_viscosity;
192 
193  if (stabilization ==
194  Parameters::Stabilization::NavierStokesStabilization::pspg_supg ||
195  stabilization ==
196  Parameters::Stabilization::NavierStokesStabilization::gls)
197  {
198  this->stabilization = stabilization;
199  }
200  else
201  throw std::runtime_error(
202  "Only SUPG/PSPG and GLS stabilization is supported at the moment.");
203 
204  this->simulation_control = simulation_control;
205 
206  this->compute_element_size();
207 
208  bool_dof_mask = create_bool_dof_mask(dof_handler.get_fe(), quadrature);
209 
210  constrained_indices.clear();
211  for (auto i : this->matrix_free.get_constrained_dofs())
212  constrained_indices.push_back(i);
213  constrained_values.resize(constrained_indices.size());
214 
215  if (this->matrix_free.get_mg_level() != numbers::invalid_unsigned_int)
216  {
217  std::vector<types::global_dof_index> interface_indices;
218  IndexSet refinement_edge_indices;
219  refinement_edge_indices = get_refinement_edges(this->matrix_free);
220  refinement_edge_indices.fill_index_vector(interface_indices);
221 
222  edge_constrained_indices.clear();
223  edge_constrained_indices.reserve(interface_indices.size());
224  edge_constrained_values.resize(interface_indices.size());
225  const IndexSet &locally_owned =
226  this->matrix_free.get_dof_handler().locally_owned_mg_dofs(
227  this->matrix_free.get_mg_level());
228  for (unsigned int i = 0; i < interface_indices.size(); ++i)
229  if (locally_owned.is_element(interface_indices[i]))
230  edge_constrained_indices.push_back(
231  locally_owned.index_within_set(interface_indices[i]));
232 
233  this->has_edge_constrained_indices =
234  Utilities::MPI::max(edge_constrained_indices.size(),
235  dof_handler.get_communicator()) > 0;
236 
237  if (this->has_edge_constrained_indices)
238  {
239  edge_constrained_cell.resize(matrix_free.n_cell_batches(), false);
240 
241  VectorType temp;
242  matrix_free.initialize_dof_vector(temp);
243 
244  for (unsigned int i = 0; i < edge_constrained_indices.size(); ++i)
245  temp.local_element(edge_constrained_indices[i]) = 1.0;
246 
247  temp.update_ghost_values();
248 
249  FECellIntegrator integrator(matrix_free);
250 
251  for (unsigned int cell = 0; cell < matrix_free.n_cell_batches();
252  ++cell)
253  {
254  integrator.reinit(cell);
255  integrator.read_dof_values(temp);
256 
257  for (unsigned int i = 0; i < integrator.dofs_per_cell; ++i)
258  if ((integrator.begin_dof_values()[i] ==
259  VectorizedArray<number>()) == false)
260  {
261  edge_constrained_cell[cell] = true;
262  break;
263  }
264  }
265 
266 #ifdef DEBUG
267  // The following lines are used when locally refined meshes are used
268  // to verify the number of edge constrained cells
269  unsigned int count = 0;
270  for (const auto i : edge_constrained_cell)
271  if (i)
272  count++;
273 
274  const unsigned int count_global =
275  Utilities::MPI::sum(count, dof_handler.get_communicator());
276 
277  const unsigned int count_cells_global =
278  Utilities::MPI::sum(matrix_free.n_cell_batches(),
279  dof_handler.get_communicator());
280 
281  if (Utilities::MPI::this_mpi_process(
282  dof_handler.get_communicator()) == 0)
283  std::cout << count_global << " " << count_cells_global << std::endl;
284 #endif
285  }

289 template <int dim, typename number>
290 void
291 NavierStokesOperatorBase<dim, number>::compute_element_size()
292 {
293  const unsigned int n_cells =
294  matrix_free.n_cell_batches() + matrix_free.n_ghost_cell_batches();
295  element_size.resize(n_cells);
296 
297  for (unsigned int cell = 0; cell < n_cells; ++cell)
298  {
299  for (auto lane = 0u;
300  lane < matrix_free.n_active_entries_per_cell_batch(cell);
301  lane++)
302  {
303  const double h_k =
304  matrix_free.get_cell_iterator(cell, lane)->measure();
305 
306  if (dim == 2)
307  element_size[cell][lane] = std::sqrt(4. * h_k / M_PI) / fe_degree;
308  else if (dim == 3)
309  element_size[cell][lane] =
310  std::pow(6 * h_k / M_PI, 1. / 3.) / fe_degree;
311  }
312  }
313 }

356 template <int dim, typename number>
357 void
358 NavierStokesOperatorBase<dim, number>::vmult(VectorType &dst,
359  const VectorType &src) const
360 {
361  this->timer.enter_subsection("operator::vmult");
362 
363  // save values for edge constrained dofs and set them to 0 in src vector
364  for (unsigned int i = 0; i < edge_constrained_indices.size(); ++i)
365  {
366  edge_constrained_values[i] =
367  src.local_element(edge_constrained_indices[i]);
368 
369  const_cast<LinearAlgebra::distributed::Vector<number> &>(src)
370  .local_element(edge_constrained_indices[i]) = 0.;
371  }
372 
373  this->matrix_free.cell_loop(
374  &NavierStokesOperatorBase::do_cell_integral_range, this, dst, src, true);
375 
376  // set constrained dofs as the sum of current dst value and src value
377  for (unsigned int i = 0; i < constrained_indices.size(); ++i)
378  dst.local_element(constrained_indices[i]) =
379  src.local_element(constrained_indices[i]);
380 
381  // restoring edge constrained dofs in src and dst
382  for (unsigned int i = 0; i < edge_constrained_indices.size(); ++i)
383  {
384  const_cast<LinearAlgebra::distributed::Vector<number> &>(src)
385  .local_element(edge_constrained_indices[i]) =
386  edge_constrained_values[i];
387  dst.local_element(edge_constrained_indices[i]) =
388  edge_constrained_values[i];
389  }
390 
391  this->timer.leave_subsection("operator::vmult");
392 }

453 template <int dim, typename number>
454 const TrilinosWrappers::SparseMatrix &
455 NavierStokesOperatorBase<dim, number>::get_system_matrix() const
456 {
457  this->timer.enter_subsection("operator::get_system_matrix");
458 
459  if (system_matrix.m() == 0 && system_matrix.n() == 0)
460  {
461  const auto &dof_handler = this->matrix_free.get_dof_handler();
462 
463  const unsigned int mg_level = this->matrix_free.get_mg_level();
464 
465  IndexSet locally_relevant_dofs;
466  IndexSet locally_owned_dofs;
467 
468  if (mg_level != numbers::invalid_unsigned_int)
469  {
470  locally_relevant_dofs =
471  DoFTools::extract_locally_relevant_level_dofs(dof_handler,
472  mg_level);
473  locally_owned_dofs = dof_handler.locally_owned_mg_dofs(mg_level);
474  }
475  else
476  {
477  locally_owned_dofs = dof_handler.locally_owned_dofs();
478  locally_relevant_dofs =
479  DoFTools::extract_locally_relevant_dofs(dof_handler);
480  }
481 
482  DynamicSparsityPattern dsp(locally_relevant_dofs);
483 
484  if (mg_level != numbers::invalid_unsigned_int)
485  {
486  // the following code does the same as
487  // MGTools::make_sparsity_pattern() but also
488  // considers bool_dof_mask for FE_Q_iso_Q1
489  std::vector<types::global_dof_index> dofs_on_this_cell;
490 
491  for (const auto &cell : dof_handler.cell_iterators_on_level(mg_level))
492  if (cell->is_locally_owned_on_level())
493  {
494  const unsigned int dofs_per_cell =
495  dof_handler.get_fe().n_dofs_per_cell();
496  dofs_on_this_cell.resize(dofs_per_cell);
497  cell->get_mg_dof_indices(dofs_on_this_cell);
498 
499  constraints.add_entries_local_to_global(dofs_on_this_cell,
500  dsp,
501  false,
502  bool_dof_mask);
503  }
504  }
505  else
506  {
507  // the following code does the same as
508  // DoFTools::make_sparsity_pattern() but also
509  // considers bool_dof_mask for FE_Q_iso_Q1
510  std::vector<types::global_dof_index> dofs_on_this_cell;
511 
512  for (const auto &cell : dof_handler.active_cell_iterators())
513  if (cell->is_locally_owned())
514  {
515  const unsigned int dofs_per_cell =
516  cell->get_fe().n_dofs_per_cell();
517  dofs_on_this_cell.resize(dofs_per_cell);
518  cell->get_dof_indices(dofs_on_this_cell);
519 
520  constraints.add_entries_local_to_global(dofs_on_this_cell,
521  dsp,
522  false,
523  bool_dof_mask);
524  }
525  }
526 
527  SparsityTools::distribute_sparsity_pattern(
528  dsp,
529  locally_owned_dofs,
530  dof_handler.get_triangulation().get_communicator(),
531  locally_relevant_dofs);
532 
533  system_matrix.reinit(locally_owned_dofs,
534  locally_owned_dofs,
535  dsp,
536  dof_handler.get_triangulation().get_communicator());
537  }
538 
539  system_matrix = 0.0;
540 
541  // the following code does the same as
542  // MatrixFreeTools::compute_matrix() but with
543  // minor corrections for FE_Q_iso_Q1
544 
545  const auto &lexicographic_numbering =
546  matrix_free.get_shape_info().lexicographic_numbering;
547 
548  unsigned int cell = numbers::invalid_unsigned_int;
549  unsigned int column = numbers::invalid_unsigned_int;
550 
551  MatrixFreeTools::
552  compute_matrix<dim, -1, 0, dim + 1, number, VectorizedArray<number>>(
553  matrix_free, constraints, system_matrix, [&](auto &integrator) {
554  if (cell != integrator.get_current_cell_index())
555  {
556  cell = integrator.get_current_cell_index();
557  column = 0;
558  }
559 
560  do_cell_integral_local(integrator);
561 
562  // remove spurious entries for FE_Q_iso_Q1
563  for (unsigned int i = 0; i < lexicographic_numbering.size(); ++i)
564  if (!bool_dof_mask[lexicographic_numbering[i]]
565  [lexicographic_numbering[column]])
566  integrator.begin_dof_values()[i] = 0.0;
567 
568  column++;
569  });
570 
571  this->timer.leave_subsection("operator::get_system_matrix");
572 
573  return this->system_matrix;
574 }

590 template <int dim, typename number>
591 void
592 NavierStokesOperatorBase<dim, number>::
593  evaluate_non_linear_term_and_calculate_tau(const VectorType &newton_step)
594 {
595  this->timer.enter_subsection("operator::evaluate_non_linear_term");
596 
597  const unsigned int n_cells = matrix_free.n_cell_batches();
598  FECellIntegrator integrator(matrix_free);
599 
600  // Set appropriate size for tables
601  nonlinear_previous_values.reinit(n_cells, integrator.n_q_points);
602  nonlinear_previous_gradient.reinit(n_cells, integrator.n_q_points);
603  nonlinear_previous_hessian_diagonal.reinit(n_cells, integrator.n_q_points);
604  stabilization_parameter.reinit(n_cells, integrator.n_q_points);
605  stabilization_parameter_lsic.reinit(n_cells, integrator.n_q_points);
606 
607  // Define 1/dt if the simulation is transient
608  double sdt = 0.0;
609 
610  bool transient =
611  (is_bdf(this->simulation_control->get_assembly_method())) ? true : false;
612 
613  if (transient)
614  {
615  const auto time_steps_vector =
616  this->simulation_control->get_time_steps_vector();
617  const double dt = time_steps_vector[0];
618  sdt = 1. / dt;
619  }
620 
621  for (unsigned int cell = 0; cell < n_cells; ++cell)
622  {
623  integrator.reinit(cell);
624  integrator.read_dof_values_plain(newton_step);
625  integrator.evaluate(EvaluationFlags::values | EvaluationFlags::gradients |
626  EvaluationFlags::hessians);
627 
628  // Get previously calculated element size needed for tau
629  const auto h = integrator.read_cell_data(this->get_element_size());
630 
631  for (const auto q : integrator.quadrature_point_indices())
632  {
633  nonlinear_previous_values(cell, q) = integrator.get_value(q);
634  nonlinear_previous_gradient(cell, q) = integrator.get_gradient(q);
635  nonlinear_previous_hessian_diagonal(cell, q) =
636  integrator.get_hessian_diagonal(q);
637 
638  // Calculate tau
639  VectorizedArray<number> u_mag_squared = 1e-12;
640  for (unsigned int k = 0; k < dim; ++k)
641  u_mag_squared +=
642  Utilities::fixed_power<2>(integrator.get_value(q)[k]);
643 
644  stabilization_parameter(cell, q) =
645  1. / std::sqrt(Utilities::fixed_power<2>(sdt) +

648  4. * this->kinematic_viscosity / (h * h)));
649 
650  stabilization_parameter_lsic(cell, q) =
651  std::sqrt(u_mag_squared) * h * 0.5;
652  }
653  }

780 NavierStokesStabilizedOperator<dim, number>::NavierStokesStabilizedOperator() =
781  default;

783 /**
784  * The expressions calculated in this cell integral are:
785  * (q,∇δu) + (v,∂t δu) + (v,(u·∇)δu) + (v,(δu·∇)u) - (∇·v,δp) + ν(∇v,∇δu) (Weak
786  * form Jacobian),
787  * plus three additional terms in the case of SUPG-PSPG stabilization:
788  * \+ (∂t δu +(u·∇)δu + (δu·∇)u + ∇δp - ν∆δu)τ·∇q (PSPG Jacobian)
789  * \+ (∂t δu +(u·∇)δu + (δu·∇)u + ∇δp - ν∆δu)τu·∇v (SUPG Jacobian Part 1)
790  * \+ (∂t u +(u·∇)u + ∇p - ν∆u - f )τδu·∇v (SUPG Jacobian Part 2),
791  * plus two additional terms in the case of full gls stabilization:
792  * \+ (∂t δu +(u·∇)δu + (δu·∇)u + ∇δp - ν∆δu)τ(−ν∆v) (GLS Jacobian)
793  * \+ (∇·δu)τ'(∇·v) (LSIC Jacobian).
794  */
795 template <int dim, typename number>
796 void
797 NavierStokesStabilizedOperator<dim, number>::do_cell_integral_local(
798  FECellIntegrator &integrator) const
799 {
800  integrator.evaluate(EvaluationFlags::values | EvaluationFlags::gradients |
801  EvaluationFlags::hessians);
802 
803  const unsigned int cell = integrator.get_current_cell_index();
804 
805  // To identify whether the problem is transient or steady
806  bool transient =
807  (is_bdf(this->simulation_control->get_assembly_method())) ? true : false;
808 
809  // Vector for BDF coefficients
810  const Vector<double> *bdf_coefs;
811  if (transient)
812  bdf_coefs = &this->simulation_control->get_bdf_coefficients();
813 
814  for (const auto q : integrator.quadrature_point_indices())
815  {
816  // Evaluate source term function
817  Tensor<1, dim, VectorizedArray<number>> source_value;
818  Point<dim, VectorizedArray<number>> point_batch =
819  integrator.quadrature_point(q);
820  source_value =
821  evaluate_function<dim, number, dim>(*(this->forcing_function),
822  point_batch);
823 
824  // Add to source term the dynamic flow control force (zero if not enabled)
825  source_value += this->beta_force;
826 
827  // Gather the original value/gradient
828  typename FECellIntegrator::value_type value = integrator.get_value(q);
829  typename FECellIntegrator::gradient_type gradient =
830  integrator.get_gradient(q);
831  typename FECellIntegrator::gradient_type hessian_diagonal =
832  integrator.get_hessian_diagonal(q);
833 
834  // Result value/gradient we will use
835  typename FECellIntegrator::value_type value_result;
836  typename FECellIntegrator::gradient_type gradient_result;
837  typename FECellIntegrator::hessian_type hessian_result;
838 
839  // Gather previous values of the velocity and the pressure
840  auto previous_values = this->nonlinear_previous_values(cell, q);
841  auto previous_gradient = this->nonlinear_previous_gradient(cell, q);
842  auto previous_hessian_diagonal =
843  this->nonlinear_previous_hessian_diagonal(cell, q);
844 
845  Tensor<1, dim + 1, VectorizedArray<number>> previous_time_derivatives;
846  if (transient)
847  previous_time_derivatives =
848  this->time_derivatives_previous_solutions(cell, q);
849 
850  // Get stabilization parameter
851  const auto tau = this->stabilization_parameter[cell][q];
852  const auto tau_lsic = this->stabilization_parameter_lsic[cell][q];
853 
854  // Weak form Jacobian
855  for (unsigned int i = 0; i < dim; ++i)
856  {
857  // ν(∇v,∇δu)
858  gradient_result[i] = this->kinematic_viscosity * gradient[i];
859  // -(∇·v,δp)
860  gradient_result[i][i] += -value[dim];
861  // +(q,∇δu)
862  value_result[dim] += gradient[i][i];
863 
864  for (unsigned int k = 0; k < dim; ++k)
865  {
866  // +(v,(u·∇)δu + (δu·∇)u)
867  value_result[i] += gradient[i][k] * previous_values[k] +
868  previous_gradient[i][k] * value[k];
869  }
870  // +(v,∂t δu)
871  if (transient)
872  value_result[i] += (*bdf_coefs)[0] * value[i];
873  }
874 
875  // PSPG Jacobian
876  for (unsigned int i = 0; i < dim; ++i)
877  {
878  for (unsigned int k = 0; k < dim; ++k)
879  {
880  // (-ν∆δu + (u·∇)δu + (δu·∇)u)·τ∇q
881  gradient_result[dim][i] +=
882  tau * (-this->kinematic_viscosity * hessian_diagonal[i][k] +
883  gradient[i][k] * previous_values[k] +
884  previous_gradient[i][k] * value[k]);
885  }
886  // +(∂t δu)·τ∇q
887  if (transient)
888  gradient_result[dim][i] += tau * (*bdf_coefs)[0] * value[i];
889  }
890  // (∇δp)τ·∇q
891  gradient_result[dim] += tau * gradient[dim];
892 
893  // SUPG Jacobian
894  for (unsigned int i = 0; i < dim; ++i)
895  {
896  for (unsigned int k = 0; k < dim; ++k)
897  {
898  // Part 1
899  for (unsigned int l = 0; l < dim; ++l)
900  {
901  // +((u·∇)δu + (δu·∇)u - ν∆δu)τ(u·∇)v
902  gradient_result[i][k] +=
903  tau * previous_values[k] *
904  (gradient[i][l] * previous_values[l] +
905  previous_gradient[i][l] * value[l] -
906  this->kinematic_viscosity * hessian_diagonal[i][l]);
907  }
908  // +(∇δp)τ(u·∇)v
909  gradient_result[i][k] +=
910  tau * previous_values[k] * (gradient[dim][i]);
911 
912  // +(∂t δu)τ(u·∇)v
913  if (transient)
914  gradient_result[i][k] +=
915  tau * previous_values[k] * ((*bdf_coefs)[0] * value[i]);
916 
917 
918  // Part 2
919  for (unsigned int l = 0; l < dim; ++l)
920  {
921  // +((u·∇)u - ν∆u)τ(δu·∇)v
922  gradient_result[i][k] +=
923  tau * value[k] *
924  (previous_gradient[i][l] * previous_values[l] -
925  this->kinematic_viscosity *
926  previous_hessian_diagonal[i][l]);
927  }
928  // +(∇p - f)τ(δu·∇)v
929  gradient_result[i][k] +=
930  tau * value[k] * (previous_gradient[dim][i] - source_value[i]);
931 
932  // +(∂t u)τ(δu·∇)v
933  if (transient)
934  gradient_result[i][k] += tau * value[k] *
935  ((*bdf_coefs)[0] * previous_values[i] +
936  previous_time_derivatives[i]);
937  }
938  }
939 
940  if (this->stabilization ==
941  Parameters::Stabilization::NavierStokesStabilization::gls)
942  {
943  // GLS Jacobian
944  for (unsigned int i = 0; i < dim; ++i)
945  {
946  for (unsigned int k = 0; k < dim; ++k)
947  {
948  for (unsigned int l = 0; l < dim; ++l)
949  {
950  // +((u·∇)δu + (δu·∇)u - ν∆δu)τ(−ν∆v)
951  hessian_result[i][k][k] +=
952  tau * -this->kinematic_viscosity *
953  (gradient[i][l] * previous_values[l] +
954  previous_gradient[i][l] * value[l] -
955  this->kinematic_viscosity * hessian_diagonal[i][l]);
956  }
957 
958  // +(∇δp)τ(−ν∆v)
959  hessian_result[i][k][k] +=
960  tau * -this->kinematic_viscosity * (gradient[dim][i]);
961 
962  // +(∂t δu)τ(−ν∆v)
963  if (transient)
964  hessian_result[i][k][k] += tau *
965  -this->kinematic_viscosity *
966  ((*bdf_coefs)[0] * value[i]);
967 
968 
969  // LSIC term
970  // (∇·δu)τ'(∇·v)
971  gradient_result[i][i] += tau_lsic * gradient[k][k];
972  }
973  }
974  }
975 
976  integrator.submit_gradient(gradient_result, q);
977  integrator.submit_value(value_result, q);
978  integrator.submit_hessian(hessian_result, q);
979  }
980 
981  integrator.integrate(EvaluationFlags::values | EvaluationFlags::gradients |
982  EvaluationFlags::hessians);
983 }

986  * The expressions calculated in this cell integral are:
987  * (q, ∇·u) + (v,∂t u) + (v,(u·∇)u) - (∇·v,p) + ν(∇v,∇u) - (v,f) (Weak form),
988  * plus two additional terms in the case of SUPG-PSPG stabilization:
989  * \+ (∂t u +(u·∇)u + ∇p - ν∆u - f)τ∇·q (PSPG term)
990  * \+ (∂t u +(u·∇)u + ∇p - ν∆u - f)τu·∇v (SUPG term),
991  * plus two additional terms in the case of full gls stabilization:
992  * \+ (∂t u +(u·∇)u + ∇p - ν∆u - f)τ(−ν∆v) (GLS term)
993  * \+ (∇·u)τ'(∇·v) (LSIC term).
994  */

1002 {

1004

1006  {

1014  (is_bdf(this->simulation_control->get_assembly_method())) ? true :
1015  false;
1016

1018  const Vector<double> *bdf_coefs;
1019  if (transient)
1020  bdf_coefs = &this->simulation_control->get_bdf_coefficients();
1021

1023  {

1026  Point<dim, VectorizedArray<number>> point_batch =
1027  integrator.quadrature_point(q);

1040  typename FECellIntegrator::gradient_type hessian_diagonal =
1041  integrator.get_hessian_diagonal(q);
1042 
1043  // Time derivatives of previous solutions
1044  Tensor<1, dim + 1, VectorizedArray<number>> previous_time_derivatives;
1045  if (transient)
1046  previous_time_derivatives =
1047  this->time_derivatives_previous_solutions(cell, q);

1049  // Get stabilization parameter
1050  const auto tau = this->stabilization_parameter[cell][q];
1051  const auto tau_lsic = this->stabilization_parameter_lsic[cell][q];
1052 
1053  // Result value/gradient we will use
1054  typename FECellIntegrator::value_type value_result;
1055  typename FECellIntegrator::gradient_type gradient_result;

1069  if (transient)

1072
