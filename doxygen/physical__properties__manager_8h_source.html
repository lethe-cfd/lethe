58 /** @brief Class that manages the physical properties
59  * model which are required to calculate the various physical properties
60  * This centralizes the place where the models are created.
61  * The class can be constructed empty and initialized from parameters
62  * or it can be constructed directly with a Parameters section.
63  * The architecture of the PhysicalPropertiesManager is made to ensure that it
64  * can store as many physical properties models as required. It first stores
65  * the physical properties of the fluids, which are always assumed to lie in
66  * material_id=0, and then stores the physical properties of the solid, which
67  * may be in cells for which material_id>0.
68  *
69  */

98  provide_simulation_control(
99  std::shared_ptr<SimulationControl> &simulation_control)
100  {
101  // At the present moment, only SpecificHeatModel can be time-dependent
102  // Consequently we only pass the SimulationControl object to the
103  // SpecificHeatModel.
104  for (unsigned int f = 0; f < number_of_fluids; ++f)
105  specific_heat[f]->provide_simulation_control(simulation_control);
106  for (unsigned int s = 0; s < number_of_solids; ++s)
107  specific_heat[s]->provide_simulation_control(simulation_control);
108  }

307  get_material_interaction_id(
308  const material_interactions_type material_interaction_type,
309  const unsigned int material_0_id,
310  const unsigned int material_1_id) const
311  {
312  if (material_interaction_type == material_interactions_type::fluid_fluid)

319  }

324  material_0_id, material_1_id);

326  .find(fluid_solid_material_interaction)
327  ->second;
328  }
329  else
330  throw(std::runtime_error(
331  "Invalid type of material interaction. The choices are <fluid-fluid|fluid-solid>"));

347  /** @brief Calculates the global id of the physical property. By default. Lethe stores all
348  * the properties of the fluids, then the properties of the solid. Fluids
349  * need to have a material id of 0, then solids are consequential.
350  *
351  * @param fluid_id the id of the fluid (0 for single phase simulations, 0 or 1 for VOF simulations)
352  *
353  * @param material_id the material id of the cell. Cells with material_id=0 are fluid cells (0 or 1) and cells with material_id>0 are solid cells
354  */
355  unsigned int
356  calculate_global_id(const unsigned int fluid_id,
357  const unsigned int material_id) const
358  {
359  if (material_id < 1)
360  return fluid_id;
361  else
362  {
363  return number_of_fluids + material_id - 1;
364  }
365  }
