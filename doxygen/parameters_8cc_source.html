915  // Density is in M L^-3, rescale
916  density *= dimensions.density_scaling;
917  isothermal_ideal_gas_density_parameters.parse_parameters(prm, dimensions);

926  }

928  {
929  rheological_model = RheologicalModel::carreau;
930  }
931  else if (op == "newtonian")

933  rheological_model = RheologicalModel::newtonian;
934  }

936  {
937  rheological_model = RheologicalModel::phase_change;
938  }

944

947  //--------------

954

957 
958
