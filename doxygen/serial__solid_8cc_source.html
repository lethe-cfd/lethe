49  , angular_velocity(&param->angular_velocity)

86 template <int dim, int spacedim>
87 std::vector<
88  std::pair<typename Triangulation<spacedim>::active_cell_iterator,
89  typename Triangulation<dim, spacedim>::active_cell_iterator>>
90 SerialSolid<dim, spacedim>::map_solid_in_background_triangulation(
91  const parallel::TriangulationBase<spacedim> &background_tr)
92 {
93  std::vector<
94  std::pair<typename Triangulation<spacedim>::active_cell_iterator,
95  typename Triangulation<dim, spacedim>::active_cell_iterator>>
96  mapped_solid;
97 
98  // Gather the amount of vertices per cell in the solid cells once to have
99  // a static memory allocation of the vector. This prevents reallocating
100  // the vector dynamically using push_back if the solid triangulation
101  // has many triangles
102  auto temporary_solid_cell = solid_tria->begin();
103  std::vector<Point<spacedim>> triangle(temporary_solid_cell->n_vertices());
104 
105  // Calculate distance from cell center to solid_cell
106  for (const auto &background_cell : background_tr.active_cell_iterators())
107  {
108  // If the cell is owned by the processor
109  if (background_cell->is_locally_owned())
110  {
111  // Calculate the characteristic size of the background cell
112  const double bg_cell_length = background_cell->diameter();
113 
114  // Calculate the center of the cell
115  Point<spacedim> bg_cell_center = background_cell->center();
116 
117  // Calculate distance from center of the cell to triangle
118  for (auto &solid_cell : solid_tria->active_cell_iterators())
119  {
120  // Gather triangle vertices
121  for (unsigned int v = 0; v < solid_cell->n_vertices(); ++v)
122  {
123  triangle[v] = solid_cell->vertex(v);
124  }
125 
126  // Calculate distance between triangle and center
127  const double distance =
128  LetheGridTools::find_point_triangle_distance(triangle,
129  bg_cell_center);
130 
131  if (distance < bg_cell_length)
132  {
133  mapped_solid.push_back(
134  std::make_pair(background_cell, solid_cell));
135  }
136  }
137  }
138  }
139 
140  reset_displacement_monitoring();
141 
142  return mapped_solid;
143 }

159 template <int dim, int spacedim>
160 void
161 SerialSolid<dim, spacedim>::setup_triangulation(const bool restart)
162 {
163  if (param->solid_mesh.type == Parameters::Mesh::Type::gmsh)
164  {
165  // Grid creation
166  GridIn<dim, spacedim> grid_in;
167  // Attach triangulation
168  grid_in.attach_triangulation(*solid_tria);
169  // Read input gmsh file
170  std::ifstream input_file(param->solid_mesh.file_name);
171 
172  grid_in.read_msh(input_file);
173  }
174  else if (param->solid_mesh.type == Parameters::Mesh::Type::dealii)
175  {
176  if (param->solid_mesh.simplex)
177  {
178  Triangulation<dim, spacedim> temporary_quad_triangulation;
179  GridGenerator::generate_from_name_and_arguments(
180  temporary_quad_triangulation,
181  param->solid_mesh.grid_type,
182  param->solid_mesh.grid_arguments);
183 
184  // initial refinement
185  const int initial_refinement = param->solid_mesh.initial_refinement;
186  temporary_quad_triangulation.refine_global(initial_refinement);
187  // flatten the triangulation
188  Triangulation<dim, spacedim> flat_temp_quad_triangulation;
189  GridGenerator::flatten_triangulation(temporary_quad_triangulation,
190  flat_temp_quad_triangulation);
191 
192  GridGenerator::convert_hypercube_to_simplex_mesh(
193  flat_temp_quad_triangulation, *solid_tria);
194  }
195  else
196  { // deal.ii creates grid and attaches the solid triangulation
197  GridGenerator::generate_from_name_and_arguments(
198  *solid_tria,
199  param->solid_mesh.grid_type,
200  param->solid_mesh.grid_arguments);
201  }
202  }
203  else
204  throw std::runtime_error(
205 
206  "Unsupported mesh type - solid mesh will not be created");
207 
208  // Rotate the triangulation
209  rotate_grid(param->solid_mesh.rotation_angle,
210  param->solid_mesh.rotation_axis);
211 
212  // Translate the triangulation
213  translate_grid(param->solid_mesh.translation);
214 
215  // Refine the solid triangulation to its initial size
216  // NB: solid_tria should not be refined if loaded from a restart file
217  // afterwards
218  if (!restart)
219  {
220  if (param->solid_mesh.simplex)
221  {
222  // Simplex triangulation refinement isn't possible yet
223  }
224  else
225  {
226  solid_tria->refine_global(param->solid_mesh.initial_refinement);
227  }
228  }
229 }

316 template <int dim, int spacedim>
317 void
318 SerialSolid<dim, spacedim>::move_solid_triangulation(const double time_step,
319  const double initial_time)
320 {
321  // First we calculate the displacement that the solid triangulation
322  // will feel during this iteration.
323  // Then, we apply this displacement to the triangulation
324  const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
325  const std::vector<bool> locally_owned_vertices =
326  GridTools::get_locally_owned_vertices(*this->solid_tria);
327  std::vector<bool> vertex_moved(this->solid_tria->n_vertices(), false);
328  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
329 
330  translational_velocity->set_time(initial_time);
331  angular_velocity->set_time(initial_time);
332 
333  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
334  {
335  current_translational_velocity[comp_i] =
336  translational_velocity->value(center_of_rotation, comp_i);
337  current_angular_velocity[comp_i] =
338  angular_velocity->value(center_of_rotation, comp_i);
339  }
340 
341  for (const auto &cell : displacement_dh.active_cell_iterators())
342  {
343  if (cell->is_locally_owned())
344  {
345  for (unsigned int vertex = 0;
346  vertex < cell->reference_cell().n_vertices();
347  ++vertex)
348  {
349  const auto dof_index = cell->vertex_dof_index(vertex, 0);
350  Point<spacedim> &vertex_position = cell->vertex(vertex);
351  const unsigned global_vertex_no = cell->vertex_index(vertex);
352  if (vertex_moved[global_vertex_no] ||
353  !locally_owned_vertices[global_vertex_no])
354  continue;
355 
356  Tensor<1, spacedim> distance_vector =
357  vertex_position - center_of_rotation;
358 
359  Tensor<1, spacedim> local_velocity =
360  current_translational_velocity;
361  if constexpr (spacedim == 2)
362  {
363  local_velocity[0] +=
364  -distance_vector[1] * current_angular_velocity[2];
365  local_velocity[1] +=
366  distance_vector[0] * current_angular_velocity[2];
367  }
368  if constexpr (spacedim == 3)
369  {
370  local_velocity +=
371  cross_product_3d(current_angular_velocity, distance_vector);
372  }
373 
374  auto vertex_displacement = time_step * local_velocity;
375 
376  vertex_position += vertex_displacement;
377 
378  for (unsigned d = 0; d < spacedim; ++d)
379  {
380  displacement[dof_index + d] =
381  displacement[dof_index + d] + vertex_displacement[d];
382  displacement_since_mapped[dof_index + d] +=
383  vertex_displacement[d];
384  }
385 
386  vertex_moved[global_vertex_no] = true;
387  }
388  }
389  }
390 
391  center_of_rotation += current_translational_velocity * time_step;
392 }

395 template <int dim, int spacedim>
396 void
397 SerialSolid<dim, spacedim>::displace_solid_triangulation()
398 {
399  const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
400  std::set<unsigned int> dof_vertex_displaced;
401  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
402 
403  for (const auto &cell : displacement_dh.active_cell_iterators())
404  {
405  if (cell->is_locally_owned())
406  {
407  for (unsigned int vertex = 0;
408  vertex < cell->reference_cell().n_vertices();
409  ++vertex)
410  {
411  if (!dof_vertex_displaced.count(cell->vertex_index(vertex)))
412  {
413  const auto dof_index = cell->vertex_dof_index(vertex, 0);
414  Point<spacedim> &vertex_position = cell->vertex(vertex);
415 
416  for (unsigned d = 0; d < spacedim; ++d)
417  {
418  vertex_position[d] =
419  vertex_position[d] + displacement[dof_index + d];
420  }
421 
422  dof_vertex_displaced.insert(cell->vertex_index(vertex));
423  }
424  }
425  }
426  }
427 }

437  std::set<unsigned int> dof_vertex_displaced;

442  {
443  for (unsigned int vertex = 0;
444  vertex < cell->reference_cell().n_vertices();
445  ++vertex)
