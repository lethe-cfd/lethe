32  * @brief Calculate the coefficients required for backward differentiation

34  *
35  * The coefficients are determined through a recursion algorithm.
36  * The algorithm is taken from: Hay, Alexander, et al. "hp-Adaptive time
37  * integration based on the BDF for viscous flows." Journal of Computational
38  * Physics 291 (2015): 151-176.
39  *
40  * @param[in] order Order of the BDF method. The BDF method of @p order
41  * \f$n\f$ requires \f$n+1\f$ coefficients.

159 /**
160  * @brief Extrapolate vector of solution to time_vector[0] using previous
161  * solution times and previous solution.
162  *
163  * @tparam DataType Type of the variable being extrapolated (e.g. double,
164  * Tensor<1,dim>)
165  *
166  * @param[in] time_vector Vector of times. The solution will be extrapolated to
167  * time 0. It should be of size @p number_of_previous_solutions+1.
168  *
169  * @param[in] solution_vector Vector of solutions. The solution will be
170  * extrapolated to index 0. It should be at least of size @p
171  * number_of_previous_solutions.
172  *
173  * @param[in] number_of_previous_solutions Number of previous solutions used
174  * to extrapolate.
175  *
176  * @param[out] extrapolated_solution Vector of extrapolated solutions.
177  */
178 
179 template <typename DataType>
180 inline void
181 bdf_extrapolate(const std::vector<double> &time_vector,
182  const std::vector<std::vector<DataType>> &solution_vector,
183  const unsigned int number_of_previous_solutions,
184  std::vector<DataType> &extrapolated_solution)
185 {
186  // If only one previous solution is used, we don't make a Lagrange
187  // polynomial but use the previous value
188  if (number_of_previous_solutions == 1)
189  {
190  for (unsigned int q = 0; q < extrapolated_solution.size(); ++q)
191  extrapolated_solution[q] = solution_vector[0][q];
192  return;
193  }
194 
195  // Otherwise we extrapolate with a Lagrange polynomial
196  for (unsigned int q = 0; q < extrapolated_solution.size(); ++q)
197  {
198  // Set extrapolated solution to zero
199  extrapolated_solution[q] = 0;
200 
201  // For all the previous solutions which we will use to extrapolate
202  for (unsigned int p = 0; p < number_of_previous_solutions; ++p)
203  {
204  // Factor is the weight of the previous solution fixed by the Lagrange
205  // Polynomial
206  double factor = 1;
207  for (unsigned int k = 0; k < number_of_previous_solutions; ++k)
208  {
209  if (p != k)
210  {
211  // The time vector also contains the time of the solution to
212  // be extrapoled to, hence the +1
213  factor *= (time_vector[0] - time_vector[k + 1]) /
214  (time_vector[p + 1] - time_vector[k + 1]);
215  }
216  }
217  // Calculate the contribution of this time step to the extrapolated
218  // solution
219  extrapolated_solution[q] += solution_vector[p][q] * factor;
220  }
221  }
222 }
