20 PhysicalPropertiesManager::establish_fields_required_by_model(
21  PhysicalPropertyModel &model)
22 {
23  // Loop through the map. The use of or (||) is there to ensure
24  // that if a field is already required, it won't be erased.
25  for (auto &f : required_fields)
26  {
27  f.second = f.second || model.depends_on(f.first);
28  }

43 void
44 PhysicalPropertiesManager::initialize(
45  Parameters::PhysicalProperties physical_properties)
46 {
47  // Keep a copy of the physical properties used to build the physical property
48  // manager
49  physical_properties_parameters = physical_properties;
50 
51  is_initialized = true;
52 
53  number_of_fluids = physical_properties.number_of_fluids;
54  number_of_solids = physical_properties.number_of_solids;
55  number_of_material_interactions =
56  physical_properties.number_of_material_interactions;
57  fluid_fluid_interactions_with_material_interaction_ids =
58  physical_properties.fluid_fluid_interactions_with_material_interaction_ids;
59  fluid_solid_interactions_with_material_interaction_ids =
60  physical_properties.fluid_solid_interactions_with_material_interaction_ids;
61  reference_temperature = physical_properties.reference_temperature;
62 
63  non_newtonian_flow = false;
64  phase_change = false;
65  constant_density = true;
66  constant_surface_tension = true;
67 
68  required_fields[field::temperature] = false;
69  required_fields[field::temperature_p1] = false;
70  required_fields[field::temperature_p2] = false;
71  required_fields[field::shear_rate] = false;
72  required_fields[field::pressure] = false;
73  required_fields[field::phase_order_cahn_hilliard] = false;
74 
75  // For each fluid, declare the physical properties
76  for (unsigned int f = 0; f < number_of_fluids; ++f)
77  {
78  density.push_back(
79  DensityModel::model_cast(physical_properties.fluids[f]));
80  establish_fields_required_by_model(*density[f]);
81 
82  // Store an indicator for the density to indicate if it is not constant
83  // This indicator is used elsewhere in the code to throw assertions
84  // if non-constant density is not implemented in a post-processing utility
85  if (!density.back()->is_constant_density_model())
86  constant_density = false;
87 
88  specific_heat.push_back(
89  SpecificHeatModel::model_cast(physical_properties.fluids[f]));
90  establish_fields_required_by_model(*specific_heat[f]);
91 
92  // Store an indicator that a phase change model is present
93  phase_change =
94  phase_change || physical_properties.fluids[f].specific_heat_model ==
95  Parameters::Material::SpecificHeatModel::phase_change;
96 
97  thermal_conductivity.push_back(
98  ThermalConductivityModel::model_cast(physical_properties.fluids[f]));
99  establish_fields_required_by_model(*thermal_conductivity[f]);
100 
101  rheology.push_back(
102  RheologicalModel::model_cast(physical_properties.fluids[f]));
103  this->establish_fields_required_by_model(*rheology[f]);
104  rheology[f]->set_dynamic_viscosity(density[f]->get_density_ref());
105 
106  tracer_diffusivity.push_back(
107  TracerDiffusivityModel::model_cast(physical_properties.fluids[f]));
108  establish_fields_required_by_model(*tracer_diffusivity[f]);
109 
110  thermal_expansion.push_back(
111  ThermalExpansionModel::model_cast(physical_properties.fluids[f]));
112  establish_fields_required_by_model(*thermal_expansion[f]);
113 
114  phase_change_parameters.push_back(
115  physical_properties.fluids[f].phase_change_parameters);
116 
117  if (rheology.back()->is_non_newtonian_rheological_model())
118  non_newtonian_flow = true;
119  }
120 
121  // For each solid, append the physical properties
122  for (unsigned int s = 0; s < number_of_solids; ++s)
123  {
124  density.push_back(
125  DensityModel::model_cast(physical_properties.solids[s]));
126  establish_fields_required_by_model(*density[s]);
127 
128  // Store an indicator for the density to indicate if it is not constant
129  // This indicator is used elsewhere in the code to throw assertions
130  // if non-constant density is not implemented in a post-processing utility
131  if (!density.back()->is_constant_density_model())
132  constant_density = false;
133 
134  specific_heat.push_back(
135  SpecificHeatModel::model_cast(physical_properties.solids[s]));
136  establish_fields_required_by_model(*specific_heat[s]);
137 
138  thermal_conductivity.push_back(
139  ThermalConductivityModel::model_cast(physical_properties.solids[s]));
140  establish_fields_required_by_model(*thermal_conductivity[s]);
141 
142  rheology.push_back(
143  RheologicalModel::model_cast(physical_properties.solids[s]));
144  this->establish_fields_required_by_model(*rheology[s]);
145  rheology.back()->set_dynamic_viscosity(density.back()->get_density_ref());
146 
147  tracer_diffusivity.push_back(
148  TracerDiffusivityModel::model_cast(physical_properties.solids[s]));
149  establish_fields_required_by_model(*tracer_diffusivity[s]);
150 
151  thermal_expansion.push_back(
152  ThermalExpansionModel::model_cast(physical_properties.solids[s]));
153  establish_fields_required_by_model(*thermal_expansion[s]);
154  }
155 
156  // For each pair of interacting materials (fluid-fluid or fluid-solid), append
157  // physical properties
158  for (unsigned int i = 0; i < number_of_material_interactions; ++i)
159  {
160  surface_tension.push_back(SurfaceTensionModel::model_cast(
161  physical_properties.material_interactions[i]));
162  establish_fields_required_by_model(*surface_tension[i]);
163  if (!surface_tension.back()->is_constant_surface_tension_model())
164  constant_surface_tension = false;
165 
166  mobility_cahn_hilliard.push_back(MobilityCahnHilliardModel::model_cast(
167  physical_properties.material_interactions[i]));
168  establish_fields_required_by_model(*mobility_cahn_hilliard[i]);
169  }
170 }
