43 /**
44  * @brief Class that stores the information required by the assembly procedure
45  * for the Cahn-Hilliard equations. Consequently, this class
46  * calculates the phase field parameter Phi (values, gradients, laplacians),
47  * the chemical potential eta (values, gradients, laplacians) and the shape
48  * function (values, gradients, laplacians) at all the Gauss points for all
49  * degrees of freedom and stores it into arrays. Additionally, the user can
50  * request that this class gathers additional fields for physics which are
51  * coupled to the Cahn-Hilliard equations, such as the velocity which is
52  * required. This class serves as a separation between the evaluation at the
53  * Gauss point of the variables of interest and their use in the assembly, which
54  * is carried out by the assembler functions. For more information on this
55  * design, the reader can consult deal.II step-9
56  * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
57  * example, the scratch is a struct instead of a templated class because of the
58  * simplicity of step-9.
59  *
60  * @tparam dim An integer that denotes the dimension of the space in which
61  * the flow is solved
62  *
63  * @ingroup solvers
64  **/
65

155  /** @brief Reinitialize the content of the scratch
156  *
157  * Using the FeValues and the content of the solutions and previous solutions
158  * , fills all of the class member of the scratch
159  *
160  * @param cell The cell over which the assembly is being carried.
161  * This cell must be compatible with the fe which is used to fill the FeValues
162  *
163  * @param current_solution The present value of the solution for [Phi,eta]
164  *
165  * @param previous_solutions The solutions at the previous time steps
166  *
167  * @param source_function The function describing the source term in Cahn-Hilliard
168  * equations
169  *
170  */
171 
172  template <typename VectorType>
173  void
174  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
175  const VectorType &current_solution,
176  const std::vector<VectorType> &previous_solutions,
177  Function<dim> *source_function)
178  {
179  this->phase_order.component = 0;
180  this->chemical_potential.component = 1;
181 
182  this->fe_values_cahn_hilliard.reinit(cell);
183 
184  quadrature_points = this->fe_values_cahn_hilliard.get_quadrature_points();
185  auto &fe_cahn_hilliard = this->fe_values_cahn_hilliard.get_fe();
186 
187  source_function->value_list(quadrature_points, source_phase_order, 0);
188  source_function->value_list(quadrature_points,
189  source_chemical_potential,
190  1);
191 
192 
193  if (dim == 2)
194  this->cell_size =
195  std::sqrt(4. * cell->measure() / M_PI) / fe_cahn_hilliard.degree;
196  else if (dim == 3)
197  this->cell_size =
198  pow(6 * cell->measure() / M_PI, 1. / 3.) / fe_cahn_hilliard.degree;
199 
200  // Gather Phi and eta (values, gradient and laplacian)
201  this->fe_values_cahn_hilliard[phase_order].get_function_values(
202  current_solution, this->phase_order_values);
203  this->fe_values_cahn_hilliard[phase_order].get_function_gradients(
204  current_solution, this->phase_order_gradients);
205  this->fe_values_cahn_hilliard[phase_order].get_function_laplacians(
206  current_solution, this->phase_order_laplacians);
207 
208  this->fe_values_cahn_hilliard[chemical_potential].get_function_values(
209  current_solution, this->chemical_potential_values);
210  this->fe_values_cahn_hilliard[chemical_potential].get_function_gradients(
211  current_solution, this->chemical_potential_gradients);
212  this->fe_values_cahn_hilliard[chemical_potential].get_function_laplacians(
213  current_solution, this->chemical_potential_laplacians);
214 
215 
216  // Gather previous phase order values
217  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
218  {
219  this->fe_values_cahn_hilliard[phase_order].get_function_values(
220  previous_solutions[p], previous_phase_order_values[p]);
221  }
222 
223  // Gather previous chemical potential values
224  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
225  {
226  this->fe_values_cahn_hilliard[chemical_potential].get_function_values(
227  previous_solutions[p], previous_chemical_potential_values[p]);
228  }
229 
230  for (unsigned int q = 0; q < n_q_points; ++q)
231  {
232  this->JxW[q] = this->fe_values_cahn_hilliard.JxW(q);
233 
234  for (unsigned int k = 0; k < n_dofs; ++k)
235  {
236  // Shape functions for the phase order
237  this->phi_phase[q][k] =
238  this->fe_values_cahn_hilliard[phase_order].value(k, q);
239  this->grad_phi_phase[q][k] =
240  this->fe_values_cahn_hilliard[phase_order].gradient(k, q);
241  this->hess_phi_phase[q][k] =
242  this->fe_values_cahn_hilliard[phase_order].hessian(k, q);
243  this->laplacian_phi_phase[q][k] = trace(this->hess_phi_phase[q][k]);
244 
245  // Shape functions for the chemical potential
246  this->phi_potential[q][k] =
247  this->fe_values_cahn_hilliard[chemical_potential].value(k, q);
248  this->grad_phi_potential[q][k] =
249  this->fe_values_cahn_hilliard[chemical_potential].gradient(k, q);
250  this->hess_phi_potential[q][k] =
251  this->fe_values_cahn_hilliard[chemical_potential].hessian(k, q);
252  this->laplacian_phi_potential[q][k] =
253  trace(this->hess_phi_potential[q][k]);
254  }
255  }
256 
257  this->is_boundary_cell =
258  cell->at_boundary(); // The attribute needs to be updated because the
259  // assembler for the angle of contact boundary
260  // condition needs to know if the cell is at the
261  // boundary
262  if (this->is_boundary_cell)
263  {
264  n_faces = cell->n_faces();
265  is_boundary_face = std::vector<bool>(n_faces, false);
266  n_faces_q_points = fe_face_values_cahn_hilliard.get_quadrature().size();
267  boundary_face_id = std::vector<unsigned int>(n_faces);
268 
269  face_JxW = std::vector<std::vector<double>>(
270  n_faces, std::vector<double>(n_faces_q_points));
271 
272  this->grad_phi_face_phase =
273  std::vector<std::vector<std::vector<Tensor<1, dim>>>>(
274  n_faces,
275  std::vector<std::vector<Tensor<1, dim>>>(
276  n_faces_q_points, std::vector<Tensor<1, dim>>(n_dofs)));
277 
278  this->face_phase_grad_values = std::vector<std::vector<Tensor<1, dim>>>(
279  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
280 
281  this->face_normal = std::vector<std::vector<Tensor<1, dim>>>(
282  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
283 
284  for (const auto face : cell->face_indices())
285  {
286  this->is_boundary_face[face] = cell->face(face)->at_boundary();
287  if (this->is_boundary_face[face])
288  {
289  fe_face_values_cahn_hilliard.reinit(cell, face);
290  boundary_face_id[face] = cell->face(face)->boundary_id();
291 
292  this->fe_face_values_cahn_hilliard[phase_order]
293  .get_function_gradients(current_solution,
294  this->face_phase_grad_values[face]);
295 
296  for (unsigned int q = 0; q < n_faces_q_points; ++q)
297  {
298  face_JxW[face][q] = fe_face_values_cahn_hilliard.JxW(q);
299  this->face_normal[face][q] =
300  this->fe_face_values_cahn_hilliard.normal_vector(q);
301  for (const unsigned int k :
302  fe_face_values_cahn_hilliard.dof_indices())
303  {
304  this->grad_phi_face_phase[face][q][k] =
305  this->fe_face_values_cahn_hilliard[phase_order]
306  .gradient(k, q);
307  }
308  }
309  }
310  }
311  }
312  }

315  template <typename VectorType>
316  void
317  reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
318  const VectorType &current_solution,
319  const Parameters::ALE<dim> &ale)
320  {
321  FEValuesExtractors::Vector velocities(0);
322  this->fe_values_fd.reinit(cell);
323 
324  this->fe_values_fd[velocities].get_function_values(current_solution,
325  velocity_values);
326 
327  if (!ale.enabled())
328  return;
329 
330 
331  // ALE enabled, so extract the ALE velocity and subtract it from the
332  // velocity obtained from the fluid dynamics
333  Tensor<1, dim> velocity_ale;
334  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
335  ale.velocity;
336  Vector<double> velocity_ale_vector(dim);
337 
338  for (unsigned int q = 0; q < n_q_points; ++q)
339  {
340  velocity_ale_function->vector_value(quadrature_points[q],
341  velocity_ale_vector);
342  for (unsigned int d = 0; d < dim; ++d)
343  velocity_ale[d] = velocity_ale_vector[d];
344 
345  velocity_values[q] -= velocity_ale;
346  }
347  }
