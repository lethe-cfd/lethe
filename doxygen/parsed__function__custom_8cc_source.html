81 void
82 ParsedFunctionCustom<n_components>::initialize()
83 {
84  if (!initialized)
85  throw ExcMessage(
86  "The required fields for muParser are not initialized yet. "
87  "This function should be called only after that is done.");
88  std::vector<std::string> const_list =
89  Utilities::split_string_list(constants_list, ',');
90  std::map<std::string, double> constants;
91  for (const auto &constant : const_list)
92  {
93  std::vector<std::string> this_c =
94  Utilities::split_string_list(constant, '=');
95  AssertThrow(this_c.size() == 2,
96  ExcMessage("The list of constants, <" + constants_list +
97  ">, is not a comma-separated list of "
98  "entries of the form 'name=value'."));
99  constants[this_c[0]] = Utilities::string_to_double(this_c[1]);
100  }
101  // set pi and Pi as synonyms for the corresponding value. note that
102  // this overrides any value a user may have given
103  constants["pi"] = numbers::PI;
104  constants["Pi"] = numbers::PI;
105 
106  const unsigned int nn = (Utilities::split_string_list(vnames)).size();
107  switch (nn)
108  {
109  case n_components:
110  // Time independent function
111  this->::internalFunctionParserCustom::ParserImplementation<
112  n_components>::initialize(vnames,
113  Utilities::split_string_list(expression,
114  ';'),
115  constants);
116  break;
117  case n_components + 1:
118  // Time dependent function
119  this->::internalFunctionParserCustom::ParserImplementation<
120  n_components>::initialize(vnames,
121  Utilities::split_string_list(expression,
122  ';'),
123  constants,
124  true);
125  break;
126  default:
127  AssertThrow(false,
128  ExcMessage(
129  "The list of variables specified is <" + vnames +
130  "> which is a list of length " +
131  Utilities::int_to_string(nn) +
132  " but it has to be a list of length equal to" +
133  " either n_components (for a time-independent function)" +
134  " or n_components+1 (for a time-dependent function)."));
135  }
136 }

180 template <int n_components>
181 Tensor<1, n_components>
182 ParsedFunctionCustom<n_components>::gradient(const Tensor<1, n_components> &p,
183  unsigned int comp) const
184 {
185  Tensor<1, n_components> grad;
186  // FourthOrder:
187  Tensor<1, n_components> q1, q2, q3, q4;
188  for (unsigned int i = 0; i < n_components; ++i)
189  {
190  q2 = p + ht[i];
191  q1 = q2 + ht[i];
192  q3 = p - ht[i];
193  q4 = q3 - ht[i];
194  grad[i] = (-this->value(q1, comp) + 8 * this->value(q2, comp) -
195  8 * this->value(q3, comp) + this->value(q4, comp)) /
196  (12 * h);
197  }
198  return grad;
199 }

202 void
203 ParsedFunctionCustom<n_components>::vector_gradient(
204  const Tensor<1, n_components> &p,
205  Tensor<2, n_components> &gradients) const
206 {
207  Point<n_components> q1, q2, q3, q4;
208  Tensor<1, n_components> v1{}, v2{}, v3{}, v4{};
209  const double h_inv_12 = 1. / (12 * h);
210  for (unsigned int i = 0; i < n_components; ++i)
211  {
212  q2 = p + ht[i];
213  q1 = q2 + ht[i];
214  q3 = p - ht[i];
215  q4 = q3 - ht[i];
216  this->vector_value(q1, v1);
217  this->vector_value(q2, v2);
218  this->vector_value(q3, v3);
219  this->vector_value(q4, v4);
220 
221  for (unsigned int comp = 0; comp < n_components; ++comp)
222  gradients[comp][i] =
223  (-v1[comp] + 8 * v2[comp] - 8 * v3[comp] + v4[comp]) * h_inv_12;
224  }
225 }
