433 PhaseChangeDarcyVOFAssembler<dim>::assemble_matrix(
434  NavierStokesScratchData<dim> &scratch_data,
435  StabilizedMethodsTensorCopyData<dim> &copy_data)
436 {
437  // Loop and quadrature information
438  const auto &JxW_vec = scratch_data.JxW;
439  const unsigned int n_q_points = scratch_data.n_q_points;
440  const unsigned int n_dofs = scratch_data.n_dofs;
441  const auto &velocities = scratch_data.velocity_values;
442  const auto &temperatures = scratch_data.temperature_values;
443  const auto &filtered_phase = scratch_data.filtered_phase_values;
444 
445  auto &local_matrix = copy_data.local_matrix;
446  auto &strong_residual = copy_data.strong_residual;
447  auto &strong_jacobian = copy_data.strong_jacobian;
448 
449  const unsigned int number_of_fluids = phase_change_parameters_vector.size();
450  std::vector<double> liquid_fractions;
451  std::vector<double> darcy_penalties;
452  liquid_fractions.reserve(number_of_fluids);
453  darcy_penalties.reserve(number_of_fluids);
454 
455  // Loop over the quadrature points
456  for (unsigned int q = 0; q < n_q_points; ++q)
457  {
458  // Store JxW in local variable for faster access;
459  const double JxW = JxW_vec[q];
460  // Current temperature values
461  double current_temperature = temperatures[q];
462  // Loop to calculate the liquid fraction and Darcy permeability of each
463  // fluid. Calculated Darcy permeability coefficients depend on the
464  // temperature and material (fluid) properties. A min(max) approach is
465  // used to calculate the liquid fractions avoiding if conditions.
466  for (unsigned int p = 0; p < number_of_fluids; p++)
467  {
468  liquid_fractions.emplace_back(
469  (std::min(1.,
470  std::max((current_temperature -
471  phase_change_parameters_vector[p].T_solidus) /
472  (phase_change_parameters_vector[p].T_liquidus -
473  phase_change_parameters_vector[p].T_solidus),
474  0.))));
475  darcy_penalties.emplace_back(
476  phase_change_parameters_vector[p].penalty_l * liquid_fractions[p] +
477  (1. - liquid_fractions[p]) *
478  phase_change_parameters_vector[p].penalty_s);
479  }
480 
481  // For a VOF two-fluid system, the global Darcy penalty coefficient
482  // takes into account the phase change parameters in both fluids
483  const double darcy_penalty =
484  ((1 - filtered_phase[q]) * darcy_penalties[0] +
485  filtered_phase[q] * darcy_penalties[1]);
486 
487  strong_residual[q] += darcy_penalty * velocities[q];
488 
489  for (unsigned int j = 0; j < n_dofs; ++j)
490  {
491  strong_jacobian[q][j] += darcy_penalty * scratch_data.phi_u[q][j];
492  }
493 
494  for (unsigned int i = 0; i < n_dofs; ++i)
495  {
496  const auto &phi_u_i = scratch_data.phi_u[q][i];
497  for (unsigned int j = 0; j < n_dofs; ++j)
498  {
499  const auto &phi_u_j = scratch_data.phi_u[q][j];
500  local_matrix(i, j) += darcy_penalty * phi_u_i * phi_u_j * JxW;
501  }
502  }
503  liquid_fractions.clear();
504  darcy_penalties.clear();
505  }

570  // Laplacian on the velocity terms
571  local_rhs(i) -= darcy_penalty * velocities[q] * phi_u_i * JxW;

679  {

682  // Surface tension gradient with respect to the temperature

730  local_rhs(i) += local_rhs_i * JxW_value;

815 template <int dim>
816 void
817 GLSNavierStokesVOFAssemblerNonNewtonianCore<dim>::assemble_matrix(
818  NavierStokesScratchData<dim> &scratch_data,
819  StabilizedMethodsTensorCopyData<dim> &copy_data)
820 {
821  // Loop and quadrature information
822  const auto &JxW_vec = scratch_data.JxW;
823  const unsigned int n_q_points = scratch_data.n_q_points;
824  const unsigned int n_dofs = scratch_data.n_dofs;
825  const double h = scratch_data.cell_size;
826 
827  // Copy data elements
828  auto &strong_residual_vec = copy_data.strong_residual;
829  auto &strong_jacobian_vec = copy_data.strong_jacobian;
830  auto &local_matrix = copy_data.local_matrix;
831 
832  // Time steps and inverse time steps which is used for stabilization constant
833  std::vector<double> time_steps_vector =
834  this->simulation_control->get_time_steps_vector();
835  const double dt = time_steps_vector[0];
836  const double sdt = 1. / dt;
837 
838  Assert(
839  scratch_data.properties_manager.density_is_constant(),
840  RequiresConstantDensity(
841  "GLSNavierStokesVOFAssemblerNonNewtonianCore<dim>::assemble_matrix"));
842 
843  // Loop over the quadrature points
844  for (unsigned int q = 0; q < n_q_points; ++q)
845  {
846  // Gather into local variables the relevant fields
847  const Tensor<1, dim> &velocity = scratch_data.velocity_values[q];
848  const Tensor<2, dim> &velocity_gradient =
849  scratch_data.velocity_gradients[q];
850  const Tensor<1, dim> &velocity_laplacian =
851  scratch_data.velocity_laplacians[q];
852  const Tensor<3, dim> &velocity_hessian =
853  scratch_data.velocity_hessians[q];
854  const Tensor<1, dim> &pressure_gradient =
855  scratch_data.pressure_gradients[q];
856 
857  // Calculate shear rate (at each q)
858  const Tensor<2, dim> shear_rate =
859  velocity_gradient + transpose(velocity_gradient);
860 
861  // Calculate the shear rate magnitude
862  double shear_rate_magnitude = calculate_shear_rate_magnitude(shear_rate);
863  // Set the shear rate magnitude to 1e-12 if it is too close to zero,
864  // since the viscosity gradient is undefined for shear_rate_magnitude = 0
865  shear_rate_magnitude =
866  shear_rate_magnitude > 1e-12 ? shear_rate_magnitude : 1e-12;
867 
868  // Calculate kinematic viscosity gradient
869  const Tensor<1, dim> kinematic_viscosity_gradient =
870  this->get_kinematic_viscosity_gradient(
871  velocity_gradient,
872  velocity_hessian,
873  shear_rate_magnitude,
874  scratch_data.grad_kinematic_viscosity_shear_rate[q]);
875 
876  // Forcing term
877  Tensor<1, dim> force = scratch_data.force[q];
878 
879  // Calculation of the magnitude of the velocity for the
880  // stabilization parameter
881  const double u_mag = std::max(velocity.norm(), 1e-12);
882 
883  // Store JxW in local variable for faster access;
884  const double JxW = JxW_vec[q];
885 
886  // Calculation of the equivalent properties at the quadrature point
887  double density_eq = scratch_data.density[q];
888  const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
889  const Tensor<1, dim> dynamic_viscosity_gradient =
890  density_eq * kinematic_viscosity_gradient;
891 
892  // Calculation of the GLS stabilization parameter. The
893  // stabilization parameter used is different if the simulation
894  // is steady or unsteady. In the unsteady case it includes the
895  // value of the time-step
896  const double tau =
897  this->simulation_control->get_assembly_method() ==
898  Parameters::SimulationControl::TimeSteppingMethod::steady ?
899  calculate_navier_stokes_gls_tau_steady(
900  u_mag, dynamic_viscosity_eq / density_eq, h) :
901  calculate_navier_stokes_gls_tau_transient(
902  u_mag, dynamic_viscosity_eq / density_eq, h, sdt);
903 
904  // Calculate the strong residual for GLS stabilization
905  auto strong_residual = density_eq * velocity_gradient * velocity +
906  pressure_gradient -
907  shear_rate * dynamic_viscosity_gradient -
908  dynamic_viscosity_eq * velocity_laplacian -
909  density_eq * force + strong_residual_vec[q];
910 
911  std::vector<Tensor<1, dim>> grad_phi_u_j_x_velocity(n_dofs);
912  std::vector<Tensor<1, dim>> velocity_gradient_x_phi_u_j(n_dofs);
913 
914  // Pressure scaling factor
915  const double pressure_scaling_factor =
916  scratch_data.pressure_scaling_factor;
917 
918  // We loop over the column first to prevent recalculation
919  // of the strong jacobian in the inner loop
920  for (unsigned int j = 0; j < n_dofs; ++j)
921  {
922  const auto &phi_u_j = scratch_data.phi_u[q][j];
923  const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
924  const auto &laplacian_phi_u_j = scratch_data.laplacian_phi_u[q][j];
925 
926  const auto &grad_phi_p_j =
927  scratch_data.grad_phi_p[q][j] * pressure_scaling_factor;
928 
929  const auto &grad_phi_u_j_non_newtonian =
930  grad_phi_u_j + transpose(grad_phi_u_j);
931 
932  strong_jacobian_vec[q][j] +=
933  density_eq * velocity_gradient * phi_u_j +
934  density_eq * grad_phi_u_j * velocity + grad_phi_p_j -
935  dynamic_viscosity_eq * laplacian_phi_u_j -
936  grad_phi_u_j_non_newtonian * dynamic_viscosity_gradient;
937 
938  // Store these temporary products in auxiliary variables for speed
939  grad_phi_u_j_x_velocity[j] = grad_phi_u_j * velocity;
940  velocity_gradient_x_phi_u_j[j] = velocity_gradient * phi_u_j;
941  }
942 
943  shear_rate_magnitude =
944  shear_rate_magnitude > 1e-3 ? shear_rate_magnitude : 1e-3;
945 
946  for (unsigned int i = 0; i < n_dofs; ++i)
947  {
948  const auto &phi_u_i = scratch_data.phi_u[q][i];
949  const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
950  const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
951  const auto &phi_p_i = scratch_data.phi_p[q][i];
952  const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
953 
954  // Store these temporary products in auxiliary variables for speed
955  const auto grad_phi_u_i_x_velocity = grad_phi_u_i * velocity;
956  const auto strong_residual_x_grad_phi_u_i =
957  strong_residual * grad_phi_u_i;
958 
959  for (unsigned int j = 0; j < n_dofs; ++j)
960  {
961  const auto &phi_u_j = scratch_data.phi_u[q][j];
962  const auto &grad_phi_u_j = scratch_data.grad_phi_u[q][j];
963  const auto &div_phi_u_j = scratch_data.div_phi_u[q][j];
964 
965  const auto &grad_phi_u_j_non_newtonian =
966  grad_phi_u_j + transpose(grad_phi_u_j);
967 
968  const auto &phi_p_j =
969  scratch_data.phi_p[q][j] * pressure_scaling_factor;
970 
971  const auto &strong_jac = strong_jacobian_vec[q][j];
972 
973  double local_matrix_ij =
974  dynamic_viscosity_eq *
975  scalar_product(grad_phi_u_j_non_newtonian, grad_phi_u_i) +
976  0.5 * scratch_data.grad_kinematic_viscosity_shear_rate[q] /
977  shear_rate_magnitude *
978  scalar_product(grad_phi_u_j_non_newtonian, shear_rate) *
979  scalar_product(shear_rate, grad_phi_u_i) +
980  density_eq * velocity_gradient_x_phi_u_j[j] * 0.5 * phi_u_i +
981  density_eq * grad_phi_u_j_x_velocity[j] * phi_u_i -
982  div_phi_u_i * phi_p_j;
983 
984  // Continuity
985  local_matrix_ij += phi_p_i * div_phi_u_j;
986 
987  // PSPG GLS term
988  local_matrix_ij += tau * (strong_jac * grad_phi_p_i);
989 
990  // The jacobian matrix for the SUPG formulation
991  // currently does not include the jacobian of the stabilization
992  // parameter tau. Our experience has shown that does not alter the
993  // number of newton iteration for convergence, but greatly
994  // simplifies assembly.
995  if (SUPG)
996  {
997  local_matrix_ij +=
998  tau * (strong_jac * grad_phi_u_i_x_velocity +
999  strong_residual_x_grad_phi_u_i * phi_u_j);
1000  }
1001  local_matrix_ij *= JxW;
1002  local_matrix(i, j) += local_matrix_ij;
1003  }
1004  }
1005  }
1006 }

1050

1053

1055  shear_rate_magnitude > 1e-12 ? shear_rate_magnitude : 1e-12;
1056 
1057  // Calculate kinematic viscosity gradient
1058  const Tensor<1, dim> kinematic_viscosity_gradient =
1059  this->get_kinematic_viscosity_gradient(

1064 
1065  // Pressure

1067  const Tensor<1, dim> pressure_gradient =

1069 
1070  // Forcing term

1072 
1073  // Calculation of the magnitude of the

1076 
1077  // Store JxW in local variable for faster access;
1078  const double JxW = JxW_vec[q];
1079 
1080  // Calculation of the equivalent properties at the quadrature point
1081  double density_eq = scratch_data.density[q];
1082  const double dynamic_viscosity_eq = scratch_data.dynamic_viscosity[q];
1083  const Tensor<1, dim> dynamic_viscosity_gradient =
1084  density_eq * kinematic_viscosity_gradient;
1085 
1086  // Calculation of the GLS stabilization parameter. The
1087  // stabilization parameter used is different if the simulation
1088  // is steady or unsteady. In the unsteady case it includes the
1089  // value of the time-step
1090  const double tau =
1091  this->simulation_control->get_assembly_method() ==
1092  Parameters::SimulationControl::TimeSteppingMethod::steady ?

1094  u_mag, dynamic_viscosity_eq / density_eq, h) :
1095  calculate_navier_stokes_gls_tau_transient(
1096  u_mag, dynamic_viscosity_eq / density_eq, h, sdt);
1097 
1098 
1099  // Calculate the strong residual for GLS stabilization
1100  auto strong_residual = density_eq * velocity_gradient * velocity +
1101  pressure_gradient -
1102  shear_rate * dynamic_viscosity_gradient -
1103  dynamic_viscosity_eq * velocity_laplacian -
1104  density_eq * force + strong_residual_vec[q];
1105 
1106  // Assembly of the right-hand side
1107  for (unsigned int i = 0; i < n_dofs; ++i)
1108  {
1109  const auto &phi_u_i = scratch_data.phi_u[q][i];
1110  const auto &grad_phi_u_i = scratch_data.grad_phi_u[q][i];
1111  const auto &phi_p_i = scratch_data.phi_p[q][i];
1112  const auto &grad_phi_p_i = scratch_data.grad_phi_p[q][i];
1113  const auto &div_phi_u_i = scratch_data.div_phi_u[q][i];
1114 
1115  // Navier-Stokes Residual
1116  local_rhs(i) +=
1117  (
1118  // Momentum
1119  -dynamic_viscosity_eq * scalar_product(shear_rate, grad_phi_u_i) -
1120  density_eq * velocity_gradient * velocity * phi_u_i +
1121  pressure * div_phi_u_i + density_eq * force * phi_u_i) *
1122  JxW;
1123 
1124  // Continuity

1126 
1127  // PSPG GLS term
1128  local_rhs(i) += -tau * (strong_residual * grad_phi_p_i) * JxW;
1129 
1130  // SUPG GLS term
1131  if (SUPG)
1132  {
1133  local_rhs(i) +=
1134  -tau * (strong_residual * (grad_phi_u_i * velocity)) * JxW;
1135  }
1136  }
