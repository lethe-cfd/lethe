21 /**
22  * @class VorticityPostprocessor Post-processor class used to calculate
23  * the vorticity field within the domain. The vorticity is defined
24  * as $\mathbf{\omega} = \nabla \times \mathbf{u}$ or, in other notation,
25  * $\mathbf{\omega} = rot(\mathbf{u})$.
26  */
27 template <int dim>
28 class VorticityPostprocessor : public DataPostprocessorVector<dim>
29 {
30 public:
31  VorticityPostprocessor()
32  : DataPostprocessorVector<dim>("vorticity", update_gradients)
33  {}
34  virtual void
35  evaluate_vector_field(
36  const DataPostprocessorInputs::Vector<dim> &input_data,
37  std::vector<Vector<double>> &computed_quantities) const override
38  {
39  AssertDimension(input_data.solution_gradients.size(),
40  computed_quantities.size());
41  for (unsigned int p = 0; p < input_data.solution_gradients.size(); ++p)
42  {
43  AssertDimension(computed_quantities[p].size(), dim);
44  if constexpr (dim == 3)
45  {
46  computed_quantities[p](0) =
47  (input_data.solution_gradients[p][2][1] -
48  input_data.solution_gradients[p][1][2]);
49  computed_quantities[p](1) =
50  (input_data.solution_gradients[p][0][2] -
51  input_data.solution_gradients[p][2][0]);
52  computed_quantities[p](2) =
53  (input_data.solution_gradients[p][1][0] -
54  input_data.solution_gradients[p][0][1]);
55  }
56  if constexpr (dim == 2)
57  {
58  computed_quantities[p][0] =
59  (input_data.solution_gradients[p][1][0] -
60  input_data.solution_gradients[p][0][1]);
61  }
62  }
63  }
64 };

66 /**
67  * @class QCriterionPostprocessor Post-processor class used to calculate
68  * the QCriterion field within the domain. The Q Criterion is defined as
69  * as $Q = ||\Omega||^2 -||S||^2$ where $S$ is the symmetric deformation tensor
70  */
71 template <int dim>
72 class QCriterionPostprocessor : public DataPostprocessorScalar<dim>
73 {
74 public:
75  QCriterionPostprocessor()
76  : DataPostprocessorScalar<dim>("q_criterion", update_gradients)
77  {}
78  virtual void
79  evaluate_vector_field(
80  const DataPostprocessorInputs::Vector<dim> &input_data,
81  std::vector<Vector<double>> &computed_quantities) const override
82  {
83  for (unsigned int p = 0; p < input_data.solution_gradients.size(); ++p)
84  {
85  AssertDimension(computed_quantities[p].size(), 1);
86  double p1 = 0.0, r1 = 0.0;
87  std::vector<Tensor<2, dim>> vorticity_vector(
88  input_data.solution_gradients.size());
89  std::vector<Tensor<2, dim>> strain_rate_tensor(
90  input_data.solution_gradients.size());
91  for (unsigned int j = 0; j < dim; j++)
92  {
93  for (unsigned int k = 0; k < dim; k++)
94  {
95  vorticity_vector[p][j][k] =
96  0.5 * ((input_data.solution_gradients[p][j][k]) -
97  input_data.solution_gradients[p][k][j]);
98  strain_rate_tensor[p][j][k] =
99  0.5 * ((input_data.solution_gradients[p][j][k]) +
100  input_data.solution_gradients[p][k][j]);
101  }
102  }
103  for (unsigned int m = 0; m < dim; m++)
104  {
105  for (unsigned int n = 0; n < dim; n++)
106  {
107  p1 += (vorticity_vector[p][m][n]) * (vorticity_vector[p][m][n]);
108  r1 +=
109  (strain_rate_tensor[p][m][n]) * (strain_rate_tensor[p][m][n]);
110  }
111  }
112  computed_quantities[p] = 0.5 * (p1 - r1);
113  }
114  }
115 };

149 /**
150  * @class Calculates the velocity in the Eulerian frame of reference
151  * for simulations which are carried out in a rotating frame.
152  * The velocity in the Eulerian frame is $\mathbf{u}_E=\mathbf{u}_L+
153  * \mathbf{\Omega} \times \mathbf{R}$ where $\Omega$ is the velocity of the
154  * frame of reference and $\mathbf{R}$ the position vector
155  */
156 template <int dim>
157 class SRFPostprocessor : public DataPostprocessorVector<dim>
158 {
159 public:
160  SRFPostprocessor(double p_omega_x, double p_omega_y, double p_omega_z)
161  : DataPostprocessorVector<dim>(std::string("velocity_eulerian"),
162  update_values | update_quadrature_points)
163  , omega_x(p_omega_x)
164  , omega_y(p_omega_y)
165  , omega_z(p_omega_z)
166  {}
167  virtual void
168  evaluate_vector_field(
169  const DataPostprocessorInputs::Vector<dim> &inputs,
170  std::vector<Vector<double>> &computed_quantities) const override
171  {
172  Tensor<1, dim> omega;
173  omega[0] = omega_x;
174  omega[1] = omega_y;
175  if (dim == 3)
176  omega[2] = omega_z;
177 
178  const unsigned int n_quadrature_points = inputs.solution_values.size();
179  for (unsigned int q = 0; q < n_quadrature_points; ++q)
180  {
181  Tensor<1, dim> velocity;
182  velocity[0] = inputs.solution_values[q][0];
183  velocity[1] = inputs.solution_values[q][1];
184  if (dim == 2)
185  velocity = velocity + omega_z * (-1.) *
186  cross_product_2d(inputs.evaluation_points[q]);
187 
188  if constexpr (dim == 3)
189  {
190  velocity[2] = inputs.solution_values[q][2];
191  velocity =
192  velocity + cross_product_3d(omega, inputs.evaluation_points[q]);
193  }
194  for (int d = 0; d < dim; ++d)
195  {
196  computed_quantities[q][d] = velocity[d];
197  }
198  }
199  }
200 
201 private:
202  double omega_x;
203  double omega_y;
204  double omega_z;
205 };
