35  *
36  * @param[in] input_data Fluid dynamics velocity and pressure information.
37  *

62  input_data.solution_gradients[p][2][0]);
63  computed_quantities[p](2) =

95  /**
96  * @brief Computes the Q-criterion from the velocity vector at given
97  * evaluation points.
98  *
99  * @param[in] input_data Fluid dynamics velocity and pressure information.
100  *
101  * @param[out] computed_quantities Computed Q-criterion value at evaluation
102  * points.
103  *
104  * @note @p input_data contains both velocity and pressure data since fluid
105  * dynamics in Lethe is solved in a monolithic way, but only the velocity
106  * gradient is used to compute the Q-criterion.
107  */
108  virtual void
109  evaluate_vector_field(
110  const DataPostprocessorInputs::Vector<dim> &input_data,
111  std::vector<Vector<double>> &computed_quantities) const override
112  {
113  for (unsigned int p = 0; p < input_data.solution_gradients.size(); ++p)
114  {
115  AssertDimension(computed_quantities[p].size(), 1);
116  double p1 = 0.0, r1 = 0.0;
117  std::vector<Tensor<2, dim>> vorticity_vector(
118  input_data.solution_gradients.size());
119  std::vector<Tensor<2, dim>> strain_rate_tensor(
120  input_data.solution_gradients.size());
121  for (unsigned int j = 0; j < dim; j++)
122  {
123  for (unsigned int k = 0; k < dim; k++)
124  {
125  vorticity_vector[p][j][k] =
126  0.5 * ((input_data.solution_gradients[p][j][k]) -
127  input_data.solution_gradients[p][k][j]);
128  strain_rate_tensor[p][j][k] =
129  0.5 * ((input_data.solution_gradients[p][j][k]) +
130  input_data.solution_gradients[p][k][j]);
131  }
132  }
133  for (unsigned int m = 0; m < dim; m++)
134  {
135  for (unsigned int n = 0; n < dim; n++)
136  {
137  p1 += (vorticity_vector[p][m][n]) * (vorticity_vector[p][m][n]);
138  r1 +=
139  (strain_rate_tensor[p][m][n]) * (strain_rate_tensor[p][m][n]);
140  }
141  }
142  computed_quantities[p] = 0.5 * (p1 - r1);
143  }
144  }
145 };

227  /**
228  * @brief Compute the Eulerian velocity in a SRF at given evaluation points.
229  *
230  * @param[in] inputs Fluid dynamics velocity and pressure information.
231  *
232  * @param[out] computed_quantities Computed Eulerian velocity vector at
233  * evaluation points.
234  *
235  * @note @p input_data contains both velocity and pressure data since fluid
236  * dynamics in Lethe is solved in a monolithic way, but only the velocity
237  * vector is used to compute the Eulerian velocity.
238  */
239  virtual void
240  evaluate_vector_field(
241  const DataPostprocessorInputs::Vector<dim> &inputs,
242  std::vector<Vector<double>> &computed_quantities) const override
243  {
244  Tensor<1, dim> omega;
245  omega[0] = omega_x;
246  omega[1] = omega_y;
247  if (dim == 3)
248  omega[2] = omega_z;
249 
250  const unsigned int n_quadrature_points = inputs.solution_values.size();
251  for (unsigned int q = 0; q < n_quadrature_points; ++q)
252  {
253  Tensor<1, dim> velocity;
254  velocity[0] = inputs.solution_values[q][0];
255  velocity[1] = inputs.solution_values[q][1];
256  if (dim == 2)
257  velocity = velocity + omega_z * (-1.) *
258  cross_product_2d(inputs.evaluation_points[q]);
259 
260  if constexpr (dim == 3)
261  {
262  velocity[2] = inputs.solution_values[q][2];
263  velocity =
264  velocity + cross_product_3d(omega, inputs.evaluation_points[q]);
265  }
266  for (int d = 0; d < dim; ++d)
267  {
268  computed_quantities[q][d] = velocity[d];
269  }
270  }
271  }
272 
273 private:
274  /// Value of the \f$x\f$ component of the SRF's velocity.
275  double omega_x;
276  /// Value of the \f$y\f$ component of the SRF's velocity.
277  double omega_y;
278  /// Value of the \f$z\f$ component of the SRF's velocity.
279  double omega_z;

696  /**
697  * @brief Compute the density of a material at given evaluation points.
698  *
699  * @param[in] inputs Fluid dynamics velocity and pressure information.
700  *
701  * @param[out] computed_quantities Computed density value at evaluation
702  * points.
703  */
704  virtual void
705  evaluate_vector_field(
706  const DataPostprocessorInputs::Vector<dim> &inputs,
707  std::vector<Vector<double>> &computed_quantities) const override
708  {
709  const unsigned int n_quadrature_points = inputs.solution_values.size();
710  std::map<field, double> field_values;
711 
712  for (unsigned int q = 0; q < n_quadrature_points; ++q)
713  {
714  AssertDimension(computed_quantities[q].size(), 1);
715 
716  field_values[field::pressure] = inputs.solution_values[q][dim];
717  computed_quantities[q] = density_model->value(field_values);
718  }
719  }
720 
721 private:
722  /// Model describing the material's density.
723  std::shared_ptr<DensityModel> density_model;

727 /**
728  * @brief Compute the local heat flux vector field within a domain.
729  *
730  * The local heat flux is defined as \f$\mathbf{q} = -k \nabla T\f$ where
731  * \f$k\f$ is the thermal conductivity of the material and \f$\nabla T\f$ the
732  * temperature gradient.
733  */
734 template <int dim>
735 class HeatFluxPostprocessor : public DataPostprocessorVector<dim>
736 {
737 public:
738  /**
739  * @brief Constructor of the heat flux postprocessor.
740  *
741  * @param[in] p_thermal_conductivity_model Thermal conductivity model of the
742  * material
743  *
744  * @param[in] p_material_string Type of material: "f" (fluid) or "s" (solid)
745  *
746  * @param[in] p_id Either fluid ID or solid ID (as initialized in the physical
747  * properties). It is used to name the postprocessed variable vector.
748  *
749  * @param[in] p_mesh_material_id Identifier corresponding to the material as
750  * identified in the mesh. By convention, fluids are given the material_id 0
751  * and solids follow with ids 1, 2, etc. This is used to identify which cell
752  * lies in which material.
753  *
754  * @param[in] p_solid_phase_present Boolean indicating if the simulation has
755  * solids. If there is no solid, the @p material_id of cells is not checked.
756  */
757  HeatFluxPostprocessor(const std::shared_ptr<ThermalConductivityModel>
758  &p_thermal_conductivity_model,
759  const std::string &p_material_string = "f",
760  const unsigned int &p_id = 0,
761  const unsigned int &p_mesh_material_id = 0,
762  const bool &p_solid_phase_present = false)
763  : DataPostprocessorVector<dim>("heat_flux_" + p_material_string +
764  Utilities::to_string(p_id, 2),
765  update_values | update_gradients)
766  , thermal_conductivity_model(p_thermal_conductivity_model)
767  , mesh_material_id(p_mesh_material_id)
768  , solid_phase_present(p_solid_phase_present)
769  {}
770 
771  /**
772  * @brief Compute local heat fluxes from temperature gradient vector field and
773  * thermal conductivity at given evaluation points.
774  *
775  * @param[in] inputs Temperature information.
776  *
777  * @param[out] computed_quantities Computed local heat flux vectors at
778  * evaluation points.
779  */
780  virtual void
781  evaluate_scalar_field(
782  const DataPostprocessorInputs::Scalar<dim> &inputs,
783  std::vector<Vector<double>> &computed_quantities) const override
784  {
785  const unsigned int n_evaluation_points = computed_quantities.size();
786  AssertDimension(inputs.solution_gradients.size(), n_evaluation_points);
787 
788  // If solid materials are present, identify each material's subdomain.
789  if (solid_phase_present)
790  {
791  // Get Cell
792  const auto cell = inputs.template get_cell<dim>();
793 
794  // Check if the cell lies in the material of interest
795  if (cell->material_id() == mesh_material_id)
796  {
797  compute_quantities(inputs, computed_quantities);
798  }
799  // Apply null values to irrelevant subdomain
800  else
801  {
802  Vector<double> null_vector(dim);
803  std::vector<Vector<double>> null_computed_quantities(
804  n_evaluation_points, null_vector);
805  computed_quantities = null_computed_quantities;
806  }
807  }
808  // Only fluids are present
809  else
810  {
811  compute_quantities(inputs, computed_quantities);
812  }
813  }
814 
815 private:
816  /**
817  * @brief Computes local heat flux vectors.
818  *
819  * @note The function is used to improve readability of the
820  * HeatFluxPostprocessor::evaluate_scalar_field function and avoid repeated
821  * lines.
822  *
823  * @param[in] inputs Temperature information.
824  *
825  * @param[out] computed_quantities Computed local heat flux vectors at
826  * evaluation points.
827  */
828  void
829  compute_quantities(const DataPostprocessorInputs::Scalar<dim> &inputs,
830  std::vector<Vector<double>> &computed_quantities) const
831  {
832  std::map<field, double> field_values;
833  for (unsigned int p = 0; p < computed_quantities.size(); ++p)
834  {
835  AssertDimension(computed_quantities[p].size(), dim);
836  field_values[field::temperature] = inputs.solution_values[p];
837 
838  for (unsigned int i = 0; i < dim; ++i)
839  {
840  computed_quantities[p][i] =
841  -thermal_conductivity_model->value(field_values) *
842  inputs.solution_gradients[p][i];
843  }
844  }
845  }
846 
847  /// Model describing of thermal conductivity of the material.
848  std::shared_ptr<ThermalConductivityModel> thermal_conductivity_model;
849  /// Identifier corresponding to the material as identified in the mesh.
850  const unsigned int mesh_material_id;
851  /// Boolean indicating if the simulation has solids.
852  const bool solid_phase_present;
853 };
