62  input_data.solution_gradients[p][2][0]);

95  /**
96  * @brief Computes the Q-criterion from the velocity vector at given
97  * evaluation points.
98  *
99  * @param[in] input_data Fluid dynamics velocity and pressure information.
100  *
101  * @param[out] computed_quantities Computed Q-criterion value at evaluation
102  * points.
103  *
104  * @note @p input_data contains both velocity and pressure data since fluid
105  * dynamics in Lethe is solved in a monolithic way, but only the velocity
106  * gradient is used to compute the Q-criterion.
107  */
108  virtual void
109  evaluate_vector_field(
110  const DataPostprocessorInputs::Vector<dim> &input_data,
111  std::vector<Vector<double>> &computed_quantities) const override
112  {
113  for (unsigned int p = 0; p < input_data.solution_gradients.size(); ++p)
114  {
115  AssertDimension(computed_quantities[p].size(), 1);
116  double p1 = 0.0, r1 = 0.0;
117  std::vector<Tensor<2, dim>> vorticity_vector(
118  input_data.solution_gradients.size());
119  std::vector<Tensor<2, dim>> strain_rate_tensor(
120  input_data.solution_gradients.size());
121  for (unsigned int j = 0; j < dim; j++)
122  {
123  for (unsigned int k = 0; k < dim; k++)
124  {
125  vorticity_vector[p][j][k] =
126  0.5 * ((input_data.solution_gradients[p][j][k]) -
127  input_data.solution_gradients[p][k][j]);
128  strain_rate_tensor[p][j][k] =
129  0.5 * ((input_data.solution_gradients[p][j][k]) +
130  input_data.solution_gradients[p][k][j]);
131  }
132  }
133  for (unsigned int m = 0; m < dim; m++)
134  {
135  for (unsigned int n = 0; n < dim; n++)
136  {
137  p1 += (vorticity_vector[p][m][n]) * (vorticity_vector[p][m][n]);
138  r1 +=
139  (strain_rate_tensor[p][m][n]) * (strain_rate_tensor[p][m][n]);
140  }
141  }
142  computed_quantities[p] = 0.5 * (p1 - r1);
143  }
144  }
145 };

227  /**
228  * @brief Compute the Eulerian velocity in a SRF at given evaluation points.
229  *
230  * @param[in] inputs Fluid dynamics velocity and pressure information.
231  *
232  * @param[out] computed_quantities Computed Eulerian velocity vector at
233  * evaluation points.
234  *
235  * @note @p input_data contains both velocity and pressure data since fluid
236  * dynamics in Lethe is solved in a monolithic way, but only the velocity
237  * vector is used to compute the Eulerian velocity.
238  */
239  virtual void
240  evaluate_vector_field(
241  const DataPostprocessorInputs::Vector<dim> &inputs,
242  std::vector<Vector<double>> &computed_quantities) const override
243  {
244  Tensor<1, dim> omega;
245  omega[0] = omega_x;
246  omega[1] = omega_y;
247  if (dim == 3)
248  omega[2] = omega_z;
249 
250  const unsigned int n_quadrature_points = inputs.solution_values.size();
251  for (unsigned int q = 0; q < n_quadrature_points; ++q)
252  {
253  Tensor<1, dim> velocity;
254  velocity[0] = inputs.solution_values[q][0];
255  velocity[1] = inputs.solution_values[q][1];
256  if (dim == 2)
257  velocity = velocity + omega_z * (-1.) *
258  cross_product_2d(inputs.evaluation_points[q]);
259 
260  if constexpr (dim == 3)
261  {
262  velocity[2] = inputs.solution_values[q][2];
263  velocity =
264  velocity + cross_product_3d(omega, inputs.evaluation_points[q]);
265  }
266  for (int d = 0; d < dim; ++d)
267  {
268  computed_quantities[q][d] = velocity[d];
269  }
270  }
271  }
272 
273 private:
274  /// Value of the \f$x\f$ component of the SRF's velocity.
275  double omega_x;
276  /// Value of the \f$y\f$ component of the SRF's velocity.
277  double omega_y;
278  /// Value of the \f$z\f$ component of the SRF's velocity.
279  double omega_z;

728 /**
729  * @brief Compute the local heat flux vector field within a domain.
730  *
731  * The local heat flux is defined as \f$\mathbf{q} = -k \nabla T\f$ where
732  * \f$k\f$ is the thermal conductivity of the material and \f$\nabla T\f$ the
733  * temperature gradient.
734  */
735 template <int dim>
736 class HeatFluxPostprocessor : public DataPostprocessorVector<dim>
737 {
738 public:
739  /**
740  * @brief Constructor of the heat flux postprocessor.
741  *
742  * @param[in] p_thermal_conductivity_model Thermal conductivity model of the
743  * material
744  *
745  * @param[in] p_material_string Type of material: "f" (fluid) or "s" (solid)
746  *
747  * @param[in] p_id Either fluid ID or solid ID (as initialized in the physical
748  * properties). It is used to name the postprocessed variable vector.
749  *
750  * @param[in] p_mesh_material_id Identifier corresponding to the material as
751  * identified in the mesh. By convention, fluids are given the material_id 0
752  * and solids follow with ids 1, 2, etc. This is used to identify which cell
753  * lies in which material.
754  *
755  * @param[in] p_solid_phase_present Boolean indicating if the simulation has
756  * solids. If there is no solid, the @p material_id of cells is not checked.
757  */
758  HeatFluxPostprocessor(const std::shared_ptr<ThermalConductivityModel>
759  &p_thermal_conductivity_model,
760  const std::string &p_material_string = "f",
761  const unsigned int &p_id = 0,
762  const unsigned int &p_mesh_material_id = 0,
763  const bool &p_solid_phase_present = false)
764  : DataPostprocessorVector<dim>("heat_flux_" + p_material_string +
765  Utilities::to_string(p_id, 2),
766  update_values | update_gradients)
767  , thermal_conductivity_model(p_thermal_conductivity_model)
768  , mesh_material_id(p_mesh_material_id)
769  , solid_phase_present(p_solid_phase_present)
770  {}
771 
772  /**
773  * @brief Compute local heat fluxes from temperature gradient vector field and
774  * thermal conductivity at given evaluation points.
775  *
776  * @param[in] inputs Temperature information.
777  *
778  * @param[out] computed_quantities Computed local heat flux vectors at
779  * evaluation points.
780  */
781  virtual void
782  evaluate_scalar_field(
783  const DataPostprocessorInputs::Scalar<dim> &inputs,
784  std::vector<Vector<double>> &computed_quantities) const override
785  {
786  const unsigned int n_evaluation_points = computed_quantities.size();
787  AssertDimension(inputs.solution_gradients.size(), n_evaluation_points);
788 
789  // If solid materials are present, identify each material's subdomain.
790  if (solid_phase_present)
791  {
792  // Get Cell
793  const auto cell = inputs.template get_cell<dim>();
794 
795  // Check if the cell lies in the material of interest
796  if (cell->material_id() == mesh_material_id)
797  {
798  compute_quantities(inputs, computed_quantities);
799  }
800  // Apply null values to irrelevant subdomain
801  else
802  {
803  Vector<double> null_vector(dim);
804  std::vector<Vector<double>> null_computed_quantities(
805  n_evaluation_points, null_vector);
806  computed_quantities = null_computed_quantities;
807  }
808  }
809  // Only fluids are present
810  else
811  {
812  compute_quantities(inputs, computed_quantities);
813  }
814  }
815 
816 private:
817  /**
818  * @brief Computes local heat flux vectors.
819  *
820  * @note The function is used to improve readability of the
821  * HeatFluxPostprocessor::evaluate_scalar_field function and avoid repeated
822  * lines.
823  *
824  * @param[in] inputs Temperature information.
825  *
826  * @param[out] computed_quantities Computed local heat flux vectors at
827  * evaluation points.
828  */
829  void
830  compute_quantities(const DataPostprocessorInputs::Scalar<dim> &inputs,
831  std::vector<Vector<double>> &computed_quantities) const
832  {
833  std::map<field, double> field_values;
834  for (unsigned int p = 0; p < computed_quantities.size(); ++p)
835  {
836  AssertDimension(computed_quantities[p].size(), dim);
837  field_values[field::temperature] = inputs.solution_values[p];
838 
839  for (unsigned int i = 0; i < dim; ++i)
840  {
841  computed_quantities[p][i] =
842  -thermal_conductivity_model->value(field_values) *
843  inputs.solution_gradients[p][i];
844  }
845  }
846  }
847 
848  /// Model describing of thermal conductivity of the material.
849  std::shared_ptr<ThermalConductivityModel> thermal_conductivity_model;
850  /// Identifier corresponding to the material as identified in the mesh.
851  const unsigned int mesh_material_id;
852  /// Boolean indicating if the simulation has solids.
853  const bool solid_phase_present;
854 };
