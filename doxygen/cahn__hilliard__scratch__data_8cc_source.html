98 template <int dim>
99 void
100 CahnHilliardScratchData<dim>::calculate_physical_properties()
101 {
102  switch (properties_manager.get_number_of_fluids())
103  {
104  case 1:
105  {
106  throw std::runtime_error("Unsupported number of fluids (<2)");
107  }
108  case 2:
109  {
110  set_field_vector(field::phase_order_cahn_hilliard,
111  this->phase_order_values,
112  this->fields);
113 
114  // Gather properties from material interactions
115  const auto material_interaction_id =
116  properties_manager.get_material_interaction_id(
117  material_interactions_type::fluid_fluid, 0, 1);
118 
119  // Gather surface tension
120  const auto surface_tension_model =
121  properties_manager.get_surface_tension(material_interaction_id);
122  surface_tension_model->vector_value(fields, surface_tension);
123 
124  // Gather mobility
125  const auto mobility_cahn_hilliard_model =
126  properties_manager.get_mobility_cahn_hilliard(
127  material_interaction_id);
128  mobility_cahn_hilliard_model->vector_value(fields,
129  mobility_cahn_hilliard);
130  mobility_cahn_hilliard_model->vector_jacobian(
131  fields, phase_order_cahn_hilliard, mobility_cahn_hilliard_gradient);
132 
133  break;
134  }
135  default:
136  throw std::runtime_error("Unsupported number of fluids (>2)");
137  }
138 }
