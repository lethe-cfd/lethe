105  simulation_parameters.fem_parameters.VOF_order);
106  mapping = std::make_shared<MappingQ<dim>>(fe->degree);
107  fe_projected_phase_fraction_gradient =
108  std::make_shared<FESystem<dim>>(FE_Q<dim>(fe->degree), dim);
109  fe_curvature = std::make_shared<FE_Q<dim>>(fe->degree);

111  std::make_shared<MappingQ<dim>>(
112  fe_projected_phase_fraction_gradient->degree);
113  curvature_mapping =
