41 /**
42  * @brief Class that stores the information required by the assembly procedure
43  * for a VOF free surface equation. Consequently, this class
44  * calculates the phase values (values, gradients, laplacians) and the shape
45  * method (values, gradients, laplacians) at all the gauss points for all
46  * degrees of freedom and stores it into arrays.
47  * This class serves as a separation between the evaluation at the gauss point
48  * of the variables of interest and their use in the assembly, which is carried
49  * out by the assembler methods.
50  *
51  * @tparam dim An integer that denotes the dimension of the space in which
52  * the flow is solved
53  *
54  * @ingroup solvers
55  **/
56

62  * @brief Constructor. The constructor creates the fe_values that will be used
63  * to fill the member variables of the scratch. It also allocated the
64  * necessary memory for all member variables. However, it does not do any
65  * evaluation, since this needs to be done at the cell level.
66  *
67  * @param simulation_control The SimulationControl object that holds
68  * information related to the control of the steady-state or transient
69  * simulation. This is used to extrapolate velocity solutions in time
70  * for transient simulation.
71  *
72  * @param fe_vof The FESystem used to solve the VOF equations
73  *
74  * @param quadrature The quadrature to use for the assembly
75  *
76  * @param mapping The mapping of the domain in which the Navier-Stokes
77  * equations are solved
78  *
79  * @param fe_fd The FESystem used to solve the Fluid Dynamics equations
80  *

136  /** @brief Reinitialize the content of the scratch
137  *
138  * Using the FeValues and the content of the solutions and previous solutions,
139  * fills all of the class member of the scratch
140  *
141  * @tparam VectorType The Vector type used for the solvers
142  *
143  * @param cell The cell over which the assembly is being carried.
144  * This cell must be compatible with the fe which is used to fill the FeValues
145  *
146  * @param current_solution The present value of the solution for the VOF
147  *
148  * @param previous_solutions The solutions at the previous time steps
149  *
150  */
151 
152  template <typename VectorType>
153  void
154  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
155  const VectorType &current_solution,
156  const std::vector<VectorType> &previous_solutions)
157  {
158  fe_values_vof.reinit(cell);
159  this->quadrature_points = fe_values_vof.get_quadrature_points();
160  auto &fe_vof = fe_values_vof.get_fe();
161 
162  if (dim == 2)
163  this->cell_size = std::sqrt(4. * cell->measure() / M_PI) / fe_vof.degree;
164  else if (dim == 3)
165  this->cell_size =
166  pow(6 * cell->measure() / M_PI, 1. / 3.) / fe_vof.degree;
167 
168  fe_values_vof.get_function_values(current_solution,
169  this->present_phase_values);
170  fe_values_vof.get_function_gradients(current_solution,
171  this->phase_gradients);
172  fe_values_vof.get_function_laplacians(current_solution,
173  this->phase_laplacians);
174 
175  fe_values_vof.get_function_gradients(previous_solutions[0],
176  this->previous_phase_gradients);
177 
178 
179  // Gather previous vof values
180  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
181  {
182  fe_values_vof.get_function_values(previous_solutions[p],
183  this->previous_phase_values[p]);
184  }
185 
186  for (unsigned int q = 0; q < this->n_q_points; ++q)
187  {
188  this->JxW[q] = fe_values_vof.JxW(q);
189 
190  for (unsigned int k = 0; k < this->n_dofs; ++k)
191  {
192  // Shape function
193  this->phi[q][k] = fe_values_vof.shape_value(k, q);
194  this->grad_phi[q][k] = fe_values_vof.shape_grad(k, q);
195  this->hess_phi[q][k] = fe_values_vof.shape_hessian(k, q);
196  this->laplacian_phi[q][k] = trace(this->hess_phi[q][k]);
197  }
198  }
199  }

201  /** @brief Reinitialize the velocity, calculated by the Fluid Dynamics
202  *
203  * @tparam VectorType The Vector type used for the solvers
204  *
205  * @param cell The cell for which the velocity is reinitialized
206  * This cell must be compatible with the Fluid Dynamics FE
207  *
208  * @param current_solution The present value of the solution for \f$[u,p]\f$
209  *
210  * @param previous_solutions Vector of \f$n\f$ @p VectorType containers of
211  * previous fluid dynamic solutions (\f$[u,p]\f$). \f$n\f$ depends on the BDF
212  * scheme selected for time-stepping.
213  *
214  * @param ale ALE parameters
215  *
216  */
217 
218  template <typename VectorType>
219  void
220  reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
221  const VectorType &current_solution,
222  const std::vector<VectorType> &previous_solutions,
223  const Parameters::ALE<dim> &ale)
224  {
225  fe_values_fd.reinit(cell);
226 
227  fe_values_fd[velocities_fd].get_function_values(current_solution,
228  velocity_values);
229  fe_values_fd[velocities_fd].get_function_gradients(
230  current_solution, velocity_gradient_values);
231 
232  for (unsigned int q = 0; q < this->n_q_points; ++q)
233  {
234  this->velocity_divergences[q] =
235  trace(this->velocity_gradient_values[q]);
236  }
237 
238  // Gather previous velocity values
239  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
240  {
241  fe_values_fd[velocities_fd].get_function_values(
242  previous_solutions[p], this->previous_velocity_values[p]);
243  }
244 
245  if (!ale.enabled())
246  return;
247 
248  // ALE enabled, so extract the ALE velocity and subtract it from the
249  // velocity obtained from the fluid dynamics
250  Tensor<1, dim> velocity_ale;
251  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
252  ale.velocity;
253  Vector<double> velocity_ale_vector(dim);
254 
255  for (unsigned int q = 0; q < n_q_points; ++q)
256  {
257  velocity_ale_function->vector_value(quadrature_points[q],
258  velocity_ale_vector);
259  for (unsigned int d = 0; d < dim; ++d)
260  velocity_ale[d] = velocity_ale_vector[d];
261 
262  velocity_values[q] -= velocity_ale;
263 
264  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
265  {
266  this->previous_velocity_values[p][q] -= velocity_ale;
267  }
268  }
269 
270  // Extrapolate velocity to t+dt using the BDF scheme if the simulation is
271  // transient
272  const auto method = this->simulation_control->get_assembly_method();
273  if (is_bdf(method))
274  {
275  // Extrapolate velocity
276  std::vector<double> time_vector =
277  this->simulation_control->get_simulation_times();
278  bdf_extrapolate(time_vector,
279  previous_velocity_values,
280  number_of_previous_solutions(method),
281  velocity_values);
282  }
283  }
