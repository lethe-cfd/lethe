163  virtual void
164  assemble_matrix_and_rhs()
165  {
166  if (some_particles_are_coupled)
167  {
168  if (this->get_current_newton_iteration() != 0)
169  {
170  force_on_ib();
171  }
172  integrate_particles();
173  update_precalculations_for_ib();
174  }
175  // Do the cut cell mapping only if it is the first Newton iteration or
176  // if the explicit position integration is false.
177  if (this->get_current_newton_iteration() == 0 ||
178  this->simulation_parameters.particlesParameters
179  ->explicit_position_integration_calculation == false)
180  {
181  if (all_spheres)
182  optimized_generate_cut_cells_map();
183  else
184  generate_cut_cells_map();
185  }
186  this->GLSNavierStokesSolver<
187  dim>::assemble_system_matrix_without_preconditioner();
188 
189 
190  sharp_edge();
191 
192  // Assemble the preconditioner
193  this->setup_preconditioner();
194  }
