425

431 private:
432  inline double
433  calculate_kinematic_viscosity(const double shear_rate_magnitude) const
434  {
435  return shear_rate_magnitude > shear_rate_min ?
436  K * std::pow(shear_rate_magnitude, n - 1) :
437  K * std::pow(shear_rate_min, n - 1);
438  }
439 
440  inline double
441  calculate_derivative(const double shear_rate_magnitude)
442  {
443  return shear_rate_magnitude > shear_rate_min ?
444  (n - 1) * K * std::pow(shear_rate_magnitude, n - 2) :
445  0;
446  }
447 
448  double K;
449  double n;
450  const double shear_rate_min;

608 /**
609  * @brief Calculates the magnitude of the shear rate tensor given as input
610  *
611  * @param shear_rate A shear rate tensor
612  */
613 template <int dim>
614 inline double
615 calculate_shear_rate_magnitude(const Tensor<2, dim> shear_rate)
616 {
617  double shear_rate_magnitude = 0;
618  for (unsigned int i = 0; i < dim; ++i)
619  {
620  for (unsigned int j = 0; j < dim; ++j)
621  {
622  shear_rate_magnitude += (shear_rate[i][j] * shear_rate[j][i]);
623  }
624  }
625  shear_rate_magnitude = sqrt(0.5 * shear_rate_magnitude);
626  return shear_rate_magnitude;
627 }
