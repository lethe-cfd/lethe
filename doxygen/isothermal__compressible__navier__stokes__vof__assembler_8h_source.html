28 /**
29  * @brief Class that assembles the core of the Navier-Stokes equation with
30  * free surface using VOF modeling.
31  * According to the following weak form:
32  * \f$\nabla \cdot (\rho \mathbf{u}) + \rho \mathbf{u} \cdot \nabla
33  * \mathbf{u} - \nabla p - \mu \nabla\cdot (\nabla \mathbf{u} +
34  * (\nabla \mathbf{u})^T) = 0 \f$ with a full GLS stabilization
35  *
36  * @tparam dim An integer that denotes the number of spatial dimensions
37  *
38  * @ingroup assemblers
39  */
40 template <int dim>
41 class GLSIsothermalCompressibleNavierStokesVOFAssemblerCore
42  : public NavierStokesAssemblerBase<dim>
43 {
44 public:
45  GLSIsothermalCompressibleNavierStokesVOFAssemblerCore(
46  std::shared_ptr<SimulationControl> simulation_control,
47  const SimulationParameters<dim> &nsparam)
48  : simulation_control(simulation_control)
49  , vof_parameters(nsparam.multiphysics.vof_parameters)
50  {}
