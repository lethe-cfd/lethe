49 // Constant resistance model
50 // M_r = - mu_r * R_eff * |F_n| * omega_hat
51 // omega_hat = (omega_i - omega_j) / (|oemaga_i - omega_j|)
52 inline Tensor<1, 3>
53 constant_rolling_resistance_torque(
54  const double effective_r,
55  const ArrayView<const double> &particle_one_properties,
56  const ArrayView<const double> &particle_two_properties,
57  const double effective_rolling_friction_coefficient,
58  const double normal_force_norm,
59  const Tensor<1, 3> & /*normal_contact_vector*/)
60 
61 {
62  // For calculation of rolling resistance torque, we need to obtain
63  // omega_ij using rotational velocities of particles one and two
64  Tensor<1, 3> particle_one_angular_velocity, particle_two_angular_velocity,
65  omega_ij, omega_ij_direction;
66  for (int d = 0; d < 3; ++d)
67  {
68  particle_one_angular_velocity[d] =
69  particle_one_properties[DEM::PropertiesIndex::omega_x + d];
70  particle_two_angular_velocity[d] =
71  particle_two_properties[DEM::PropertiesIndex::omega_x + d];
72  }
73 
74  omega_ij = particle_one_angular_velocity - particle_two_angular_velocity;
75  omega_ij_direction = omega_ij / (omega_ij.norm() + DBL_MIN);
76 
77  // Calculation of rolling resistance torque
78  return (-effective_rolling_friction_coefficient * effective_r *
79  normal_force_norm * omega_ij_direction);
80 }

82 /**
83  * @brief
84  */
85 // Viscous resistance model
86 // M_r = - mu_r * R_eff * |F_n| * |V_omega| * omega_hat
87 // omega_hat = (omega_i - omega_j) / (|oemaga_i - omega_j|)
88 // V_omega = omega_i × (R_i * n_ij) - omega_j × (R_j * n_ji)
89 inline Tensor<1, 3>
90 viscous_rolling_resistance_torque(
91  const double effective_r,
92  const ArrayView<const double> &particle_one_properties,
93  const ArrayView<const double> &particle_two_properties,
94  const double effective_rolling_friction_coefficient,
95  const double normal_force_norm,
96  const Tensor<1, 3> &normal_contact_vector)
97 
98 {
99  // For calculation of rolling resistance torque, we need to obtain
100  // omega_ij using rotational velocities of particles one and two
101  Tensor<1, 3> particle_one_angular_velocity, particle_two_angular_velocity,
102  omega_ij, omega_ij_direction;
103 
104  for (int d = 0; d < 3; ++d)
105  {
106  particle_one_angular_velocity[d] =
107  particle_one_properties[DEM::PropertiesIndex::omega_x + d];
108  particle_two_angular_velocity[d] =
109  particle_two_properties[DEM::PropertiesIndex::omega_x + d];
110  }
111 
112  omega_ij = particle_one_angular_velocity - particle_two_angular_velocity;
113  omega_ij_direction = omega_ij / (omega_ij.norm() + DBL_MIN);
114 
115  Tensor<1, 3> v_omega =
116  cross_product_3d(particle_one_angular_velocity,
117  particle_one_properties[DEM::PropertiesIndex::dp] * 0.5 *
118  normal_contact_vector) -
119  cross_product_3d(particle_two_angular_velocity,
120  particle_two_properties[DEM::PropertiesIndex::dp] * 0.5 *
121  -normal_contact_vector);
122 
123  // Calculation of rolling resistance torque
124  return (-effective_rolling_friction_coefficient * effective_r *
125  normal_force_norm * v_omega.norm() * omega_ij_direction);
126 }
