43 /*
44  * Fields on which physical property can depend. All fields are assumed
45  * to be at time t+dt other than those for which a _p suffix is explicitly
46  * given. The suffix _p1 implies that the field is at time t (previous one step)

48  * steps).
49  */

74 /**
75  * @brief Abstract class that defines the interface for a physical property model
76  * Physical property model provides an abstract interface to calculate the
77  * value of a physical property or a vector of physical property value for
78  * given field value. By default, the interface does not require that all (or
79  * any) fields be specified. This is why a map is used to pass the dependent
80  * variable. To allow for the calculation of the appropriate jacobian matrix
81  * (when that is necessary) the interface also provides a jacobian function,
82  * which must provide the derivative with respect to the field specified as an
83  * argument.
84  */

167  /**
168  * @brief numerical_jacobian Calculates the jacobian through a forward finite difference (Euler) approximation.
169  * This approach, although not preferable, is meant as a fall-back when
170  * calculating the jacobian manually is too difficult.
171  * @param fields_values The values of the various fields
172  * @param id Field id of the field with respect to which the jacobian should be calculated
173  * @return
174  */
175  inline double
176  numerical_jacobian(const std::map<field, double> &field_values,
177  const field id)
178  {
179  double f_x = this->value(field_values);
180  auto x_dx = field_values;
181  double dx = std::max(1e-6 * x_dx[id], 1e-8);
182  x_dx[id] = x_dx[id] + dx;
183  double f_xdx = this->value(x_dx);
184  return (f_xdx - f_x) / dx;
185  }
186 
187  /**
188  * @brief vector_numerical_jacobian Calculates the vector of jacobian through forward finite difference (Euler) approximation.
189  * This approach, although not preferable, is meant as a fall-back when
190  * calculating the jacobian manually is too difficult.
191  * @param field_vectors
192  * @param field_id
193  * @param jacobian_vector
194  * @return
195  */
196  inline void
197  vector_numerical_jacobian(
198  const std::map<field, std::vector<double>> &field_vectors,
199  const field id,
200  std::vector<double> &jacobian_vector)
201  {
202  const unsigned int n_pts = jacobian_vector.size();
203 
204  // Evaluate the properties using the values of the field vector
205  std::vector<double> f_x(n_pts);
206  vector_value(field_vectors, f_x);
207 
208  // Make a copy of the field vector for the field we wil perturbate
209  std::map<field, std::vector<double>> perturbed_field_vectors =
210  field_vectors;
211  std::vector<double> &x = perturbed_field_vectors.at(id);
212  std::vector<double> dx(n_pts);
213 
214  // Perturb the field by dx
215  for (unsigned int i = 0; i < n_pts; ++i)
216  {
217  dx[i] = std::max(1e-6 * x[i], 1e-8);
218  x[i] += dx[i];
219  }
220 
221  // Evaluate the properties using the perturbed value of the field vector
222  std::vector<double> f_xdx(n_pts);
223  vector_value(perturbed_field_vectors, f_xdx);
224 
225  // Fill jacobian
226  for (unsigned int i = 0; i < n_pts; ++i)
227  jacobian_vector[i] = (f_xdx[i] - f_x[i]) / dx[i];
228  }
