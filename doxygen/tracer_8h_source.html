16  *
17  * Implementation of tracer as an auxiliary physics.
18  * Equation solved:
19  * dT/dt + u * gradT = D * div(grad T) + f
20  * with T the tracer function, D the diffusivity and f the forcing
21  *
22  * Author: Bruno Blais, Polytechnique Montreal, 2020-
23  */

54 class Tracer : public AuxiliaryPhysics<dim, GlobalVectorType>

56 public:
57  Tracer(MultiphysicsInterface<dim> *multiphysics_interface,
58  const SimulationParameters<dim> &p_simulation_parameters,
59  std::shared_ptr<parallel::DistributedTriangulationBase<dim>>
60  p_triangulation,
61  std::shared_ptr<SimulationControl> p_simulation_control)
62  : AuxiliaryPhysics<dim, GlobalVectorType>(
63  p_simulation_parameters.non_linear_solver.at(PhysicsID::tracer))
64  , multiphysics(multiphysics_interface)
65  , computing_timer(p_triangulation->get_communicator(),
66  this->pcout,
67  TimerOutput::summary,
68  TimerOutput::wall_times)
69  , simulation_parameters(p_simulation_parameters)
70  , triangulation(p_triangulation)
71  , simulation_control(p_simulation_control)
72  , dof_handler(*triangulation)
73  {
74  if (simulation_parameters.mesh.simplex)
75  {
76  // for simplex meshes
77  fe = std::make_shared<FE_SimplexP<dim>>(
78  simulation_parameters.fem_parameters.tracer_order);
79  mapping = std::make_shared<MappingFE<dim>>(*fe);
80  cell_quadrature = std::make_shared<QGaussSimplex<dim>>(fe->degree + 1);
81  }
82  else
83  {
84  // Usual case, for quad/hex meshes
85  fe = std::make_shared<FE_Q<dim>>(
86  simulation_parameters.fem_parameters.tracer_order);
87  mapping = std::make_shared<MappingQ<dim>>(fe->degree);
88  cell_quadrature = std::make_shared<QGauss<dim>>(fe->degree + 1);
89  }
90 
91  // Allocate solution transfer
92  solution_transfer = std::make_shared<
93  parallel::distributed::SolutionTransfer<dim, GlobalVectorType>>(
94  dof_handler);
95 
96  // Set size of previous solutions using BDF schemes information
97  previous_solutions.resize(maximum_number_of_previous_solutions());
98 
99  // Prepare previous solutions transfer
100  previous_solutions_transfer.reserve(previous_solutions.size());
101  for (unsigned int i = 0; i < previous_solutions.size(); ++i)
102  {
103  previous_solutions_transfer.emplace_back(
104  parallel::distributed::SolutionTransfer<dim, GlobalVectorType>(
105  this->dof_handler));
106  }
107 
108  // Change the behavior of the timer for situations when you don't want
109  // outputs
110  if (simulation_parameters.timer.type == Parameters::Timer::Type::none)
111  this->computing_timer.disable_output();
112  }
