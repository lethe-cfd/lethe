21 attach_grid_to_triangulation(
22  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
23  const Parameters::Mesh &mesh_parameters)
24

26  // GMSH input
27  if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
28  {
29  if (mesh_parameters.simplex)
30  {
31  auto comm = triangulation.get_communicator();
32  std::string file_name = mesh_parameters.file_name;
33 
34  auto construction_data = TriangulationDescription::Utilities::
35  create_description_from_triangulation_in_groups<dim, spacedim>(
36  [file_name](dealii::Triangulation<dim, spacedim> &basetria) {
37  GridIn<dim, spacedim> grid_in;
38  grid_in.attach_triangulation(basetria);
39  std::ifstream input_file(file_name);
40 
41  grid_in.read_msh(input_file);
42  },
43  [](dealii::Triangulation<dim, spacedim> &basetria,
44  const MPI_Comm comm,
45  const unsigned int /*group_size*/) {
46  GridTools::partition_triangulation(
47  Utilities::MPI::n_mpi_processes(comm), basetria);
48  },
49  comm,
50  Utilities::MPI::n_mpi_processes(comm) /* group size */,
51  dealii::Triangulation<dim, spacedim>::none);
52 
53  triangulation.create_triangulation(construction_data);
54  }
55  else
56  {
57  GridIn<dim, spacedim> grid_in;
58  grid_in.attach_triangulation(triangulation);
59  std::ifstream input_file(mesh_parameters.file_name);
60  grid_in.read_msh(input_file);
61  }
62  }
63  // Dealii grids
64  else if (mesh_parameters.type == Parameters::Mesh::Type::dealii)
65  {
66  if (mesh_parameters.simplex)
67  {
68  Triangulation<dim, spacedim> temporary_quad_triangulation;
69  GridGenerator::generate_from_name_and_arguments(
70  temporary_quad_triangulation,
71  mesh_parameters.grid_type,
72  mesh_parameters.grid_arguments);
73 
74  // initial refinement
75  const unsigned int initial_refinement =
76  mesh_parameters.initial_refinement;
77  temporary_quad_triangulation.refine_global(initial_refinement);
78 
79  // flatten the triangulation
80  Triangulation<dim, spacedim> flat_temp_quad_triangulation;
81  GridGenerator::flatten_triangulation(temporary_quad_triangulation,
82  flat_temp_quad_triangulation);
83 
84  Triangulation<dim, spacedim> temporary_tri_triangulation(
85  Triangulation<dim, spacedim>::limit_level_difference_at_vertices);
86  GridGenerator::convert_hypercube_to_simplex_mesh(
87  flat_temp_quad_triangulation, temporary_tri_triangulation);
88 
89  GridTools::partition_triangulation_zorder(
90  Utilities::MPI::n_mpi_processes(triangulation.get_communicator()),
91  temporary_tri_triangulation);
92  GridTools::partition_multigrid_levels(temporary_tri_triangulation);
93 
94  // extract relevant information from distributed triangulation
95  auto construction_data = TriangulationDescription::Utilities::
96  create_description_from_triangulation(
97  temporary_tri_triangulation,
98  triangulation.get_communicator(),
99  TriangulationDescription::Settings::
100  construct_multigrid_hierarchy);
101  triangulation.create_triangulation(construction_data);
102  }
103  else
104  {
105  GridGenerator::generate_from_name_and_arguments(
106  triangulation,
107  mesh_parameters.grid_type,
108  mesh_parameters.grid_arguments);
109  }
110  }
111  // Customizable cylinder mesh
112  else if (mesh_parameters.type == Parameters::Mesh::Type::cylinder)
113  {
114  if (mesh_parameters.simplex)
115  {
116  throw std::runtime_error(
117  "Unsupported mesh type - custom cylinder mesh with simplex is not supported. Use a dealii cylinder to use simplex mesh.");
118  }
119  else if (dim != 3)
120  {
121  throw std::runtime_error(
122  "Unsupported mesh type - custom cylinder mesh is only supported in 3d.");
123  }
124 
125  if constexpr (dim == 3)
126  {
127  // Separate arguments of the string
128  std::vector<std::string> arguments;
129  std::stringstream s_stream(mesh_parameters.grid_arguments);
130  while (s_stream.good())
131  {
132  std::string substr;
133  getline(s_stream, substr, ':');
134  arguments.push_back(substr);
135  }
136 
137  // Arguments declaration
138  unsigned int subdivisions;
139  double radius, half_height;
140  if (arguments.size() != 3)
141  {
142  throw std::runtime_error(
143  "Mandatory cylinder parameters are (x subdivisions: radius : half height)");
144  }
145  else
146  {
147  std::vector<double> arguments_double =
148  dealii::Utilities::string_to_double(arguments);
149  subdivisions = static_cast<int>(arguments_double[0]);
150  radius = arguments_double[1];
151  half_height = arguments_double[2];
152  }
153 
154  if (mesh_parameters.grid_type == "classic")
155  {
156  // Create a subdivided cylinder from deal.ii
157  GridGenerator::subdivided_cylinder(triangulation,
158  subdivisions,
159  radius,
160  half_height);
161  }
162  else
163  {
164  // Create a temporary 2d mesh
165  Triangulation<2, spacedim - 1> temporary_triangulation;
166 
167  // Create a spherical manifold for 2d mesh
168  Point<2> center(0.0, 0.0);
169  const SphericalManifold<2, spacedim - 1> m0(center);
170 
171  if (mesh_parameters.grid_type == "regularized" ||
172  mesh_parameters.grid_type == "squared")
173  {
174  // Create a square mesh
175  double real_radius = radius * std::sin(M_PI_4);
176  GridGenerator::hyper_cube(temporary_triangulation,
177  -real_radius,
178  real_radius,
179  true);
180 
181  // Assign boundary 0 to perimeter as for cylinder
182  for (const auto &cell :
183  temporary_triangulation.active_cell_iterators())
184  {
185  if (cell->is_locally_owned())
186  {
187  // Looping through all the faces of the cell
188  for (const auto &face : cell->face_iterators())
189  {
190  // Check to see if the face is located at boundary
191  if (face->at_boundary())
192  {
193  face->set_boundary_id(0);
194  }
195  }
196  }
197  }
198  }
199  else if (mesh_parameters.grid_type == "balanced")
200  {
201  GridGenerator::hyper_ball_balanced(temporary_triangulation,
202  center,
203  radius);
204  }
205  else
206  {
207  throw std::runtime_error(
208  "Unknown grid type. Choices are <classic|balanced|squared|regularized>.");
209  }
210 
211  temporary_triangulation.reset_all_manifolds();
212  temporary_triangulation.set_all_manifold_ids_on_boundary(0);
213  temporary_triangulation.set_manifold(0, m0);
214 
215  if (mesh_parameters.grid_type == "regularized")
216  {
217  // Pre-refinement to reduce mesh size at corners before
218  // regularization
219  temporary_triangulation.refine_global(2);
220  GridTools::regularize_corner_cells(temporary_triangulation);
221 
222  // Flatten the triangulation
223  Triangulation<2, spacedim - 1> flat_temporary_triangulation;
224  flat_temporary_triangulation.copy_triangulation(
225  temporary_triangulation);
226  temporary_triangulation.clear();
227  GridGenerator::flatten_triangulation(
228  flat_temporary_triangulation, temporary_triangulation);
229  }
230 
231  // Extrude the 2d temporary mesh to 3d cylinder
232  GridGenerator::extrude_triangulation(temporary_triangulation,
233  subdivisions + 1,
234  2.0 * half_height,
235  triangulation,
236  true);
237 
238  // Rotate mesh in x-axis and set the (0,0,0) at the barycenter
239  // to be comparable to dealii cylinder meshes
240  Tensor<1, spacedim> axis_vector({0.0, 1.0, 0.0});
241  GridTools::rotate(axis_vector, M_PI_2, triangulation);
242  Tensor<1, spacedim> shift_vector({-half_height, 0.0, 0.0});
243  GridTools::shift(shift_vector, triangulation);
244 
245  // Force the manifold id to be zero in the case of the balanced
246  // cylinder
247  if (mesh_parameters.grid_type == "balanced")
248  triangulation.reset_manifold(1);
249 
250  // Add a cylindrical manifold on the final unrefined mesh
251  const CylindricalManifold<3, spacedim> m1(0);
252  triangulation.set_manifold(0, m1);
253  }
254  }
255  }
256 
257  // Colorized cylinder shell
258  else if (mesh_parameters.type ==
259  Parameters::Mesh::Type::colorized_cylinder_shell)
260  {
261  if (mesh_parameters.simplex)
262  {
263  throw std::runtime_error(
264  "Unsupported mesh type - colorized cylinder shell with simplex is not supported.");
265  }
266  else if (dim != 3)
267  {
268  throw std::runtime_error(
269  "Unsupported mesh type - colorized cylinder shell is only supported in 3d.");
270  }
271 
272  // First generate the regular deal.II cylinder_shell using the
273  // mesh_parameters arguments
274  GridGenerator::generate_from_name_and_arguments(
275  triangulation, "cylinder_shell", mesh_parameters.grid_arguments);
276 
277  // Now that we have our grid, colorize the boundary conditions.
278  // Inner cylinder has boundary id 0
279  // Outer cylinder has boundary id 1
280  // Bottom (Z-) part of the cylinder has boundary id 2
281  // Top (Z+) part of the cylinder has boundary id 3
282 
283  // Split the argument to extract the radius
284  std::vector<std::string> split_arguments =
285  Utilities::split_string_list(mesh_parameters.grid_arguments, ":");
286 
287  const double length = Utilities::string_to_double(split_arguments[0]);
288 
289  // Tolerance along z
290  const double eps_z = 1e-6 * length;
291 
292  // Gather the inner radius from the faces instead of the argument, this is
293  // more robust for some aspect ratios. First initialize the outer to 0 and
294  // the inner to a large value
295  double inner_radius = DBL_MAX;
296  double outer_radius = 0.;
297 
298  // Loop over the cells once to acquire the min and max radius at the face
299  // centers Otherwise, for some cell ratio, the center of the faces can be
300  // at a radius which is significantly different from the one prescribed.
301  for (const auto &cell : triangulation.active_cell_iterators())
302  for (const unsigned int f : GeometryInfo<3>::face_indices())
303  {
304  if (!cell->face(f)->at_boundary())
305  continue;
306 
307  const auto face_center = cell->face(f)->center();
308  const double z = face_center[2];
309 
310  if ((std::fabs(z) > eps_z) &&
311  (std::fabs(z - length) > eps_z)) // Not a zmin or zmax boundary
312  {
313  const double radius =
314  std::sqrt(face_center[0] * face_center[0] +
315  face_center[1] * face_center[1]);
316  inner_radius = std::min(inner_radius, radius);
317  outer_radius = std::max(outer_radius, radius);
318  }
319  }
320 
321  // Use the gathered radius to define the medium radial distance.
322  double mid_radial_distance = 0.5 * (outer_radius - inner_radius);
323 
324  for (const auto &cell : triangulation.active_cell_iterators())
325  for (const unsigned int f : GeometryInfo<3>::face_indices())
326  {
327  if (!cell->face(f)->at_boundary())
328  continue;
329 
330  const auto face_center = cell->face(f)->center();
331 
332  const double radius = std::sqrt(face_center[0] * face_center[0] +
333  face_center[1] * face_center[1]);
334 
335  const double z = face_center[2];
336 
337  if (std::fabs(z) < eps_z) // z = 0 set boundary 2
338  {
339  cell->face(f)->set_boundary_id(2);
340  }
341  else if (std::fabs(z - length) < eps_z) // z = length set boundary 3
342  {
343  cell->face(f)->set_boundary_id(3);
344  }
345  else if (std::fabs(radius - inner_radius) >
346  mid_radial_distance) // r = outer_radius set boundary 1
347  {
348  cell->face(f)->set_boundary_id(1);
349  }
350  else if (std::fabs(radius - inner_radius) <
351  mid_radial_distance) // r = inner_radius set boundary 0
352  {
353  cell->face(f)->set_boundary_id(0);
354  }
355  else
356  throw std::runtime_error(
357  "There was an error while setting up this mesh");
358  }
359  }
360 
361  // Periodic Hills grid
362  else if (mesh_parameters.type == Parameters::Mesh::Type::periodic_hills &&
363  !mesh_parameters.simplex)
364  {
365  if (mesh_parameters.simplex)
366  {
367  throw std::runtime_error(
368  "Unsupported mesh type - periodic hills mesh with simplex is not supported");
369  }
370  else
371  {
372  PeriodicHillsGrid<dim, spacedim> grid(mesh_parameters.grid_arguments);
373  grid.make_grid(triangulation);
374  }
375  }
376  else
377  throw std::runtime_error(
378  "Unsupported mesh type - mesh will not be created");
379 }

382 template <int dim, int spacedim>
383 void
384 setup_periodic_boundary_conditions(
385  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
386  const BoundaryConditions::BoundaryConditions<spacedim> &boundary_conditions)
387 
388 {
389  // Setup periodic boundary conditions
390  for (unsigned int i_bc = 0; i_bc < boundary_conditions.size; ++i_bc)
391  {
392  if (boundary_conditions.type[i_bc] ==
393  BoundaryConditions::BoundaryType::periodic)
394  {
395  auto triangulation_ptr = &triangulation;
396 
397  std::vector<GridTools::PeriodicFacePair<
398  typename Triangulation<dim, spacedim>::cell_iterator>>
399  periodicity_vector;
400  GridTools::collect_periodic_faces(
401  *dynamic_cast<Triangulation<dim, spacedim> *>(triangulation_ptr),
402  boundary_conditions.id[i_bc],
403  boundary_conditions.periodic_id[i_bc],
404  boundary_conditions.periodic_direction[i_bc],
405  periodicity_vector);
406  triangulation.add_periodicity(periodicity_vector);
407  }
408  }

411 template <int dim, int spacedim>
412 void
413 read_mesh_and_manifolds(
414  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation,
415  const Parameters::Mesh &mesh_parameters,
416  const Parameters::Manifolds &manifolds_parameters,
417  const bool &restart,
418  const BoundaryConditions::BoundaryConditions<spacedim> &boundary_conditions)
419 {
420  attach_grid_to_triangulation(triangulation, mesh_parameters);
421 
422  setup_periodic_boundary_conditions(triangulation, boundary_conditions);
423 
424  // If the mesh is a GMSH mesh, we need to manually set the manifold id
425  // of the face to be that of the boundary
426  // ID, otherwise, we will be unable to use external manifold.
427  // This is done manually by looping through all faces and giving them a
428  // manifold id if there is a manifold associated with this number.
429  if (mesh_parameters.type == Parameters::Mesh::Type::gmsh)
430  {
431  // Gather all the manifold ids within a set
432  std::set<int> manifold_ids;
433  for (unsigned int i = 0; i < manifolds_parameters.size; ++i)
434  manifold_ids.insert(manifolds_parameters.id[i]);
435 
436  // Reset all the manifolds manually and force them to zero
437  triangulation.reset_all_manifolds();
438 
439  // If the parameter file forces the occurrence of manifold,
440  // loop over the faces of the triangulation. If the face of the
441  // triangulation has a boundary id which corresponds to a manifold id
442  // identified within the parameter file, then fix the manifold id of this
443  // face manually to be that of the boundary id. In the past, this was done
444  // by default for every face, but since 2023-12 this throws (rightfully)
445  // an error in deal.II
446  if (manifolds_parameters.size > 0)
447  {
448  for (const auto &face : triangulation.active_face_iterators())
449  {
450  if (face->at_boundary() &&
451  manifold_ids.find(face->boundary_id()) != manifold_ids.end())
452  face->set_all_manifold_ids(face->boundary_id());
453  }
454  }
455  }
456 
457  // Finally attach the manifolds to the triangulation
458  // Right now this should only occur for GMSH mesh, but the function is generic
459  // enough.
460  attach_manifolds_to_triangulation(triangulation, manifolds_parameters);
461 
462  if (mesh_parameters.simplex)
463  {
464  // Refinement isn't possible yet
465  }
466  else
467  {
468  if (mesh_parameters.refine_until_target_size)
469  {
470  double minimal_cell_size =
471  GridTools::minimal_cell_diameter(triangulation);
472  double target_size = mesh_parameters.target_size;
473  unsigned int number_refinement =
474  floor(std::log(minimal_cell_size / target_size) / std::log(2));
475  triangulation.refine_global(number_refinement);
476  }
477  else if (!restart)
478  {
479  const int initial_refinement = mesh_parameters.initial_refinement;
480  triangulation.refine_global(initial_refinement);
481  refine_triangulation_at_boundaries(
482  mesh_parameters.boundaries_to_refine,
483  mesh_parameters.initial_refinement_at_boundaries,
484  triangulation);
485  }
486  }
487 }
