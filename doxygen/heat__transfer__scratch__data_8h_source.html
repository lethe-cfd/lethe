19  * Equation solved:
20  * rho * Cp * (dT/dt + u.gradT) = k div(gradT) + nu/rho * (gradu : gradu)
21  *
22  * Polytechnique Montreal, 2020-

52 /**
53  * @brief Class that stores the information required by the assembly procedure
54  * for a heat transfer advection-diffusion equation. Consequently, this class
55  * calculates the heat transfer (values, gradients, laplacians) and the shape
56  * function (values, gradients, laplacians) at all the gauss points for all
57  * degrees of freedom and stores it into arrays. Additionnaly, the user can
58  * request that this class gathers additional fields for physics which are
59  * coupled to the heat transfer equation, such as the velocity which is
60  * required. This class serves as a seperation between the evaluation at the
61  * gauss point of the variables of interest and their use in the assembly, which
62  * is carried out by the assembler methods. For more information on this
63  * design, the reader can consult deal.II step-9
64  * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
65  * example, the scratch is a struct instead of a templated class because of the
66  * simplicity of step-9.
67  *
68  * @tparam dim An integer that denotes the dimension of the space in which
69  * the flow is solved
70  *
71  * @ingroup solvers
72  **/
73

77 public:

79  * @brief Constructor. The constructor creates the fe_values that will be used
80  * to fill the member variables of the scratch. It also allocated the
81  * necessary memory for all member variables. However, it does not do any
82  * evalution, since this needs to be done at the cell level.
83  *
84  * @param properties_manager The physical properties Manager (see physical_properties_manager.h)
85  *
86  * @param fe_ht The FESystem used to solve the Heat Transfer equations
87  *
88  * @param quadrature The quadrature to use for the assembly
89  *
90  * @param mapping The mapping of the domain in which the Navier-Stokes equations are solved
91  *
92  * @param fe_fd The FESystem used to solve the Fluid Dynamics equations
93  *
94  * @param face_quadrature_formula The face quadrature formula
95  *
96  */

165  /** @brief Reinitialize the content of the scratch
166  *
167  * Using the FeValues and the content of the solutions, previous solutions ,
168  * fills all of the class member of the scratch
169  *
170  * @tparam VectorType The Vector type used for the solvers
171  *
172  * @param cell The cell over which the assembly is being carried.
173  * This cell must be compatible with the fe which is used to fill the FeValues
174  *
175  * @param current_solution The present value of the solution for the Heat Transfer
176  *
177  * @param previous_solutions The solutions at the previous time steps
178  *
179  * @param source_function The function describing the Heat Transfer source term
180  */
181 
182  template <typename VectorType>
183  void
184  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
185  const VectorType &current_solution,
186  const std::vector<GlobalVectorType> &previous_solutions,
187  Function<dim> *source_function)
188  {
189  material_id = cell->material_id();
190  this->fe_values_T.reinit(cell);
191 
192  quadrature_points = this->fe_values_T.get_quadrature_points();
193  auto &fe_T = this->fe_values_T.get_fe();
194 
195  source_function->value_list(quadrature_points, source);
196 
197  if (dim == 2)
198  this->cell_size = std::sqrt(4. * cell->measure() / M_PI) / fe_T.degree;
199  else if (dim == 3)
200  this->cell_size = pow(6 * cell->measure() / M_PI, 1. / 3.) / fe_T.degree;
201 
202  // Gather temperature (values, gradient and laplacian)
203  this->fe_values_T.get_function_values(current_solution,
204  this->present_temperature_values);
205  this->fe_values_T.get_function_gradients(current_solution,
206  this->temperature_gradients);
207  this->fe_values_T.get_function_laplacians(
208  current_solution, this->present_temperature_laplacians);
209 
210  // Gather previous temperature values
211  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
212  {
213  this->fe_values_T.get_function_values(previous_solutions[p],
214  previous_temperature_values[p]);
215 
216  this->fe_values_T.get_function_gradients(
217  previous_solutions[p], previous_temperature_gradients[p]);
218  }
219 
220  for (unsigned int q = 0; q < n_q_points; ++q)
221  {
222  this->JxW[q] = this->fe_values_T.JxW(q);
223 
224  for (unsigned int k = 0; k < n_dofs; ++k)
225  {
226  // Shape function
227  this->phi_T[q][k] = this->fe_values_T.shape_value(k, q);
228  this->grad_phi_T[q][k] = this->fe_values_T.shape_grad(k, q);
229  this->hess_phi_T[q][k] = this->fe_values_T.shape_hessian(k, q);
230  this->laplacian_phi_T[q][k] = trace(this->hess_phi_T[q][k]);
231  }
232  }
233 
234 
235  // Arrays related to faces must be re-initialized for each cell, since they
236  // might depend on reference cell
237  // Only carry out this initialization if the cell is a boundary cell,
238  // otherwise these are wasted calculations
239  this->is_boundary_cell = cell->at_boundary();
240  if (cell->at_boundary())
241  {
242  n_faces = cell->n_faces();
243  is_boundary_face = std::vector<bool>(n_faces, false);
244  n_faces_q_points = fe_face_values_ht.get_quadrature().size();
245  boundary_face_id = std::vector<unsigned int>(n_faces);
246 
247  face_JxW = std::vector<std::vector<double>>(
248  n_faces, std::vector<double>(n_faces_q_points));
249 
250 
251  this->phi_face_T = std::vector<std::vector<std::vector<double>>>(
252  n_faces,
253  std::vector<std::vector<double>>(n_faces_q_points,
254  std::vector<double>(n_dofs)));
255 
256  this->temperature_face_value = std::vector<std::vector<double>>(
257  n_faces, std::vector<double>(n_faces_q_points));
258 
259  for (const auto face : cell->face_indices())
260  {
261  this->is_boundary_face[face] = cell->face(face)->at_boundary();
262  if (this->is_boundary_face[face])
263  {
264  fe_face_values_ht.reinit(cell, face);
265  boundary_face_id[face] = cell->face(face)->boundary_id();
266  this->fe_face_values_ht.get_function_values(
267  current_solution, this->temperature_face_value[face]);
268 
269  for (unsigned int q = 0; q < n_faces_q_points; ++q)
270  {
271  face_JxW[face][q] = fe_face_values_ht.JxW(q);
272  for (const unsigned int k : fe_face_values_ht.dof_indices())
273  {
274  this->phi_face_T[face][q][k] =
275  this->fe_face_values_ht.shape_value(k, q);
276  }
277  }
278  }
279  }
280  }

283  /** @brief Reinitialize the velocity, calculated by the fluid dynamics while also taking into account ALE
284  *
285  * @tparam VectorType The Vector type used for the solvers
286  *
287  * @param cell The cell for which the velocity is reinitialized
288  * This cell must be compatible with the Fluid Dynamics FE
289  *
290  * @param current_solution The present value of the solution for [u,p]
291  *
292  * @param ale The ALE parameters which include the ALE function
293  *
294  */
295 
296  template <typename VectorType>
297  void
298  reinit_velocity(const typename DoFHandler<dim>::active_cell_iterator &cell,
299  const VectorType &current_solution,
300  const Parameters::ALE<dim> &ale)
301  {
302  this->fe_values_fd.reinit(cell);
303 
304  this->fe_values_fd[velocities].get_function_values(current_solution,
305  velocity_values);
306 
307  if (!ale.enabled())
308  return;
309 
310  // ALE enabled, so extract the ALE velocity and subtract it from the
311  // velocity obtained from the fluid dynamics
312  Tensor<1, dim> velocity_ale;
313  std::shared_ptr<Functions::ParsedFunction<dim>> velocity_ale_function =
314  ale.velocity;
315  Vector<double> velocity_ale_vector(dim);
316 
317  for (unsigned int q = 0; q < n_q_points; ++q)
318  {
319  velocity_ale_function->vector_value(quadrature_points[q],
320  velocity_ale_vector);
321  for (unsigned int d = 0; d < dim; ++d)
322  velocity_ale[d] = velocity_ale_vector[d];
323 
324  velocity_values[q] -= velocity_ale;
325  }
326  }
