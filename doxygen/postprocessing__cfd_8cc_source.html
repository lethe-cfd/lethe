37 std::pair<double, double>
38 calculate_pressure_drop(const DoFHandler<dim> &dof_handler,
39  std::shared_ptr<Mapping<dim>> mapping,
40  const VectorType &evaluation_point,
41  const Quadrature<dim> &cell_quadrature_formula,
42  const Quadrature<dim - 1> &face_quadrature_formula,
43  const unsigned int inlet_boundary_id,
44  const unsigned int outlet_boundary_id)
45 {
46  FESystem<dim, dim> fe = dof_handler.get_fe();
47  FEValues<dim> fe_values(*mapping,
48  fe,
49  cell_quadrature_formula,
50  update_values | update_quadrature_points |
51  update_JxW_values | update_gradients |
52  update_hessians);
53  FEFaceValues<dim> fe_face_values(fe,
54  face_quadrature_formula,
55  update_values | update_quadrature_points |
56  update_normal_vectors | update_JxW_values);
57 
58  const FEValuesExtractors::Vector velocities(0);
59  const FEValuesExtractors::Scalar pressure(dim);
60 
61  const unsigned int face_n_q_points = face_quadrature_formula.size();
62 
63  std::vector<Tensor<1, dim>> velocity_values(face_n_q_points);
64  std::vector<double> pressure_values(face_n_q_points);
65 
66  double static_pressure_outlet_boundary = 0.;
67  double static_pressure_inlet_boundary = 0.;
68  double static_pressure_drop = 0.;
69 
70  double dynamic_pressure_outlet_boundary = 0.;
71  double dynamic_pressure_inlet_boundary = 0.;
72  double dynamic_pressure_drop = 0.;
73 
74  double outlet_surface = 0.;
75  double inlet_surface = 0.;
76 
77  double temp_surface = 0.;
78  double temp_static_pressure_at_boundary = 0.;
79  double temp_dynamic_pressure_at_boundary = 0.;
80 
81  for (const auto &cell : dof_handler.active_cell_iterators())
82  {
83  if (cell->is_locally_owned())
84  {
85  for (unsigned int face_id = 0;
86  face_id < GeometryInfo<dim>::faces_per_cell;
87  face_id++)
88  {
89  if (cell->face(face_id)->at_boundary())
90  {
91  unsigned int boundary_id = cell->face(face_id)->boundary_id();
92  if (boundary_id != inlet_boundary_id &&
93  boundary_id != outlet_boundary_id)
94  continue;
95  fe_face_values.reinit(cell, face_id);
96  fe_face_values[velocities].get_function_values(
97  evaluation_point, velocity_values);
98  fe_face_values[pressure].get_function_values(evaluation_point,
99  pressure_values);
100  temp_surface = 0.;
101  temp_static_pressure_at_boundary = 0.;
102  temp_dynamic_pressure_at_boundary = 0.;
103  for (unsigned int q = 0; q < face_n_q_points; q++)
104  {
105  temp_surface += fe_face_values.JxW(q);
106  // Integration of velocity^2 and
107  // pressure, since the pressure in Lethe has units of
108  // Length^2/Time^2
109  temp_static_pressure_at_boundary +=
110  fe_face_values.JxW(q) * pressure_values[q];
111  temp_dynamic_pressure_at_boundary +=
112  0.5 * fe_face_values.JxW(q) * velocity_values[q] *
113  velocity_values[q];
114  }
115  if (boundary_id == inlet_boundary_id)
116  {
117  inlet_surface += temp_surface;
118  dynamic_pressure_inlet_boundary +=
119  temp_dynamic_pressure_at_boundary;
120  static_pressure_inlet_boundary +=
121  temp_static_pressure_at_boundary;
122  }
123  if (boundary_id == outlet_boundary_id)
124  {
125  outlet_surface += temp_surface;
126  dynamic_pressure_outlet_boundary +=
127  temp_dynamic_pressure_at_boundary;
128  static_pressure_outlet_boundary +=
129  temp_static_pressure_at_boundary;
130  }
131  }
132  }
133  }
134  }
135 
136  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
137  static_pressure_inlet_boundary =
138  Utilities::MPI::sum(static_pressure_inlet_boundary, mpi_communicator);
139  static_pressure_outlet_boundary =
140  Utilities::MPI::sum(static_pressure_outlet_boundary, mpi_communicator);
141 
142  dynamic_pressure_inlet_boundary =
143  Utilities::MPI::sum(dynamic_pressure_inlet_boundary, mpi_communicator);
144  dynamic_pressure_outlet_boundary =
145  Utilities::MPI::sum(dynamic_pressure_outlet_boundary, mpi_communicator);
146 
147  inlet_surface = Utilities::MPI::sum(inlet_surface, mpi_communicator);
148  outlet_surface = Utilities::MPI::sum(outlet_surface, mpi_communicator);
149 
150  static_pressure_outlet_boundary =
151  static_pressure_outlet_boundary / outlet_surface;
152  static_pressure_inlet_boundary =
153  static_pressure_inlet_boundary / inlet_surface;
154  static_pressure_drop =
155  static_pressure_inlet_boundary - static_pressure_outlet_boundary;
156 
157  dynamic_pressure_outlet_boundary =
158  dynamic_pressure_outlet_boundary / outlet_surface;
159  dynamic_pressure_inlet_boundary =
160  dynamic_pressure_inlet_boundary / inlet_surface;
161  dynamic_pressure_drop =
162  dynamic_pressure_inlet_boundary - dynamic_pressure_outlet_boundary;
163 
164  return {static_pressure_drop, static_pressure_drop + dynamic_pressure_drop};
165 }

229 template <int dim, typename VectorType>
230 double
231 calculate_CFL(const DoFHandler<dim> &dof_handler,
232  const VectorType &evaluation_point,
233  const double time_step,
234  const Quadrature<dim> &quadrature_formula,
235  const Mapping<dim> &mapping)
236 {
237  FESystem<dim, dim> fe = dof_handler.get_fe();
238  FEValues<dim> fe_values(mapping,
239  fe,
240  quadrature_formula,
241  update_values | update_quadrature_points |
242  update_JxW_values);
243 
244  const FEValuesExtractors::Vector velocities(0);
245  const unsigned int n_q_points = quadrature_formula.size();
246 
247 
248  std::vector<Tensor<1, dim>> present_velocity_values(n_q_points);
249 
250  // Element size
251  double h;
252 
253  // Element degree
254  double degree = double(fe.degree);
255 
256  // CFL
257  double CFL = 0;
258 
259  for (const auto &cell : dof_handler.active_cell_iterators())
260  {
261  if (cell->is_locally_owned())
262  {
263  if (dim == 2)
264  h = std::sqrt(4. * cell->measure() / M_PI) / degree;
265  else if (dim == 3)
266  h = pow(6 * cell->measure() / M_PI, 1. / 3.) / degree;
267  fe_values.reinit(cell);
268  fe_values[velocities].get_function_values(evaluation_point,
269  present_velocity_values);
270  for (unsigned int q = 0; q < n_q_points; ++q)
271  {
272  const double localCFL =
273  present_velocity_values[q].norm() / h * time_step;
274  CFL = std::max(CFL, localCFL);
275  }
276  }
277  }
278  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
279  CFL = Utilities::MPI::max(CFL, mpi_communicator);
280  return (CFL);
281 }

332 template <int dim, typename VectorType>
333 double
334 calculate_enstrophy(const DoFHandler<dim> &dof_handler,
335  const VectorType &evaluation_point,
336  const Quadrature<dim> &quadrature_formula,
337  const Mapping<dim> &mapping)
338 {
339  const FESystem<dim, dim> fe = dof_handler.get_fe();
340  FEValues<dim> fe_values(mapping,
341  fe,
342  quadrature_formula,
343  update_values | update_gradients |
344  update_quadrature_points | update_JxW_values);
345 
346  const FEValuesExtractors::Vector velocities(0);
347 
348  const unsigned int n_q_points = quadrature_formula.size();
349 
350  std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
351  double en = 0.0;
352  double domain_volume =
353  GridTools::volume(dof_handler.get_triangulation(), mapping);
354 
355  for (const auto &cell : dof_handler.active_cell_iterators())
356  {
357  if (cell->is_locally_owned())
358  {
359  fe_values.reinit(cell);
360 
361  fe_values[velocities].get_function_gradients(
362  evaluation_point, present_velocity_gradients);
363 
364  for (unsigned int q = 0; q < n_q_points; q++)
365  {
366  // Find the values of gradient of ux and uy (the finite element
367  // solution) at the quadrature points
368  double ux_y = present_velocity_gradients[q][0][1];
369  double uy_x = present_velocity_gradients[q][1][0];
370 
371  if (dim == 2)
372  {
373  en += 0.5 * (uy_x - ux_y) * (uy_x - ux_y) * fe_values.JxW(q) /
374  domain_volume;
375  }
376  else
377  {
378  double uz_y = present_velocity_gradients[q][2][1];
379  double uy_z = present_velocity_gradients[q][1][2];
380  double ux_z = present_velocity_gradients[q][0][2];
381  double uz_x = present_velocity_gradients[q][2][0];
382  en += 0.5 * (uz_y - uy_z) * (uz_y - uy_z) * fe_values.JxW(q) /
383  domain_volume;
384  en += 0.5 * (ux_z - uz_x) * (ux_z - uz_x) * fe_values.JxW(q) /
385  domain_volume;
386  en += 0.5 * (uy_x - ux_y) * (uy_x - ux_y) * fe_values.JxW(q) /
387  domain_volume;
388  }
389  }
390  }
391  }
392  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
393  en = Utilities::MPI::sum(en, mpi_communicator);
394  return (en);
395 }

538 template <int dim, typename VectorType>
539 double
540 calculate_apparent_viscosity(const DoFHandler<dim> &dof_handler,
541  const VectorType &evaluation_point,
542  const Quadrature<dim> &quadrature_formula,
543  const Mapping<dim> &mapping,
544  PhysicalPropertiesManager &properties_manager)
545 {
546  double integral_viscosity_x_shear_rate = 0;
547  double integral_shear_rate = 0;
548  double shear_rate_magnitude;
549  Tensor<2, dim> shear_rate;
550  double kinematic_viscosity;
551  const auto rheological_model = properties_manager.get_rheology();
552 
553  const FESystem<dim, dim> fe = dof_handler.get_fe();
554  FEValues<dim> fe_values(mapping,
555  fe,
556  quadrature_formula,
557  update_values | update_gradients |
558  update_quadrature_points | update_JxW_values);
559 
560  const FEValuesExtractors::Vector velocities(0);
561 
562  const unsigned int n_q_points = quadrature_formula.size();
563 
564  std::vector<Tensor<2, dim>> present_velocity_gradients(n_q_points);
565 
566  for (const auto &cell : dof_handler.active_cell_iterators())
567  {
568  if (cell->is_locally_owned())
569  {
570  fe_values.reinit(cell);
571 
572  fe_values[velocities].get_function_gradients(
573  evaluation_point, present_velocity_gradients);
574 
575  for (unsigned int q = 0; q < n_q_points; q++)
576  {
577  shear_rate = present_velocity_gradients[q] +
578  transpose(present_velocity_gradients[q]);
579 
580  double shear_rate_x_velocity_gradient = 0;
581  for (int i = 0; i < dim; ++i)
582  {
583  for (int j = 0; j < dim; ++j)
584  {
585  shear_rate_x_velocity_gradient +=
586  shear_rate[i][j] * present_velocity_gradients[q][j][i];
587  }
588  }
589 
590  shear_rate_magnitude = calculate_shear_rate_magnitude(shear_rate);
591 
592  std::map<field, double> field_values;
593  field_values[field::shear_rate] = shear_rate_magnitude;
594 
595  kinematic_viscosity = rheological_model->value(field_values);
596 
597  integral_viscosity_x_shear_rate +=
598  kinematic_viscosity * shear_rate_x_velocity_gradient *
599  fe_values.JxW(q);
600  integral_shear_rate +=
601  shear_rate_x_velocity_gradient * fe_values.JxW(q);
602  }
603  }
604  }
605  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
606  integral_viscosity_x_shear_rate =
607  Utilities::MPI::sum(integral_viscosity_x_shear_rate, mpi_communicator);
608  integral_shear_rate =
609  Utilities::MPI::sum(integral_shear_rate, mpi_communicator);
610  return integral_viscosity_x_shear_rate / integral_shear_rate;
611 }

663 template <int dim, typename VectorType>
664 std::vector<std::vector<Tensor<1, dim>>>
665 calculate_forces(
666  const DoFHandler<dim> &dof_handler,
667  const VectorType &evaluation_point,
668  PhysicalPropertiesManager &properties_manager,
669  const BoundaryConditions::NSBoundaryConditions<dim> &boundary_conditions,
670  const Quadrature<dim - 1> &face_quadrature_formula,
671  const Mapping<dim> &mapping)
672 {
673  const FESystem<dim, dim> fe = dof_handler.get_fe();
674 
675  // Rheological model for viscosity properties
676  double kinematic_viscosity;
677  const auto rheological_model = properties_manager.get_rheology();
678 
679 
680  const unsigned int n_q_points = face_quadrature_formula.size();
681  const FEValuesExtractors::Vector velocities(0);
682  const FEValuesExtractors::Scalar pressure(dim);
683  std::vector<double> pressure_values(n_q_points);
684  std::vector<Tensor<2, dim>> velocity_gradients(n_q_points);
685  Tensor<1, dim> normal_vector;
686  Tensor<2, dim> shear_rate;
687  Tensor<2, dim> fluid_stress;
688  Tensor<2, dim> fluid_viscous_stress;
689  Tensor<2, dim> fluid_pressure;
690 
691  std::vector<Tensor<1, dim>> viscous_force_vector(boundary_conditions.size);
692  std::vector<Tensor<1, dim>> pressure_force_vector(boundary_conditions.size);
693  std::vector<Tensor<1, dim>> force_vector(boundary_conditions.size);
694 
695 
696  FEFaceValues<dim> fe_face_values(mapping,
697  fe,
698  face_quadrature_formula,
699  update_values | update_quadrature_points |
700  update_gradients | update_JxW_values |
701  update_normal_vectors);
702  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
703 
704  for (const auto &cell : dof_handler.active_cell_iterators())
705  {
706  if (cell->is_locally_owned())
707  {
708  if (cell->at_boundary())
709  {
710  for (const auto face : cell->face_indices())
711  {
712  if (cell->face(face)->at_boundary())
713  {
714  const auto boundary_id =
715  std::find(begin(boundary_conditions.id),
716  end(boundary_conditions.id),
717  cell->face(face)->boundary_id());
718 
719 
720  if (boundary_id != end(boundary_conditions.id))
721  {
722  unsigned int vector_index =
723  boundary_id - begin(boundary_conditions.id);
724  fe_face_values.reinit(cell, face);
725 
726  std::vector<Point<dim>> q_points =
727  fe_face_values.get_quadrature_points();
728  fe_face_values[velocities].get_function_gradients(
729  evaluation_point, velocity_gradients);
730  fe_face_values[pressure].get_function_values(
731  evaluation_point, pressure_values);
732  for (unsigned int q = 0; q < n_q_points; q++)
733  {
734  normal_vector = -fe_face_values.normal_vector(q);
735  for (int d = 0; d < dim; ++d)
736  {
737  fluid_pressure[d][d] = pressure_values[q];
738  }
739  shear_rate = velocity_gradients[q] +
740  transpose(velocity_gradients[q]);
741 
742  const double shear_rate_magnitude =
743  calculate_shear_rate_magnitude(shear_rate);
744 
745  std::map<field, double> field_values;
746  field_values[field::shear_rate] =
747  shear_rate_magnitude;
748 
749  kinematic_viscosity =
750  rheological_model->value(field_values);
751  fluid_viscous_stress =
752  -kinematic_viscosity * shear_rate;
753  fluid_stress =
754  -fluid_viscous_stress - fluid_pressure;
755 
756  viscous_force_vector[vector_index] -=
757  fluid_viscous_stress * normal_vector *
758  fe_face_values.JxW(q);
759  pressure_force_vector[vector_index] -=
760  fluid_pressure * normal_vector *
761  fe_face_values.JxW(q);
762  force_vector[vector_index] +=
763  fluid_stress * normal_vector *
764  fe_face_values.JxW(q);
765  }
766  }
767  }
768  }
769  }
770  }
771  }
772 
773  for (unsigned int i_bc = 0; i_bc < boundary_conditions.size; ++i_bc)
774  {
775  viscous_force_vector[i_bc] =
776  Utilities::MPI::sum(viscous_force_vector[i_bc], mpi_communicator);
777  pressure_force_vector[i_bc] =
778  Utilities::MPI::sum(pressure_force_vector[i_bc], mpi_communicator);
779  force_vector[i_bc] =
780  Utilities::MPI::sum(force_vector[i_bc], mpi_communicator);
781  }
782  std::vector<std::vector<Tensor<1, dim>>> forces{force_vector,
783  viscous_force_vector,
784  pressure_force_vector};
785  return forces;
786 }

862  const FEValuesExtractors::Scalar pressure(dim);
863  std::vector<double> pressure_values(n_q_points);
864  std::vector<Tensor<2, dim>> velocity_gradients(n_q_points);
865  Tensor<1, dim> normal_vector;
866  Tensor<2, dim> shear_rate;
867  Tensor<2, dim> fluid_stress;
868  Tensor<2, dim> fluid_pressure;
869  // torque tensor had to be considered in 3D at all time...

875  fe,

882  {
883  unsigned int boundary_id = boundary_conditions.id[i_bc];
884  torque = 0;
885  Point<dim> center_of_rotation =
886  boundary_conditions.bcFunctions[boundary_id].center_of_rotation;
887  for (const auto &cell : dof_handler.active_cell_iterators())
888  {
889  if (cell->is_locally_owned())
890  {
891  for (const auto face : cell->face_indices())
892  {
893  if (cell->face(face)->at_boundary())
894  {
895  fe_face_values.reinit(cell, face);
896  if (cell->face(face)->boundary_id() == boundary_id)
897  {
898  std::vector<Point<dim>> q_points =
899  fe_face_values.get_quadrature_points();
900  fe_face_values[velocities].get_function_gradients(
901  evaluation_point, velocity_gradients);
902  fe_face_values[pressure].get_function_values(
903  evaluation_point, pressure_values);
904  for (unsigned int q = 0; q < n_q_points; q++)
905  {
906  normal_vector = -fe_face_values.normal_vector(q);
907  for (int d = 0; d < dim; ++d)
908  {
909  fluid_pressure[d][d] = pressure_values[q];
910  }
911  shear_rate = velocity_gradients[q] +
912  transpose(velocity_gradients[q]);
913  const double shear_rate_magnitude =
914  calculate_shear_rate_magnitude(shear_rate);
915 
916  std::map<field, double> field_values;
917  field_values[field::shear_rate] =
918  shear_rate_magnitude;
919 
920  kinematic_viscosity =
921  rheological_model->value(field_values);
922 
923  fluid_stress = kinematic_viscosity * shear_rate -
924  fluid_pressure;
925  auto force = fluid_stress * normal_vector *
926  fe_face_values.JxW(q);
927 
928  auto distance = q_points[q] - center_of_rotation;
929  if (dim == 2)
930  {
931  torque[0] = 0.;
932  torque[1] = 0.;
933  torque[2] += distance[0] * force[1] -
934  distance[1] * force[0];
935  }
936  else if (dim == 3)
937  {
938  torque[0] += distance[1] * force[2] -
939  distance[2] * force[1];
940  torque[1] += distance[2] * force[0] -
941  distance[0] * force[2];
942  torque[2] += distance[0] * force[1] -
943  distance[1] * force[0];
944  }
945  }
946  }
947  }
948  }
949  }
950  }
951  torque_vector[i_bc] = Utilities::MPI::sum(torque, mpi_communicator);
952  }
953  return torque_vector;

1011 // Find the l2 norm of the error between the finite element sol'n and the exact
1012 // sol'n for both the velocity and the pressure
1013 // Mean pressure is removed from both the analytical and the simulation solution
1014 template <int dim, typename VectorType>
1015 std::pair<double, double>
1016 calculate_L2_error(const DoFHandler<dim> &dof_handler,
1017  const VectorType &evaluation_point,
1018  const Function<dim> *exact_solution,
1019  const Quadrature<dim> &quadrature_formula,
1020  const Mapping<dim> &mapping)
1021 {
1022  const FESystem<dim, dim> fe = dof_handler.get_fe();
1023  FEValues<dim> fe_values(mapping,
1024  fe,
1025  quadrature_formula,
1026  update_values | update_gradients |
1027  update_quadrature_points | update_JxW_values);
1028 
1029  const FEValuesExtractors::Vector velocities(0);
1030  const FEValuesExtractors::Scalar pressure(dim);
1031  const unsigned int n_q_points = quadrature_formula.size();
1032 
1033  std::vector<Vector<double>> q_exactSol(n_q_points, Vector<double>(dim + 1));
1034 
1035  std::vector<Tensor<1, dim>> local_velocity_values(n_q_points);
1036  std::vector<double> local_pressure_values(n_q_points);
1037 
1038  double pressure_integral = 0;
1039  double exact_pressure_integral = 0;
1040 
1041  // loop over elements to calculate average pressure
1042  for (const auto &cell : dof_handler.active_cell_iterators())
1043  {
1044  if (cell->is_locally_owned())
1045  {
1046  fe_values.reinit(cell);
1047 
1048  fe_values[pressure].get_function_values(evaluation_point,
1049  local_pressure_values);
1050  // Get the exact solution at all gauss points
1051  exact_solution->vector_value_list(fe_values.get_quadrature_points(),
1052  q_exactSol);
1053 
1054  for (unsigned int q = 0; q < n_q_points; q++)
1055  {
1056  pressure_integral += local_pressure_values[q] * fe_values.JxW(q);
1057  exact_pressure_integral += q_exactSol[q][dim] * fe_values.JxW(q);
1058  }
1059  }
1060  }
1061  const MPI_Comm mpi_communicator = dof_handler.get_communicator();
1062  pressure_integral = Utilities::MPI::sum(pressure_integral, mpi_communicator);
1063  exact_pressure_integral =
1064  Utilities::MPI::sum(exact_pressure_integral, mpi_communicator);
1065 
1066  // double global_volume =
1067  // GridTools::volume(dof_handler.get_triangulation(),*mapping);
1068  double global_volume =
1069  GridTools::volume(dof_handler.get_triangulation(), mapping);
1070  double average_pressure = pressure_integral / global_volume;
1071  double average_exact_pressure = exact_pressure_integral / global_volume;
1072 
1073 
1074  double l2errorU = 0.;
1075  double l2errorP = 0.;
1076 
1077  for (const auto &cell : dof_handler.active_cell_iterators())
1078  {
1079  if (cell->is_locally_owned())
1080  {
1081  fe_values.reinit(cell);
1082  fe_values[velocities].get_function_values(evaluation_point,
1083  local_velocity_values);
1084  fe_values[pressure].get_function_values(evaluation_point,
1085  local_pressure_values);
1086 
1087  // Get the exact solution at all gauss points
1088  exact_solution->vector_value_list(fe_values.get_quadrature_points(),
1089  q_exactSol);
1090 
1091  for (unsigned int q = 0; q < n_q_points; q++)
1092  {
1093  // Find the values of x and u_h (the finite element solution) at
1094  // the quadrature points
1095  double ux_sim = local_velocity_values[q][0];
1096  double ux_exact = q_exactSol[q][0];
1097 
1098  double uy_sim = local_velocity_values[q][1];
1099  double uy_exact = q_exactSol[q][1];
1100 
1101  l2errorU +=
1102  (ux_sim - ux_exact) * (ux_sim - ux_exact) * fe_values.JxW(q);
1103  l2errorU +=
1104  (uy_sim - uy_exact) * (uy_sim - uy_exact) * fe_values.JxW(q);
1105 
1106  if (dim == 3)
1107  {
1108  double uz_sim = local_velocity_values[q][2];
1109  double uz_exact = q_exactSol[q][2];
1110  l2errorU += (uz_sim - uz_exact) * (uz_sim - uz_exact) *
1111  fe_values.JxW(q);
1112  }
1113 
1114  double p_sim = local_pressure_values[q] - average_pressure;
1115  double p_exact = q_exactSol[q][dim] - average_exact_pressure;
1116  l2errorP +=
1117  (p_sim - p_exact) * (p_sim - p_exact) * fe_values.JxW(q);
1118  }
1119  }
1120  }
1121  l2errorU = Utilities::MPI::sum(l2errorU, mpi_communicator);
1122  l2errorP = Utilities::MPI::sum(l2errorP, mpi_communicator);
1123 
1124  return std::make_pair(std::sqrt(l2errorU), std::sqrt(l2errorP));
1125 }
