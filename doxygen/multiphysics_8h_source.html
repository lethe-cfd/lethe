58 /**
59  * @brief Utility function used for parsing physics-based
60  * parameters
61  *
62  */
63 inline PhysicsID
64 get_physics_id(std::string physics_name)
65 {
66  if (physics_name == "fluid dynamics")
67  return PhysicsID::fluid_dynamics;
68  else if (physics_name == "heat transfer")
69  return PhysicsID::heat_transfer;
70  else if (physics_name == "tracer")
71  return PhysicsID::tracer;
72  else if (physics_name == "VOF")
73  return PhysicsID::VOF;
74  else
75  return PhysicsID::cahn_hilliard;
76 }
