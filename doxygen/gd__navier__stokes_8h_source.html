289 template <class BSPreconditioner>
290 void
291 BlockSchurPreconditioner<BSPreconditioner>::vmult(
292  GlobalBlockVectorType &dst,
293  const GlobalBlockVectorType &src) const
294 {
295  // TimerOutput computing_timer(std::cout,
296  // TimerOutput::summary,
297  // TimerOutput::wall_times);
298 
299  GlobalVectorType utmp(src.block(0));
300  {
301  // computing_timer.enter_section("Pressure");
302  SolverControl solver_control(
303  linear_solver_parameters.max_iterations,
304  std::max(
305  1e-3 * src.block(1).l2_norm(),
306  // linear_solver_parameters.relative_residual*src.block(0).l2_norm(),
307  linear_solver_parameters.minimum_residual));
308  SolverCG<GlobalVectorType> cg(solver_control);
309 
310  dst.block(1) = 0.0;
311  cg.solve(pressure_mass_matrix,
312  dst.block(1),
313  src.block(1),
314  *pmass_preconditioner);
315  dst.block(1) *= -(kinematic_viscosity + gamma);
316  // computing_timer.exit_section("Pressure");
317  }
318 
319  {
320  // computing_timer.enter_section("Operations");
321  stokes_matrix.block(0, 1).vmult(utmp, dst.block(1));
322  utmp *= -1.0;
323  utmp += src.block(0);
324  // computing_timer.exit_section("Operations");
325  }
326  {
327  // computing_timer.enter_section("A Matrix");
328  SolverControl solver_control(
329  linear_solver_parameters.max_iterations,
330  std::max(1e-1 * src.block(0).l2_norm(),
331  linear_solver_parameters.minimum_residual));
332 
333 
334 
335  // TrilinosWrappers::SolverBicgstab solver(solver_control);
336 #ifndef LETHE_USE_LDV
337  TrilinosWrappers::SolverGMRES solver(solver_control);
338 #else
339  SolverGMRES<GlobalVectorType> solver(solver_control);
340 #endif
341 
342  // A_inverse.solve(stokes_matrix.block(0, 0),dst.block(0), utmp,
343  // mp_preconditioner);
344  solver.solve(stokes_matrix.block(0, 0),
345  dst.block(0),
346  utmp,
347  *amat_preconditioner);
348  // computing_timer.exit_section("A Matrix");
349  }
350 }
