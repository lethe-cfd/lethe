167  /**
168  * @brief set_initial_conditions
169  *
170  * @param initial_condition_type Type of method use to impose initial condition.
171  *
172  * @param restart Indicator if the simulation is being restarted or not.
173  *
174  **/
175 
176  virtual void
177  set_initial_condition(Parameters::InitialConditionType initial_condition_type,
178  bool restart = false)
179  {
180  unsigned int ref_iter = 0;
181  do
182  {
183  if (ref_iter > 0)
184  this->refine_mesh();
185 
186  set_initial_condition_fd(initial_condition_type, restart);
187  if (!restart)
188  {
189  multiphysics->set_initial_conditions();
190  this->postprocess_fd(true);
191  multiphysics->postprocess(true);
192  }
193  ref_iter++;
194  }
195  while (
196  ref_iter <
197  (this->simulation_parameters.mesh_adaptation.initial_refinement + 1) &&
198  restart == false);
199  }

340  /**
341  * @brief Turns regions of the mesh where the material_id=1 to a solid block by injecting it into the constraints
342  * This is achieved by imposing $\mathbf{u}=0$ within the cells which have a
343  * material_id=1. In addition, solid cells which are not connected to the
344  * fluid by any means also get a pressure dirichlet boundary condition which
345  * fixes the pressure to 0. This ensures that the linear system is well-posed.
346  * Right now, this routine only supports the usage of 1 solid domain, but
347  * eventually it could be extended to more than one. By default, the fluid
348  * domain is assumed to have a material_id=0 and the rest of the domains have
349  * a material_id>0.
350  *
351  * @param non_zero_constraints If this parameter is true, it indicates that it is the non_zero constraints
352  * which are being constrainted for the solid domain. If this is set to
353  * false, the zero constraints are constrained in the solid domain
354  */
