227  virtual void
228  set_initial_condition(Parameters::InitialConditionType initial_condition_type,
229  bool restart = false)
230  {
231  unsigned int ref_iter = 0;
232  do
233  {
234  if (ref_iter > 0)
235  this->refine_mesh();
236 
237  set_initial_condition_fd(initial_condition_type, restart);
238  if (!restart)
239  {
240  multiphysics->set_initial_conditions();
241  this->postprocess_fd(true);
242  multiphysics->postprocess(true);
243  }
244  ref_iter++;
245  }
246  while (
247  ref_iter <
248  (this->simulation_parameters.mesh_adaptation.initial_refinement + 1) &&
249  restart == false);
250  }

402  /**
403  * @brief Turn regions of the mesh where the @p material_id>0 into a solid
404  * block by injecting velocity and pressure DOFs into the zero constraints.
405  *
406  * It is achieved by imposing \f$\mathbf{u}=0\f$ within the cells which have a
407  * @p material_id>0. In addition, solid cells which are not connected to the

658  * are locally owned.
659  *
660  * @param[in] non_zero_constraints If this parameter is true, it indicates
661  * that non-zero constraints are applied in the solid domain. If
662  * this is set to false, homogeneous constraints are applied in the solid
663  * domain.

665  * @param[in] local_dof_indices Vector of a cell's local DOF indices.
666  *
667  * @param[out] zero_constraints Homogeneous constraints holding object.
668  */
669  inline void
670  constrain_pressure(
671  const bool &non_zero_constraints,
672  const std::vector<types::global_dof_index> &local_dof_indices,
673  AffineConstraints<double> &zero_constraints)
674  {
675  for (unsigned int i = 0; i < local_dof_indices.size(); ++i)
676  {
677  const unsigned int component =
678  this->fe->system_to_component_index(i).first;
679 
680  // Only pressure DOFs have an additional Dirichlet condition
681  if (component == dim) // pressure DOFs
682  {
683  // We only apply the constraint on the locally owned pressure DOFs
684  // since we have no way of verifying if the locally relevant DOFs
685  // are connected to a fluid cell.
686  bool dof_is_locally_owned = false;
687 
688  // For the GLS-family of solvers, we only have a single index set
689  if constexpr (std::is_same_v<DofsType, IndexSet>)
690  {
691  dof_is_locally_owned =
692  this->locally_owned_dofs.is_element(local_dof_indices[i]);
693  }
694 
695  // For the GD-family of solvers, we have two index sets. One for
696  // velocities and one for pressure.
697  if constexpr (std::is_same_v<DofsType, std::vector<IndexSet>>)
698  {
699  dof_is_locally_owned =
700  this->locally_owned_dofs[1].is_element(local_dof_indices[i]);
701  }
702 
703  if (dof_is_locally_owned)
704  {
705  if (non_zero_constraints)
706  {
707  this->nonzero_constraints.add_line(local_dof_indices[i]);
708  this->nonzero_constraints.set_inhomogeneity(
709  local_dof_indices[i], 0);
710  }
711  else
712  {
713  zero_constraints.add_line(local_dof_indices[i]);
714  }
715  }
716  }
717  }
