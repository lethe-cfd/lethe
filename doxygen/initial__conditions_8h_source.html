174  template <int dim>
175  void
176  InitialConditions<dim>::parse_parameters(ParameterHandler &prm)
177  {
178  prm.enter_subsection("initial conditions");
179  {
180  const std::string op = prm.get("type");
181  if (op == "L2projection")
182  type = InitialConditionType::L2projection;
183  else if (op == "viscous")
184  type = InitialConditionType::viscous;
185  else if (op == "nodal")
186  type = InitialConditionType::nodal;
187  else if (op == "ramp")
188  type = InitialConditionType::ramp;
189 
190  kinematic_viscosity = prm.get_double("kinematic viscosity");
191  prm.enter_subsection("uvwp");
192  uvwp.parse_parameters(prm);
193  prm.leave_subsection();
194 
195  prm.enter_subsection("temperature");
196  temperature.parse_parameters(prm);
197  prm.leave_subsection();
198 
199  prm.enter_subsection("tracer");
200  tracer.parse_parameters(prm);
201  prm.leave_subsection();
202 
203  prm.enter_subsection("VOF");
204  {
205  VOF.parse_parameters(prm);
206  prm.enter_subsection("projection step");
207  {
208  enable_projection_step = prm.get_bool("enable");
209  projection_step_diffusion_factor = prm.get_double("diffusion factor");
210  }
211  prm.leave_subsection();

213  prm.leave_subsection();
214

216

218  cahn_hilliard.parse_parameters(prm);
219  prm.leave_subsection();
220  }
221  prm.leave_subsection();
