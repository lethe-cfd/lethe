38  {
39  // Gather into local variables the relevant fields
40  const Tensor<1, dim> velocity = scratch_data.velocity_values[q];
41  const double velocity_divergence = scratch_data.velocity_divergences[q];
42 
43  // Store JxW in local variable for faster access;
44  const double JxW = JxW_vec[q];
45 
46  // We use the previous phase gradient for the shock capture
47  const Tensor<1, dim> previous_phase_gradient =
48  scratch_data.previous_phase_gradients[q];
49 
50  const double phase_gradient_norm = previous_phase_gradient.norm();

52  // Calculation of the magnitude of the velocity for the
53  // stabilization parameter and the compression term for the phase
54  // indicator
55  const double u_mag = std::max(velocity.norm(), 1e-12);
56 
57  // Implementation of a Discontinuity-Capturing Directional Dissipation
58  // (DCDD) shock capturing scheme. For more information see Tezduyar, T. E.
59  // (2003). Computation of moving boundaries and interfaces and
60  // stabilization parameters. International Journal for Numerical Methods
61  // in Fluids, 43(5), 555-575. Our implementation is based on equations
62  // (70) and (79), which are adapted for the VOF solver.
63  const double tolerance = 1e-12;
64 
65 
66  // In Tezduyar 2003, this is denoted r
67  Tensor<1, dim> gradient_unit_vector =
68  previous_phase_gradient / (phase_gradient_norm + tolerance);
69 
70  // Calculate the artificial viscosity of the shock capture
71  const double vdcdd =
72  (0.5 * h * h) * velocity.norm() * phase_gradient_norm;
73 
74  // We remove the diffusion aligned with the velocity
75  // as is done in the original article. In Tezduyar 2003, this is denoted
76  // s.
77  Tensor<1, dim> velocity_unit_vector =
78  velocity / (velocity.norm() + 1e-12);
79  const Tensor<2, dim> k_corr =
80  (gradient_unit_vector * velocity_unit_vector) *
81  outer_product(velocity_unit_vector, velocity_unit_vector);
82  const Tensor<2, dim> gradient_unit_tensor =
83  outer_product(gradient_unit_vector, gradient_unit_vector);
84  const Tensor<2, dim> dcdd_factor = gradient_unit_tensor - k_corr;
85 
86  // Calculation of the GLS stabilization parameter. The
87  // stabilization parameter used is different if the simulation is
88  // steady or unsteady. In the unsteady case it includes the value
89  // of the time-step. Hypothesis : advection dominated problem
90  // (Pe>3) [Bochev et al., Stability of the SUPG finite element
91  // method for transient advection-diffusion problems, CMAME 2004]
92  const double tau =
93  is_steady(method) ?
94  1. / std::sqrt(std::pow(2. * u_mag / h, 2) +
95  9 * std::pow(4 * diffusivity / (h * h), 2)) :
96  1. / std::sqrt(std::pow(sdt, 2) + std::pow(2. * u_mag / h, 2) +
97  9 * std::pow(4 * diffusivity / (h * h), 2));
98 
99  for (unsigned int j = 0; j < n_dofs; ++j)
100  {
101  const auto phi_phase_j = scratch_data.phi[q][j];
102  const auto grad_phi_phase_j = scratch_data.grad_phi[q][j];
103  const auto laplacian_phi_phase_j = scratch_data.laplacian_phi[q][j];
104 
105  // Strong Jacobian associated with the GLS
106  // stabilization
107  strong_jacobian_vec[q][j] +=
108  velocity * grad_phi_phase_j - diffusivity * laplacian_phi_phase_j;
109 
110  if (compressible)
111  strong_jacobian_vec[q][j] += phi_phase_j * velocity_divergence;
112  }
113 
114 
115 
116  for (unsigned int i = 0; i < n_dofs; ++i)
117  {
118  const auto phi_phase_i = scratch_data.phi[q][i];
119  const auto grad_phi_phase_i = scratch_data.grad_phi[q][i];
120 
121 
122  for (unsigned int j = 0; j < n_dofs; ++j)
123  {
124  const auto phi_phase_j = scratch_data.phi[q][j];
125  const auto grad_phi_phase_j = scratch_data.grad_phi[q][j];
126 
127  // Weak form for advection-diffusion:
128  // u * grad(phase) + phase * grad(u) - diffusivity *
129  // laplacian(phase) = 0
130  local_matrix(i, j) +=
131  (phi_phase_i * velocity * grad_phi_phase_j +
132  diffusivity * grad_phi_phase_i * grad_phi_phase_j) *
133  JxW;
134 
135  // Add compressibility term if the VOF is compressible
136  if (compressible)
137  local_matrix(i, j) +=
138  phi_phase_i * phi_phase_j * velocity_divergence * JxW;
139 
140  // Addition to the cell matrix for GLS stabilization
141  local_matrix(i, j) += tau * strong_jacobian_vec[q][j] *
142  (grad_phi_phase_i * velocity) * JxW;
143 
144  // DCDD shock capturing
145  if (DCDD)
146  {
147  local_matrix(i, j) +=
148  (vdcdd * scalar_product(grad_phi_phase_j,
149  dcdd_factor * grad_phi_phase_i)) *
150  JxW;
151  }
152  }
153  }
154  } // end loop on quadrature points
