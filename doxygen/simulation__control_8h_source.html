26 /**
27  * @brief The SimulationControl class is responsible for the control of steady-state and transient
28  * simulations carried out with Lethe. This base class is a pure virtual class
29  * and cannot be instantiated. However, it stores the core variables which are
30  * necessary for its serialization (write and read)
31  **/
32 
33 class SimulationControl
34 {
35 protected:
36  // Time stepping method used for the simulation
37  Parameters::SimulationControl::TimeSteppingMethod method;
38 
39  // Current method used for the present assembly
40  // This is used to differentiate the substeps of SDIRK methods or
41  // to start-up BDF simulations
42  Parameters::SimulationControl::TimeSteppingMethod assembly_method;
43 
44  // Time of the current iteration being solved for
45  double current_time;
46 
47  // Time of the previous iteration
48  double previous_time;
49 
50  // Time step linking the previous iteration and the current time
51  double time_step;
52 
53  // Initial time step given in the parameter
54  double initial_time_step;
55 
56  // Simulation end time
57  double end_time;
58 
59  // Time step vector. This vector accumulates the time steps of the previous
60  // iterations. This is required for multiple steps methods such as the bdfs.
61  std::vector<double> time_step_vector;
62 
63  // Iteration. Iterations start at 0, but the first actual iteration is
64  // iteration 1.
65  unsigned int iteration_number;
66 
67  // Number of mesh adaptation iteration
68  unsigned int number_mesh_adapt;
69 
70  // Courant-Friedrich-Levy (CFL) condition
71  // Since the simulation control is unaware of the information propagation
72  // mechanism (for instance the velocity), the current CFL must be set by the
73  // solver itself
74  double CFL;
75 
76  // Maximal CFL condition. This is used to control adaptative time stepping. In
77  // the case of constant time stepping or steady-state simulations, this
78  // parameter remains unused.
79  double max_CFL;
80 
81  // Current value of the norm of the rhs
82  double residual;
83 
84  // Residual at which the simulation stops when adjoint time-stepping is used
85  double stop_tolerance;
86 
87  // Number of time steps stored
88  // BDF methods require a number of previous time steps. This number is known a
89  // priori and depends on the method used. We do not keep all the time steps to
90  // prevent the accumulation within a large vector.
91  static const unsigned int n_previous_time_steps = 4;
92 
93  // Output iteration frequency
94  // Controls the output of the simulation results when the output is controlled
95  // by the iteration number.
96  unsigned int output_frequency;
97 
98  // Output time frequency
99  // Controls the output of the simulation results when the output is controlled
100  // by the time
101  double output_time_frequency;
102 
103  // Log iteration frequency
104  // Controls the frequency at which status of the simulation is written to
105  // the terminal
106  unsigned int log_frequency;
107 
108  // Log precision
109  // Controls the number of significant digits displayed on the standard outputs
110  unsigned int log_precision;
111 
112  // Number of mesh subdivision to be used when outputting the results
113  unsigned int subdivision;
114 
115  // Number of parallel file to generate
116  unsigned int group_files;
117 
118  // Output name
119  std::string output_name;
120 
121  // Output path
122  std::string output_path;
123 
124  // Output boundaries
125  // Control if the boundaries of the domain are outputted when writing results
126  //
127  bool output_boundaries;
128 
129  // Indicator to tell if this is the first assembly of a step
130  bool first_assembly;
131 
132  // The method use to start high order bdf scheme
133  Parameters::SimulationControl::BDFStartupMethods bdf_start_method;
134 
135  // The time scaling used to do small time-steps at the startup of the

424  std::vector<double>
425  get_simulation_times() const
426  {
427  // Create a vector of the previous times
428  std::vector<double> times(n_previous_time_steps + 1);
429  for (unsigned int i = 0; i < n_previous_time_steps + 1; ++i)
430  {
431  times[i] = current_time;
432  for (unsigned int j = 0; j < i; ++j)
433  times[i] -= time_step_vector[j];
434  }
435  return times;
436  }

447 class SimulationControlTransient : public SimulationControl
448 {
449 protected:
450  // Enable adaptative time stepping
451  bool adapt;
452 
453  // Time step scaling for adaptative time stepping
454  double adaptative_time_step_scaling;
455 
456  /**
457  * @brief Calculates the next value of the time step. If adaptation
458  * is enabled, the time step is calculated in order to ensure
459  * that the CFL condition is bound by the maximal CFL value.
460  * The new time step is equal to adaptative_time_step_scaling * the previous
461  * time step. If this surpasses the simulation time or if it surpasses the
462  * maximal CFL value, the time step is scaled down to ensure that this is
463  * respected.
464  */
465  virtual double
466  calculate_time_step() override;
467 
468

473  print_progression(const ConditionalOStream &pcout) override;

476  * @brief Proceeds with the simulation until the end condition is reached
477  */
478  virtual bool
479  integrate() override;
480 
481 
482  /**
483  * @brief Ends the simulation when the end time is reached
484  */
485  virtual bool
486  is_at_end() override;
