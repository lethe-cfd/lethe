136 template <typename VectorType>
137 PhysicsSolver<VectorType>::PhysicsSolver(
138  const Parameters::NonLinearSolver non_linear_solver_parameters)
139  : pcout(std::cout, Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
140 {
141  switch (non_linear_solver_parameters.solver)
142  {
143  case Parameters::NonLinearSolver::SolverType::newton:
144  non_linear_solver =
145  new NewtonNonLinearSolver<VectorType>(this,
146  non_linear_solver_parameters);
147  break;
148  case Parameters::NonLinearSolver::SolverType::kinsol_newton:
149  non_linear_solver = new KinsolNewtonNonLinearSolver<VectorType>(
150  this, non_linear_solver_parameters);
151  break;
152  case Parameters::NonLinearSolver::SolverType::inexact_newton:
153  non_linear_solver = new InexactNewtonNonLinearSolver<VectorType>(
154  this, non_linear_solver_parameters);
155  break;
156  default:
157  break;
158  }
159 }
