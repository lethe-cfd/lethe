193  template <int n_components>
194  void
195  ParserImplementation<n_components>::initialize(
196  const std::string &variables,
197  const std::vector<std::string> &expressions,
198  const std::map<std::string, double> &constants,
199  const bool time_dependent)
200  {
201  this->parser_data.clear(); // this will reset all thread-local objects
202 
203  this->constants = constants;
204  this->var_names = dealii::Utilities::split_string_list(variables, ',');
205  this->expressions = expressions;
206  AssertThrow(((time_dependent) ? n_components + 1 : n_components) ==
207  this->var_names.size(),
208  dealii::ExcMessage("Wrong number of variables"));
209 
210  // Now we define how many variables we expect to read in. We distinguish
211  // between two cases: Time dependent problems, and not time dependent
212  // problems. In the first case the number of variables is given by the
213  // components plus one. In the other case, the number of variables is
214  // equal to the dimension. Once we parsed the variables string, if none of
215  // this is the case, then an exception is thrown.
216  if (time_dependent)
217  this->n_vars = n_components + 1;
218  else
219  this->n_vars = n_components;
220 
221  // Create a parser object for the current thread we can then query in
222  // value() and vector_value(). This is not strictly necessary because a
223  // user may never call these functions on the current thread, but it gets
224  // us error messages about wrong formulas right away
225  this->init_muparser();
226  this->initialized = true;
227  }

231  template <int n_components>
232  void
233  ParserImplementation<n_components>::init_muparser() const
234  {
235  // check that we have not already initialized the parser on the
236  // current thread, i.e., that the current function is only called
237  // once per thread
238  ParserData &data = this->parser_data.get();
239  Assert(data.parsers.empty() && data.vars.empty(),
240  dealii::StandardExceptions::ExcInternalError());
241 
242  // initialize the objects for the current thread
243  data.parsers.reserve(n_components);
244  data.vars.resize(this->var_names.size());
245  for (unsigned int component = 0; component < n_components; ++component)
246  {
247  data.parsers.emplace_back(std::make_unique<Parser>());
248  mu::Parser &parser = dynamic_cast<Parser &>(*data.parsers.back());
249 
250  for (const auto &constant : this->constants)
251  parser.DefineConst(constant.first, constant.second);
252 
253  for (unsigned int iv = 0; iv < this->var_names.size(); ++iv)
254  parser.DefineVar(this->var_names[iv], &data.vars[iv]);
255 
256  // define some compatibility functions:
257  parser.DefineFun("if", mu_if, true);
258  parser.DefineOprt("|", mu_or, 1);
259  parser.DefineOprt("&", mu_and, 2);
260  parser.DefineFun("int", mu_int, true);
261  parser.DefineFun("ceil", mu_ceil, true);
262  parser.DefineFun("cot", mu_cot, true);
263  parser.DefineFun("csc", mu_csc, true);
264  parser.DefineFun("floor", mu_floor, true);
265  parser.DefineFun("sec", mu_sec, true);
266  parser.DefineFun("log", mu_log, true);
267  parser.DefineFun("pow", mu_pow, true);
268  parser.DefineFun("erfc", mu_erfc, true);
269  // Disable optimizations (by passing false) that assume the functions
270  // will always return the same value:
271  parser.DefineFun("rand_seed", mu_rand_seed, false);
272  parser.DefineFun("rand", mu_rand, false);
273 
274  try
275  {
276  // muparser expects that functions have no
277  // space between the name of the function and the opening
278  // parenthesis. this is awkward because it is not backward
279  // compatible to the library we used to use before muparser
280  // (the fparser library) but also makes no real sense.
281  // consequently, in the expressions we set, remove any space
282  // we may find after function names
283  std::string transformed_expression = this->expressions[component];
284 
285  for (const auto &current_function_name : get_function_names())
286  {
287  const unsigned int function_name_length =
288  current_function_name.size();
289 
290  std::string::size_type pos = 0;
291  while (true)
292  {
293  // try to find any occurrences of the function name
294  pos =
295  transformed_expression.find(current_function_name, pos);
296  if (pos == std::string::npos)
297  break;
298 
299  // replace whitespace until there no longer is any
300  while (
301  (pos + function_name_length <
302  transformed_expression.size()) &&
303  ((transformed_expression[pos + function_name_length] ==
304  ' ') ||
305  (transformed_expression[pos + function_name_length] ==
306  '\t')))
307  transformed_expression.erase(
308  transformed_expression.begin() + pos +
309  function_name_length);
310 
311  // move the current search position by the size of the
312  // actual function name
313  pos += function_name_length;
314  }
315  }
316 
317  // now use the transformed expression
318  parser.SetExpr(transformed_expression);
319  }
320  catch (mu::ParserError &e)
321  {
322  std::cerr << "Message: <" << e.GetMsg() << ">\n";
323  std::cerr << "Formula: <" << e.GetExpr() << ">\n";
324  std::cerr << "Token: <" << e.GetToken() << ">\n";
325  std::cerr << "Position: <" << e.GetPos() << ">\n";
326  std::cerr << "Errc: <" << e.GetCode() << ">" << std::endl;
327  AssertThrow(false, ExcParseError(e.GetCode(), e.GetMsg()));
328  }
329  }
