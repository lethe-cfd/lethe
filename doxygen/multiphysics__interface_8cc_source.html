220 void
221 MultiphysicsInterface<dim>::inspect_multiphysics_models_dependencies(
222  const SimulationParameters<dim> &nsparam)
223 {
224  bool buoyancy_force_enabled = nsparam.multiphysics.buoyancy_force;
225  bool heat_transfer_enabled = nsparam.multiphysics.heat_transfer;
226  bool marangoni_effect_enabled =
227  nsparam.multiphysics.vof_parameters.surface_tension_force
228  .enable_marangoni_effect;
229  bool surface_tension_force_enabled =
230  nsparam.multiphysics.vof_parameters.surface_tension_force.enable;
231  bool fluid_dynamics_enabled = nsparam.multiphysics.fluid_dynamics;
232  bool interface_sharpening_enabled =
233  nsparam.multiphysics.vof_parameters.sharpening.enable;
234  bool VOF_enabled = nsparam.multiphysics.VOF;
235 
236  // To avoid getting unused parameter warning
237  _unused(buoyancy_force_enabled && heat_transfer_enabled &&
238  marangoni_effect_enabled && surface_tension_force_enabled &&
239  fluid_dynamics_enabled && interface_sharpening_enabled &&
240  VOF_enabled);
241 
242  // Dependence of buoyant force on fluid dynamics
243  AssertThrow(!(buoyancy_force_enabled == true &&
244  fluid_dynamics_enabled == false),
245  BuoyancyWithoutFluidDynamicsError(buoyancy_force_enabled));
246 
247  // Dependence of buoyant force on heat transfer
248  AssertThrow(!(buoyancy_force_enabled == true &&
249  heat_transfer_enabled == false),
250  BuoyancyWithoutHeatTransferError(buoyancy_force_enabled));
251 
252  // Dependence of Marangoni effect on fluid dynamics
253  AssertThrow(!(marangoni_effect_enabled == true &&
254  fluid_dynamics_enabled == false),
255  MarangoniWithoutFluidDynamicsError(marangoni_effect_enabled));
256 
257  // Dependence of Marangoni effect on heat transfer
258  AssertThrow(!(marangoni_effect_enabled == true &&
259  heat_transfer_enabled == false),
260  MarangoniWithoutHeatTransferError(marangoni_effect_enabled));
261 
262  // Dependence of Marangoni effect on surface tension force
263  AssertThrow(!(marangoni_effect_enabled == true &&
264  surface_tension_force_enabled == false),
265  MarangoniWithoutSurfaceTensionForceError(
266  marangoni_effect_enabled));
267 
268  // Dependence of surface tension force on VOF
269  AssertThrow(!(surface_tension_force_enabled == true && VOF_enabled == false),
270  SurfaceTensionForceWithoutVOFError(
271  surface_tension_force_enabled));
272 
273  // Dependence of Marangoni effect on VOF
274  AssertThrow(!(marangoni_effect_enabled == true && VOF_enabled == false),
275  MarangoniWithoutVOFError(marangoni_effect_enabled));
276 
277  // Dependence of interface sharpening on VOF
278  AssertThrow(!(interface_sharpening_enabled == true && VOF_enabled == false),
279  InterfaceSharpeningWithoutVOFError(interface_sharpening_enabled));
280 }
