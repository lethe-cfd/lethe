97  /**
98  * @brief Call for the solution of the physics that should be solved
99  *
100  * @param fluid_dynamics_has_been_solved Boolean that states if the fluid dynamics has been
101  * already solved or not. See the map `solve_pre_fluid` to know which
102  * subphysics are solved before the fluid dynamics.
103  *
104  * @param time_stepping_method Time-Stepping method with which the assembly is called
105  */
106  void
107  solve(const bool fluid_dynamics_has_been_solved,
108  const Parameters::SimulationControl::TimeSteppingMethod
109  time_stepping_method)
110  {
111  // Loop through all the elements in the physics map. Consequently, iphys is
112  // an std::pair where iphys.first is the PhysicsID and iphys.second is the
113  // AuxiliaryPhysics pointer. This is how the map can be traversed
114  // sequentially.
115  for (auto &iphys : physics)
116  {
117  // If iphys.first should be solved BEFORE fluid dynamics
118  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
119  solve_physics(iphys.first, time_stepping_method);
120 
121  // If iphys.first should be solved AFTER fluid dynamics OR if is not
122  // present in solve_pre_fluid map
123  else if (fluid_dynamics_has_been_solved &&
124  (!solve_pre_fluid[iphys.first] ||
125  solve_pre_fluid.count(iphys.first) == 0))
126  solve_physics(iphys.first, time_stepping_method);
127  }
128 
129  for (auto &iphys : block_physics)
130  {
131  // If iphys.first should be solved BEFORE fluid dynamics
132  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
133  solve_block_physics(iphys.first, time_stepping_method);
134 
135  // If iphys.first should be solved AFTER fluid dynamics OR if is not
136  // present in solve_pre_fluid map
137  else if (fluid_dynamics_has_been_solved &&
138  (!solve_pre_fluid[iphys.first] ||
139  solve_pre_fluid.count(iphys.first) == 0))
140  solve_block_physics(iphys.first, time_stepping_method);
141  }
142  }

144  /**
145  * @brief Call for the solution of a single physic
146  *
147  * @param time_stepping_method Time-Stepping method with which the assembly is called
148  */
149  void
150  solve_physics(const PhysicsID physics_id,
151  const Parameters::SimulationControl::TimeSteppingMethod
152  time_stepping_method)
153  {
154  // Announce physic solved (verbosity = non_linear_solver.verbosity)
155  if (verbosity.at(physics_id) != Parameters::Verbosity::quiet)
156  announce_physics(physics_id);
157 
158  AssertThrow(std::find(active_physics.begin(),
159  active_physics.end(),
160  physics_id) != active_physics.end(),
161  ExcInternalError());
162 
163  physics[physics_id]->time_stepping_method = time_stepping_method;
164  physics[physics_id]->solve_non_linear_system(false);
165  physics[physics_id]->modify_solution();
166  }

228  /**
229  * @brief Rearrange vector solution correctly for transient simulations for
230  * all auxiliary physics.
231  *
232  * @param fluid_dynamics_has_been_solved Boolean that states if the fluid dynamics has been
233  * already solved or not. See the map `solve_pre_fluid` to know which
234  * subphysics are solved before the fluid dynamics.
235  */
236  void
237  percolate_time_vectors(const bool fluid_dynamics_has_been_solved)
238  {
239  // Loop through all the elements in the physics map. Consequently, iphys is
240  // an std::pair where iphys.first is the PhysicsID and iphys.second is the
241  // AuxiliaryPhysics pointer. This is how the map can be traversed
242  // sequentially.
243  for (auto &iphys : physics)
244  {
245  // If iphys.first should be percolated BEFORE fluid dynamics is solved
246  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
247  iphys.second->percolate_time_vectors();
248 
249  // If iphys.first should be percolated AFTER fluid dynamics is solved OR
250  // if is not present in solve_pre_fluid map
251  else if (fluid_dynamics_has_been_solved &&
252  (!solve_pre_fluid[iphys.first] ||
253  solve_pre_fluid.count(iphys.first) == 0))
254  iphys.second->percolate_time_vectors();
255  }
256  for (auto &iphys : block_physics)
257  {
258  // If iphys.first should be percolated BEFORE fluid dynamics is solved
259  if (!fluid_dynamics_has_been_solved && solve_pre_fluid[iphys.first])
260  iphys.second->percolate_time_vectors();
261 
262  // If iphys.first should be percolated AFTER fluid dynamics is solved OR
263  // if is not present in solve_pre_fluid map
264  else if (fluid_dynamics_has_been_solved &&
