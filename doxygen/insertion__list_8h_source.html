38  /**
39  * @brief The InsertionList class inserts particles using a list specific position.
40  * This allows the insertion of any number of particles at a well-controled
41  * location which is especially useful from a testing perspective. The code
42  * ensures that the number of positions provided in the x,y (and possibly z)
43  * direction is coherent. If more particles than the number of position in the
44  * list are requested, the class will continue inserting particles at the
45  * insertion frequency using the list of position. There is no mechanism in
46  * place that prevents the overlap of these new particles with previous ones.
47  *
48  * @param particle_handler The particle handler of particles which are being
49  * inserted
50  * @param triangulation Triangulation to access the cells in which the
51  * particles are inserted
52  * @param dem_parameters DEM parameters declared in the .prm file
53  */
54  virtual void
55  insert(Particles::ParticleHandler<dim> &particle_handler,
56  const parallel::distributed::Triangulation<dim> &triangulation,
57  const DEMSolverParameters<dim> &dem_parameters) override;
