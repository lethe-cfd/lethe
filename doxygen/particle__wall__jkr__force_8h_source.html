154  /**
155  * @brief Carries out calculation of the rolling resistance torque using the constant model
156  *
157  * @param particle_one_properties Particle one properties
158  * @param particle_two_properties Particle two properties
159  * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
160  * @param normal_force_norm Normal force norm
161  *
162  * @return rolling resistance torque
163  */
164  inline Tensor<1, 3>
165  constant_resistance(const ArrayView<const double> &particle_properties,
166  const double effective_rolling_friction_coefficient,
167  const double normal_force_norm,
168  const Tensor<1, 3> & /*normal_contact_vector*/)
169  {
170  // Getting the angular velocity of particle in the vector format
171  Tensor<1, 3> angular_velocity;
172  for (int d = 0; d < 3; ++d)
173  {
174  angular_velocity[d] =
175  particle_properties[DEM::PropertiesIndex::omega_x + d];
176  }
177 
178  // Calculation of particle-wall angular velocity (norm of the
179  // particle angular velocity)
180  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
181 
182  double omega_value = angular_velocity.norm();
183  if (omega_value != 0)
184  {
185  particle_wall_angular_velocity = angular_velocity / omega_value;
186  }
187 
188  // Calculation of rolling resistance torque
189  Tensor<1, 3> rolling_resistance_torque =
190  -effective_rolling_friction_coefficient *
191  (particle_properties[DEM::PropertiesIndex::dp] * 0.5) *
192  normal_force_norm * particle_wall_angular_velocity;
193 
194  return rolling_resistance_torque;
195  }

197  /**
198  * @brief Carries out calculation of the rolling resistance torque using the viscous model
199  *
200  * @param particle_one_properties Particle one properties
201  * @param particle_two_properties Particle two properties
202  * @param effective_rolling_friction_coefficient Effective rolling friction coefficient
203  * @param normal_force_norm Normal force norm
204  *
205  * @return rolling resistance torque
206  */
207  inline Tensor<1, 3>
208  viscous_resistance(const ArrayView<const double> &particle_properties,
209  const double effective_rolling_friction_coefficient,
210  const double normal_force_norm,
211  const Tensor<1, 3> &normal_contact_vector)
212  {
213  // Getting the angular velocity of particle in the vector format
214  Tensor<1, 3> angular_velocity;
215  for (int d = 0; d < 3; ++d)
216  {
217  angular_velocity[d] =
218  particle_properties[DEM::PropertiesIndex::omega_x + d];
219  }
220 
221  // Calculation of particle-wall angular velocity (norm of the
222  // particle angular velocity)
223  Tensor<1, 3> particle_wall_angular_velocity({0.0, 0.0, 0.0});
224 
225  double omega_value = angular_velocity.norm();
226  if (omega_value != 0)
227  {
228  particle_wall_angular_velocity = angular_velocity / omega_value;
229  }
230 
231  Tensor<1, 3> v_omega =
232  cross_product_3d(angular_velocity,
233  particle_properties[DEM::PropertiesIndex::dp] * 0.5 *
234  normal_contact_vector);
235 
236  // Calculation of rolling resistance torque
237  Tensor<1, 3> rolling_resistance_torque =
238  -effective_rolling_friction_coefficient *
239  particle_properties[DEM::PropertiesIndex::dp] * 0.5 * normal_force_norm *
240  v_omega.norm() * particle_wall_angular_velocity;
241 
242  return rolling_resistance_torque;
243  }
