120 PeriodicHillsGrid<dim, spacedim>::PeriodicHillsGrid(
121  const std::string &grid_arguments)
122 {
123  this->grid_arguments = grid_arguments;
124 
125  // Separate arguments of the string
126  std::vector<std::string> arguments;
127  std::stringstream s_stream(grid_arguments);
128  while (s_stream.good())
129  {
130  std::string substr;
131  getline(s_stream, substr, ';');
132  arguments.push_back(substr);
133  }
134 
135  std::vector<double> arguments_double =
136  dealii::Utilities::string_to_double(arguments);
137  spacing_y = arguments_double[0];
138  alpha = arguments_double[1];
139  repetitions_x = arguments_double[2];
140  repetitions_y = arguments_double[3];
141  if (dim == 3)
142  repetitions_z = arguments_double[4];
143 
144  if (abs(alpha - 1) < 1e-6)
145  alpha = int(alpha);

194 /**
195  * \brief vector_value. This vector_value function is used to construct the
196  * geometry manifold. It changes the new point (np) of the hill grid to the
197  * original point (op) of the transformed hyper_rectangle grid. This function
198  * is mandatory to use FunctionManifold. First, it finds the minimum value of
199  * y depending the x position and then calculates the op with the inverse of
200  * the transformation done by hill_geometry function.
201  *
202  * \param np. A point in space.
203  *
204  * @param values. The vector of values which will be calculated at the position p.
205  */
206 template <int dim, int spacedim>
207 void
208 PeriodicHillsPullBack<dim, spacedim>::vector_value(const Point<spacedim> &np,
209  Vector<double> &values) const
210 {
211  double x = np[0];
212  double max_y = 3.035;
213  double max_x = 9.0;
214  double flat_region_length = 5.142;
215  double left_hill = 1.929;
216  double right_hill = 7.071;
217  double min_y;
218 
219  // Reversing elongation and shifting of x lines
220  if (alpha != 1)
221  {
222  if (x < left_hill * alpha)
223  x = (x / alpha);
224  else if (x > alpha * left_hill + flat_region_length)
225  x = (x - flat_region_length - alpha * left_hill) / alpha + right_hill;
226  else
227  x = x - (alpha * left_hill) + left_hill;
228  }
229 
230  if (alpha > 1)
231  {
232  if (x < max_x / 2)
233  x = (-(1 - 0.5) +
234  std::sqrt(std::pow((1 - 0.5), 2) - (4 * (1 / max_x) * -x))) /
235  (2 / max_x);
236  else if (x > max_x / 2 && x < max_x)
237  x = (-(1 + 1.5) + std::sqrt(std::pow((1 + 1.5), 2) -
238  (4 * (-1 / max_x) * (-0.5 * max_x - x)))) /
239  (2 * -1 / max_x);
240  }
241 
242  // Reversing polynomial transformation and shifting of y lines
243  if (spacedim == 2)
244  {
245  min_y =
246  PeriodicHillsGrid<dim, spacedim>::hill_geometry(Point<spacedim>(x, 0),
247  spacing_y,
248  alpha)[1];
249  }
250  else if (spacedim == 3)
251  {
252  min_y = PeriodicHillsGrid<dim, spacedim>::hill_geometry(
253  Point<spacedim>(x, 0, np[2]), spacing_y, alpha)[1];
254  values(2) = np[2];
255  }
256 
257  double y = (np[1] - min_y) / (1 - min_y / max_y);
258 
259  if (y < max_y / 2 && spacing_y != 0)
260  y = (-(1 - 0.5 * spacing_y) + std::sqrt(std::pow((1 - 0.5 * spacing_y), 2) -
261  (4 * (spacing_y / max_y) * -y))) /
262  (2 * spacing_y / max_y);
263  else if (y > max_y / 2 && y < max_y && spacing_y != 0)
264  y =
265  (-(1 + 1.5 * spacing_y) +
266  std::sqrt(std::pow((1 + 1.5 * spacing_y), 2) -
267  (4 * (-spacing_y / max_y) * (-0.5 * spacing_y * max_y - y)))) /
268  (2 * -spacing_y / max_y);
269 
270  values(0) = x;
271  values(1) = y;
272 }

351 /**
352  * @brief The hill_geometry function calculates all the domain of the geometry with 6
353  * polynomials depending the x position. (See Hill Geometry Definition file :
354  * https://turbmodels.larc.nasa.gov/Other_LES_Data/2dhill_periodic.html)
355  * This code has nondimensionalized geometry, but the coefficients provided
356  * need a hill height of 28.
357  * This function also does a gradual shifting of the horizontal lines
358  * prior to have smaller element on the bottom of the geometry where results
359  * are more important.
360  *
361  * @param p A point in space which will be adapted to the periodic hill geometry
362  *
363  * @param param Non-linear solver parameters
364  *
365  */
366 template <int dim, int spacedim>
367 Point<spacedim>
368 PeriodicHillsGrid<dim, spacedim>::hill_geometry(const Point<spacedim> &p,
369  double spacing_y,
370  double alpha)
371 {
372  double H = 28; // Height dimension to use with polynomials
373  double x = p[0] * H, y = p[1] * H;
374  double max_y = 3.035 * H;
375  double max_x = 9.0 * H;
376  double flat_region_length = 5.142 * H;
377  double left_hill = 1.929 * H;
378  double right_hill = 7.071 * H;
379  double pos_x_left;
380  double pos_x_right;
381  double pos_y_bottom;
382  double pos_y_top;
383  double pos_y;
384  double y_0;
385  double diff;
386 
387  // Gradual spacing and swifting depending on x position if the geometry
388  // is elongated
389  if (alpha > 1)
390  {
391  if (x < max_x / 2)
392  {
393  pos_x_left = x / -max_x + 0.5;
394  x -= pos_x_left * x;
395  }
396  else if (x > max_x / 2 && x < max_x)
397  {
398  pos_x_right = x / max_x - 0.5;
399  x += pos_x_right * (max_x - x);
400  }
401  }
402 
403  // Polynomial coefficients :
404  const double a1 = 2.800000000000E+01, b1 = 0.000000000000E+00,
405  c1 = 6.775070969851E-03, d1 = -2.124527775800E-03;
406  const double a2 = 2.507355893131E+01, b2 = 9.754803562315E-01,
407  c2 = -1.016116352781E-01, d2 = 1.889794677828E-03;
408  const double a3 = 2.579601052357E+01, b3 = +8.206693007457E-01,
409  c3 = -9.055370274339E-02, d3 = 1.626510569859E-03;
410  const double a4 = 4.046435022819E+01, b4 = -1.379581654948E+00,
411  c4 = 1.945884504128E-02, d4 = -2.070318932190E-04;
412  const double a5 = 1.792461334664E+01, b5 = +8.743920332081E-01,
413  c5 = -5.567361123058E-02, d5 = 6.277731764683E-04;
414  const double a6 = 5.639011190988E+01, b6 = -2.010520359035E+00,
415  c6 = 1.644919857549E-02, d6 = 2.674976141766E-05;
416 
417  double new_x = (max_x - x); // x for the left side of the geometry
418 
419  // Gradual spacing and swifting depending on y position
420  if (y < max_y / 2 && (std::abs(spacing_y) > 0))
421  {
422  pos_y_bottom = y / -max_y + 0.5;
423  y -= spacing_y * pos_y_bottom * y;
424  }
425  else if (y > max_y / 2 && y < max_y && (std::abs(spacing_y) > 0))
426  {
427  pos_y_top = y / max_y - 0.5;
428  y += spacing_y * pos_y_top * (max_y - y);
429  }
430 
431  pos_y = y / -max_y + 1;
432 
433  // Polynomial equations :
434  if (x >= 0 && x < 9)
435  {
436  y += pos_y * (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
437 
438  // Checking if y is under 28 and correction
439  y_0 = (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
440  diff = y_0 - 28.0;
441  if (diff > 0)
442  y -= pos_y * diff;
443  }
444  else if (x >= 9 && x < 14)
445  y += pos_y * (a2 + b2 * x + c2 * std::pow(x, 2) + d2 * std::pow(x, 3));
446  else if (x >= 14 && x < 20)
447  y += pos_y * (a3 + b3 * x + c3 * std::pow(x, 2) + d3 * std::pow(x, 3));
448  else if (x >= 20 && x < 30)
449  y += pos_y * (a4 + b4 * x + c4 * std::pow(x, 2) + d4 * std::pow(x, 3));
450  else if (x >= 30 && x < 40)
451  y += pos_y * (a5 + b5 * x + c5 * std::pow(x, 2) + d5 * std::pow(x, 3));
452  else if (x >= 40 && x < 54)
453  {
454  y += pos_y * (a6 + b6 * x + c6 * std::pow(x, 2) + d6 * std::pow(x, 3));
455  if (y < 0)
456  y = 0;
457  }
458  else if (x <= 252 && x > 243)
459  {
460  y += pos_y * (a1 + b1 * new_x + c1 * std::pow(new_x, 2) +
461  d1 * std::pow(new_x, 3));
462  y_0 = (a1 + b1 * x + c1 * std::pow(x, 2) + d1 * std::pow(x, 3));
463  diff = y_0 - 28.0;
464  if (diff > 0)
465  y -= pos_y * diff;
466  }
467  else if (x <= 243 && x > 238)
468  y += pos_y *
469  (a2 + b2 * new_x + c2 * std::pow(new_x, 2) + d2 * std::pow(new_x, 3));
470  else if (x <= 238 && x > 232)
471  y += pos_y *
472  (a3 + b3 * new_x + c3 * std::pow(new_x, 2) + d3 * std::pow(new_x, 3));
473  else if (x <= 232 && x > 222)
474  y += pos_y *
475  (a4 + b4 * new_x + c4 * std::pow(new_x, 2) + d4 * std::pow(new_x, 3));
476  else if (x <= 222 && x > 212)
477  y += pos_y *
478  (a5 + b5 * new_x + c5 * std::pow(new_x, 2) + d5 * std::pow(new_x, 3));
479  else if (x <= 212 && x > 198)
480  {
481  y += pos_y * (a6 + b6 * new_x + c6 * std::pow(new_x, 2) +
482  d6 * std::pow(new_x, 3));
483  if (y < 0)
484  y = 0;
485  }
486  else
487  y += 0;
488 
489  // Elongation of the geometry with the alpha factor
490  // Note : The length of the flat region is always the same length
491  if (x < left_hill)
492  x = alpha * x;
493  else if (x > right_hill)
494  x = alpha * (x - right_hill) + flat_region_length + alpha * left_hill;
495  else
496  x = (x - left_hill) + (alpha * left_hill);
497 
498  Point<spacedim> q;
499  q[0] = (x / H);
500  q[1] = (y / H);
501 
502  if (spacedim == 3)
503  q[2] = p[2];
504 
505  return q;
506 }

508 /**
509  * @brief make_grid. The make_grid function generates a hyper rectangle of the size of the domain
510  * and then transforms it to the hill geometry. It also constructs the
511  * geometry manifold with FunctionManifold and finally sets the manifold.
512  *
513  * @param triangulation. The triangulation object on which the grid is generated
514  */
515 template <int dim, int spacedim>
516 void
517 PeriodicHillsGrid<dim, spacedim>::make_grid(
518  Triangulation<dim, spacedim> &triangulation)
519 {
520  std::vector<unsigned int> repetitions(2);
521  repetitions[0] = repetitions_x;
522  repetitions[1] = repetitions_y;
523 
524  if (dim == 2)
525  {
526  if (alpha != 1 && (repetitions_x > 1 || repetitions_y > 1))
527  throw std::logic_error(
528  "When alpha parameter is not 1, repetition parameters should be all set to 1");
529 
530  GridGenerator::subdivided_hyper_rectangle(triangulation,
531  repetitions,
532  Point<dim>(0.0, 0.0),
533  Point<dim>(9.0, 3.035),
534  true);
535  }
536  else if (dim == 3)
537  {
538  if (alpha != 1 &&
539  (repetitions_x > 1 || repetitions_y > 1 || repetitions_z > 1))
540  throw std::logic_error(
541  "When alpha parameter is not 1, repetition parameters should be all set to 1");
542  repetitions.push_back(repetitions_z);
543  GridGenerator::subdivided_hyper_rectangle(triangulation,
544  repetitions,
545  Point<dim>(0.0, 0.0, 0.0),
546  Point<dim>(9.0, 3.035, 4.5),
547  true);
548  }
549 
550  // Transformation of the geometry with the hill geometry
551  GridTools::transform(
552  [this](const Point<spacedim> &p) {
553  return this->hill_geometry(p, spacing_y, alpha);
554  },
555  triangulation);
556 
557  // Manifold construction
558  static const FunctionManifold<dim, spacedim, spacedim> manifold_func(
559  std::make_unique<PeriodicHillsPushForward<dim, spacedim>>(spacing_y, alpha),
560  std::make_unique<PeriodicHillsPullBack<dim, spacedim>>(spacing_y, alpha));
561  triangulation.set_manifold(1, manifold_func);
562  triangulation.set_all_manifold_ids(1);
563 }
