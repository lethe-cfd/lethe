52 /**
53  * @brief Calculates the area of intersection between a circular (2D) particle and a circle
54  *
55  * @param r_particle Radius of the particle
56  *
57  * @param r_circle Radius of the circle
58  *
59  * @param neighbor_distance Distance between the particle and the circle
60  */
61 inline double
62 particle_circle_intersection_2d(double r_particle,
63  double r_circle,
64  double neighbor_distance)
65 {
66  return pow(r_particle, 2) * Utilities::fixed_power<-1, double>(
67  cos((pow(neighbor_distance, 2) +
68  pow(r_particle, 2) - pow(r_circle, 2)) /
69  (2 * neighbor_distance * r_particle))) +
70  Utilities::fixed_power<2, double>(r_circle) *
71  Utilities::fixed_power<-1, double>(
72  cos((pow(neighbor_distance, 2) - pow(r_particle, 2) +
73  pow(r_circle, 2)) /
74  (2 * neighbor_distance * r_circle))) -
75  0.5 * sqrt((-neighbor_distance + r_particle + r_circle) *
76  (neighbor_distance + r_particle - r_circle) *
77  (neighbor_distance - r_particle + r_circle) *
78  (neighbor_distance + r_particle + r_circle));
79 }

81 /**
82  * @brief Calculates the volume of intersection between a spherical (3D) particle and a sphere
83  *
84  * @param r_particle Radius of the particle
85  *
86  * @param r_sphere Radius of the sphere
87  *
88  * @param neighbor_distance Distance between the particle and the sphere
89  */
90 
91 inline double
92 particle_sphere_intersection_3d(double r_particle,
93  double r_sphere,
94  double neighbor_distance)
95 {
96  return M_PI *
97  Utilities::fixed_power<2, double>(r_sphere + r_particle -
98  neighbor_distance) *
99  (Utilities::fixed_power<2, double>(neighbor_distance) +
100  (2 * neighbor_distance * r_particle) -
101  (3 * Utilities::fixed_power<2, double>(r_particle)) +
102  (2 * neighbor_distance * r_sphere) + (6 * r_sphere * r_particle) -
103  (3 * Utilities::fixed_power<2, double>(r_sphere))) /
104  (12 * neighbor_distance);
105 }

151  /**
152  * @brief This function calculates and returns the periodic offset distance of the domain which is needed
153  * for the periodic boundary conditions using the QCM or SPM for void fraction
154  * with the GLS VANS/CFD-DEM solver. The distance is based on one of the
155  * periodic boundaries and all particle location shifted by this distance is
156  * according to this periodic boundary.
157  *
158  * @param boundary_id The id of one of the periodic boundaries
159  *
160  * @return The periodic offset distance
161  */
162  inline Tensor<1, dim>
163  get_periodic_offset_distance(unsigned int boundary_id) const
164  {
165  Tensor<1, dim> offset;
166 
167  // Iterating over the active cells in the triangulation
168  for (const auto &cell : (*this->triangulation).active_cell_iterators())
169  {
170  if (cell->is_locally_owned() || cell->is_ghost())
171  {
172  if (cell->at_boundary())
173  {
174  // Iterating over cell faces
175  for (unsigned int face_id = 0; face_id < cell->n_faces();
176  ++face_id)
177  {
178  unsigned int face_boundary_id =
179  cell->face(face_id)->boundary_id();
180 
181  // Check if face is on the boundary, if so, get
182  // the periodic offset distance for one pair of periodic
183  // faces only since periodic boundaries are aligned with the
184  // direction and only axis are currently allowed
185  if (face_boundary_id == boundary_id)
186  {
187  Point<dim> face_center = cell->face(face_id)->center();
188  auto periodic_cell = cell->periodic_neighbor(face_id);
189  unsigned int periodic_face_id =
190  cell->periodic_neighbor_face_no(face_id);
191  Point<dim> periodic_face_center =
192  periodic_cell->face(periodic_face_id)->center();
193 
194  offset = periodic_face_center - face_center;
195 
196  return offset;
197  }
198  }
199  }
200  }
201  }
202 
203  // A zero tensor is returned in case no cells are found on the periodic
204  // boundaries on this processor. This processor won't handle particle in
205  // cells at periodic boundaries, so it won't affect any computation.
206  return offset;
207  }
