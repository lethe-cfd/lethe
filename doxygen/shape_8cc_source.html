53 template <int dim>
54 Point<dim>
55 Shape<dim>::align_and_center(const Point<dim> &evaluation_point) const
56 {
57  // Translation and rotation to standard position and orientation for
58  // distance calculations
59  Point<dim> center_of_rotation = position;
60  Point<dim> rotated_point;
61  Point<dim> translated_point;
62 
63  // Rotation from the solid orientation
64  // Angular position around x, y and z axis
65  Tensor<1, 3> theta = orientation;
66 
67  // The centralized point is the one to be rotated, and it is updated after
68  // each rotation around one axis The centralized rotated point is the result
69  // of each rotation, and it is initialized in case no rotation is performed
70  Point<dim> centralized_point;
71  centralized_point = evaluation_point - center_of_rotation;
72  Point<dim> centralized_rotated = centralized_point;
73 
74  // Selection of the first axis around which to rotate:
75  // x -> 0, y -> 1, z -> 2
76  // In 2D, only rotation around the z axis is possible
77  if constexpr (dim == 2)
78  {
79  if (std::abs(theta[2]) > 1e-10 * this->effective_radius)
80  {
81  Tensor<2, 2> rotation_matrix =
82  Physics::Transformations::Rotations::rotation_matrix_2d(-theta[2]);
83 
84  // Multiplication
85  centralized_rotated.clear();
86  for (unsigned int j = 0; j < dim; ++j)
87  {
88  for (unsigned int k = 0; k < dim; ++k)
89  {
90  centralized_rotated[j] +=
91  rotation_matrix[j][k] * centralized_point[k];
92  }
93  }
94  centralized_point = centralized_rotated;
95  }
96  }
97  else // (dim == 3)
98  {
99  for (unsigned int i = 0; i < 3; ++i)
100  {
101  if (std::abs(theta[i]) > 1e-10 * this->effective_radius)
102  {
103  Tensor<1, 3> axis;
104  axis[i] = 1.0;
105  Tensor<2, 3> rotation_matrix =
106  Physics::Transformations::Rotations::rotation_matrix_3d(
107  axis, -theta[i]);
108 
109  // Multiplication
110  centralized_rotated.clear();
111  for (unsigned int j = 0; j < dim; ++j)
112  {
113  for (unsigned int k = 0; k < dim; ++k)
114  {
115  centralized_rotated[j] +=
116  rotation_matrix[j][k] * centralized_point[k];
117  }
118  }
119  centralized_point = centralized_rotated;
120  }
121  }
122  }
123 
124  rotated_point = centralized_rotated + center_of_rotation;
125 
126  // Translation from the solid position
127  translated_point = rotated_point - position;
128 
129  return translated_point;
130 }

236 void
237 Shape<dim>::closest_surface_point(
238  const Point<dim> &p,
239  Point<dim> &closest_point,
240  const typename DoFHandler<dim>::active_cell_iterator &cell_guess)
241 {
242  Tensor<1, dim> actual_gradient;
243  double distance_from_surface;
244  actual_gradient = this->gradient_with_cell_guess(p, cell_guess);
245  distance_from_surface = this->value_with_cell_guess(p, cell_guess);
246  // Check if the gradient is well-defined. If the point is on the surface,
247  // the gradient can be badly defined for some shapes. We return the point
248  // directly in these cases since it would be on the surface.
249  closest_point = p - (actual_gradient / (actual_gradient.norm() +
250  1e-16 * this->effective_radius)) *

318 template <int dim>
319 Tensor<1, dim>
320 Sphere<dim>::gradient(const Point<dim> &evaluation_point,
321  const unsigned int /*component*/) const
322 {
323  // We make sure that the evaluation point and the sphere center are different,
324  // because if they are the same the analytical gradient is not defined: the
325  // function returns a NaN. We use the numerical gradient if the points are the
326  // same.
327  if ((evaluation_point - this->position).norm() <
328  1e-12 * this->effective_radius)
329  return AutoDerivativeFunction<dim>::gradient(evaluation_point);
330 #if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 4)
331  const Tensor<1, dim> center_to_point = evaluation_point - this->position;
332  const Tensor<1, dim> grad = center_to_point / center_to_point.norm();
333  return grad;
334 #else
335  return sphere_function->gradient(evaluation_point);
336 #endif
337 }

542 template <int dim>
543 void
544 Superquadric<dim>::closest_surface_point(const Point<dim> &p,
545  Point<dim> &closest_point) const
546 {
547  auto point_in_string = this->point_to_string(p);
548  auto iterator = this->closest_point_cache.find(point_in_string);
549  if (iterator == this->closest_point_cache.end())
550  {
551  // The initial guess is chosen as the centered point (i.e. evaluation
552  // point, in the superquadric referential). It should already be somewhat
553  // close to the closest point, and it is already in the right octant.
554  Point<dim> current_point = this->align_and_center(p);
555 
556  unsigned int iteration = 0;
557  const unsigned int iteration_max = 1e2;
558 
559  Point<dim> dx{}, distance_gradient{};
560  double current_distance = superquadric(current_point);
561 
562  double relaxation = 1;
563  while (iteration < iteration_max && abs(current_distance) > epsilon)
564  {
565  distance_gradient = superquadric_gradient(current_point);
566  // limit the step size
567 
568  if (distance_gradient.norm() < epsilon)
569  // Gradient can be null if the evaluation point is exactly on the
570  // centroid of the shape. In this case it is also the closest point.
571  break;
572  // This is a modified Newton method that limits the step size when the
573  // gradient norm is smaller then 1.
574  dx = -relaxation * (current_distance * distance_gradient) /
575  distance_gradient.norm() /
576  std::max(1.0, distance_gradient.norm());
577 
578  current_point = current_point + dx;
579  current_distance = superquadric(current_point);
580 
581  iteration++;
582  }
583  closest_point = this->reverse_align_and_center(current_point);
584  }
585  else
586  closest_point = iterator->second;
587 }

591 Superquadric<dim>::value(const Point<dim> &evaluation_point,
592  const unsigned int /*component*/) const
593 {

595

597  auto iterator = this->value_cache.find(point_in_string);
598  if (iterator == this->value_cache.end())
599  {
600  // The closest point has to be found first, because it is used in value

602  Point<dim> closest_point{};
603  this->closest_surface_point(evaluation_point, closest_point);

607  return (closest_point - evaluation_point).norm() -
608  this->layer_thickening;
609  else

611  this->layer_thickening;
612  }

1178  intermediate_q[1] * b_intermediate1[1]});
1179  Point<2> b;
1180  b = w - b_intermediate2;
1181  double k = (intermediate_q[1] > 0) ? 1 : ((intermediate_q[1] < 0) ? -1 : 0);
1182  double d = std::min(scalar_product<1, 2, double>(a, a),
1183  scalar_product<1, 2, double>(b, b));
1184  double s = std::max(k * (w[0] * intermediate_q[1] - w[1] * intermediate_q[0]),
1185  k * (w[1] - intermediate_q[1]));

1543 template <int dim>
1544 std::pair<double, Tensor<1, dim>>
1545 CompositeShape<dim>::apply_boolean_operations(
1546  std::map<unsigned int, double> constituent_shapes_values,
1547  std::map<unsigned int, Tensor<1, dim>> constituent_shapes_gradients) const
1548 {
1549  // The boolean operations between the shapes are applied in order
1550  // The last computed values are considered to be the correct values to return
1551  double levelset = constituent_shapes_values[0];
1552  Tensor<1, dim> gradient = constituent_shapes_gradients[0];
1553  for (auto const &[operation_id, op_triplet] : operations)
1554  {
1555  BooleanOperation operation;
1556  unsigned int first_id;
1557  unsigned int second_id;
1558  std::tie(operation, first_id, second_id) = op_triplet;
1559 
1560  double value_first_component, value_second_component;
1561  Tensor<1, dim> gradient_first_component{};
1562  Tensor<1, dim> gradient_second_component{};
1563  value_first_component = constituent_shapes_values.at(first_id);
1564  value_second_component = constituent_shapes_values.at(second_id);
1565  gradient_first_component = constituent_shapes_gradients.at(first_id);
1566  gradient_second_component = constituent_shapes_gradients.at(second_id);
1567  switch (operation)
1568  {
1569  case BooleanOperation::Union:
1570  if (value_first_component < value_second_component)
1571  {
1572  constituent_shapes_values[operation_id] = value_first_component;
1573  constituent_shapes_gradients[operation_id] =
1574  gradient_first_component;
1575  }
1576  else
1577  {
1578  constituent_shapes_values[operation_id] =
1579  value_second_component;
1580  constituent_shapes_gradients[operation_id] =
1581  gradient_second_component;
1582  }
1583  break;
1584  case BooleanOperation::Difference:
1585  if (-value_first_component > value_second_component)
1586  {
1587  constituent_shapes_values[operation_id] =
1588  -value_first_component;
1589  constituent_shapes_gradients[operation_id] =
1590  -gradient_first_component;
1591  }
1592  else
1593  {
1594  constituent_shapes_values[operation_id] =
1595  value_second_component;
1596  constituent_shapes_gradients[operation_id] =
1597  gradient_second_component;
1598  }
1599  break;
1600  default: // BooleanOperation::Intersection
1601  if (value_first_component > value_second_component)
1602  {
1603  constituent_shapes_values[operation_id] = value_first_component;
1604  constituent_shapes_gradients[operation_id] =
1605  gradient_first_component;
1606  }
1607  else
1608  {
1609  constituent_shapes_values[operation_id] =
1610  value_second_component;
1611  constituent_shapes_gradients[operation_id] =
1612  gradient_second_component;
1613  }
1614  }
1615  levelset = constituent_shapes_values[operation_id];
1616  gradient = constituent_shapes_gradients[operation_id];
1617  }
1618  return {levelset, gradient};
1619 }

1761 template <int dim>
1762 Tensor<1, dim>
1763 RBFShape<dim>::gradient(const Point<dim> &evaluation_point,
1764  const unsigned int /*component*/) const
1765 {
1766  Point<dim> centered_point = evaluation_point;
1767  double bounding_box_distance = bounding_box->value(centered_point);
1768  Tensor<1, dim> bounding_box_gradient = bounding_box->gradient(centered_point);
1769  if (bounding_box_distance > 0.)
1770  return bounding_box_gradient;
1771 
1772  double distance, normalized_distance;
1773  Point<dim> relative_position{};
1774  // We use chain derivation to express the gradient of the basis in regards to
1775  // the position d(basis)/dx = d(basis) /d(normalized_distance)
1776  // * d(normalized_distance)/d(distance)
1777  // * d(distance) /dx
1778  double dbasis_drnorm_derivative;
1779  double drnorm_dr_derivative;
1780  Tensor<1, dim> dr_dx_derivative{};
1781  Tensor<1, dim> gradient{};
1782 
1783  if (iterable_nodes.size() == 0)
1784  {
1785  throw std::logic_error(
1786  "Every location inside the bounding box should be covered by the radius "
1787  "of at least one RBF node. It this isn't the case, an error has been "
1788  "introduced in the code. ");
1789  }
1790  // Here we loop on every portion (of RBF nodes located in active cells close
1791  // to the evaluation point) and on every RBF node located in these active
1792  // cells.
1793  for (size_t portion_id = 0; portion_id < iterable_nodes.size(); portion_id++)
1794  {
1795  for (const size_t &node_id : *(std::get<2>(iterable_nodes[portion_id])))
1796  {
1797  // Calculation of dr/dx
1798  relative_position =
1799  (evaluation_point - rotated_nodes_positions[node_id]);
1800  distance = (relative_position).norm();
1801  normalized_distance = distance / support_radii[node_id];
1802  if (distance > 1e-16 * this->effective_radius)
1803  dr_dx_derivative = relative_position / distance;
1804  else
1805  {
1806  // If the evaluation point overlaps with the node position, we
1807  // assume that the contribution of this node is 0. This assumption
1808  // can be made because radial basis functions are symmetrical. If
1809  // can be made because radial basis functions are symmetrical. If
1810  // the basis function is not differentiable at its node (e.g.
1811  // linear function), this approximation will still hold since the
1812  // approximated distance field is already imperfect and
1813  // neighboring nodes will add their contribution at the evaluation
1814  // point.
1815  for (int d = 0; d < dim; d++)
1816  dr_dx_derivative[d] = 0;
1817  }
1818  // Calculation of dr_norm/dr
1819  drnorm_dr_derivative = 1.0 / support_radii[node_id];
1820  // Calculation of d(basis)/dr
1821  // We cast the basis function value to the proper RBFBasisFunction
1822  dbasis_drnorm_derivative = evaluate_basis_function_derivative(
1823  static_cast<enum RBFShape<dim>::RBFBasisFunction>(
1824  round(basis_functions[node_id])),
1825  normalized_distance);
1826  // Sum
1827  gradient += dbasis_drnorm_derivative * drnorm_dr_derivative *
1828  dr_dx_derivative * weights[node_id];
1829  }
1830  }
1831  return gradient;
1832 }
