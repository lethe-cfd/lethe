28 /**
29  * @brief Class that assembles the core of the Navier-Stokes equation with
30  * free surface using VOF modeling.
31  * According to the following weak form:
32  * \f$ \nabla \cdot \mathbf{u} + \rho \mathbf{u} \cdot \nabla
33  * \mathbf{u} + \nabla p - \mu \nabla\cdot (\nabla \mathbf{u} +
34  * (\nabla \mathbf{u})^T) = 0 \f$ with an SUPG and PSPG stabilization
35  *
36  * @tparam dim An integer that denotes the number of spatial dimensions
37  *
38  * @ingroup assemblers
39  */
40 template <int dim>
41 class GLSNavierStokesVOFAssemblerCore : public NavierStokesAssemblerBase<dim>
42 {
43 public:
44  GLSNavierStokesVOFAssemblerCore(
45  std::shared_ptr<SimulationControl> simulation_control,
46  const SimulationParameters<dim> &nsparam)
47  : simulation_control(simulation_control)
48  , vof_parameters(nsparam.multiphysics.vof_parameters)
49  {}

285  /**
286  * @brief Calculates an approximation of the gradient of the viscosity
287  * @param velocity_gradient The velocity gradient tensor on the quadrature point
288  @param velocity_hessians The velocity hessian tensor on the quadrture point
289  @param non_newtonian_viscosity The viscosity at which the gradient is calculated
290  @param d_gamma_dot Th difference in the shear rate magnitude to approximate the
291  viscosity variation with a slight change in the shear_rate magnitude
292  */
293  inline Tensor<1, dim>
294  get_kinematic_viscosity_gradient(
295  const Tensor<2, dim> &velocity_gradient,
296  const Tensor<3, dim> &velocity_hessians,
297  const double shear_rate_magnitude,
298  const double grad_kinematic_viscosity_shear_rate) const
299  {
300  // Calculates an approximation of the shear rate magnitude gradient using
301  // the derived form, since it does not change with rheological models
302  Tensor<1, dim> grad_shear_rate;
303  for (unsigned int d = 0; d < dim; ++d)
304  {
305  if (dim == 2)
306  {
307  for (unsigned int k = 0; k < dim; ++k)
308  {
309  grad_shear_rate[d] +=
310  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
311  shear_rate_magnitude;
312  }
313  grad_shear_rate[d] +=
314  (velocity_gradient[0][1] + velocity_gradient[1][0]) *
315  (velocity_hessians[0][d][1] + velocity_hessians[1][d][0]) /
316  shear_rate_magnitude;
317  }
318  else
319  {
320  for (unsigned int k = 0; k < dim; ++k)
321  {
322  grad_shear_rate[d] +=
323  2 * (velocity_gradient[k][k] * velocity_hessians[k][d][k]) /
324  shear_rate_magnitude +
325  (velocity_gradient[(k + 1) % dim][(k + 2) % dim] +
326  velocity_gradient[(k + 2) % dim][(k + 1) % dim]) *
327  (velocity_hessians[(k + 1) % dim][d][(k + 2) % dim] +
328  velocity_hessians[(k + 2) % dim][d][(k + 1) % dim]) /
329  shear_rate_magnitude;
330  }
331  }
332  }
333  return grad_shear_rate * grad_kinematic_viscosity_shear_rate;
334  };
