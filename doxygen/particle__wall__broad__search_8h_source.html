166  /**
167  * @brief Stores the candidate particle-wall contact pairs.
168  *
169  * @param particle_iterator The particle iterator to start boundary id.

171  * store.
172  * @param contact_pair_candidates A map which will contain all the

174  */
175  inline void
176  store_candidates(
177  const typename Particles::ParticleHandler<
178  dim>::particle_iterator_range::iterator &particle_iterator,
179  const boundary_cells_info_struct<dim> &boundary_cells_content,
180  typename DEM::dem_data_structures<dim>::particle_wall_candidates
181  &contact_pair_candidates)
182  {
183  // Find the contact candidate container of the particle
184  const types::particle_index particle_id = particle_iterator->get_id();
185  auto candidates_container_it = contact_pair_candidates.find(particle_id);
186 
187  // Reserve arbitrary vector capacity and store if the particle does not have
188  // contact candidate yet
189  if (candidates_container_it == contact_pair_candidates.end())
190  {
191  auto pair_it_bool = contact_pair_candidates.emplace(
192  particle_id,
193  ankerl::unordered_dense::map<
194  DEM::global_face_id,
195  std::tuple<Particles::ParticleIterator<dim>,
196  Tensor<1, dim>,
197  Point<dim>,
198  DEM::global_face_id>>());
199 
200  candidates_container_it = pair_it_bool.first;
201  }
202 
203  // Store particle ids from the selected particle iterator
204  candidates_container_it->second.emplace(
205  boundary_cells_content.global_face_id,
206  std::make_tuple(particle_iterator,
207  boundary_cells_content.normal_vector,
208  boundary_cells_content.point_on_face,
209  boundary_cells_content.boundary_id));
210  }
