56 void
57 ParticlesForceChains<dim, contact_model, rolling_friction_model>::
58  multi_general_cell(Triangulation<1, 3> &tria,
59  const std::vector<Point<3>> &vertices)
60 {
61  const unsigned int n_cells = vertices.size() / 2;
62  std::vector<CellData<1>> cells(n_cells, CellData<1>());
63  for (unsigned int i = 0; i < n_cells; ++i)
64  {
65  cells[i].vertices[0] = 2 * i;
66  cells[i].vertices[1] = 2 * i + 1;
67  cells[i].material_id = 0;
68  };
69  tria.create_triangulation(vertices, cells, SubCellData());
70 }

72 template <
73  int dim,
74  Parameters::Lagrangian::ParticleParticleContactForceModel contact_model,
75  Parameters::Lagrangian::RollingResistanceMethod rolling_friction_model>
76 void
77 ParticlesForceChains<dim, contact_model, rolling_friction_model>::
78  calculate_force_chains(DEMContactManager<dim> &container_manager)
79 {
80  auto &local_adjacent_particles = container_manager.local_adjacent_particles;
81  // Lines 89 to 101 kept for future ghost particles implementation.
82  // auto &ghost_adjacent_particles =
83  // container_manager.ghost_adjacent_particles; auto
84  // &local_periodic_adjacent_particles =
85  // container_manager.local_periodic_adjacent_particles;
86  // auto &ghost_periodic_adjacent_particles =
87  // container_manager.ghost_periodic_adjacent_particles;
88  // auto &ghost_local_periodic_adjacent_particles =
89  // container_manager.ghost_local_periodic_adjacent_particles;
90 
91  // Define local variables which will be used within the contact calculation
92  // Namely: normal and tangential contact forces, tangential and rolling
93  // torques, normal unit vector of the contact and contact relative velocity
94  // in the normal direction
95  Tensor<1, 3> normal_unit_vector;
96  Tensor<1, 3> normal_force;
97  Tensor<1, 3> tangential_force;
98  Tensor<1, 3> particle_one_tangential_torque;
99  Tensor<1, 3> particle_two_tangential_torque;
100  Tensor<1, 3> rolling_resistance_torque;
101  double normal_relative_velocity_value;
102  Tensor<1, 3> tangential_relative_velocity;
103 
104  // Contact forces calculations of local-local and local-ghost particle
105  // pairs are performed in separate loops
106 
107  // Looping over local_adjacent_particles values with iterator
108  // adjacent_particles_list
109  for (auto &&adjacent_particles_list :
110  local_adjacent_particles | boost::adaptors::map_values)
111  {
112  if (!adjacent_particles_list.empty())
113  {
114  // Gather information about particle 1 and set it up.
115  auto first_contact_info = adjacent_particles_list.begin();
116  auto particle_one = first_contact_info->second.particle_one;
117  auto particle_one_properties = particle_one->get_properties();
118 
119 
120  // Fix particle one location for 2d and 3d
121  Point<3> particle_one_location = [&] {
122  if constexpr (dim == 3)
123  {
124  return particle_one->get_location();
125  }
126  if constexpr (dim == 2)
127  {
128  return (point_nd_to_3d(particle_one->get_location()));
129  }
130  }();
131 
132  for (auto &&contact_info :
133  adjacent_particles_list | boost::adaptors::map_values)
134  {
135  // Getting information (location and properties) of particle 2 in
136  // contact with particle 1
137  auto particle_two = contact_info.particle_two;
138  auto particle_two_properties = particle_two->get_properties();
139 
140  // Get particle 2 location in dimension independent way
141  Point<3> particle_two_location = [&] {
142  if constexpr (dim == 3)
143  {
144  return particle_two->get_location();
145  }
146  if constexpr (dim == 2)
147  {
148  return (point_nd_to_3d(particle_two->get_location()));
149  }
150  }();
151 
152  // Calculation of normal overlap
153  double normal_overlap =
154  0.5 * (particle_one_properties[PropertiesIndex::dp] +
155  particle_two_properties[PropertiesIndex::dp]) -
156  particle_one_location.distance(particle_two_location);
157 
158  if (normal_overlap > 0.0)
159  {
160  // This means that the adjacent particles are in contact
161  // Since the normal overlap is already calculated, we update
162  // this element of the container here. The rest of information
163  // are updated using the following function
164  this->update_contact_information(
165  contact_info,
166  tangential_relative_velocity,
167  normal_relative_velocity_value,
168  normal_unit_vector,
169  particle_one_properties,
170  particle_two_properties,
171  particle_one_location,
172  particle_two_location,
173  0);
174  if constexpr (contact_model ==
175  Parameters::Lagrangian::
176  ParticleParticleContactForceModel::DMT)
177  {
178  this->calculate_DMT_contact(
179  contact_info,
180  tangential_relative_velocity,
181  normal_relative_velocity_value,
182  normal_unit_vector,
183  normal_overlap,
184  particle_one_properties,
185  particle_two_properties,
186  normal_force,
187  tangential_force,
188  particle_one_tangential_torque,
189  particle_two_tangential_torque,
190  rolling_resistance_torque);
191  }
192 
193  if constexpr (contact_model ==
194  Parameters::Lagrangian::
195  ParticleParticleContactForceModel::linear)
196  {
197  this->calculate_linear_contact(
198  contact_info,
199  tangential_relative_velocity,
200  normal_relative_velocity_value,
201  normal_unit_vector,
202  normal_overlap,
203  particle_one_properties,
204  particle_two_properties,
205  normal_force,
206  tangential_force,
207  particle_one_tangential_torque,
208  particle_two_tangential_torque,
209  rolling_resistance_torque);
210  }
211 
212  if constexpr (contact_model ==
213  Parameters::Lagrangian::
214  ParticleParticleContactForceModel::hertz)
215  {
216  this->calculate_hertz_contact(
217  contact_info,
218  tangential_relative_velocity,
219  normal_relative_velocity_value,
220  normal_unit_vector,
221  normal_overlap,
222  particle_one_properties,
223  particle_two_properties,
224  normal_force,
225  tangential_force,
226  particle_one_tangential_torque,
227  particle_two_tangential_torque,
228  rolling_resistance_torque);
229  }
230 
231  if constexpr (contact_model ==
232  Parameters::Lagrangian::
233  ParticleParticleContactForceModel::hertz_JKR)
234  {
235  this->calculate_hertz_JKR_contact(
236  contact_info,
237  tangential_relative_velocity,
238  normal_relative_velocity_value,
239  normal_unit_vector,
240  normal_overlap,
241  particle_one_properties,
242  particle_two_properties,
243  normal_force,
244  tangential_force,
245  particle_one_tangential_torque,
246  particle_two_tangential_torque,
247  rolling_resistance_torque);
248  }
249 
250  if constexpr (contact_model ==
251  Parameters::Lagrangian::
252  ParticleParticleContactForceModel::
253  hertz_mindlin_limit_force)
254  {
255  this->calculate_hertz_mindlin_limit_force_contact(
256  contact_info,
257  tangential_relative_velocity,
258  normal_relative_velocity_value,
259  normal_unit_vector,
260  normal_overlap,
261  particle_one_properties,
262  particle_two_properties,
263  normal_force,
264  tangential_force,
265  particle_one_tangential_torque,
266  particle_two_tangential_torque,
267  rolling_resistance_torque);
268  }
269 
270  if constexpr (contact_model ==
271  Parameters::Lagrangian::
272  ParticleParticleContactForceModel::
273  hertz_mindlin_limit_overlap)
274  {
275  this->calculate_hertz_mindlin_limit_overlap_contact(
276  contact_info,
277  tangential_relative_velocity,
278  normal_relative_velocity_value,
279  normal_unit_vector,
280  normal_overlap,
281  particle_one_properties,
282  particle_two_properties,
283  normal_force,
284  tangential_force,
285  particle_one_tangential_torque,
286  particle_two_tangential_torque,
287  rolling_resistance_torque);
288  }
289 
290  vertices.push_back(particle_one_location);
291  vertices.push_back(particle_two_location);
292  force_normal.push_back(sqrt(normal_force.norm()));
293  }
294  }
295  }
296  }
297 }

299 template <
300  int dim,
301  Parameters::Lagrangian::ParticleParticleContactForceModel contact_model,
302  Parameters::Lagrangian::RollingResistanceMethod rolling_friction_model>
303 void
304 ParticlesForceChains<dim, contact_model, rolling_friction_model>::
305  write_force_chains(MPI_Comm mpi_communicator,
306  const std::string folder,
307  const unsigned int iter)
308 {
309  Triangulation<1, 3> triangulation;
310  multi_general_cell(triangulation, vertices);
311  DoFHandler<1, 3> force_dh(triangulation);
312  DataOut<1, 3> data_out;
313  data_out.attach_dof_handler(force_dh);
314 
315  Vector<float> force_values(triangulation.n_active_cells());
316  for (unsigned int i = 0; i < force_values.size(); ++i)
317  {
318  force_values[i] = force_normal[i];
319  }
320  data_out.add_data_vector(force_values, "force");
321 
322  data_out.build_patches();
323 
324  const std::string face_filename =
325  (folder + "force_chains." + Utilities::int_to_string(iter, 5) + ".vtu");
326  data_out.write_vtu_in_parallel(face_filename.c_str(), mpi_communicator);
327 }
