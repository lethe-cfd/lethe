82 /**
83  * @brief Refine a mesh around specific boundary ids
84  *
85  * @param[in] n_refinement The number of times the boundary should be refined
86  *
87  * @param[in] boundary_ids The boundary ids for which the cells should be
88  * refined
89  *
90  * @param[in, out] triangulation The triangulation on which refinement must be
91  * carried out
92  *
93  */
94 template <int dim, int spacedim = dim>
95 void
96 refine_triangulation_at_boundaries(
97  const std::vector<int> boundary_ids,
98  const unsigned int n_refinement,
99  parallel::DistributedTriangulationBase<dim, spacedim> &triangulation)
100 {
101  // For the amount of refinements required
102  for (unsigned int r = 0; r < n_refinement; ++r)
103  {
104  // Loop over the cells and flag all cells which are within the list of
105  // boundary ids
106  for (const auto &cell : triangulation.active_cell_iterators())
107  {
108  if (cell->is_locally_owned())
109  {
110  for (const auto f : cell->face_indices())
111 
112  {
113  if (cell->face(f)->at_boundary())
114  {
115  if (std::find(boundary_ids.begin(),
116  boundary_ids.end(),
117  cell->face(f)->boundary_id()) !=
118  boundary_ids.end())
119  {
120  cell->set_refine_flag();
121  }
122  }
123  }
124  }
125  }
126  triangulation.execute_coarsening_and_refinement();
127  }
128 }
