70 /**
71  * @brief Calculates a filtered phase fraction for Cahn-Hilliard simulations.
72  * In this case, a simple clamping is performed on the phase fraction parameter
73  * for it to remain in the [-1,1] interval.
74  */
75 class CahnHilliardClipFilter : public CahnHilliardFilterBase
76 {
77 public:
78  CahnHilliardClipFilter()
79  {}
80 
81  /**
82  * @brief Calculates the value of the filtered phase fraction
83  * @param unfiltered_phase Value of the phase fraction before applying the
84  * filter
85  * @return Value of the phase fraction after applying the filter
86  */
87  virtual double
88  filter_phase(const double &unfiltered_phase) override
89  {
90  return (std::abs(unfiltered_phase) < 1) ? unfiltered_phase :
91  sgn(unfiltered_phase);
92  }
93 };

95 /**
96  * @brief Calculates a filtered phase fraction for Cahn-Hilliard simulations.
97  * The filtered phase is defined as \f$\phi_f = \tanh(\beta \phi)\f$.
98  */
99 class CahnHilliardTanhFilter : public CahnHilliardFilterBase
100 {
101 public:
102  CahnHilliardTanhFilter(const double beta)
103  : beta(beta)
104  {}
105 
106  /**
107  * @brief Calculates the value of the filtered phase fraction
108  * @param unfiltered_phase Value of the phase fraction before applying the
109  * filter
110  * @return Value of the phase fraction after applying the filter
111  */
112  virtual double
113  filter_phase(const double &unfiltered_phase) override
114  {
115  return tanh(beta * unfiltered_phase);
116  }
117 
118 private:
119  // User-defined parameter that influences how sharp the filtering is.
120  const double beta;
121 };
