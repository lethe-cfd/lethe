71 /**

73  * In this case, a simple clamping is performed on the phase fraction parameter
74  * for it to remain in the [-1,1] interval.

78 public:

82  /**
83  * @brief Calculate the value of the filtered phase fraction.
84  * @param[in] unfiltered_phase Value of the phase fraction before applying the
85  * filter.
86  * @return Value of the phase fraction after applying the filter.
87  */
88  virtual double
89  filter_phase(const double &unfiltered_phase) override
90  {
91  return (std::abs(unfiltered_phase) < 1) ? unfiltered_phase :
92  sgn(unfiltered_phase);
93  }
94 };

96 /**
97  * @brief Calculate a filtered phase fraction for Cahn-Hilliard simulations.
98  *
99  * The filtered phase is defined as \f$\phi_f = \tanh(\beta \phi)\f$ with
100  * \f$ \phi \f$ the unfiltered phase field.
101  */
102 class CahnHilliardTanhFilter : public CahnHilliardFilterBase
103 {
104 public:
105  CahnHilliardTanhFilter(const double beta)
106  : beta(beta)
107  {}
108 
109  /**
110  * @brief Calculate the value of the filtered phase fraction.
111  * @param[in] unfiltered_phase Value of the phase fraction before applying the
112  * filter.
113  * @return Value of the phase fraction after applying the filter.
114  */
115  virtual double
116  filter_phase(const double &unfiltered_phase) override
117  {
118  return tanh(beta * unfiltered_phase);
119  }
120 
121 private:
122  /**
123  * User-defined parameter that influences how sharp the filtering is.
124  */
125  const double beta;
126 };
