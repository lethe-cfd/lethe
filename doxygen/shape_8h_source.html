137  /**
138  * @brief Return the smoothed maximum of two variables used for shape contact calculation.
139  * @param a first variable
140  * @param b second variable
141  */
142  double
143  smooth_max(double a, double b, double smooth_factor = 10)
144  {
145  return (a * std::exp(a * smooth_factor) + b * std::exp(b * smooth_factor)) /
146  ((std::exp(a * smooth_factor) + std::exp(b * smooth_factor)));

149  /**
150  * @brief Return the distance, the center point, and the unit normal vector between the current shape and the shape given in the argument. By default, the function does not calculate the distance between the two shapes if their bounding boxes are not in contact. This behavior can be modified by setting exact_distance_outside_of_contact to true.
151  * @param shape The shape with which the distance is evaluated
152  * @param cell The cell that is likely to contain the evaluation point
153  * @param candidate_points This is the initial guess points used in the calculation.
154  * @param precision This is the precision of the distance between the two shapes.
155  * @param exact_distance_outside_of_contact This is a boolean to force the exact distance evaluation if the shapes are not in contact.
156  */
157  virtual std::tuple<double, Tensor<1, dim>, Point<dim>>
158  distance_to_shape_with_cell_guess(
159  Shape<dim> &shape,
160  const typename DoFHandler<dim>::active_cell_iterator &cell,
161  std::vector<Point<dim>> &candidate_points,
162  double precision = 1e-6,
163  bool exact_distance_outside_of_contact = false)
164  {
165  // Initialized the default values.
166  double distance = DBL_MAX;
167  Tensor<1, dim> normal;
168  Point<dim> contact_point;
169  bool bounding_box_contact = true;
170 
171  // Check the bounding boxes. First we check the spherical bounding boxes
172  // then the rectangular boxes.
173  if (exact_distance_outside_of_contact == false)
174  {
175  Point<dim> bounding_box_center_one;
176  Point<dim> bounding_box_center_two;
177  double radius_one = this->bounding_box_half_length.norm();
178  double radius_two = shape.bounding_box_half_length.norm();
179  // In the case of the plane, the radius of the bounding box is zero, so
180  // we replace it with the radius of the other object.
181  if (radius_one == 0)
182  {
183  radius_one = radius_two;
184  }
185  if (radius_two == 0)
186  {
187  radius_two = radius_one;
188  }
189  if constexpr (dim == 3)
190  {
191  bounding_box_center_one =
192  this->position +
193  this->rotation_matrix * this->bounding_box_center;
194  bounding_box_center_two =
195  shape.position +
196  shape.rotation_matrix * shape.bounding_box_center;
197  }
198  else
199  {
200  bounding_box_center_one =
201  this->position +
202  tensor_nd_to_2d(this->rotation_matrix *
203  point_nd_to_3d(this->bounding_box_center));
204  bounding_box_center_two =
205  shape.position +
206  tensor_nd_to_2d(shape.rotation_matrix *
207  point_nd_to_3d(shape.bounding_box_center));
208  }
209  if (((bounding_box_center_one - bounding_box_center_two).norm() -
210  radius_one - radius_two) <= 0)
211  {
212  bounding_box_contact = this->bounding_box_contact(shape);
213  }
214  else
215  {
216  bounding_box_contact = false;
217  }
218  }

220  // The following algorithm does a minimization of the level set obtained by
221  // the intersection of two shapes. A usual gradient descent does not work as
222  // such. This algorithm uses a cartesian search combined with an additional
223  // guess based on the gradient of the level set and a numerical gradient
224  // descent.
225  if (bounding_box_contact)
226  {
227  std::vector<Tensor<1, dim>> search_direction(2 * dim);
228  // Define the cartesian direction search directions.
229  search_direction[0][0] = 1.;
230  search_direction[1][0] = -1.;
231  search_direction[2][1] = 1.;
232  search_direction[3][1] = -1.;
233  if constexpr (dim == 3)
234  {
235  search_direction[4][2] = 1.;
236  search_direction[5][2] = -1.;
237  }
238  // Loop over all the initial candidate points.
239  for (unsigned int i = 0; i < candidate_points.size(); ++i)
240  {
241  // Initialize variables used in the calculation.
242  Point<dim> current_point = candidate_points[i];
243  Point<dim> dx{}, distance_gradient{}, previous_position{},
244  previous_gradient{};
245  // Initialize the iteration counter.
246  unsigned int iteration = 0;
247  const unsigned int iteration_max = 2e2;
248 
249  // The initial step size is set to 25% of the effective radius of
250  // the shape. Tests showed that this initial step size is generally
251  // adequate.
252  double max_step = shape.effective_radius * 0.25;
253  double previous_step_size = max_step;
254 
255  // Initialize the value. In this minimization of the intersection of
256  // two level set we use the smooth max function for the union as it
257  // significantly smooths the minimization problem as we get close to
258  // the local minimum of the intersection.
259  double value_first_component =
260  this->value_with_cell_guess(current_point, cell);
261  double value_second_component =
262  shape.value_with_cell_guess(current_point, cell);
263  double current_distance =
264  smooth_max(value_first_component, value_second_component);
265 
266  previous_position = current_point;
267  // The initial previous value and position
268  double previous_value = DBL_MAX;
269  unsigned int consecutive_center = 0;
270 
271  // Iterate to find the minimum of the intersection.
272  while (iteration < iteration_max && previous_step_size > precision)
273  {
274  // Check the local gradient direction.
275  value_first_component =
276  this->value_with_cell_guess(current_point, cell);
277  value_second_component =
278  shape.value_with_cell_guess(current_point, cell);
279  if (value_first_component > value_second_component)
280  {
281  distance_gradient =
282  this->gradient_with_cell_guess(current_point, cell);
283  }
284  else
285  {
286  distance_gradient =
287  shape.gradient_with_cell_guess(current_point, cell);
288  }
289  Tensor<1, dim> direction = distance_gradient;
290  dx = -max_step * direction / direction.norm();
291  if (distance_gradient.norm() == 0)
292  {
293  dx = -max_step * direction;
294  }
295  // Check the first candidate point in the direction of the
296  // gradient.
297  Point<dim> new_point = current_point + dx;
298  value_first_component =
299  this->value_with_cell_guess(new_point, cell);
300  value_second_component =
301  shape.value_with_cell_guess(new_point, cell);
302  double new_distance =
303  smooth_max(value_first_component, value_second_component);
304  // Check if the guess is better than the previous guess. If it
305  // is not better, we do the cartesian directional search.
306  if (new_distance > current_distance - precision * precision)
307  {
308  // Initialize the container for the value.
309  std::vector<double> diff_results;
310  diff_results.resize(search_direction.size());
311  Point<dim> best_point;
312  double best_dist = current_distance;
313  for (unsigned int d = 0; d < search_direction.size(); ++d)
314  {
315  Tensor<1, dim> perturbation =
316  search_direction[d] * max_step;
317  value_first_component = this->value_with_cell_guess(
318  current_point + perturbation, cell);
319  value_second_component = shape.value_with_cell_guess(
320  current_point + perturbation, cell);
321  new_distance = smooth_max(value_first_component,
322  value_second_component);
323  diff_results[d] =
324  (new_distance - current_distance) / max_step;
325  // Check if the Cartesian search is better than the last
326  // candidate.
327  if (new_distance < best_dist - precision * precision)
328  {
329  best_dist = new_distance;
330  best_point = current_point + perturbation;
331  }
332  }
333  // Store the current results.
334  if (best_dist < current_distance - precision * precision)
335  {
336  current_distance = best_dist;
337  current_point = best_point;
338  }
339  Point<dim> extra_guess;
340  // If none of the cartesian candidates are better, we use
341  // the value to find the point that minimizes the numerical
342  // gradient we evaluated with the cartesian guess.
343  if (current_distance >
344  previous_value - precision * precision)
345  {
346  // Define the guess point.
347  new_point[0] =
348  previous_position[0] -
349  ((diff_results[0] - diff_results[1]) / 2) /
350  ((diff_results[0] + diff_results[1]) / max_step);
351  new_point[1] =
352  previous_position[1] -
353  ((diff_results[2] - diff_results[3]) / 2) /
354  ((diff_results[2] + diff_results[3]) / max_step);
355  if constexpr (dim == 3)
356  {
357  new_point[2] =
358  previous_position[2] -
359  ((diff_results[4] - diff_results[5]) / 2) /
360  ((diff_results[4] + diff_results[5]) /
361  max_step);
362  }
363  value_first_component =
364  this->value_with_cell_guess(new_point, cell);
365  value_second_component =
366  shape.value_with_cell_guess(new_point, cell);
367  new_distance = smooth_max(value_first_component,
368  value_second_component);
369  // Check if it is better.
370  if (new_distance <
371  previous_value - precision * precision)
372  {
373  // If it is, we reduce the step size to the size of
374  // the step made by this guess.
375  max_step = (new_point - previous_position).norm();
376  current_distance = new_distance;
377  current_point = new_point;
378  }
379  }
380  // If the guess is not better, we reduce the search size and
381  // store the value. If one of the points is better, we reset
382  // the consecutive_center counter. This variable helps to
383  // converge faster when one of the guesses is very close to
384  // the optimal point.
385  if (current_distance >
386  previous_value - precision * precision)
387  {
388  consecutive_center += 1;
389  previous_step_size = max_step;
390  max_step *= 1.0 / std::pow(2.0, consecutive_center);
391  current_point = previous_position;
392  current_distance = previous_value;
393  }
394  else
395  {
396  max_step *= 1;
397  consecutive_center = 0;
398  }
399  }
400  else
401  {
402  current_point = new_point;
403  current_distance = new_distance;
404  max_step *= 1;
405  consecutive_center = 0;
406  }
407 
408  // Store the previous values.
409  previous_value = current_distance;
410  previous_position = current_point;
411  iteration++;
412  }
413  // Store the minimum obtained with this initial guess and compare it
414  // with the results for other initial points.
415  if (distance > current_distance)
416  {
417  distance = current_distance;
418  contact_point = current_point;
419  // The normal is defined using the closest surface point on the
420  // two shapes.
421  Point<dim> closest_surface_point_on_this_shape;
422  Point<dim> closest_surface_point_on_the_input_shape;
423  this->closest_surface_point(current_point,
424  closest_surface_point_on_this_shape,
425  cell);
426  shape.closest_surface_point(
427  current_point,
428  closest_surface_point_on_the_input_shape,
429  cell);
430  if (distance > 0)
431  {
432  normal = (closest_surface_point_on_the_input_shape -
433  closest_surface_point_on_this_shape) /
434  ((closest_surface_point_on_the_input_shape -
435  closest_surface_point_on_this_shape)
436  .norm() +
437  DBL_MIN);
438  }
439  else
440  {
441  normal = (closest_surface_point_on_this_shape -
442  closest_surface_point_on_the_input_shape) /
443  ((closest_surface_point_on_this_shape -
444  closest_surface_point_on_the_input_shape)
445  .norm() +
446  DBL_MIN);
447  }
448  }
449  }
450  }
451  return std::make_tuple(distance, normal, contact_point);
452  }

648  new_point[1] =
649  previous_position[1] -
650  ((diff_results[2] - diff_results[3]) / 2) /
651  ((diff_results[2] + diff_results[3]) / max_step);
652  if constexpr (dim == 3)
653  {
654  new_point[2] =
655  previous_position[2] -
656  ((diff_results[4] - diff_results[5]) / 2) /
657  ((diff_results[4] + diff_results[5]) /
658  max_step);
659  }
660  value_first_component = this->value(new_point);
661  value_second_component = shape.value(new_point);
662  new_distance = smooth_max(value_first_component,
663  value_second_component);
664  // Check if it is better.
665  if (new_distance <
666  previous_value - precision * precision)
667  {
668  // If it is, we reduce the step size to the size of
669  // the step made by this guess.
670  max_step = (new_point - previous_position).norm();
671  current_distance = new_distance;
672  current_point = new_point;
673  }
674  }
675  // If the guess is not better, we reduce the search size and
676  // store the value. If one of the points is better, we reset
677  // the consecutive_center counter. This variable helps to
678  // converge faster when one of the guesses is very close to
679  // the optimal point.
680  if (current_distance >
681  previous_value - precision * precision)
682  {
683  consecutive_center += 1;
684  previous_step_size = max_step;
685  max_step *= 1.0 / std::pow(2.0, consecutive_center);
686  current_point = previous_position;
687  current_distance = previous_value;
688  }
689  else
690  {
691  max_step *= 1;
692  consecutive_center = 0;
693  }
694  }
695  else
696  {
697  current_point = new_point;
698  current_distance = new_distance;
699  max_step *= 1;
700  consecutive_center = 0;
701  }
702 
703  // Store the previous values.
704  previous_value = current_distance;
705  previous_position = current_point;
706  iteration++;
707  }
708  // Store the minimum obtained with this initial guess and compare it
709  // with the results for other initial points.
710  if (distance > current_distance)
711  {
712  distance = current_distance;
713  contact_point = current_point;
714  Point<dim> closest_surface_point_on_this_shape;
715  Point<dim> closest_surface_point_on_the_input_shape;
716  // The normal is defined using the closest surface point on the
717  // two shapes.
718  this->closest_surface_point(
719  current_point, closest_surface_point_on_this_shape);
720  shape.closest_surface_point(
721  current_point, closest_surface_point_on_the_input_shape);
722  if (distance > 0)
723  {
724  normal = (closest_surface_point_on_the_input_shape -
725  closest_surface_point_on_this_shape) /
726  ((closest_surface_point_on_the_input_shape -
727  closest_surface_point_on_this_shape)
728  .norm() +
729  DBL_MIN);
730  }
731  else
732  {
733  normal = (closest_surface_point_on_this_shape -
734  closest_surface_point_on_the_input_shape) /
735  ((closest_surface_point_on_this_shape -
736  closest_surface_point_on_the_input_shape)
737  .norm() +
738  DBL_MIN);
739  }
740  }
741  }
742  }
743  return std::make_tuple(distance, normal, contact_point);

1085  */

1088  {
1089  double curvature = 0;
1090  double epsilone = 1e-6 * this->effective_radius;
1091  for (unsigned int d = 0; d < dim; ++d)
1092  {
1093  Tensor<1, dim> perturbation;
1094  perturbation[d] = epsilone;
1095  Tensor<1, dim> gradient_p = this->gradient(p + perturbation);
1096  gradient_p = gradient_p / (gradient_p.norm() + DBL_MIN);
1097  Tensor<1, dim> gradient_m = this->gradient(p - perturbation);
1098  gradient_m = gradient_m / (gradient_m.norm() + DBL_MIN);
1099  curvature += (gradient_p - gradient_m)[d] / (2 * epsilone);
1100  }
1101  // Here we bound the curvature radius obtained
1102  double radius_of_curvature = 1.0 / abs(curvature);
1103  return std::min(std::max(radius_of_curvature,
1104  this->effective_radius * 1e-2),
1105  this->effective_radius * 1e2);
1106  }
