62  /**
63  * The cell_weight() function indicates to the triangulation how much
64  * computational work is expected to happen on this cell, and consequently
65  * how the domain needs to be partitioned so that every MPI rank receives a
66  * roughly equal amount of work (potentially not an equal number of cells).
67  * While the function is called from the outside, it is connected to the
68  * corresponding signal from inside this class, therefore it can be private.
69  * This function is the key component that allow us to dynamically balance the
70  * computational load. The function attributes a weight to
71  * every cell that represents the computational work on this cell. Here the
72  * majority of work is expected to happen on the particles, therefore the
73  * return value of this function (representing "work for this cell") is
74  * calculated based on the number of particles in the current cell.
75  * The function is connected to the cell_weight() signal inside the
76  * triangulation, and will be called once per cell, whenever the triangulation
77  * repartitions the domain between ranks (the connection is created inside the
78  * particles_generation() function of this class).
79  */

247  /**
248  * @brief Checks all the conditions that require a contact search step. The
249  * check of conditions is done in order of suspected frequency occurrence.
250  *
251  * @param counter The counter of DEM iterations in a CFD iteration.
252  */
253  inline bool
254  contact_search_step(const unsigned int counter)
255  {
256  if (contact_detection_step)
257  {
258  // Contact search step according to the contact detection method
259  return true;
260  }
261  else if (counter == (coupling_frequency - 1) &&
262  (contact_search_counter == 0))
263  {
264  // Ensure that the contact search is executed at least once per CFD time
265  // step.
266  return true;
267  }
268  else if (has_sparse_contacts && (counter == 1))
269  {
270  // First mobility status identification of the CFD time step (from the
271  // velocity computed at the first DEM time step (counter = 0) of the CFD
272  // time step) The contact search is executed to make sure the mobility
273  // status of cell match the particles that are in.
274  return true;
275  }
276  else if (load_balance_step)
277  {
278  // Needs to update contacts since particles/cells may have been
279  // distributed to a different subdomain
280  return true;
281  }
282  else if ((this->simulation_control->is_at_start() && (counter == 0)) ||
283  checkpoint_step)
284  {
285  // First contact search of the simulation
286  return true;
287  }
288  else if (has_periodic_boundaries && particle_displaced_in_pbc)
289  {
290  // Particles have been displaced in periodic boundaries
291  return true;
292  }
293  else
294  {
295  return false;
296  }
297  }
