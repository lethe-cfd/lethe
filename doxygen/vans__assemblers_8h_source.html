288 /**
289  * @brief Class that assembles the drag force using the Koch and Hill drag model for the
290  * VANS equations where the momentum exchange coefficient
291  * beta = ((18 * mu * cell_void_fraction^2 *
292  (1 - cell_void_fraction)) / pow(dp, 2)) *
293  (f0 + 0.5 * f3 * cell_void_fraction * re) *
294  Vp /(1 - cell_void_fraction) where f0 and f3 are functions given by:
295  if ((1 - cell_void_fraction) < 0.4)
296  {
297  f0 = (1 + 3 * sqrt((1 - cell_void_fraction) / 2) +
298  (135.0 / 64) * (1 - cell_void_fraction) *
299  log(1 - cell_void_fraction) +
300  16.14 * (1 - cell_void_fraction)) /
301  (1 + 0.681 * (1 - cell_void_fraction) -
302  8.48 * (1 - cell_void_fraction)^2 +
303  8.14 * (1 - cell_void_fraction)^3);
304  }
305  else if ((1 - cell_void_fraction) >= 0.4)
306  {
307  f0 = 10 * (1 - cell_void_fraction) / cell_void_fraction^3;
308  }
309 
310  f3 = 0.0673 + 0.212 * (1 - cell_void_fraction) +
311  0.0232 / pow(cell_void_fraction, 5);
312  * @tparam dim An integer that denotes the number of spatial dimensions
313  *
314  * @ingroup assemblers
315  */

377 /**
378  * @brief Class that assembles the drag force using Gidaspow model for the
379  * VANS equations where the momentum_transfer_coefficient is calculated
380  according to the following Marchelli et al. (2020):
381  if (cell_void_fraction > 0.8)
382  {
383  momentum_transfer_coefficient =
384  (18 * pow(cell_void_fraction, -3.65) *
385  (1 + 0.15 * pow(Re_p[particle_number], 0.687))) *
386  (particle_properties[DEM::PropertiesIndex::mass] * viscosity /
387  (pow(particle_properties[DEM::PropertiesIndex::dp], 2) *
388  particle_density));
389  }
390  else
391  {
392  // Assuming the sphericity of particles = 1
393  momentum_transfer_coefficient =
394  (150 * (1 - cell_void_fraction) / pow(cell_void_fraction, 2) +
395  1.75 * Re_p[particle_number] / pow(cell_void_fraction, 2)) *
396  (particle_properties[DEM::PropertiesIndex::mass] * viscosity /
397  (pow(particle_properties[DEM::PropertiesIndex::dp], 2) *
398  particle_density));
399  }
