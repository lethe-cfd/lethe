74  /**
75  * The cell_weight() function indicates to the triangulation how much
76  * computational work is expected to happen on this cell, and consequently
77  * how the domain needs to be partitioned so that every MPI rank receives a
78  * roughly equal amount of work (potentially not an equal number of cells).
79  * While the function is called from the outside, it is connected to the
80  * corresponding signal from inside this class, therefore it can be private.
81  * This function is the key component that allow us to dynamically balance the
82  * computational load. The function attributes a weight to
83  * every cell that represents the computational work on this cell. Here the
84  * majority of work is expected to happen on the particles, therefore the
85  * return value of this function (representing "work for this cell") is
86  * calculated based on the number of particles in the current cell.
87  * The function is connected to the cell_weight() signal inside the
88  * triangulation, and will be called once per cell, whenever the triangulation
89  * repartitions the domain between ranks (the connection is created inside the
90  * particles_generation() function of this class).
91  */
92 # if (DEAL_II_VERSION_MAJOR < 10 && DEAL_II_VERSION_MINOR < 6)
93  unsigned int

95  const typename parallel::distributed::Triangulation<dim>::cell_iterator

97  const typename parallel::distributed::Triangulation<dim>::CellStatus status)
98  const;
99 # else
100  unsigned int
101  cell_weight(
102  const typename parallel::distributed::Triangulation<dim>::cell_iterator
103  &cell,
104  const CellStatus status) const;
105 # endif

107  /**
108  * Similar to the cell_weight() function, this function is used when the cell
109  * weight is adapted to the mobility status. For instance, if the
110  * cell is inactive, its computational load will be significantly lower than
111  * if it is a mobile cell since there is no force calculation and no velocity
112  * integration for the particles that lie within it. The weight of the cells
113  * must thus be adapted to the status of the cell.
114  *
115  * cell load = cell weight + load balancing factor * n particles * particle
116  * weight
117  *
118  * @param cell The cell for which the load is calculated
119  *
120  * @param status The status of the cell used to inform functions in derived
121  * classes how the cell with the given cell iterator is going to change
122  *
123  * @param mobility_status The mobility status of the cell
124  */
