50 /**
51  * @brief Class that stores the information required by the assembly procedure
52  * for a Navier-Stokes equation. Consequently, this class calculates
53  * the velocity (values, gradients, laplacians) and the shape function
54  * (values, gradients, laplacians) at all the gauss points for all degrees
55  * of freedom and stores it into arrays. Additionally, the use can request
56  * that this class gathers additional fields for physics which are coupled
57  * to the Navier-Stokes equation, such as the VOF. This class
58  * serves as a separation between the evaluation at the gauss point of the
59  * variables of interest and their use in the assembly, which is carried out
60  * by the assembler functions. For more information on this design, the reader
61  * can consult deal.II step-9
62  * "https://www.dealii.org/current/doxygen/deal.II/step_9.html". In this latter
63  * example, the scratch is a struct instead of a templated class because of the
64  * simplicity of step-9.
65  *
66  * @tparam dim An integer that denotes the dimension of the space in which
67  * the flow is solved
68  * @ingroup solvers
69  **/

214  /** @brief Reinitialize the content of the scratch
215  *
216  * Using the FeValues and the content of the solutions and previous solutions,
217  * fills all of the class member of the scratch
218  *
219  * @param cell The cell over which the assembly is being carried.
220  * This cell must be compatible with the fe which is used to fill the FeValues
221  *
222  * @param current_solution The present value of the solution for [u,p]
223  *
224  * @param previous_solutions The solutions at the previous time steps
225  *
226  *
227  * @param forcing_function The function describing the momentum/mass source term
228  *
229  * @param beta_force The additional force for flow control. TODO : Deprecate this argument and pass it
230  * to the constructor of the assembler
231  *
232  */
233 
234  template <typename VectorType>
235  void
236  reinit(const typename DoFHandler<dim>::active_cell_iterator &cell,
237  const VectorType &current_solution,
238  const std::vector<VectorType> &previous_solutions,
239  std::shared_ptr<Function<dim>> forcing_function,
240  Tensor<1, dim> beta_force,
241  const double pressure_scaling_factor)
242  {
243  this->fe_values.reinit(cell);
244 
245  quadrature_points = this->fe_values.get_quadrature_points();
246  auto &fe = this->fe_values.get_fe();
247 
248  forcing_function->vector_value_list(quadrature_points, this->rhs_force);
249 
250  // Establish the force vector
251  for (unsigned int q = 0; q < n_q_points; ++q)
252  {
253  for (int d = 0; d < dim; ++d)
254  {
255  const unsigned int component_i =
256  fe.system_to_component_index(d).first;
257  this->force[q][d] = this->rhs_force[q](component_i);
258  }
259 
260  const unsigned int component_mass =
261  fe.system_to_component_index(dim).first;
262  this->mass_source[q] = this->rhs_force[q](component_mass);
263 
264  // Correct force to include the dynamic forcing term for flow
265  // control
266  force[q] = force[q] + beta_force;
267  }
268 
269  for (const unsigned int k : fe_values.dof_indices())
270  {
271  components[k] = fe.system_to_component_index(k).first;
272  }
273 
274  if (dim == 2)
275  this->cell_size = std::sqrt(4. * cell->measure() / M_PI) / fe.degree;
276  else if (dim == 3)
277  this->cell_size = pow(6 * cell->measure() / M_PI, 1. / 3.) / fe.degree;
278 
279  // Gather velocity (values, gradient and laplacian)
280  this->fe_values[velocities].get_function_values(current_solution,
281  this->velocity_values);
282  this->fe_values[velocities].get_function_gradients(
283  current_solution, this->velocity_gradients);
284  this->fe_values[velocities].get_function_laplacians(
285  current_solution, this->velocity_laplacians);
286  if (gather_hessian)
287  this->fe_values[velocities].get_function_hessians(
288  current_solution, this->velocity_hessians);
289 
290  for (unsigned int q = 0; q < this->n_q_points; ++q)
291  {
292  this->velocity_divergences[q] = trace(this->velocity_gradients[q]);
293  }
294 
295  // Gather pressure (values, gradient)
296  fe_values[pressure].get_function_values(current_solution,
297  this->pressure_values);
298  fe_values[pressure].get_function_gradients(current_solution,
299  this->pressure_gradients);
300  this->pressure_scaling_factor = pressure_scaling_factor;
301 
302  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
303  {
304  this->fe_values[velocities].get_function_values(
305  previous_solutions[p], previous_velocity_values[p]);
306  }
307 
308  // Only gather the pressure when a pressure history is necessary
309  // (compressible Navier-Stokes)
310  if (!this->properties_manager.density_is_constant())
311  for (unsigned int p = 0; p < previous_solutions.size(); ++p)
312  {
313  this->fe_values[pressure].get_function_values(
314  previous_solutions[p], previous_pressure_values[p]);
315  }
316 
317  for (unsigned int q = 0; q < n_q_points; ++q)
318  {
319  this->JxW[q] = this->fe_values.JxW(q);
320  for (unsigned int k = 0; k < n_dofs; ++k)
321  {
322  // Velocity
323  this->phi_u[q][k] = this->fe_values[velocities].value(k, q);
324  this->div_phi_u[q][k] =
325  this->fe_values[velocities].divergence(k, q);
326  this->grad_phi_u[q][k] = this->fe_values[velocities].gradient(k, q);
327  this->hess_phi_u[q][k] = this->fe_values[velocities].hessian(k, q);
328  for (int d = 0; d < dim; ++d)
329  this->laplacian_phi_u[q][k][d] = trace(this->hess_phi_u[q][k][d]);
330  // Pressure
331  this->phi_p[q][k] = this->fe_values[pressure].value(k, q);
332  this->grad_phi_p[q][k] = this->fe_values[pressure].gradient(k, q);
333  }
334  }
335 
336  is_boundary_cell = cell->at_boundary();
337  if (is_boundary_cell)
338  {
339  n_faces = cell->n_faces();
340  is_boundary_face = std::vector<bool>(n_faces, false);
341  n_faces_q_points = fe_face_values.get_quadrature().size();
342  boundary_face_id = std::vector<unsigned int>(n_faces);
343 
344  face_JxW = std::vector<std::vector<double>>(
345  n_faces, std::vector<double>(n_faces_q_points));
346 
347 
348  // Velocity and pressure values
349  // First vector is face number, second quadrature point
350  this->face_velocity_values = std::vector<std::vector<Tensor<1, dim>>>(
351  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
352 
353  this->face_velocity_divergences = std::vector<std::vector<double>>(
354  n_faces, std::vector<double>(n_faces_q_points));
355 
356  this->face_velocity_gradients =
357  std::vector<std::vector<Tensor<2, dim>>>(
358  n_faces, std::vector<Tensor<2, dim>>(n_faces_q_points));
359 
360  this->face_velocity_laplacians =
361  std::vector<std::vector<Tensor<1, dim>>>(
362  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
363 
364  this->face_pressure_values = std::vector<std::vector<double>>(
365  n_faces, std::vector<double>(n_faces_q_points));
366 
367  this->face_pressure_gradients =
368  std::vector<std::vector<Tensor<1, dim>>>(
369  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
370 
371  this->face_normal = std::vector<std::vector<Tensor<1, dim>>>(
372  n_faces, std::vector<Tensor<1, dim>>(n_faces_q_points));
373 
374  this->face_quadrature_points = std::vector<std::vector<Point<dim>>>(
375  n_faces, std::vector<Point<dim>>(n_faces_q_points));
376 
377  this->face_div_phi_u = std::vector<std::vector<std::vector<double>>>(
378  n_faces,
379  std::vector<std::vector<double>>(n_faces_q_points,
380  std::vector<double>(n_dofs)));
381 
382  this->face_phi_u =
383  std::vector<std::vector<std::vector<Tensor<1, dim>>>>(
384  n_faces,
385  std::vector<std::vector<Tensor<1, dim>>>(
386  n_faces_q_points, std::vector<Tensor<1, dim>>(n_dofs)));
387 
388  this->face_hess_phi_u =
389  std::vector<std::vector<std::vector<Tensor<3, dim>>>>(
390  n_faces,
391  std::vector<std::vector<Tensor<3, dim>>>(
392  n_faces_q_points, std::vector<Tensor<3, dim>>(n_dofs)));
393 
394  this->face_laplacian_phi_u =
395  std::vector<std::vector<std::vector<Tensor<1, dim>>>>(
396  n_faces,
397  std::vector<std::vector<Tensor<1, dim>>>(
398  n_faces_q_points, std::vector<Tensor<1, dim>>(n_dofs)));
399 
400  this->face_grad_phi_u =
401  std::vector<std::vector<std::vector<Tensor<2, dim>>>>(
402  n_faces,
403  std::vector<std::vector<Tensor<2, dim>>>(
404  n_faces_q_points, std::vector<Tensor<2, dim>>(n_dofs)));
405 
406  this->face_phi_p = std::vector<std::vector<std::vector<double>>>(
407  n_faces,
408  std::vector<std::vector<double>>(n_faces_q_points,
409  std::vector<double>(n_dofs)));
410 
411  this->face_grad_phi_p =
412  std::vector<std::vector<std::vector<Tensor<1, dim>>>>(
413  n_faces,
414  std::vector<std::vector<Tensor<1, dim>>>(
415  n_faces_q_points, std::vector<Tensor<1, dim>>(n_dofs)));
416  for (const auto face : cell->face_indices())
417  {
418  is_boundary_face[face] = cell->face(face)->at_boundary();
419  if (is_boundary_face[face])
420  {
421  fe_face_values.reinit(cell, face);
422  n_dofs = fe_face_values.get_fe().n_dofs_per_cell();
423  boundary_face_id[face] = cell->face(face)->boundary_id();
424  // Shape functions
425  // First vector is face number, second quadrature point, third
426  // DOF
427 
428  // Gather velocity (values, gradient and laplacian)
429  this->fe_face_values[velocities].get_function_values(
430  current_solution, this->face_velocity_values[face]);
431  this->fe_face_values[velocities].get_function_gradients(
432  current_solution, this->face_velocity_gradients[face]);
433  this->fe_face_values[velocities].get_function_laplacians(
434  current_solution, this->face_velocity_laplacians[face]);
435 
436  // Gather pressure (values, gradient)
437  this->fe_values[pressure].get_function_values(
438  current_solution, this->face_pressure_values[face]);
439  this->fe_values[pressure].get_function_gradients(
440  current_solution, this->face_pressure_gradients[face]);
441 
442  for (unsigned int q = 0; q < n_faces_q_points; ++q)
443  {
444  this->face_JxW[face][q] = this->fe_face_values.JxW(q);
445  this->face_normal[face][q] =
446  this->fe_face_values.normal_vector(q);
447  this->face_quadrature_points[face][q] =
448  this->fe_face_values.quadrature_point(q);
449  for (const unsigned int k : fe_face_values.dof_indices())
450  {
451  // Velocity
452  this->face_phi_u[face][q][k] =
453  this->fe_face_values[velocities].value(k, q);
454  this->face_div_phi_u[face][q][k] =
455  this->fe_face_values[velocities].divergence(k, q);
456  this->face_grad_phi_u[face][q][k] =
457  this->fe_face_values[velocities].gradient(k, q);
458  this->face_hess_phi_u[face][q][k] =
459  this->fe_face_values[velocities].hessian(k, q);
460  for (int d = 0; d < dim; ++d)
461  this->face_laplacian_phi_u[face][q][k][d] =
462  trace(this->face_hess_phi_u[face][q][k][d]);
463  // Pressure
464  this->face_phi_p[face][q][k] =
465  this->fe_face_values[pressure].value(k, q);
466  this->face_grad_phi_p[face][q][k] =
467  this->fe_face_values[pressure].gradient(k, q);
468  }
469  }
470  }
471  }
472  }

664  *

668  * [epsilon]
669  *

673  void
674  reinit_particle_fluid_interactions(
675  const typename DoFHandler<dim>::active_cell_iterator &cell,
676  const VectorType /*current_solution*/,
677  const VectorType previous_solution,
678  const VectorType void_fraction_solution,
679  const Particles::ParticleHandler<dim> &particle_handler,
680  DoFHandler<dim> &dof_handler,
681  DoFHandler<dim> &void_fraction_dof_handler)
682  {
683  const FiniteElement<dim> &fe = this->fe_values.get_fe();
684  const FiniteElement<dim> &fe_void_fraction =
685  this->fe_values_void_fraction->get_fe();
686 
687  bool interpolated_void_fraction = this->interpolated_void_fraction;
688  const unsigned int dofs_per_cell = fe.dofs_per_cell;
689  std::vector<types::global_dof_index> fluid_dof_indices(dofs_per_cell);
690  std::vector<types::global_dof_index> void_fraction_dof_indices(
691  fe_void_fraction.dofs_per_cell);
692 
693  pic = particle_handler.particles_in_cell(this->fe_values.get_cell());
694 
695  average_particle_velocity = 0;
696  cell_volume = this->fe_values.get_cell()->measure();
697 
698  // Get the local dof indices for velocity and void fraction field
699  // interpolation at particle's position
700  const auto &dh_cell =
701  typename DoFHandler<dim>::cell_iterator(*this->fe_values.get_cell(),
702  &dof_handler);
703  dh_cell->get_dof_indices(fluid_dof_indices);
704 
705  const auto &void_fraction_dh_cell = typename DoFHandler<dim>::cell_iterator(
706  *this->fe_values_void_fraction->get_cell(), &void_fraction_dof_handler);
707  void_fraction_dh_cell->get_dof_indices(void_fraction_dof_indices);
708 
709  // Loop over particles in cell
710  double total_particle_volume = 0;
711  {
712  unsigned int particle_no = 0;
713  for (auto &particle : pic)
714  {
715  auto particle_properties = particle.get_properties();
716  // Set the particle_fluid_interactions properties and vectors to 0
717  for (int d = 0; d < dim; ++d)
718  {
719  particle_properties[DEM::PropertiesIndex::fem_force_x + d] = 0.;
720  particle_properties[DEM::PropertiesIndex::fem_torque_x + d] = 0.;
721  undisturbed_flow_force[d] = 0.;
722  }
723 
724  fluid_velocity_at_particle_location[particle_no] = 0;
725 
726  // Stock the values of particle velocity in a tensor
727  particle_velocity[particle_no][0] =
728  particle_properties[DEM::PropertiesIndex::v_x];
729  particle_velocity[particle_no][1] =
730  particle_properties[DEM::PropertiesIndex::v_y];
731  if (dim == 3)
732  particle_velocity[particle_no][2] =
733  particle_properties[DEM::PropertiesIndex::v_z];
734 
735  // Interpolate velocity and void fraction at particle position
736  // Reference location of the particle
737  auto reference_location = particle.get_reference_location();
738 
739  for (unsigned int j = 0; j < dofs_per_cell; ++j)
740  {
741  const auto comp_j = fe.system_to_component_index(j).first;
742  if (comp_j < dim)
743  {
744  auto &evaluation_point = previous_solution;
745  fluid_velocity_at_particle_location[particle_no][comp_j] +=
746  evaluation_point[fluid_dof_indices[j]] *
747  fe.shape_value(j, reference_location);
748  }
749  }
750 
751  cell_void_fraction[particle_no] = 0;
752  if (interpolated_void_fraction == true)
753  {
754  for (unsigned int j = 0; j < fe_void_fraction.dofs_per_cell; ++j)
755  {
756  cell_void_fraction[particle_no] +=
757  void_fraction_solution[void_fraction_dof_indices[j]] *
758  fe_void_fraction.shape_value(j, reference_location);
759  }
760  }
761  else
762  {
763  total_particle_volume +=
764  M_PI * pow(particle_properties[DEM::PropertiesIndex::dp], dim) /
765  (2 * dim);
766  }
767 
768  average_particle_velocity += particle_velocity[particle_no];
769  particle_no++;
770  }
771  number_of_particles = particle_no;
772  }
773 
774  if (interpolated_void_fraction == false)
775  {
776  double cell_void_fraction_bulk = 0;
777  cell_void_fraction_bulk =
778  (cell_volume - total_particle_volume) / cell_volume;
779 
780  for (unsigned int j = 0; j < number_of_particles; ++j)
781  cell_void_fraction[j] = cell_void_fraction_bulk;
782  }
783 
784  if (number_of_particles > 0)
785  average_particle_velocity =
786  average_particle_velocity / number_of_particles;
787 
788  // Relative velocity and particle Reynolds
789  unsigned int particle_no = 0;
790  average_fluid_particle_relative_velocity = 0;
791  double kinematic_viscosity =
792  properties_manager.get_kinematic_viscosity_scale();
793 
794  for (auto &particle : pic)
795  {
796  auto particle_properties = particle.get_properties();
797  fluid_particle_relative_velocity_at_particle_location[particle_no] =
798  fluid_velocity_at_particle_location[particle_no] -
799  particle_velocity[particle_no];
800  average_fluid_particle_relative_velocity +=
801  fluid_particle_relative_velocity_at_particle_location[particle_no];
802 
803  Re_particle[particle_no] =
804  1e-3 +
805  cell_void_fraction[particle_no] *
806  fluid_particle_relative_velocity_at_particle_location[particle_no]
807  .norm() *
808  particle_properties[DEM::PropertiesIndex::dp] /
809  (kinematic_viscosity + DBL_MIN);
810  particle_no++;
811  }
812  if (particle_no > 0)
813  {
814  average_fluid_particle_relative_velocity =
815  average_fluid_particle_relative_velocity / particle_no;
816  }
817 
818  // If there are no particles in the cell,there is no need to try and gather
819  // the rest of the information
820  if (number_of_particles == 0)
821  return;
822 
823  // If particles are in the cell, gather the rest of the
824  // information
825 
826  // Create local vector that will be used to spawn an in-situ quadrature to
827  // interpolate at the location of the particles
828  std::vector<Point<dim>> particle_reference_location(number_of_particles);
829  std::vector<double> particle_weights(number_of_particles, 1);
830 
831  // Resize arrays to be of the right size
832  fluid_velocity_laplacian_at_particle_location.resize(number_of_particles);
833  fluid_velocity_curls_at_particle_location_2d.resize(number_of_particles);
834  fluid_velocity_curls_at_particle_location_3d.resize(number_of_particles);
835  fluid_pressure_gradients_at_particle_location.resize(number_of_particles);
836 
837  // Loop over particles in cell and cache their reference location
838  {
839  unsigned int particle_no = 0;
840  for (auto &particle : pic)
841  {
842  // Store particle positions and weights
843  // Reference location of the particle
844  particle_reference_location[particle_no] =
845  particle.get_reference_location();
846  particle_no++;
847  }
848  }
849 
850  // Create a quadrature that is based on the particle reference location
851  Quadrature<dim> q_local(particle_reference_location, particle_weights);
852  FEValues<dim> fe_values_local_particles(
853  fe, q_local, update_gradients | update_values | update_hessians);
854 
855  // Evaluate the relevant information at the
856  // quadrature points to do the interpolation.
857  fe_values_local_particles.reinit(cell);
858 
859  fe_values_local_particles[velocities].get_function_laplacians(
860  previous_solution, fluid_velocity_laplacian_at_particle_location);
861 
862  if constexpr (dim == 2)
863  {
864  fe_values_local_particles[velocities].get_function_curls(
865  previous_solution, fluid_velocity_curls_at_particle_location_2d);
866  }
867  else if constexpr (dim == 3)
868  {
869  fe_values_local_particles[velocities].get_function_curls(
870  previous_solution, fluid_velocity_curls_at_particle_location_3d);
871  }
872 
873  fe_values_local_particles[pressure].get_function_gradients(
874  previous_solution, fluid_pressure_gradients_at_particle_location);
875  }
