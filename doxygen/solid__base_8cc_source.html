53 template <int dim, int spacedim>
54 SolidBase<dim, spacedim>::SolidBase(
55  std::shared_ptr<Parameters::NitscheObject<spacedim>> &param,
56  std::shared_ptr<parallel::DistributedTriangulationBase<spacedim>> fluid_tria,
57  std::shared_ptr<Mapping<spacedim>> fluid_mapping)
58  : mpi_communicator(MPI_COMM_WORLD)
59  , n_mpi_processes(Utilities::MPI::n_mpi_processes(mpi_communicator))
60  , this_mpi_process(Utilities::MPI::this_mpi_process(mpi_communicator))
61  , fluid_tria(fluid_tria)
62  , fluid_mapping(fluid_mapping)
63  , param(param)
64  , velocity(&param->solid_velocity)
65  , temperature(&param->solid_temperature)
66 {
67  if (param->solid_mesh.simplex)
68  {
69  // for simplex meshes
70  fe = std::make_shared<FE_SimplexP<dim, spacedim>>(1);
71  solid_mapping = std::make_shared<MappingFE<dim, spacedim>>(*fe);
72  quadrature =
73  std::make_shared<QGaussSimplex<dim>>(param->number_quadrature_points);
74  solid_tria = std::make_shared<
75  parallel::fullydistributed::Triangulation<dim, spacedim>>(
76  this->mpi_communicator);
77 
78  displacement_fe =
79  std::make_shared<FESystem<dim, spacedim>>(FE_SimplexP<dim, spacedim>(1),
80  spacedim);
81  }
82  else
83  {
84  // Usual case, for quad/hex meshes
85  fe = std::make_shared<FE_Q<dim, spacedim>>(1);
86  solid_mapping = std::make_shared<MappingQGeneric<dim, spacedim>>(1);
87  quadrature =
88  std::make_shared<QGauss<dim>>(param->number_quadrature_points);
89  solid_tria =
90  std::make_shared<parallel::distributed::Triangulation<dim, spacedim>>(
91  mpi_communicator,
92  typename Triangulation<dim, spacedim>::MeshSmoothing(
93  Triangulation<dim, spacedim>::smoothing_on_refinement |
94  Triangulation<dim, spacedim>::smoothing_on_coarsening));
95 
96  displacement_fe =
97  std::make_shared<FESystem<dim, spacedim>>(FE_Q<dim, spacedim>(1),
98  spacedim);
99  }
100 
101  // Dof handler associated with particles
102  solid_dh.clear();
103  solid_dh.reinit(*this->solid_tria);
104 
105  // Dof handler associated with mesh displacement
106  displacement_dh.clear();
107  displacement_dh.reinit(*this->solid_tria);
108 }

123 template <int dim, int spacedim>
124 void
125 SolidBase<dim, spacedim>::setup_triangulation(const bool restart)
126 {
127  if (param->solid_mesh.type == Parameters::Mesh::Type::gmsh)
128  {
129  if (param->solid_mesh.simplex)
130  {
131  auto comm = solid_tria->get_communicator();
132  std::string file_name = param->solid_mesh.file_name;
133 
134  auto construction_data = TriangulationDescription::Utilities::
135  create_description_from_triangulation_in_groups<dim, spacedim>(
136  [file_name](dealii::Triangulation<dim, spacedim> &basetria) {
137  GridIn<dim, spacedim> grid_in;
138  grid_in.attach_triangulation(basetria);
139  std::ifstream input_file(file_name);
140 
141  grid_in.read_msh(input_file);
142  },
143  [](dealii::Triangulation<dim, spacedim> &basetria,
144  const MPI_Comm comm,
145  const unsigned int /*group_size*/) {
146  GridTools::partition_triangulation(
147  Utilities::MPI::n_mpi_processes(comm), basetria);
148  },
149  comm,
150  Utilities::MPI::n_mpi_processes(comm) /* group size */,
151  dealii::Triangulation<dim, spacedim>::none);
152 
153  solid_tria->create_triangulation(construction_data);
154  }
155  else
156  { // Grid creation
157  GridIn<dim, spacedim> grid_in;
158  // Attach triangulation to the grid
159  grid_in.attach_triangulation(*solid_tria);
160  // Read input gmsh file
161  std::ifstream input_file(param->solid_mesh.file_name);
162  grid_in.read_msh(input_file);
163  }
164  }
165  else if (param->solid_mesh.type == Parameters::Mesh::Type::dealii)
166  {
167  if (param->solid_mesh.simplex)
168  { // TODO Using dealii generated meshes with simplices in Nitsche solver
169  // generates and error. "boundary_id !=
170  // numbers::internal_face_boundary_id"
171  Triangulation<dim, spacedim> temporary_quad_triangulation;
172  GridGenerator::generate_from_name_and_arguments(
173  temporary_quad_triangulation,
174  param->solid_mesh.grid_type,
175  param->solid_mesh.grid_arguments);
176 
177  // initial refinement
178  const int initial_refinement = param->solid_mesh.initial_refinement;
179  temporary_quad_triangulation.refine_global(initial_refinement);
180  // flatten the triangulation
181  Triangulation<dim, spacedim> flat_temp_quad_triangulation;
182  GridGenerator::flatten_triangulation(temporary_quad_triangulation,
183  flat_temp_quad_triangulation);
184 
185  Triangulation<dim, spacedim> temporary_tri_triangulation(
186  Triangulation<dim, spacedim>::limit_level_difference_at_vertices);
187  GridGenerator::convert_hypercube_to_simplex_mesh(
188  flat_temp_quad_triangulation, temporary_tri_triangulation);
189 
190  GridTools::partition_triangulation_zorder(
191  Utilities::MPI::n_mpi_processes(solid_tria->get_communicator()),
192  temporary_tri_triangulation);
193  GridTools::partition_multigrid_levels(temporary_tri_triangulation);
194 
195  // extract relevant information from distributed triangulation
196  auto construction_data = TriangulationDescription::Utilities::
197  create_description_from_triangulation(
198  temporary_tri_triangulation,
199  solid_tria->get_communicator(),
200  TriangulationDescription::Settings::
201  construct_multigrid_hierarchy);
202  solid_tria->create_triangulation(construction_data);
203  }
204  else
205  { // deal.ii creates grid and attaches the solid triangulation
206  GridGenerator::generate_from_name_and_arguments(
207  *solid_tria,
208  param->solid_mesh.grid_type,
209  param->solid_mesh.grid_arguments);
210  }
211  }
212  else
213  throw std::runtime_error(
214  "Unsupported mesh type - solid mesh will not be created");
215 
216  // Rotate the triangulation
217  rotate_grid(param->solid_mesh.rotation_angle,
218  param->solid_mesh.rotation_axis);
219 
220  // Translate the triangulation
221  translate_grid(param->solid_mesh.translation);
222  // Refine the solid triangulation to its initial size
223  // NB: solid_tria should not be refined if loaded from a restart file
224  // afterwards
225  if (!restart)
226  {
227  if (param->solid_mesh.simplex)
228  {
229  // Simplex triangulation refinement isn't possible yet
230  }
231  else
232  {
233  solid_tria->refine_global(param->solid_mesh.initial_refinement);
234  }
235  solid_dh.distribute_dofs(*fe);
236  }
237 }

330 template <int dim, int spacedim>
331 void
332 SolidBase<dim, spacedim>::setup_particles()
333 {
334  setup_particles_handler();
335 
336  // Initialize FEValues
337  std::vector<Point<spacedim>> quadrature_points_vec;
338  quadrature_points_vec.reserve(quadrature->size() *
339  solid_tria->n_locally_owned_active_cells());
340  unsigned int n_properties = (dim == 2 && spacedim == 3) ? 1 + spacedim : 1;
341 
342  std::vector<std::vector<double>> properties;
343  properties.reserve(quadrature->size() *
344  solid_tria->n_locally_owned_active_cells() * n_properties);
345 
346  UpdateFlags update_flags = update_JxW_values | update_quadrature_points;
347  if (dim == 2 && spacedim == 3)
348  update_flags = update_flags | update_normal_vectors;
349 
350  FEValues<dim, spacedim> fe_v(*solid_mapping, *fe, *quadrature, update_flags);
351 
352  // Fill quadrature points vector and properties, used to fill
353  // solid_particle_handler
354  for (const auto &cell : solid_dh.active_cell_iterators())
355  if (cell->is_locally_owned())
356  {
357  fe_v.reinit(cell);
358  const auto &points = fe_v.get_quadrature_points();
359  const auto &JxW = fe_v.get_JxW_values();
360  if (dim == 2 && spacedim == 3)
361  {
362  const auto &normal_vectors = fe_v.get_normal_vectors();
363  for (unsigned int q = 0; q < points.size(); ++q)
364  {
365  quadrature_points_vec.emplace_back(points[q]);
366  std::vector<double> prop_i = {JxW[q],
367  normal_vectors[q][0],
368  normal_vectors[q][1]};
369  if (spacedim == 3)
370  prop_i.push_back(normal_vectors[q][2]);
371  properties.emplace_back(prop_i);
372  }
373  }
374  else
375  {
376  for (unsigned int q = 0; q < points.size(); ++q)
377  {
378  quadrature_points_vec.emplace_back(points[q]);
379  std::vector<double> prop_i = {JxW[q]};
380  properties.emplace_back(prop_i);
381  }
382  }
383  }
384 
385  // Compute fluid bounding box
386  std::vector<std::vector<BoundingBox<spacedim>>> global_fluid_bounding_boxes;
387 
388 
389 
390  // Use the more general boost rtree bounding boxes
391  std::vector<BoundingBox<spacedim>> all_boxes;
392  all_boxes.reserve(fluid_tria->n_locally_owned_active_cells());
393  for (const auto &cell : fluid_tria->active_cell_iterators())
394  if (cell->is_locally_owned())
395  all_boxes.emplace_back(cell->bounding_box());
396  const auto tree = pack_rtree(all_boxes);
397  const auto local_boxes = extract_rtree_level(
398  tree, std::max(int(log10(fluid_tria->n_locally_owned_active_cells())), 1));
399 
400  global_fluid_bounding_boxes =
401  Utilities::MPI::all_gather(mpi_communicator, local_boxes);
402 
403  // Fill solid particle handler
404  solid_particle_handler->insert_global_particles(quadrature_points_vec,
405  global_fluid_bounding_boxes,
406  properties);
407 
408  // Number of particles used to assess that no particle is lost
409  initial_number_of_particles = solid_particle_handler->n_global_particles();
410 }

490 template <int dim, int spacedim>
491 void
492 SolidBase<dim, spacedim>::integrate_velocity(double time_step,
493  double initial_time)
494 {
495  const unsigned int sub_particles_iterations = param->particles_sub_iterations;
496  AssertThrow(sub_particles_iterations >= 1,
497  ExcMessage("Sub particles iterations must be 1 or larger"));
498 
499 
500 
501  double sub_iteration_relaxation = 1. / sub_particles_iterations;
502  time_step = time_step * sub_iteration_relaxation;
503  // Particle sub iterations divide the time step in a number of "sub
504  // iterations". This allows the solver to use a larger CFL without
505  // necessitating the use of the more complex particle location detection
506  // approaches. The number of sub particles iterations must be chosen so that
507  // the time_step / sub_particles_iterations * velocity / cell size is smaller
508  // than unity
509  for (unsigned int it = 0; it < sub_particles_iterations; ++it)
510  {
511  for (auto particle = solid_particle_handler->begin();
512  particle != solid_particle_handler->end();
513  ++particle)
514  {
515  // Calculate the next position of the particle using the RK4
516  // time integration scheme
517  // x(t+dt) = x(t) + 1/6 (k1+2*k2+2*k3+k4)
518  // k1 = dt * v(t,x(t))
519  // k2 = dt * v(t+0.5*dt,x+k1/2)
520  // k3 = dt * v(t+0.5*dt,x+k2/2)
521  // k4 = dt * vt(t+dt,x+k3)
522  // The four stages (1 to 4) are built successively
523  // For each stage, the time of the function must be "reset"
524  // to ensure adequate evaluation of the RK4 scheme.
525  // See https://en.wikipedia.org/wiki/Runge%E2%80%93Kutta_methods for
526  // more details
527 
528  Point<spacedim> particle_location = particle->get_location();
529 
530  Tensor<1, spacedim> k1;
531  velocity->set_time(initial_time);
532  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
533  k1[comp_i] = velocity->value(particle_location, comp_i);
534 
535  Point<spacedim> p1 = particle_location + time_step / 2 * k1;
536  Tensor<1, spacedim> k2;
537  velocity->set_time(initial_time + time_step / 2);
538  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
539  k2[comp_i] = velocity->value(p1, comp_i);
540 
541  Point<spacedim> p2 = particle_location + time_step / 2 * k2;
542  Tensor<1, spacedim> k3;
543  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
544  k3[comp_i] = velocity->value(p2, comp_i);
545 
546  Point<spacedim> p3 = particle_location + time_step * k3;
547  Tensor<1, spacedim> k4;
548  velocity->set_time(initial_time + time_step);
549  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
550  k4[comp_i] = velocity->value(p3, comp_i);
551 
552  particle_location += time_step / 6 * (k1 + 2 * k2 + 2 * k3 + k4);
553  particle->set_location(particle_location);
554  }
555  solid_particle_handler->sort_particles_into_subdomains_and_cells();
556  initial_time += time_step;
557  }
558 
559  if (initial_number_of_particles !=
560  solid_particle_handler->n_global_particles())
561  {
562  if (this_mpi_process == 0)
563  {
564  std::cout << "Warning - Nitsche Particles have been lost"
565  << std::endl;
566  std::cout << "Initial number of particles : "
567  << initial_number_of_particles << std::endl;
568  std::cout << "Current number of particles : "
569  << solid_particle_handler->n_global_particles()
570  << std::endl;
571  std::cout << "Try increasing the number of particles sub iterations"
572  << std::endl;
573 
574  // Stops the simulation if particles have been lost
575  // Future improvement would be to dynamically increase the number of
576  // particles subiteration if particles have been lost
577  if (param->stop_particles_lost)
578  {
579  std::cerr
580  << std::endl
581  << std::endl
582  << "----------------------------------------------------"
583  << std::endl;
584  std::cerr
585  << "Exception on processing: " << std::endl
586  << "Nitsche Particles have been lost" << std::endl
587  << "Aborting!" << std::endl
588  << "----------------------------------------------------"
589  << std::endl;
590  std::exit(1);
591  }
592  }
593  }
594 }

596 template <int dim, int spacedim>
597 void
598 SolidBase<dim, spacedim>::move_solid_triangulation(const double time_step,
599  const double initial_time)
600 {
601  // First we calculate the displacement that the solid triangulation
602  // will feel during this iteration.
603  // Then, we apply this displacement to the triangulation
604  const unsigned int dofs_per_cell = displacement_fe->dofs_per_cell;
605  const std::vector<bool> locally_owned_vertices =
606  GridTools::get_locally_owned_vertices(*this->solid_tria);
607  std::vector<bool> vertex_moved(this->solid_tria->n_vertices(), false);
608  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
609 
610 
611  for (const auto &cell : displacement_dh.active_cell_iterators())
612  {
613  if (cell->is_locally_owned())
614  {
615  for (unsigned int vertex = 0;
616  vertex < cell->reference_cell().n_vertices();
617  ++vertex)
618  {
619  // Calculate the next position of the particle using the RK4
620  // time integration scheme
621  // x(t+dt) = x(t) + 1/6 (k1+2*k2+2*k3+k4)
622  // k1 = dt * v(t,x(t))
623  // k2 = dt * v(t+0.5*dt,x+k1/2)
624  // k3 = dt * v(t+0.5*dt,x+k2/2)
625  // k4 = dt * vt(t+dt,x+k3)
626  // The four stages (1 to 4) are built successively
627  // For each stage, the time of the function must be "reset"
628  // to ensure adequate evaluation of the RK4 scheme.
629  // See https://en.wikipedia.org/wiki/Runge%E2%80%93Kutta_methods
630  // for more details
631 
632  const auto dof_index = cell->vertex_dof_index(vertex, 0);
633  Point<spacedim> &vertex_position = cell->vertex(vertex);
634  const unsigned global_vertex_no = cell->vertex_index(vertex);
635  if (vertex_moved[global_vertex_no] ||
636  !locally_owned_vertices[global_vertex_no])
637  continue;
638 
639  Tensor<1, spacedim> k1;
640  velocity->set_time(initial_time);
641  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
642  k1[comp_i] = velocity->value(vertex_position, comp_i);
643 
644  Point<spacedim> p1 = vertex_position + time_step / 2 * k1;
645  Tensor<1, spacedim> k2;
646  velocity->set_time(initial_time + time_step / 2);
647  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
648  k2[comp_i] = velocity->value(p1, comp_i);
649 
650  Point<spacedim> p2 = vertex_position + time_step / 2 * k2;
651  Tensor<1, spacedim> k3;
652  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
653  k3[comp_i] = velocity->value(p2, comp_i);
654 
655  Point<spacedim> p3 = vertex_position + time_step * k3;
656  Tensor<1, spacedim> k4;
657  velocity->set_time(initial_time + time_step);
658  for (unsigned int comp_i = 0; comp_i < spacedim; ++comp_i)
659  k4[comp_i] = velocity->value(p3, comp_i);
660 
661  auto vertex_displacement =
662  time_step / 6 * (k1 + 2 * k2 + 2 * k3 + k4);
663 
664  vertex_position += vertex_displacement;
665 
666  for (unsigned d = 0; d < spacedim; ++d)
667  displacement[dof_index + d] =
668  displacement[dof_index + d] + vertex_displacement[d];
669 
670  vertex_moved[global_vertex_no] = true;
671  }
672  }
673  }
674  this->solid_tria->communicate_locally_moved_vertices(locally_owned_vertices);
675 }
