156 /**
157  * @brief Quartic mobility model.
158  *
159  * The mobility function is the following : \f$M(\phi) = D(1-\phi^2)^2 \f$
160  * where D is the mobility constant.
161  */
162 class MobilityCahnHilliardModelQuartic : public MobilityCahnHilliardModel
163 {
164 public:
165  /**
166  * @brief Default constructor.
167  *
168  * @param[in] p_mobility_cahn_hilliard_constant the user defined mobility
169  * constant.
170  */
171  MobilityCahnHilliardModelQuartic(
172  const double p_mobility_cahn_hilliard_constant)
173  : mobility_cahn_hilliard_constant(p_mobility_cahn_hilliard_constant)
174  {
175  this->model_depends_on[field::phase_order_cahn_hilliard] = true;
176  }
177 
178  /**

180  */

183  /**
184  * @brief Return the mobility constant.
185  * @return Value of the mobility constant.
186  */
187  double
188  get_mobility_constant() override
189  {
190  return mobility_cahn_hilliard_constant;
191  }
192 
193  /**
194  * @brief Calculate the mobility.
195  * @param[in] fields_value Value of the various fields on which the mobility
196  * may depend.
197  * @return Value of the mobility calculated with the fields_value.
198  */
199  double
200  value(const std::map<field, double> &fields_value) override
201  {
202  AssertThrow(
203  fields_value.find(field::phase_order_cahn_hilliard) != fields_value.end(),
204  PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
205  "phase_order_cahn_hilliard"));
206  const double &phase_order_cahn_hilliard =
207  fields_value.at(field::phase_order_cahn_hilliard);
208 
209  // The phase order values are clamped to avoid unphysical mobilities in the
210  // bulk phases.
211  return mobility_cahn_hilliard_constant *
212  std::pow((1 - std::min(1.0,
213  phase_order_cahn_hilliard *
214  phase_order_cahn_hilliard)),
215  2);
216  }
217 
218  /**
219  * @brief Calculate the vector of mobility.
220  * @param[in] field_vectors Vectors of the fields on which the mobility
221  * may depend.
222  * @param[out] property_vector Vector of the mobility values.
223  */
224  void
225  vector_value(const std::map<field, std::vector<double>> &field_vectors,
226  std::vector<double> &property_vector) override
227  {
228  AssertThrow(
229  field_vectors.find(field::phase_order_cahn_hilliard) !=
230  field_vectors.end(),
231  PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
232  "phase_order_cahn_hilliard"));
233  const std::vector<double> &phase_order_cahn_hilliard =
234  field_vectors.at(field::phase_order_cahn_hilliard);
235  for (unsigned int i = 0; i < property_vector.size(); ++i)
236  {
237  property_vector[i] =
238  mobility_cahn_hilliard_constant *
239  std::pow((1 - std::min(1.0,
240  phase_order_cahn_hilliard[i] *
241  phase_order_cahn_hilliard[i])),
242  2);
243  }
244  }

246  /**
247  * @brief Calculate the jacobian (the partial derivative) of the
248  * mobility with respect to a field.
249  * @param[in] field_values Value of the various fields on which the mobility
250  * may depend.
251  * @param[in] id Indicator of the field with respect to which the Jacobian
252  * should be calculated.
253  * @return Value of the partial derivative of the mobility with respect to
254  * the field.
255  */
256 
257  double
258  jacobian(const std::map<field, double> &fields_value, field /*id*/) override
259  {
260  AssertThrow(
261  fields_value.find(field::phase_order_cahn_hilliard) != fields_value.end(),
262  PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
263  "phase_order_cahn_hilliard"));
264  const double &phase_order_cahn_hilliard =
265  fields_value.at(field::phase_order_cahn_hilliard);
266 
267  return -4 *
268  (std::max(-1.0, std::min(phase_order_cahn_hilliard, 0.0)) +
269  std::min(1.0, std::max(phase_order_cahn_hilliard, 0.0))) *
270  mobility_cahn_hilliard_constant *
271  (1 -
272  std::min(1.0,
273  phase_order_cahn_hilliard * phase_order_cahn_hilliard));
274  }

286  void
287  vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
288  const field /*id*/,
289  std::vector<double> &jacobian_vector) override
290  {
291  AssertThrow(
292  field_vectors.find(field::phase_order_cahn_hilliard) !=
293  field_vectors.end(),
294  PhysicialPropertyModelFieldUndefined("MobilityCahnHilliardModelQuartic",
295  "phase_order_cahn_hilliard"));
296  const std::vector<double> &phase_order_cahn_hilliard =
297  field_vectors.at(field::phase_order_cahn_hilliard);
298  for (unsigned int i = 0; i < jacobian_vector.size(); ++i)
299  jacobian_vector[i] =
300  -4 *
301  (std::max(-1.0, std::min(phase_order_cahn_hilliard[i], 0.0)) +
302  std::min(1.0, std::max(phase_order_cahn_hilliard[i], 0.0))) *
303  mobility_cahn_hilliard_constant *
304  (1 -
305  std::min(1.0,
306  phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]));
307  }
308
