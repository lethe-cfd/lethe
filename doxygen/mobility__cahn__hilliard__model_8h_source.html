156 /**

159  * The mobility function is the following : \f$M(\phi) = D(1-\phi^2)^2 \f$

161  */
162 class MobilityCahnHilliardModelQuartic : public MobilityCahnHilliardModel
163 {
164 public:

193  /**
194  * @brief Calculate the mobility.
195  * @param[in] fields_value Value of the various fields on which the mobility
196  * may depend.
197  * @return Value of the mobility calculated with the fields_value.
198  */
199  double
200  value(const std::map<field, double> &fields_value) override
201  {
202  const double &phase_order_cahn_hilliard =
203  fields_value.at(field::phase_order_cahn_hilliard);
204 
205  // The phase order values are clamped to avoid unphysical mobilities in the
206  // bulk phases.
207  return mobility_cahn_hilliard_constant *
208  std::pow((1 - std::min(1.0,
209  phase_order_cahn_hilliard *
210  phase_order_cahn_hilliard)),
211  2);
212  }
213 
214  /**
215  * @brief Calculate the vector of mobility.
216  * @param[in] field_vectors Vectors of the fields on which the mobility
217  * may depend.
218  * @param[out] property_vector Vector of the mobility values.
219  */
220  void
221  vector_value(const std::map<field, std::vector<double>> &field_vectors,
222  std::vector<double> &property_vector) override
223  {
224  const std::vector<double> &phase_order_cahn_hilliard =
225  field_vectors.at(field::phase_order_cahn_hilliard);
226  for (unsigned int i = 0; i < property_vector.size(); ++i)
227  {
228  property_vector[i] =
229  mobility_cahn_hilliard_constant *
230  std::pow((1 - std::min(1.0,
231  phase_order_cahn_hilliard[i] *
232  phase_order_cahn_hilliard[i])),
233  2);
234  }
235  }
236 
237  /**
238  * @brief Calculate the jacobian (the partial derivative) of the
239  * mobility with respect to a field.
240  * @param[in] field_values Value of the various fields on which the mobility
241  * may depend.
242  * @param[in] id Indicator of the field with respect to which the Jacobian
243  * should be calculated.
244  * @return Value of the partial derivative of the mobility with respect to
245  * the field.
246  */
247 
248  double
249  jacobian(const std::map<field, double> &fields_value, field /*id*/) override
250  {
251  const double &phase_order_cahn_hilliard =
252  fields_value.at(field::phase_order_cahn_hilliard);
253 
254  return -4 *
255  (std::max(-1.0, std::min(phase_order_cahn_hilliard, 0.0)) +
256  std::min(1.0, std::max(phase_order_cahn_hilliard, 0.0))) *
257  mobility_cahn_hilliard_constant *
258  (1 -
259  std::min(1.0,
260  phase_order_cahn_hilliard * phase_order_cahn_hilliard));
261  }
262 
263  /**
264  * @brief Calculate the derivative of the mobility with respect to a field.
265  * @param[in] field_vectors Vector for the values of the fields used to
266  * evaluate the mobility.
267  * @param[in] id Identifier of the field with respect to which a derivative
268  * should be calculated.
269  * @param[out] jacobian_vector Vector of the values of the derivatives of the
270  * mobility with respect to the field id.
271  */
272 
273  void
274  vector_jacobian(const std::map<field, std::vector<double>> &field_vectors,
275  const field /*id*/,
276  std::vector<double> &jacobian_vector) override
277  {
278  const std::vector<double> &phase_order_cahn_hilliard =
279  field_vectors.at(field::phase_order_cahn_hilliard);
280  for (unsigned int i = 0; i < jacobian_vector.size(); ++i)
281  jacobian_vector[i] =
282  -4 *
283  (std::max(-1.0, std::min(phase_order_cahn_hilliard[i], 0.0)) +
284  std::min(1.0, std::max(phase_order_cahn_hilliard[i], 0.0))) *
285  mobility_cahn_hilliard_constant *
286  (1 -
287  std::min(1.0,
288  phase_order_cahn_hilliard[i] * phase_order_cahn_hilliard[i]));
289  }

292  const double mobility_cahn_hilliard_constant;
